391 void InitializeSolutionStep(
392 ModelPart& r_model_part,
393 TSystemMatrixType& A,
394 TSystemVectorType& Dx,
395 TSystemVectorType& b
396 ) override
397 {
398 ProcessInfo& CurrentProcessInfo = r_model_part.GetProcessInfo();
399
400 Scheme<TSparseSpace, TDenseSpace>::InitializeSolutionStep(r_model_part, A, Dx, b);
401
402 double DeltaTime = CurrentProcessInfo[DELTA_TIME];
403
404 if (DeltaTime == 0)
405 KRATOS_THROW_ERROR( std::logic_error, "detected delta_time = 0 in the Bossak Scheme ... check if the time step is created correctly for the current model part", "" )
406
407 //initializing constants
408 ma0 = 1.0 / (mBetaNewmark * pow(DeltaTime, 2));
409 ma1 = mGammaNewmark / (mBetaNewmark * DeltaTime);
410 ma2 = 1.0 / (mBetaNewmark * DeltaTime);
411 ma3 = 1.0 / (2.0 * mBetaNewmark) - 1.0;
412 ma4 = mGammaNewmark / mBetaNewmark - 1.0;
413 ma5 = DeltaTime * 0.5 * (mGammaNewmark / mBetaNewmark - 2.0);
414 mam = (1.0 - mAlphaBossak) / (mBetaNewmark * pow(DeltaTime, 2));
415 }

424 int Check(const ModelPart& r_model_part) const override
425 {
426 KRATOS_TRY
427
428 int err = Scheme<TSparseSpace, TDenseSpace>::Check(r_model_part);
429 if (err != 0) return err;
430
431 //check that variables are correctly allocated
432 for (const auto& r_node : r_model_part.Nodes())
433 {
434 if (r_node.SolutionStepsDataHas(DISPLACEMENT) == false)
435 KRATOS_THROW_ERROR( std::logic_error, "DISPLACEMENT variable is not allocated for node ", r_node.Id() )
436 if (r_node.SolutionStepsDataHas(VELOCITY) == false)
437 KRATOS_THROW_ERROR( std::logic_error, "DISPLACEMENT variable is not allocated for node ", r_node.Id() )
438 if (r_node.SolutionStepsDataHas(ACCELERATION) == false)
439 KRATOS_THROW_ERROR( std::logic_error, "DISPLACEMENT variable is not allocated for node ", r_node.Id() )
440 }
441
442 //check that dofs exist
443 for (const auto& r_node : r_model_part.Nodes())
444 {
445 if (r_node.HasDofFor(DISPLACEMENT_X) == false)
446 KRATOS_THROW_ERROR( std::invalid_argument, "missing DISPLACEMENT_X dof on node ", r_node.Id() )
447 if (r_node.HasDofFor(DISPLACEMENT_Y) == false)
448 KRATOS_THROW_ERROR( std::invalid_argument, "missing DISPLACEMENT_Y dof on node ", r_node.Id() )
449 if (r_node.HasDofFor(DISPLACEMENT_Z) == false)
450 KRATOS_THROW_ERROR( std::invalid_argument, "missing DISPLACEMENT_Z dof on node ", r_node.Id() )
451 }
452
453
454 //check for admissible value of the AlphaBossak
455 if (mAlphaBossak > 0.0 || mAlphaBossak < -0.3)
456 KRATOS_THROW_ERROR( std::logic_error, "Value not admissible for AlphaBossak. Admissible values should be between 0.0 and -0.3. Current value is ", mAlphaBossak )
457
458 //check for minimum value of the buffer index
459 //verify buffer size
460 if (r_model_part.GetBufferSize() < 2)
461 KRATOS_THROW_ERROR( std::logic_error, "insufficient buffer size. Buffer size should be greater than 2. Current size is", r_model_part.GetBufferSize() )
462
463
464 return 0;
465 KRATOS_CATCH( "" )
466 }
