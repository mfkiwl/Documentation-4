170 void SetUpSystem(
171 ModelPart& rModelPart
172 ) override
173 {
178 // We create a set of dofs of the displacement slave dofs with LM associated
179 std::unordered_map<IndexType, IndexSetType> set_nodes_with_lm_associated;
180 if (rModelPart.HasSubModelPart("Contact"))
181 set_nodes_with_lm_associated.reserve(rModelPart.GetSubModelPart("Contact").NumberOfNodes());
182 // Allocating auxiliary parameters
183 IndexType node_id;
184 // We start the dof loop
185 for (auto& i_dof : BaseType::mDofSet) {
186 node_id = i_dof.Id();
187 if (IsLMDof(i_dof))
188 set_nodes_with_lm_associated.insert({node_id, IndexSetType({})});
189 }
190
191 // Auxiliary keys
192 const IndexType key_lm_x = VECTOR_LAGRANGE_MULTIPLIER_X.Key();
193 const IndexType key_lm_y = VECTOR_LAGRANGE_MULTIPLIER_Y.Key();
194 const IndexType key_lm_z = VECTOR_LAGRANGE_MULTIPLIER_Z.Key();
195
196 // We see which LM block
197 for (auto& i_dof : BaseType::mDofSet) {
198 node_id = i_dof.Id();
199 auto it = set_nodes_with_lm_associated.find(node_id);
200 if ( it != set_nodes_with_lm_associated.end()) {
201 const auto& r_variable = i_dof.GetVariable();
202 auto& aux_set = (it->second);
203 if (i_dof.IsFixed()) {
204 if (r_variable == DISPLACEMENT_X) {
205 aux_set.insert(key_lm_x);
206 } else if (r_variable == DISPLACEMENT_Y) {
207 aux_set.insert(key_lm_y);
208 } else if (r_variable == DISPLACEMENT_Z) {
209 aux_set.insert(key_lm_z);
210 }
211 }
212 }
213 }
214
215 // We do now the loop over the dofs
216 for (auto& i_dof : BaseType::mDofSet) {
217 if (i_dof.IsFree()) {
218 node_id = i_dof.Id();
219 auto it = set_nodes_with_lm_associated.find(node_id);
220 if (it != set_nodes_with_lm_associated.end()) {
221 auto& aux_set = it->second;
222 if (aux_set.find((i_dof.GetVariable()).Key()) != aux_set.end()) {
223 i_dof.FixDof();
224 }
225 }
226 }
227 }
228
229 BaseType::SetUpSystem(rModelPart);
230 }
