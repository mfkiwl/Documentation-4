84template<class TVec>
85inline void ShapeFunc(double Xi, double Eta, TVec& rN)
86{
87 rN(0) = 0.25 * (1.0 - Xi) * (1.0 - Eta); // node 1
88 rN(1) = 0.25 * (1.0 + Xi) * (1.0 - Eta); // node 2
89 rN(2) = 0.25 * (1.0 + Xi) * (1.0 + Eta); // node 3
90 rN(3) = 0.25 * (1.0 - Xi) * (1.0 + Eta); // node 4
91}

93template<class TMat>
94inline void ShapeFunc_NaturalDerivatives(double Xi, const double Eta, TMat& rDN)
95{
96 rDN(0, 0) = -(1.0 - Eta) * 0.25;
97 rDN(1, 0) = (1.0 - Eta) * 0.25;
98 rDN(2, 0) = (1.0 + Eta) * 0.25;
99 rDN(3, 0) = -(1.0 + Eta) * 0.25;
100
101 rDN(0, 1) = -(1.0 - Xi) * 0.25;
102 rDN(1, 1) = -(1.0 + Xi) * 0.25;
103 rDN(2, 1) = (1.0 + Xi) * 0.25;
104 rDN(3, 1) = (1.0 - Xi) * 0.25;
105}
