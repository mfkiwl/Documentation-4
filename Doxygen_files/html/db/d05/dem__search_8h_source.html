53class PointDistance2
54{
55 public:
56 inline double operator()( T const& p1, T const& p2 )
57 {
58 double dist = 0.0;
59
60 double tmp1 = p1[0] - p2[0];
61 double tmp2 = p1[1] - p2[1];
62 double tmp3 = p1[2] - p2[2];
63
64 dist += tmp1*tmp1 + tmp2*tmp2 + tmp3*tmp3;
65
66// dist = sqrt(dist) - p2.radius - p1.radius;
67
68 return dist;
69 }
