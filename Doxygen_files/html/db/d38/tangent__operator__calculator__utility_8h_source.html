108 static void CalculateTangentTensor(
109 ConstitutiveLaw::Parameters& rValues,
110 ConstitutiveLaw* pConstitutiveLaw,
111 const ConstitutiveLaw::StressMeasure& rStressMeasure = ConstitutiveLaw::StressMeasure_Cauchy,
112 const bool ConsiderPertubationThreshold = true,
113 const IndexType ApproximationOrder = 2
114 )
115 {
116 // Ensure the proper flag
117 Flags& cl_options = rValues.GetOptions();
118 const bool use_element_provided_strain = cl_options.Is(ConstitutiveLaw::USE_ELEMENT_PROVIDED_STRAIN);
119
120 if (use_element_provided_strain) {
121 CalculateTangentTensorSmallDeformationProvidedStrain(rValues, pConstitutiveLaw, rStressMeasure, ConsiderPertubationThreshold, ApproximationOrder);
122 } else {
123 CalculateTangentTensorSmallDeformationNotProvidedStrain(rValues, pConstitutiveLaw, rStressMeasure, ConsiderPertubationThreshold, ApproximationOrder);
124 }

133 static void CalculateTangentTensorSmallDeformationProvidedStrain(
134 ConstitutiveLaw::Parameters& rValues,
135 ConstitutiveLaw *pConstitutiveLaw,
136 const ConstitutiveLaw::StressMeasure& rStressMeasure = ConstitutiveLaw::StressMeasure_Cauchy,
137 const bool ConsiderPertubationThreshold = true,
138 const IndexType ApproximationOrder = 2
139 )
140 {
141 // Converged values to be storaged
142 const Vector unperturbed_strain_vector_gp = Vector(rValues.GetStrainVector());
143 const Vector unperturbed_stress_vector_gp = Vector(rValues.GetStressVector());
144 const auto &r_properties = rValues.GetMaterialProperties();
145 const bool symmetrize_operator = (r_properties.Has(SYMMETRIZE_TANGENT_OPERATOR)) ? r_properties[SYMMETRIZE_TANGENT_OPERATOR] : false;
146
147 // The number of components
148 const SizeType num_components = unperturbed_stress_vector_gp.size();
149
150 // The tangent tensor
151 Matrix& r_tangent_tensor = rValues.GetConstitutiveMatrix();
152 r_tangent_tensor.clear();
153 Matrix auxiliary_tensor = ZeroMatrix(num_components,num_components);
154
155 // Calculate the perturbation
156 double pertubation = PerturbationThreshold;
157 if (r_properties.Has(PERTURBATION_SIZE)) {
158 pertubation = r_properties[PERTURBATION_SIZE];
159 if (pertubation == -1.0)
160 pertubation = std::sqrt(tolerance);
161 } else {
162 for (IndexType i_component = 0; i_component < num_components; ++i_component) {
163 double component_perturbation;
164 CalculatePerturbation(unperturbed_strain_vector_gp, i_component, component_perturbation);
165 pertubation = std::max(component_perturbation, pertubation);
166 }
167 // We check that the perturbation has a threshold value of PerturbationThreshold
168 if (ConsiderPertubationThreshold && pertubation < PerturbationThreshold) pertubation = PerturbationThreshold;
169 }
170
171 // Loop over components of the strain
172 Vector& r_perturbed_strain = rValues.GetStrainVector();
173 Vector& r_perturbed_integrated_stress = rValues.GetStressVector();
174 if (ApproximationOrder == 1) {
175 for (IndexType i_component = 0; i_component < num_components; ++i_component) {
176
177 // Apply the perturbation
178 PerturbateStrainVector(r_perturbed_strain, unperturbed_strain_vector_gp, pertubation, i_component);
179
180 // We continue with the calculations
181 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
182
183 // Compute tangent moduli
184 const Vector delta_stress = r_perturbed_integrated_stress - unperturbed_stress_vector_gp;
185 CalculateComponentsToTangentTensorFirstOrder(auxiliary_tensor, r_perturbed_strain-unperturbed_strain_vector_gp, delta_stress, i_component);
186
187 // Reset the values to the initial ones
188 noalias(r_perturbed_strain) = unperturbed_strain_vector_gp;
189 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
190 }
191 } else if (ApproximationOrder == 2) {
192 for (IndexType i_component = 0; i_component < num_components; ++i_component) {
193 // Apply the perturbation (positive)
194 PerturbateStrainVector(r_perturbed_strain, unperturbed_strain_vector_gp, pertubation, i_component);
195
196 // We continue with the calculations
197 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
198
199 // Compute stress (plus)
200 const Vector strain_plus = r_perturbed_strain;
201 const Vector stress_plus = r_perturbed_integrated_stress;
202
203 // Reset the values to the initial ones
204 noalias(r_perturbed_strain) = unperturbed_strain_vector_gp;
205 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
206
207 // Apply the perturbation (negative)
208 PerturbateStrainVector(r_perturbed_strain, unperturbed_strain_vector_gp, - pertubation, i_component);
209
210 // We continue with the calculations
211 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
212
213 // Compute stress (minus)
214 const Vector strain_minus = r_perturbed_strain;
215 const Vector stress_minus = r_perturbed_integrated_stress;
216
217 // Finally we compute the components
218 CalculateComponentsToTangentTensorSecondOrder(auxiliary_tensor, strain_plus, strain_minus, stress_plus, stress_minus, i_component);
219
220 // Reset the values to the initial ones
221 noalias(r_perturbed_strain) = unperturbed_strain_vector_gp;
222 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
223 }
224 } else if (ApproximationOrder == 4) { // Second order but with another approach. It is computed as first order but taking into account the second derivatives
225 for (IndexType i_component = 0; i_component < num_components; ++i_component) {
226 // Apply the perturbation (positive)
227 PerturbateStrainVector(r_perturbed_strain, unperturbed_strain_vector_gp, pertubation, i_component);
228
229 // We continue with the calculations
230 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
231
232 // Compute stress (plus)
233 const Vector strain_plus = r_perturbed_strain;
234 const Vector stress_plus = r_perturbed_integrated_stress;
235
236 // Reset the values to the initial ones
237 noalias(r_perturbed_strain) = unperturbed_strain_vector_gp;
238 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
239
240 // Apply the perturbation twice
241 PerturbateStrainVector(r_perturbed_strain, unperturbed_strain_vector_gp, 2.0*pertubation, i_component);
242
243 // We continue with the calculations
244 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
245
246 // Compute stress (minus)
247 const Vector strain_2_plus = r_perturbed_strain;
248 const Vector stress_2_plus = r_perturbed_integrated_stress;
249
250 // Finally we compute the components
251 const SizeType voigt_size = stress_plus.size();
252 for (IndexType row = 0; row < voigt_size; ++row) {
253 auxiliary_tensor(row, i_component) = (stress_plus[row] - unperturbed_stress_vector_gp[row]) / pertubation - (stress_2_plus[row] - 2.0 * stress_plus[row] + unperturbed_stress_vector_gp[row]) / (2.0 * pertubation);
254 }
255
256 // Reset the values to the initial ones
257 noalias(r_perturbed_strain) = unperturbed_strain_vector_gp;
258 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
259 }
260 }
261 if (symmetrize_operator)
262 noalias(r_tangent_tensor) = 0.5*(auxiliary_tensor + trans(auxiliary_tensor));
263 else
264 noalias(r_tangent_tensor) = auxiliary_tensor;
265 }

273 static void CalculateTangentTensorSmallDeformationNotProvidedStrain(
274 ConstitutiveLaw::Parameters& rValues,
275 ConstitutiveLaw *pConstitutiveLaw,
276 const ConstitutiveLaw::StressMeasure& rStressMeasure = ConstitutiveLaw::StressMeasure_Cauchy,
277 const bool ConsiderPertubationThreshold = true,
278 const IndexType ApproximationOrder = 2
279 )
280 {
281 // Converged values to be storaged
282 const Vector unperturbed_strain_vector_gp = Vector(rValues.GetStrainVector());
283 const Vector unperturbed_stress_vector_gp = Vector(rValues.GetStressVector());
284
285 const auto &r_properties = rValues.GetMaterialProperties();
286 const bool symmetrize_operator = (r_properties.Has(SYMMETRIZE_TANGENT_OPERATOR)) ? r_properties[SYMMETRIZE_TANGENT_OPERATOR] : false;
287
288 // The number of components
289 const SizeType num_components = unperturbed_stress_vector_gp.size();
290
291 // Converged values to be storaged (only used in case of elements that not provide the strain)
292 const Matrix unperturbed_deformation_gradient_gp = Matrix(rValues.GetDeformationGradientF());
293 const double det_unperturbed_deformation_gradient_gp = double(rValues.GetDeterminantF());
294
295 // The tangent tensor
296 Matrix& r_tangent_tensor = rValues.GetConstitutiveMatrix();
297 r_tangent_tensor.clear();
298 Matrix auxiliary_tensor = ZeroMatrix(num_components,num_components);
299
300 const std::size_t size1 = unperturbed_deformation_gradient_gp.size1();
301 const std::size_t size2 = unperturbed_deformation_gradient_gp.size2();
302
303 KRATOS_ERROR_IF_NOT(ApproximationOrder == 1 || ApproximationOrder == 2) << "The approximation order for the perturbation is " << ApproximationOrder << ". Options are 1 and 2" << std::endl;
304
305 // Calculate the perturbation
306 double pertubation = PerturbationThreshold;
307 if (r_properties.Has(PERTURBATION_SIZE)) {
308 pertubation = r_properties[PERTURBATION_SIZE];
309 } else {
310 for (IndexType i_component = 0; i_component < num_components; ++i_component) {
311 double component_perturbation;
312 CalculatePerturbation(unperturbed_strain_vector_gp, i_component, component_perturbation);
313 pertubation = std::max(component_perturbation, pertubation);
314 }
315 // We check that the perturbation has a threshold value of PerturbationThreshold
316 if (ConsiderPertubationThreshold && pertubation < PerturbationThreshold) pertubation = PerturbationThreshold;
317 }
318
319 // Loop over components of the strain
320 Vector& r_perturbed_strain = rValues.GetStrainVector();
321 Vector& r_perturbed_integrated_stress = rValues.GetStressVector();
322 Matrix& r_perturbed_deformation_gradient = const_cast<Matrix&>(rValues.GetDeformationGradientF());
323 double& r_perturbed_det_deformation_gradient = const_cast<double&>(rValues.GetDeterminantF());
324
325 if (ApproximationOrder == 1) {
326 for (IndexType i_component = 0; i_component < size1; ++i_component) {
327 for (IndexType j_component = i_component; j_component < size2; ++j_component) {
328 // Apply the perturbation
329 PerturbateDeformationGradient(r_perturbed_deformation_gradient, unperturbed_deformation_gradient_gp, pertubation, i_component, j_component);
330
331 // We continue with the calculations
332 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
333
334 // Compute delta stress
335 const Vector delta_stress = r_perturbed_integrated_stress - unperturbed_stress_vector_gp;
336
337 // Finally we compute the components
338 const IndexType voigt_index = CalculateVoigtIndex(delta_stress.size(), i_component, j_component);
339 CalculateComponentsToTangentTensorFirstOrder(auxiliary_tensor, r_perturbed_strain-unperturbed_strain_vector_gp, delta_stress, voigt_index);
340
341 // Reset the values to the initial ones
342 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
343 noalias(r_perturbed_deformation_gradient) = unperturbed_deformation_gradient_gp;
344 r_perturbed_det_deformation_gradient = det_unperturbed_deformation_gradient_gp;
345 }
346 }
347 } else if (ApproximationOrder == 2) {
348 for (IndexType i_component = 0; i_component < size1; ++i_component) {
349 for (IndexType j_component = i_component; j_component < size2; ++j_component) {
350 // Apply the perturbation (positive)
351 PerturbateDeformationGradient(r_perturbed_deformation_gradient, unperturbed_deformation_gradient_gp, pertubation, i_component, j_component);
352
353 // We continue with the calculations
354 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
355
356 // Compute stress (plus)
357 const Vector strain_plus = r_perturbed_strain;
358 const Vector stress_plus = r_perturbed_integrated_stress;
359
360 // Reset the values to the initial ones
361 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
362 noalias(r_perturbed_deformation_gradient) = unperturbed_deformation_gradient_gp;
363 r_perturbed_det_deformation_gradient = det_unperturbed_deformation_gradient_gp;
364
365 // Apply the perturbation (negative)
366 PerturbateDeformationGradient(r_perturbed_deformation_gradient, unperturbed_deformation_gradient_gp, - pertubation, i_component, j_component);
367
368 // We continue with the calculations
369 IntegratePerturbedStrain(rValues, pConstitutiveLaw, rStressMeasure);
370
371 // Compute stress (minus)
372 const Vector strain_minus = r_perturbed_strain;
373 const Vector stress_minus = r_perturbed_integrated_stress;
374
375 // Finally we compute the components
376 const IndexType voigt_index = CalculateVoigtIndex(stress_plus.size(), i_component, j_component);
377 CalculateComponentsToTangentTensorSecondOrder(auxiliary_tensor, strain_plus, strain_minus, stress_plus, stress_minus, voigt_index);
378
379 // Reset the values to the initial ones
380 noalias(r_perturbed_integrated_stress) = unperturbed_stress_vector_gp;
381 noalias(r_perturbed_deformation_gradient) = unperturbed_deformation_gradient_gp;
382 r_perturbed_det_deformation_gradient = det_unperturbed_deformation_gradient_gp;
383 }
384 }
385 }
386 if (symmetrize_operator)
387 noalias(r_tangent_tensor) = 0.5*(auxiliary_tensor + trans(auxiliary_tensor));
388 else
389 noalias(r_tangent_tensor) = auxiliary_tensor;
390 }

460 static void CalculatePerturbation(
461 const Vector& rStrainVector,
462 const IndexType Component,
463 double& rPerturbation
464 )
465 {
466 double perturbation_1, perturbation_2;
467 if (std::abs(rStrainVector[Component]) > tolerance) {
468 perturbation_1 = PerturbationCoefficient1 * rStrainVector[Component];
469 } else {
470 double min_strain_component;
471 GetMinAbsValue(rStrainVector, min_strain_component);
472 perturbation_1 = PerturbationCoefficient1 * min_strain_component;
473 }
474 double max_strain_component;
475 GetMaxAbsValue(rStrainVector, max_strain_component);
476 perturbation_2 = PerturbationCoefficient2 * max_strain_component;
477 rPerturbation = std::max(perturbation_1, perturbation_2);
478 }

487 static void CalculatePerturbationFiniteDeformation(
488 const Matrix& rDeformationGradient,
489 const IndexType ComponentI,
490 const IndexType ComponentJ,
491 double& rPerturbation
492 )
493 {
494 double perturbation_1, perturbation_2;
495 if (std::abs(rDeformationGradient(ComponentI, ComponentJ)) > tolerance) {
496 perturbation_1 = PerturbationCoefficient1 * rDeformationGradient(ComponentI, ComponentJ);
497 } else {
498 double min_strain_component;
499 GetMinAbsValue(rDeformationGradient, min_strain_component);
500 perturbation_1 = PerturbationCoefficient1 * min_strain_component;
501 }
502 double max_strain_component;
503 GetMaxAbsValue(rDeformationGradient, max_strain_component);
504 perturbation_2 = PerturbationCoefficient2 * max_strain_component;
505 rPerturbation = std::max(perturbation_1, perturbation_2);
506 }

534 static void PerturbateDeformationGradient(
535 Matrix& rPerturbedDeformationGradient,
536 const Matrix& rDeformationGradientGP,
537 const double Perturbation,
538 const IndexType ComponentI,
539 const IndexType ComponentJ
540 )
541 {
542 Matrix aux_perturbation_matrix = IdentityMatrix(rDeformationGradientGP.size1());
543 if (ComponentI == ComponentJ) {
544 aux_perturbation_matrix(ComponentI, ComponentJ) += Perturbation;
545 } else {
546 aux_perturbation_matrix(ComponentI, ComponentJ) += 0.5 * Perturbation;
547 aux_perturbation_matrix(ComponentJ, ComponentI) += 0.5 * Perturbation;
548 }
549 noalias(rPerturbedDeformationGradient) = prod(aux_perturbation_matrix, rDeformationGradientGP);
550 }

558 static void IntegratePerturbedStrain(
559 ConstitutiveLaw::Parameters& rValues,
560 ConstitutiveLaw* pConstitutiveLaw,
561 const ConstitutiveLaw::StressMeasure& rStressMeasure = ConstitutiveLaw::StressMeasure_Cauchy
562 )
563 {
564 Flags& cl_options = rValues.GetOptions();
565
566 // In order to avoid recursivity...
567 const bool flag_back_up_1 = cl_options.Is(ConstitutiveLaw::COMPUTE_CONSTITUTIVE_TENSOR);
568 const bool flag_back_up_2 = cl_options.Is(ConstitutiveLaw::COMPUTE_STRESS);
569
570 cl_options.Set(ConstitutiveLaw::COMPUTE_CONSTITUTIVE_TENSOR, false);
571 cl_options.Set(ConstitutiveLaw::COMPUTE_STRESS, true);
572
573 pConstitutiveLaw->CalculateMaterialResponse(rValues, rStressMeasure);
574
575 cl_options.Set(ConstitutiveLaw::COMPUTE_CONSTITUTIVE_TENSOR, flag_back_up_1);
576 cl_options.Set(ConstitutiveLaw::COMPUTE_STRESS, flag_back_up_2);
577 }

584 static void GetMaxAbsValue(
585 const Vector& rArrayValues,
586 double& rMaxValue
587 )
588 {
589 const SizeType dimension = rArrayValues.size();
590
591 IndexType counter = 0;
592 double aux = 0.0;
593 for (IndexType i = 0; i < dimension; ++i) {
594 if (std::abs(rArrayValues[i]) > aux) {
595 aux = std::abs(rArrayValues[i]);
596 ++counter;
597 }
598 }
599
600 rMaxValue = aux;
601 }

611 )

615 const SizeType size2 = rMatrixValues.size2();
616

620 IndexType counter = 0;

622 for (IndexType i = 0; i < size1; ++i) {

696 static void CalculateComponentsToTangentTensorFirstOrder(
697 Matrix& rTangentTensor,
698 const Vector& rVectorStrain,
699 const Vector& rDeltaStress,
700 const IndexType Component
701 )
702 {
703 const double perturbation = rVectorStrain[Component];
704 const SizeType voigt_size = rDeltaStress.size();
705 for (IndexType row = 0; row < voigt_size; ++row) {
706 rTangentTensor(row, Component) = rDeltaStress[row] / perturbation;

723 const Vector& rStressPlus,
724 const Vector& rStressMinus,
