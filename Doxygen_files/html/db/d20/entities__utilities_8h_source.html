77 template<class TEntityType>
78 void InitializeEntities(ModelPart& rModelPart)
79 {
80 KRATOS_TRY
81
82 // Array of entities
83 auto& r_entities_array = GetEntities<TEntityType>(rModelPart);
84
85 // The current process info
86 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
87
88 // Initialize
89 block_for_each(
90 r_entities_array,
91 [&r_current_process_info](TEntityType& rEntity)
92 {
93 // Detect if the entity is active or not. If the user did not make any choice the entity
94 // It is active by default
95 const bool entity_is_active = (rEntity.IsDefined(ACTIVE)) ? rEntity.Is(ACTIVE) : true;
96 if (entity_is_active) {
97 rEntity.Initialize(r_current_process_info);
98 }
99 }
100 );
101
102 KRATOS_CATCH("")
103 }
