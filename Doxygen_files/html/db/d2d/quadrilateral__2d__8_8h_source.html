979 *( 1.0 - rPoint[0] + rPoint[1] ) ) / 4.0;

984 return -(( 1.0 - rPoint[0] )*( 1.0 + rPoint[1] )
985 *( 1.0 + rPoint[0] - rPoint[1] ) ) / 4.0;
