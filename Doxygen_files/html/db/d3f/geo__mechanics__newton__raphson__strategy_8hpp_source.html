157 double CalculateReferenceDofsNorm(DofsArrayType& rDofSet)
158 {
159 double ReferenceDofsNorm = 0.0;
160
161 int NumThreads = ParallelUtilities::GetNumThreads();
162 OpenMPUtils::PartitionVector DofSetPartition;
163 OpenMPUtils::DivideInPartitions(rDofSet.size(), NumThreads, DofSetPartition);
164
165 #pragma omp parallel reduction(+:ReferenceDofsNorm)
166 {
167 int k = OpenMPUtils::ThisThread();
168
169 typename DofsArrayType::iterator DofsBegin = rDofSet.begin() + DofSetPartition[k];
170 typename DofsArrayType::iterator DofsEnd = rDofSet.begin() + DofSetPartition[k+1];
171
172 for (typename DofsArrayType::iterator itDof = DofsBegin; itDof != DofsEnd; ++itDof)
173 {
174 if (itDof->IsFree())
175 {
176 const double& temp = itDof->GetSolutionStepValue();
177 ReferenceDofsNorm += temp*temp;
178 }
179 }
180 }
181
182 return sqrt(ReferenceDofsNorm);
183 }
