134 ResidualBasedBossakScheme(double rAlpham=0,double rDynamic=1)
135 :Scheme<TSparseSpace,TDenseSpace>()
136 {
137 //For pure Newmark Scheme
138 mAlpha.f= 0;
139 mAlpha.m= rAlpham;
140
141 mNewmark.beta= (1.0+mAlpha.f-mAlpha.m)*(1.0+mAlpha.f-mAlpha.m)*0.25;
142 mNewmark.gamma= 0.5+mAlpha.f-mAlpha.m;
143
144 mNewmark.static_dynamic= rDynamic;
145
146 //std::cout << " MECHANICAL SCHEME: The Bossak Time Integration Scheme [alpha_m= "<<mAlpha.m<<" beta= "<<mNewmark.beta<<" gamma= "<<mNewmark.gamma<<"]"<<std::endl;

208 void Update(
209 ModelPart& r_model_part,
210 DofsArrayType& rDofSet,
211 TSystemMatrixType& A,
212 TSystemVectorType& Dx,
213 TSystemVectorType& b ) override
214 {
215 KRATOS_TRY
216
217 //std::cout << " Update " << std::endl;
218 //update of displacement (by DOF)
219 for (typename DofsArrayType::iterator i_dof = rDofSet.begin(); i_dof != rDofSet.end(); ++i_dof)
220 {
221 if (i_dof->IsFree() )
222 {
223 i_dof->GetSolutionStepValue() += Dx[i_dof->EquationId()];
224 }
225 }
226
227 //updating time derivatives (nodally for efficiency)
228 array_1d<double, 3 > DeltaDisplacement;
229 for (ModelPart::NodeIterator i = r_model_part.NodesBegin();
230 i != r_model_part.NodesEnd(); ++i)
231 {
232
233 noalias(DeltaDisplacement) = (i)->FastGetSolutionStepValue(DISPLACEMENT) - (i)->FastGetSolutionStepValue(DISPLACEMENT, 1);
234
235
236 array_1d<double, 3 > & CurrentVelocity = (i)->FastGetSolutionStepValue(VELOCITY, 0);
237 array_1d<double, 3 > & PreviousVelocity = (i)->FastGetSolutionStepValue(VELOCITY, 1);
238
239 array_1d<double, 3 > & CurrentAcceleration = (i)->FastGetSolutionStepValue(ACCELERATION, 0);
240 array_1d<double, 3 > & PreviousAcceleration = (i)->FastGetSolutionStepValue(ACCELERATION, 1);
241
242 UpdateVelocity (CurrentVelocity, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
243
244 UpdateAcceleration (CurrentAcceleration, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
245
246 }
247
248 KRATOS_CATCH( "" )
249 }

255 //predicts the solution for the current step:
256 // x = xold + vold * Dt
257
258 void Predict(
259 ModelPart& r_model_part,
260 DofsArrayType& rDofSet,
261 TSystemMatrixType& A,
262 TSystemVectorType& Dx,
263 TSystemVectorType& b
264 ) override
265 {
266 //std::cout << " Prediction " << std::endl;
267 array_1d<double, 3 > DeltaDisplacement;
268
269 //double DeltaTime = r_model_part.GetProcessInfo()[DELTA_TIME];
270
271
272 for (ModelPart::NodeIterator i = r_model_part.NodesBegin();
273 i != r_model_part.NodesEnd(); ++i)
274 {
275 //predicting displacement = PreviousDisplacement + PreviousVelocity * DeltaTime;
276 //ATTENTION::: the prediction is performed only on free nodes
277
278 array_1d<double, 3 > & PreviousVelocity = (i)->FastGetSolutionStepValue(VELOCITY, 1);
279 array_1d<double, 3 > & PreviousDisplacement = (i)->FastGetSolutionStepValue(DISPLACEMENT, 1);
280 array_1d<double, 3 > & CurrentDisplacement = (i)->FastGetSolutionStepValue(DISPLACEMENT);
281 array_1d<double, 3 > & ImposedDisplacement = (i)->FastGetSolutionStepValue(IMPOSED_DISPLACEMENT);
282
283
284 if ((i->pGetDof(DISPLACEMENT_X))->IsFixed() == false)
285 {
286 CurrentDisplacement[0] = PreviousDisplacement[0];// + DeltaTime * PreviousVelocity[0];
287 }
288 else
289 {
290 CurrentDisplacement[0] = PreviousDisplacement[0] + ImposedDisplacement[0];//to impose fixed displacements;
291 //PreviousDisplacement[0] = 0;
292 }
293
294 if (i->pGetDof(DISPLACEMENT_Y)->IsFixed() == false)
295 {
296 CurrentDisplacement[1] = PreviousDisplacement[1]; //+ DeltaTime * PreviousVelocity[1];
297 }
298 else
299 {
300 CurrentDisplacement[1] = PreviousDisplacement[1] + ImposedDisplacement[1];//to impose fixed displacements;
301 //PreviousDisplacement[1] = 0;
302 }
303
304
305 if (i->HasDofFor(DISPLACEMENT_Z))
306 {
307 if (i->pGetDof(DISPLACEMENT_Z)->IsFixed() == false)
308 {
309 CurrentDisplacement[2] = PreviousDisplacement[2]; // + DeltaTime * PreviousVelocity[2];
310 }
311 else
312 {
313 CurrentDisplacement[2] = PreviousDisplacement[2] + ImposedDisplacement[2];//to impose fixed displacements;
314 //PreviousDisplacement[2] = 0;
315 }
316 }
317
318
319 // std::cout<<" DispPre "<<PreviousDisplacement<<" ID "<<i->Id()<<std::endl;
320 // std::cout<<" DispCur "<<CurrentDisplacement<<" ID "<<i->Id()<<std::endl;
321
322 if (i->HasDofFor(PRESSURE))
323 {
324 double& PreviousPressure = (i)->FastGetSolutionStepValue(PRESSURE, 1);
325 double& CurrentPressure = (i)->FastGetSolutionStepValue(PRESSURE);
326
327 if ((i->pGetDof(PRESSURE))->IsFixed() == false)
328 CurrentPressure = PreviousPressure;
329
330 //std::cout<<" PressureCur [1] "<<CurrentPressure<<" PressurePre [1] "<<PreviousPressure<<" ID "<<i->Id()<<std::endl;
331 }
332
333
334
335 //updating time derivatives ::: please note that displacements and its time derivatives
336 //can not be consistently fixed separately
337
338 noalias(DeltaDisplacement) = CurrentDisplacement - PreviousDisplacement;
339
340 array_1d<double, 3 > & PreviousAcceleration = (i)->FastGetSolutionStepValue(ACCELERATION, 1);
341 array_1d<double, 3 > & CurrentVelocity = (i)->FastGetSolutionStepValue(VELOCITY);
342 array_1d<double, 3 > & CurrentAcceleration = (i)->FastGetSolutionStepValue(ACCELERATION);
343
344 UpdateVelocity (CurrentVelocity, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
345
346 UpdateAcceleration (CurrentAcceleration, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
347
348 }

433 void InitializeSolutionStep(
434 ModelPart& r_model_part,
435 TSystemMatrixType& A,
436 TSystemVectorType& Dx,
437 TSystemVectorType& b) override
438 {
439 KRATOS_TRY
440
441 ProcessInfo CurrentProcessInfo= r_model_part.GetProcessInfo();
442
443
444 Scheme<TSparseSpace,TDenseSpace>::InitializeSolutionStep(r_model_part,A,Dx,b);
445
446
447 double DeltaTime = CurrentProcessInfo[DELTA_TIME];
448
449 //if (DeltaTime == 0)
450 //KRATOS_THROW_ERROR( std::logic_error, "detected delta_time = 0 in the Solution Scheme ... check if the time step is created correctly for the current model part", "" )
451
452 if (DeltaTime == 0){
453 std::cout<<" WARNING: detected delta_time = 0 in the Solution Scheme "<<std::endl;
454 std::cout<<" DELTA_TIME set to 1 considering a Quasistatic step with one step only "<<std::endl;
455 std::cout<<" PLEASE : check if the time step is created correctly for the current model part "<<std::endl;
456
457 CurrentProcessInfo[DELTA_TIME] = 1;
458 DeltaTime = CurrentProcessInfo[DELTA_TIME];
459 }
460
461
462 //initializing Newmark constants
463 mNewmark.c0 = ( 1.0 / (mNewmark.beta * DeltaTime * DeltaTime) );
464 mNewmark.c1 = ( mNewmark.gamma / (mNewmark.beta * DeltaTime) );
465 mNewmark.c2 = ( 1.0 / (mNewmark.beta * DeltaTime) );
466 mNewmark.c3 = ( 0.5 / (mNewmark.beta) - 1.0 );
467 mNewmark.c4 = ( (mNewmark.gamma / mNewmark.beta) - 1.0 );
468 mNewmark.c5 = ( DeltaTime * 0.5 * ( ( mNewmark.gamma / mNewmark.beta ) - 2 ) );
469
470
471 //std::cout<<" Newmark Variables "<<mNewmark.c0<<" "<<mNewmark.c1<<" "<<mNewmark.c2<<" "<<mNewmark.c3<<" "<<mNewmark.c4<<" "<<mNewmark.c5<<std::endl;
472
473 KRATOS_CATCH( "" )
474 }

783
791 int Check(ModelPart& r_model_part) override
792 {
793 KRATOS_TRY
794
795 int err = Scheme<TSparseSpace, TDenseSpace>::Check(r_model_part);
796 if(err!=0) return err;
797
798 //check for variables keys
799 //verify that the variables are correctly initialized
800 if(DISPLACEMENT.Key() == 0)
801 KRATOS_THROW_ERROR( std::invalid_argument,"DISPLACEMENT has Key zero! (check if the application is correctly registered", "" )
802 if(VELOCITY.Key() == 0)
803 KRATOS_THROW_ERROR( std::invalid_argument,"VELOCITY has Key zero! (check if the application is correctly registered", "" )
804 if(ACCELERATION.Key() == 0)
805 KRATOS_THROW_ERROR( std::invalid_argument,"ACCELERATION has Key zero! (check if the application is correctly registered", "" )
806
807 //check that variables are correctly allocated
808 for(ModelPart::NodesContainerType::iterator it=r_model_part.NodesBegin();
809 it!=r_model_part.NodesEnd(); it++)
810 {
811 if (it->SolutionStepsDataHas(DISPLACEMENT) == false)
812 KRATOS_THROW_ERROR( std::logic_error, "DISPLACEMENT variable is not allocated for node ", it->Id() )
813 if (it->SolutionStepsDataHas(VELOCITY) == false)
814 KRATOS_THROW_ERROR( std::logic_error, "DISPLACEMENT variable is not allocated for node ", it->Id() )
815 if (it->SolutionStepsDataHas(ACCELERATION) == false)
816 KRATOS_THROW_ERROR( std::logic_error, "DISPLACEMENT variable is not allocated for node ", it->Id() )
817 }
818
819 //check that dofs exist
820 for(ModelPart::NodesContainerType::iterator it=r_model_part.NodesBegin();
821 it!=r_model_part.NodesEnd(); it++)
822 {
823 if(it->HasDofFor(DISPLACEMENT_X) == false)
824 KRATOS_THROW_ERROR( std::invalid_argument,"missing DISPLACEMENT_X dof on node ",it->Id() )
825 if(it->HasDofFor(DISPLACEMENT_Y) == false)
