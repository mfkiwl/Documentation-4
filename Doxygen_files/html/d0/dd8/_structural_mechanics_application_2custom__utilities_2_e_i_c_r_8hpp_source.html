219 inline static MatrixType Compute_Pt(const SizeType NumNodes)
220 {
221 const RealType a = RealType(NumNodes - 1) / RealType(NumNodes);
222 const RealType b = -1.0 / RealType(NumNodes);
223
224 const SizeType num_dofs = NumNodes * 6;
225
226 MatrixType P(IdentityMatrix(num_dofs, num_dofs));
227
228 for (SizeType i = 0; i < NumNodes; i++) {
229 const SizeType j = i * 6;
230
231 // diagonal block
232 P(j , j) = a;
233 P(j + 1, j + 1) = a;
234 P(j + 2, j + 2) = a;
235
236 // out-of-diagonal block
237 for (SizeType k = i + 1; k < NumNodes; k++) {
238 const SizeType w = k * 6;
239
240 P(j , w) = b;
241 P(j + 1, w + 1) = b;
242 P(j + 2, w + 2) = b;
243
244 P(w , j) = b;
245 P(w + 1, j + 1) = b;
246 P(w + 2, j + 2) = b;
247 }
248 }
249
250 return P;
251 }

263 const SizeType num_dofs = num_nodes * 6;

274 S(j + 5, 2) = 1.0;

287 inline static MatrixType Compute_H(const VectorType& rDisplacements)
288 {
289 const SizeType num_dofs = rDisplacements.size();
290 const SizeType num_nodes = num_dofs / 6;
291
292 MatrixType H(IdentityMatrix(num_dofs, num_dofs));
293
294 BoundedMatrixType3x3 omega(3, 3);
295 BoundedMatrixType3x3 Hi(3, 3);
296
297 for (SizeType i = 0; i < num_nodes; i++) {
298 const SizeType index = i * 6;
299 Vector3Type rv = project(rDisplacements, range(index + 3, index + 6));
300
301 double angle = norm_2(rv);
302
303 if (angle >= 2.0 * Globals::Pi) {
304 angle = std::fmod(angle, 2.0 * Globals::Pi);
305 }
306
307 double eta;
308 if (angle < 0.05) {
309 double angle2 = angle * angle;
310 double angle4 = angle2 * angle2;
311 double angle6 = angle4 * angle2;
312 eta = 1.0 / 12.0 + 1.0 / 270.0 * angle2 + 1.0 / 30240.0 * angle4 + 1.0 / 1209600.0 * angle6;
313 } else {
314 eta = (1.0 - 0.5 * angle * std::tan(0.5 * Globals::Pi - 0.5 * angle)) / (angle * angle);
315 }
316
317 Spin(rv, omega);
318
319 noalias(Hi) = IdentityMatrix(3, 3);
320 noalias(Hi) -= 0.5 * omega;
321 noalias(Hi) += eta * prod(omega, omega);
322
323 range i_range(index + 3, index + 6);
324 project(H, i_range, i_range) = Hi;
325 }
326
327 return H;
328 }

339 inline static MatrixType Compute_L(const VectorType& rDisplacements,
340 const VectorType& rForces,
341 const MatrixType& rH)
342 {
343 const SizeType num_dofs = rDisplacements.size();
344 const SizeType num_nodes = num_dofs / 6;
345
346 MatrixType L(num_dofs, num_dofs, 0.0);
347
348 Vector3Type rotation_vector;
349 Vector3Type moment_vector;
350 BoundedMatrixType3x3 omega(3, 3);
351 BoundedMatrixType3x3 omega_2(3, 3);
352 BoundedMatrixType3x3 Li(3, 3);
353 BoundedMatrixType3x3 LiTemp1(3, 3);
354 BoundedMatrixType3x3 LiTemp2(3, 3);
355
356 for (SizeType i = 0; i < num_nodes; i++) {
357 const SizeType index = i * 6;
358 range i_range(index + 3, index + 6);
359 noalias(rotation_vector) = project(rDisplacements, i_range);
360 noalias(moment_vector) = project(rForces, i_range);
361
362 double angle = norm_2(rotation_vector);
363
364 if (angle >= 2.0 * Globals::Pi) {
365 angle = std::fmod(angle, 2.0 * Globals::Pi);
366 }
367
368 const double angle2 = angle * angle;
369 const double angle4 = angle2 * angle2;
370 const double angle6 = angle4 * angle2;
371
372 double eta;
373 double mu;
374 if (angle < 0.05) {
375 eta = 1.0 / 12.0 + angle2 / 270.0 + angle4 / 30240.0 + angle6 / 1209600.0;
376 mu = 1.0 / 360.0 + angle2 / 7560.0 + angle4 / 201600.0 + angle6 / 5987520.0;
377 } else {
378 eta = (1.0 - 0.5 * angle * std::tan(0.5 * Globals::Pi - 0.5 * angle)) / (angle * angle);
379 double sin_h_angle = std::sin(0.5 * angle);
380 mu = (angle2 + 4.0 * std::cos(angle) + angle * std::sin(angle) - 4.0) / (4.0 * angle4 * sin_h_angle * sin_h_angle);
381 }
382
383 Spin(rotation_vector, omega);
384 noalias(omega_2) = prod(omega, omega);
385
386 noalias(LiTemp2) = outer_prod(moment_vector, rotation_vector);
387
388 noalias(Li) = inner_prod(rotation_vector, moment_vector) * IdentityMatrix(3, 3);
389 noalias(Li) += outer_prod(rotation_vector, moment_vector);
390 noalias(Li) -= LiTemp2;
391
392 noalias(LiTemp1) = mu * prod(omega_2, LiTemp2);
393 Spin(moment_vector, LiTemp2, 0.5);
394 noalias(LiTemp1) -= LiTemp2;
395
396 noalias(LiTemp1) += eta * Li;
397
398 noalias(Li) = prod(LiTemp1, project(rH, i_range, i_range));
399
400 project(L, i_range, i_range) = Li;
401 }
402
403 return L;
404 }
