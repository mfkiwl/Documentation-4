117  double CalculateValue() override {

132  double CalculateElementMass(Element& elem_i, const std::size_t DomainSize){
133  // We get the element geometry
134  auto& r_this_geometry = elem_i.GetGeometry();
135  const std::size_t local_space_dimension = r_this_geometry.LocalSpaceDimension();
136 
137  double volume_area = 0.0;
138  const IntegrationMethod this_integration_method = r_this_geometry.GetDefaultIntegrationMethod();
139  const GeometryType::IntegrationPointsArrayType& integration_points = r_this_geometry.IntegrationPoints(this_integration_method);
140  for(std::size_t i_point = 0; i_point<integration_points.size(); ++i_point)
141  {
142  Matrix J0;
143  GeometryUtils::JacobianOnInitialConfiguration(r_this_geometry, integration_points[i_point], J0);
144  volume_area += integration_points[i_point].Weight() * MathUtils<double>::GeneralizedDet(J0);
145  }
146 
147 
148  double element_mass = 0;
149  if (local_space_dimension == 2 && DomainSize == 3 && elem_i.GetProperties().Has(THICKNESS) && elem_i.GetProperties().Has(PT))
150  element_mass = volume_area * elem_i.GetProperties().GetValue(PT) * elem_i.GetProperties().GetValue(DENSITY);
151  else if (local_space_dimension == 2 && DomainSize == 3 && elem_i.GetProperties().Has(THICKNESS))
152  element_mass = volume_area * elem_i.GetProperties().GetValue(THICKNESS) * elem_i.GetProperties().GetValue(DENSITY);
153  else if (local_space_dimension == 3 && DomainSize == 3)
154  element_mass = volume_area * elem_i.GetProperties().GetValue(DENSITY);
155  else
156  KRATOS_ERROR << "MassOptResponse::CalculateElementMass: the element is neither shell nor solid element !" << std::endl;
157 
158  return element_mass;
159  }

198  void CalculateElementShapeGradients(Element& elem_i, const std::size_t DomainSize){
199 
200  // We get the element geometry
201  auto& r_this_geometry = elem_i.GetGeometry();
202  const std::size_t number_of_nodes = r_this_geometry.size();
203 
204  // We copy the current coordinates and move the coordinates to the initial configuration
205  std::vector<array_1d<double, 3>> current_coordinates(number_of_nodes);
206  for (std::size_t i_node = 0; i_node < number_of_nodes; ++i_node) {
207  noalias(current_coordinates[i_node]) = r_this_geometry[i_node].Coordinates();
208  noalias(r_this_geometry[i_node].Coordinates()) = r_this_geometry[i_node].GetInitialPosition().Coordinates();
209  }
210  double mass_before_fd = CalculateElementMass(elem_i,DomainSize);
211 
212  for (SizeType i_node = 0; i_node < number_of_nodes; ++i_node){
213  auto& node_grads = r_this_geometry[i_node].FastGetSolutionStepValue(D_MASS_D_X);
214 
215  r_this_geometry[i_node].X() += mDelta;
216  r_this_geometry[i_node].X0() += mDelta;
217  double mass_after_fd = CalculateElementMass(elem_i,DomainSize);
218  node_grads[0] += (mass_after_fd - mass_before_fd) / mDelta;
219  r_this_geometry[i_node].X() -= mDelta;
220  r_this_geometry[i_node].X0() -= mDelta;
221 
222  r_this_geometry[i_node].Y() += mDelta;
223  r_this_geometry[i_node].Y0() += mDelta;
224  mass_after_fd = CalculateElementMass(elem_i,DomainSize);
225  node_grads[1] += (mass_after_fd - mass_before_fd) / mDelta;
226  r_this_geometry[i_node].Y() -= mDelta;
227  r_this_geometry[i_node].Y0() -= mDelta;
228 
229  r_this_geometry[i_node].Z() += mDelta;
230  r_this_geometry[i_node].Z0() += mDelta;
231  mass_after_fd = CalculateElementMass(elem_i,DomainSize);
232  node_grads[2] += (mass_after_fd - mass_before_fd) / mDelta;
233  r_this_geometry[i_node].Z() -= mDelta;
234  r_this_geometry[i_node].Z0() -= mDelta;
235  }
236 
237  // We restore the current configuration
238  for (std::size_t i_node = 0; i_node < number_of_nodes; ++i_node) {
239  noalias(r_this_geometry[i_node].Coordinates()) = current_coordinates[i_node];
240  }

244  void CalculateElementMaterialGradients(Element& elem_i, const std::size_t DomainSize){
245 
246  // We get the element geometry
247  auto& r_this_geometry = elem_i.GetGeometry();
248  const std::size_t number_of_nodes = r_this_geometry.size();
249 
250  double curr_density = elem_i.GetProperties().GetValue(DENSITY);
251  elem_i.GetProperties().SetValue(DENSITY,1.0);
252  double elem_dens_grad = CalculateElementMass(elem_i,DomainSize);
253  elem_i.GetProperties().SetValue(DENSITY,curr_density);
254 
255  for (SizeType i_node = 0; i_node < number_of_nodes; ++i_node){
256  const auto& d_pd_d_fd = r_this_geometry[i_node].FastGetSolutionStepValue(D_PD_D_FD);
257  #pragma omp atomic
258  r_this_geometry[i_node].FastGetSolutionStepValue(D_MASS_D_FD) += d_pd_d_fd * elem_dens_grad / number_of_nodes;
259  }
