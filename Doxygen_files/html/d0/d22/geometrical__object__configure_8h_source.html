156 {
157 array_1d<double, 3> rObj_2_to_rObj_1;
158 noalias(rObj_2_to_rObj_1) = rObj_1->GetGeometry()[0] - rObj_2->GetGeometry()[0];
159 double distance_2 = inner_prod(rObj_2_to_rObj_1, rObj_2_to_rObj_1);

162 SphericParticle* p_particle2 = dynamic_cast<SphericParticle*>(&*rObj_2);

184 static inline bool IntersectionBox(const PointerType& rObject, const PointType& rLowPoint, const PointType& rHighPoint)
185 {
186 array_1d<double, 3> center_of_particle = rObject->GetGeometry()[0];
187
188 SphericParticle* p_particle = dynamic_cast<SphericParticle*>(&*rObject);
189 const double& radius = p_particle->GetSearchRadius();
190
191 bool intersect = (
192 floatle(rLowPoint[0] - radius,center_of_particle[0]) &&
193 floatle(rLowPoint[1] - radius,center_of_particle[1]) &&
194 floatle(rLowPoint[2] - radius,center_of_particle[2]) &&
195 floatge(rHighPoint[0] + radius,center_of_particle[0]) &&
196 floatge(rHighPoint[1] + radius,center_of_particle[1]) &&
197 floatge(rHighPoint[2] + radius,center_of_particle[2]));
198
199 return intersect;
200 }

217 static inline void Distance(const PointerType& rObj_1, const PointerType& rObj_2, double& distance)
218 {
219 array_1d<double, 3> center_of_particle1 = rObj_1->GetGeometry()[0];
220 array_1d<double, 3> center_of_particle2 = rObj_2->GetGeometry()[0];
221
222 distance = std::sqrt((center_of_particle1[0] - center_of_particle2[0]) * (center_of_particle1[0] - center_of_particle2[0]) +
223 (center_of_particle1[1] - center_of_particle2[1]) * (center_of_particle1[1] - center_of_particle2[1]) +
224 (center_of_particle1[2] - center_of_particle2[2]) * (center_of_particle1[2] - center_of_particle2[2]) );
225 }
