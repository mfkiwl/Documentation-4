195 void Build(
196 typename TSchemeType::Pointer pScheme,
197 ModelPart& rModelPart,
198 TSystemMatrixType& A,
199 TSystemVectorType& b) override
200 {
201 KRATOS_TRY
202
203 KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
204
205 // Getting the elements from the model
206 const int nelements = static_cast<int>(rModelPart.Elements().size());
207
208 // Getting the array of the conditions
209 const int nconditions = static_cast<int>(rModelPart.Conditions().size());
210
211 const ProcessInfo& CurrentProcessInfo = rModelPart.GetProcessInfo();
212 ModelPart::ElementsContainerType::iterator el_begin = rModelPart.ElementsBegin();
213 ModelPart::ConditionsContainerType::iterator cond_begin = rModelPart.ConditionsBegin();
214
215 //contributions to the system
216 LocalSystemMatrixType LHS_Contribution = LocalSystemMatrixType(0, 0);
217 LocalSystemVectorType RHS_Contribution = LocalSystemVectorType(0);
218
219 //vector containing the localization in the system of the different
220 //terms
221 Element::EquationIdVectorType EquationId;
222
223 // Assemble all elements
224 const auto timer = BuiltinTimer();
225
226 #pragma omp parallel firstprivate(nelements,nconditions, LHS_Contribution, RHS_Contribution, EquationId )
227 {
228 # pragma omp for schedule(guided, 512) nowait
229 for (int k = 0; k < nelements; k++) {
230 auto it_elem = el_begin + k;
231
232 if (it_elem->IsActive()) {
233 // Calculate elemental contribution
234 pScheme->CalculateSystemContributions(*it_elem, LHS_Contribution, RHS_Contribution, EquationId, CurrentProcessInfo);
235
236 // Assemble the elemental contribution
237 Assemble(A, b, LHS_Contribution, RHS_Contribution, EquationId);
238 }
239
240 }
241
242 #pragma omp for schedule(guided, 512)
243 for (int k = 0; k < nconditions; k++) {
244 auto it_cond = cond_begin + k;
245
246 if (it_cond->IsActive()) {
247 // Calculate elemental contribution
248 pScheme->CalculateSystemContributions(*it_cond, LHS_Contribution, RHS_Contribution, EquationId, CurrentProcessInfo);
249
250 // Assemble the elemental contribution
251 Assemble(A, b, LHS_Contribution, RHS_Contribution, EquationId);
252 }
253 }
254 }
255
256 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", this->GetEchoLevel() >= 1) << "Build time: " << timer.ElapsedSeconds() << std::endl;
257
258 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", (this->GetEchoLevel() > 2 && rModelPart.GetCommunicator().MyPID() == 0)) << "Finished parallel building" << std::endl;

271 void BuildLHS(
272 typename TSchemeType::Pointer pScheme,
273 ModelPart& rModelPart,
274 TSystemMatrixType& rA
275 ) override
276 {
277 KRATOS_TRY
278
279 KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
280
281 // Getting the elements from the model
282 const int nelements = static_cast<int>(rModelPart.Elements().size());
283
284 // Getting the array of the conditions
285 const int nconditions = static_cast<int>(rModelPart.Conditions().size());
286
287 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
288 const auto it_elem_begin = rModelPart.ElementsBegin();
289 const auto it_cond_begin = rModelPart.ConditionsBegin();
290
291 // Contributions to the system
292 LocalSystemMatrixType lhs_contribution(0, 0);
293
294 // Vector containing the localization in the system of the different terms
295 Element::EquationIdVectorType equation_id;
296
297 // Assemble all elements
298 const auto timer = BuiltinTimer();
299
300 #pragma omp parallel firstprivate(nelements, nconditions, lhs_contribution, equation_id )
301 {
302 # pragma omp for schedule(guided, 512) nowait
303 for (int k = 0; k < nelements; ++k) {
304 auto it_elem = it_elem_begin + k;
305
306 // Detect if the element is active or not. If the user did not make any choice the element is active by default
307 bool element_is_active = true;
308 if (it_elem->IsDefined(ACTIVE))
309 element_is_active = it_elem->Is(ACTIVE);
310
311 if (element_is_active) {
312 // Calculate elemental contribution
313 pScheme->CalculateLHSContribution(*it_elem, lhs_contribution, equation_id, r_current_process_info);
314
315 // Assemble the elemental contribution
316 AssembleLHS(rA, lhs_contribution, equation_id);
317 }
318 }
319
320 #pragma omp for schedule(guided, 512)
321 for (int k = 0; k < nconditions; ++k) {
322 auto it_cond = it_cond_begin + k;
323
324 // Detect if the element is active or not. If the user did not make any choice the element is active by default
325 bool condition_is_active = true;
326 if (it_cond->IsDefined(ACTIVE))
327 condition_is_active = it_cond->Is(ACTIVE);
328
329 if (condition_is_active) {
330 // Calculate elemental contribution
331 pScheme->CalculateLHSContribution(*it_cond, lhs_contribution, equation_id, r_current_process_info);
332
333 // Assemble the elemental contribution
334 AssembleLHS(rA, lhs_contribution, equation_id);
335 }
336 }
337 }
338
339 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", this->GetEchoLevel() >= 1) << "Build time LHS: " << timer.ElapsedSeconds() << std::endl;
340
341 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", this->GetEchoLevel() > 2) << "Finished parallel building LHS" << std::endl;
342
343 KRATOS_CATCH("")

537 void BuildAndSolveLinearizedOnPreviousIteration(
538 typename TSchemeType::Pointer pScheme,
539 ModelPart& rModelPart,
540 TSystemMatrixType& rA,
541 TSystemVectorType& rDx,
542 TSystemVectorType& rb,
543 const bool MoveMesh
544 ) override
545 {
546 Timer::Start("Linearizing on Old iteration");
547
548 KRATOS_INFO_IF("BlockBuilderAndSolver", this->GetEchoLevel() > 0) << "Linearizing on Old iteration" << std::endl;
549
550 KRATOS_ERROR_IF(rModelPart.GetBufferSize() == 1) << "BlockBuilderAndSolver: \n"
551 << "The buffer size needs to be at least 2 in order to use \n"
552 << "BuildAndSolveLinearizedOnPreviousIteration \n"
553 << "current buffer size for modelpart: " << rModelPart.Name() << std::endl
554 << "is :" << rModelPart.GetBufferSize()
555 << " Please set IN THE STRATEGY SETTINGS "
556 << " UseOldStiffnessInFirstIteration=false " << std::endl;
557
558 DofsArrayType fixed_dofs;
559 for(auto& r_dof : BaseType::mDofSet){
560 if(r_dof.IsFixed()){
561 fixed_dofs.push_back(&r_dof);
562 r_dof.FreeDof();
563 }
564 }
565
566 //TODO: Here we need to take the vector from other ones because
567 // We cannot create a trilinos vector without a communicator. To be improved!
568 TSystemVectorType dx_prediction(rDx);
569 TSystemVectorType rhs_addition(rb); //we know it is zero here, so we do not need to set it
570
571 // Here we bring back the database to before the prediction,
572 // but we store the prediction increment in dx_prediction.
573 // The goal is that the stiffness is computed with the
574 // converged configuration at the end of the previous step.
575 block_for_each(BaseType::mDofSet, [&](Dof<double>& rDof){
576 // NOTE: this is initialized to - the value of dx prediction
577 dx_prediction[rDof.EquationId()] = -(rDof.GetSolutionStepValue() - rDof.GetSolutionStepValue(1));
578
579 });
580
581 // Use UpdateDatabase to bring back the solution to how it was at the end of the previous step
582 pScheme->Update(rModelPart, BaseType::mDofSet, rA, dx_prediction, rb);
583 if (MoveMesh) {
584 VariableUtils().UpdateCurrentPosition(rModelPart.Nodes(),DISPLACEMENT,0);
585 }
586
587 Timer::Stop("Linearizing on Old iteration");
588
589 Timer::Start("Build");
590
591 this->Build(pScheme, rModelPart, rA, rb);
592
593 Timer::Stop("Build");
594
595 // Put back the prediction into the database
596 TSparseSpace::InplaceMult(dx_prediction, -1.0); //change sign to dx_prediction
597 TSparseSpace::UnaliasedAdd(rDx, 1.0, dx_prediction);
598
599 // Use UpdateDatabase to bring back the solution
600 // to where it was taking into account BCs
601 // it is done here so that constraints are correctly taken into account right after
602 pScheme->Update(rModelPart, BaseType::mDofSet, rA, dx_prediction, rb);
603 if (MoveMesh) {
604 VariableUtils().UpdateCurrentPosition(rModelPart.Nodes(),DISPLACEMENT,0);
605 }
606
607 // Apply rb -= A*dx_prediction
608 TSparseSpace::Mult(rA, dx_prediction, rhs_addition);
609 TSparseSpace::UnaliasedAdd(rb, -1.0, rhs_addition);
610
611 for(auto& dof : fixed_dofs)
612 dof.FixDof();
613
614 if (!rModelPart.MasterSlaveConstraints().empty()) {
615 const auto timer_constraints = BuiltinTimer();
616 Timer::Start("ApplyConstraints");
617 this->ApplyConstraints(pScheme, rModelPart, rA, rb);
618 Timer::Stop("ApplyConstraints");
619 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", this->GetEchoLevel() >=1) << "Constraints build time: " << timer_constraints.ElapsedSeconds() << std::endl;
620 }
621 this->ApplyDirichletConditions(pScheme, rModelPart, rA, rDx, rb);
622
623 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", ( this->GetEchoLevel() == 3)) << "Before the solution of the system" << "\nSystem Matrix = " << rA << "\nUnknowns vector = " << rDx << "\nRHS vector = " << rb << std::endl;
624
625 const auto timer = BuiltinTimer();
626
627 Timer::Start("Solve");
628
629 this->SystemSolveWithPhysics(rA, rDx, rb, rModelPart);
630
631 Timer::Stop("Solve");
632 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", this->GetEchoLevel() >=1) << "System solve time: " << timer.ElapsedSeconds() << std::endl;
633
634 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", ( this->GetEchoLevel() == 3)) << "After the solution of the system" << "\nSystem Matrix = " << rA << "\nUnknowns vector = " << rDx << "\nRHS vector = " << rb << std::endl;
635 }

718 void SetUpDofSet(
719 typename TSchemeType::Pointer pScheme,
720 ModelPart& rModelPart

722 {
723 KRATOS_TRY;
724
725 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", ( this->GetEchoLevel() > 1 && rModelPart.GetCommunicator().MyPID() == 0)) << "Setting up the dofs" << std::endl;
726
727 //Gets the array of elements from the modeler
728 ElementsArrayType& r_elements_array = rModelPart.Elements();
729 const int number_of_elements = static_cast<int>(r_elements_array.size());
730
731 DofsVectorType dof_list, second_dof_list; // NOTE: The second dof list is only used on constraints to include master/slave relations
732
733 unsigned int nthreads = ParallelUtilities::GetNumThreads();
734
735 typedef std::unordered_set < NodeType::DofType::Pointer, DofPointerHasher> set_type;
736
737 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", ( this->GetEchoLevel() > 2)) << "Number of threads" << nthreads << "\n" << std::endl;
738
739 KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", ( this->GetEchoLevel() > 2)) << "Initializing element loop" << std::endl;
740
746 set_type dof_global_set;
747 dof_global_set.reserve(number_of_elements*20);
748
749 #pragma omp parallel firstprivate(dof_list, second_dof_list)
750 {
751 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
752
753 // We cleate the temporal set and we reserve some space on them
754 set_type dofs_tmp_set;
755 dofs_tmp_set.reserve(20000);
756
757 // Gets the array of elements from the modeler
758 #pragma omp for schedule(guided, 512) nowait
759 for (int i = 0; i < number_of_elements; ++i) {
760 auto it_elem = r_elements_array.begin() + i;
761
762 // Gets list of Dof involved on every element
763 pScheme->GetDofList(*it_elem, dof_list, r_current_process_info);
764 dofs_tmp_set.insert(dof_list.begin(), dof_list.end());
765 }
766
767 // Gets the array of conditions from the modeler
768 ConditionsArrayType& r_conditions_array = rModelPart.Conditions();
769 const int number_of_conditions = static_cast<int>(r_conditions_array.size());
770 #pragma omp for schedule(guided, 512) nowait
771 for (int i = 0; i < number_of_conditions; ++i) {
772 auto it_cond = r_conditions_array.begin() + i;
773
774 // Gets list of Dof involved on every element
775 pScheme->GetDofList(*it_cond, dof_list, r_current_process_info);
776 dofs_tmp_set.insert(dof_list.begin(), dof_list.end());
777 }
778
779 // Gets the array of constraints from the modeler
780 auto& r_constraints_array = rModelPart.MasterSlaveConstraints();
781 const int number_of_constraints = static_cast<int>(r_constraints_array.size());
782 #pragma omp for schedule(guided, 512) nowait
783 for (int i = 0; i < number_of_constraints; ++i) {
784 auto it_const = r_constraints_array.begin() + i;
785
786 // Gets list of Dof involved on every element
787 it_const->GetDofList(dof_list, second_dof_list, r_current_process_info);
788 dofs_tmp_set.insert(dof_list.begin(), dof_list.end());
789 dofs_tmp_set.insert(second_dof_list.begin(), second_dof_list.end());
790 }
791
792 // We merge all the sets in one thread
793 #pragma omp critical
794 {
795 dof_global_set.insert(dofs_tmp_set.begin(), dofs_tmp_set.end());
796 }

1300 virtual void ConstructMasterSlaveConstraintsStructure(ModelPart& rModelPart)
1301 {
1302 if (rModelPart.MasterSlaveConstraints().size() > 0) {
1303 Timer::Start("ConstraintsRelationMatrixStructure");
1304 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
1305
1306 // Constraint initial iterator
1307 const auto it_const_begin = rModelPart.MasterSlaveConstraints().begin();
1308 std::vector<std::unordered_set<IndexType>> indices(BaseType::mDofSet.size());
1309
1310 std::vector<LockObject> lock_array(indices.size());
1311
1312 #pragma omp parallel
1313 {
1314 Element::EquationIdVectorType slave_ids(3);
1315 Element::EquationIdVectorType master_ids(3);
1316 std::unordered_map<IndexType, std::unordered_set<IndexType>> temp_indices;
1317
1318 #pragma omp for schedule(guided, 512) nowait
1319 for (int i_const = 0; i_const < static_cast<int>(rModelPart.MasterSlaveConstraints().size()); ++i_const) {
1320 auto it_const = it_const_begin + i_const;
1321 it_const->EquationIdVector(slave_ids, master_ids, r_current_process_info);
1322
1323 // Slave DoFs
1324 for (auto &id_i : slave_ids) {
1325 temp_indices[id_i].insert(master_ids.begin(), master_ids.end());
1326 }
1327 }
1328
1329 // Merging all the temporal indexes
1330 for (int i = 0; i < static_cast<int>(temp_indices.size()); ++i) {
1331 lock_array[i].lock();
1332 indices[i].insert(temp_indices[i].begin(), temp_indices[i].end());
1333 lock_array[i].unlock();
1334 }
1335 }
1336
1337 mSlaveIds.clear();
1338 mMasterIds.clear();
1339 for (int i = 0; i < static_cast<int>(indices.size()); ++i) {
1340 if (indices[i].size() == 0) // Master dof!
1341 mMasterIds.push_back(i);
1342 else // Slave dof
1343 mSlaveIds.push_back(i);
1344 indices[i].insert(i); // Ensure that the diagonal is there in T
1345 }
1346
1347 // Count the row sizes
1348 const std::size_t nnz = block_for_each<SumReduction<std::size_t>>(indices, [](auto& rIndices) {return rIndices.size();});
1349
1350 mT = TSystemMatrixType(indices.size(), indices.size(), nnz);
1351 mConstantVector.resize(indices.size(), false);
1352
1353 double *Tvalues = mT.value_data().begin();
1354 IndexType *Trow_indices = mT.index1_data().begin();
1355 IndexType *Tcol_indices = mT.index2_data().begin();
1356
1357 // Filling the index1 vector - DO NOT MAKE PARALLEL THE FOLLOWING LOOP!
1358 Trow_indices[0] = 0;
1359 for (int i = 0; i < static_cast<int>(mT.size1()); i++)
1360 Trow_indices[i + 1] = Trow_indices[i] + indices[i].size();
1361
1362 IndexPartition<std::size_t>(mT.size1()).for_each([&](std::size_t Index){
1363 const IndexType row_begin = Trow_indices[Index];
1364 const IndexType row_end = Trow_indices[Index + 1];
1365 IndexType k = row_begin;
1366 for (auto it = indices[Index].begin(); it != indices[Index].end(); ++it) {
1367 Tcol_indices[k] = *it;
1368 Tvalues[k] = 0.0;
1369 k++;
1370 }
1371
1372 indices[Index].clear(); //deallocating the memory
1373
1374 std::sort(&Tcol_indices[row_begin], &Tcol_indices[row_end]);
1375 });
1376
1377 mT.set_filled(indices.size() + 1, nnz);
1378
1379 Timer::Stop("ConstraintsRelationMatrixStructure");
1380 }

1383 virtual void BuildMasterSlaveConstraints(ModelPart& rModelPart)
1384 {
1385 KRATOS_TRY
1386
1387 TSparseSpace::SetToZero(mT);
1388 TSparseSpace::SetToZero(mConstantVector);
1389
1390 // The current process info
1391 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
1392
1393 // Contributions to the system
1394 Matrix transformation_matrix = LocalSystemMatrixType(0, 0);
1395 Vector constant_vector = LocalSystemVectorType(0);
1396
1397 // Vector containing the localization in the system of the different terms
1398 Element::EquationIdVectorType slave_equation_ids, master_equation_ids;
1399
1400 const int number_of_constraints = static_cast<int>(rModelPart.MasterSlaveConstraints().size());
1401
1402 // We clear the set
1403 mInactiveSlaveDofs.clear();
1404
1405 #pragma omp parallel firstprivate(transformation_matrix, constant_vector, slave_equation_ids, master_equation_ids)
1406 {
1407 std::unordered_set<IndexType> auxiliar_inactive_slave_dofs;
1408
1409 #pragma omp for schedule(guided, 512)
1410 for (int i_const = 0; i_const < number_of_constraints; ++i_const) {
1411 auto it_const = rModelPart.MasterSlaveConstraints().begin() + i_const;
1412
1413 // Detect if the constraint is active or not. If the user did not make any choice the constraint
1414 // It is active by default
1415 bool constraint_is_active = true;
1416 if (it_const->IsDefined(ACTIVE))
1417 constraint_is_active = it_const->Is(ACTIVE);
1418
1419 if (constraint_is_active) {
1420 it_const->CalculateLocalSystem(transformation_matrix, constant_vector, r_current_process_info);
1421 it_const->EquationIdVector(slave_equation_ids, master_equation_ids, r_current_process_info);
1422
1423 for (IndexType i = 0; i < slave_equation_ids.size(); ++i) {
1424 const IndexType i_global = slave_equation_ids[i];
1425
1426 // Assemble matrix row
1427 AssembleRowContribution(mT, transformation_matrix, i_global, i, master_equation_ids);
1428
1429 // Assemble constant vector
1430 const double constant_value = constant_vector[i];
1431 double& r_value = mConstantVector[i_global];
1432 AtomicAdd(r_value, constant_value);
1433 }
1434 } else { // Taking into account inactive constraints
1435 it_const->EquationIdVector(slave_equation_ids, master_equation_ids, r_current_process_info);
1436 auxiliar_inactive_slave_dofs.insert(slave_equation_ids.begin(), slave_equation_ids.end());
1437 }
1438 }
1439
1440 // We merge all the sets in one thread
1441 #pragma omp critical
1442 {
1443 mInactiveSlaveDofs.insert(auxiliar_inactive_slave_dofs.begin(), auxiliar_inactive_slave_dofs.end());
1444 }
1445 }
1446
1447 // Setting the master dofs into the T and C system
1448 for (auto eq_id : mMasterIds) {
1449 mConstantVector[eq_id] = 0.0;
1450 mT(eq_id, eq_id) = 1.0;
1451 }
1452
1453 // Setting inactive slave dofs in the T and C system
1454 for (auto eq_id : mInactiveSlaveDofs) {
1455 mConstantVector[eq_id] = 0.0;
1456 mT(eq_id, eq_id) = 1.0;
1457 }
1458
1459 KRATOS_CATCH("")
1460 }

1463 typename TSchemeType::Pointer pScheme,
1464 TSystemMatrixType& A,
1465 ModelPart& rModelPart)
1466 {
1467 //filling with zero the matrix (creating the structure)
1468 Timer::Start("MatrixStructure");
1469
1470 const ProcessInfo& CurrentProcessInfo = rModelPart.GetProcessInfo();
1471
1472 const std::size_t equation_size = BaseType::mEquationSystemSize;
1473
1474 std::vector< LockObject > lock_array(equation_size);
1475
1476 std::vector<std::unordered_set<std::size_t> > indices(equation_size);
1477
1478 block_for_each(indices, [](std::unordered_set<std::size_t>& rIndices){
1479 rIndices.reserve(40);
1480 });
1481
1482 Element::EquationIdVectorType ids(3, 0);
1483
1484 block_for_each(rModelPart.Elements(), ids, [&](Element& rElem, Element::EquationIdVectorType& rIdsTLS){
1485 pScheme->EquationId(rElem, rIdsTLS, CurrentProcessInfo);
1486 for (std::size_t i = 0; i < rIdsTLS.size(); i++) {
1487 lock_array[rIdsTLS[i]].lock();
1488 auto& row_indices = indices[rIdsTLS[i]];
1489 row_indices.insert(rIdsTLS.begin(), rIdsTLS.end());
1490 lock_array[rIdsTLS[i]].unlock();
1491 }
1492 });
1493
1494 block_for_each(rModelPart.Conditions(), ids, [&](Condition& rCond, Element::EquationIdVectorType& rIdsTLS){
1495 pScheme->EquationId(rCond, rIdsTLS, CurrentProcessInfo);
1496 for (std::size_t i = 0; i < rIdsTLS.size(); i++) {
1497 lock_array[rIdsTLS[i]].lock();
1498 auto& row_indices = indices[rIdsTLS[i]];
1499 row_indices.insert(rIdsTLS.begin(), rIdsTLS.end());
1500 lock_array[rIdsTLS[i]].unlock();
1501 }
1502 });
1503
1504 if (rModelPart.MasterSlaveConstraints().size() != 0) {
1505 struct TLS
1506 {
1507 Element::EquationIdVectorType master_ids = Element::EquationIdVectorType(3,0);
1508 Element::EquationIdVectorType slave_ids = Element::EquationIdVectorType(3,0);
1509 };
