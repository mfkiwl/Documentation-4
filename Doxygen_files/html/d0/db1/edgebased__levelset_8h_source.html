293
294 double ComputeTimeStep(const double CFLNumber, const double MaxDt)
295 {
296 KRATOS_TRY
297
298 // save the maximum time step
299 max_dt = MaxDt;
300
301 // local variable for time step size
302 double delta_t = 1e10; // max_dt;
303
304 mdelta_t_avg = 1e10; // max_dt;
305
306 // getting value of current velocity and of viscosity
307 mr_matrix_container.FillVectorFromDatabase(BODY_FORCE, mBodyForce, mr_model_part.Nodes());
308 mr_matrix_container.FillScalarFromDatabase(VISCOSITY, mViscosity, mr_model_part.Nodes());
309 mr_matrix_container.FillVectorFromDatabase(VELOCITY, mvel_n1, mr_model_part.Nodes());
310 mr_matrix_container.FillScalarFromDatabase(POROSITY, mEps, mr_model_part.Nodes());
311 mr_matrix_container.FillScalarFromDatabase(LIN_DARCY_COEF, mA, mr_model_part.Nodes());
312 mr_matrix_container.FillScalarFromDatabase(NONLIN_DARCY_COEF, mB, mr_model_part.Nodes());
313 mr_matrix_container.FillVectorFromDatabase(STRUCTURE_VELOCITY, mStrVel, mr_model_part.Nodes());
314
315 //*******************
316 // loop over all nodes
317 unsigned int n_nodes = mvel_n1.size();
318 for (unsigned int i_node = 0; i_node < n_nodes; i_node++)
319 {
320 const array_1d<double, TDim> &v_i = mvel_n1[i_node];
321 const double havg_i = mHavg[i_node];
322 const double hmin_i = mHmin[i_node];
323 const double eps_i = mEps[i_node];
324 const double nu = mViscosity[i_node];
325
326 double vel_norm = 0.0;
327 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
328 {
329 vel_norm += v_i[l_comp] * v_i[l_comp];
330 }
331 vel_norm = sqrt(vel_norm);
332
333 vel_norm /= eps_i;
334
335 // use CFL condition to compute time step size
336 double delta_t_i = CFLNumber * 1.0 / (2.0 * vel_norm / hmin_i + 4.0 * nu / (hmin_i * hmin_i));
337 double delta_t_i_avg = 1.0 / (2.0 * vel_norm / havg_i + 4.0 * nu / (havg_i * havg_i));
338
339 // considering the most restrictive case of neighbor's velocities with similar direction but opposite sense.
340 // loop over all neighbours
341 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
342 {
343 // get global index of neighbouring node j
344 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
345
346 const array_1d<double, TDim> &v_j = mvel_n1[j_neighbour];
347
348 double v_diff_norm = 0.0;
349 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
350 {
351 double temp = v_i[l_comp] - v_j[l_comp];
352 v_diff_norm += temp * temp;
353 }
354 v_diff_norm = sqrt(v_diff_norm);
355 v_diff_norm /= eps_i;
356 double delta_t_j = CFLNumber * 1.0 / (2.0 * v_diff_norm / hmin_i + 4.0 * nu / (hmin_i * hmin_i));
357
358 if (delta_t_j < delta_t_i)
359 delta_t_i = delta_t_j;
360 }
361
362 // choose the overall minimum of delta_t_i
363 if (delta_t_i < delta_t)
364 delta_t = delta_t_i;
365
366 if (delta_t_i_avg < mdelta_t_avg)
367 mdelta_t_avg = delta_t_i_avg;
368 }
369
370 return delta_t;
371
372 KRATOS_CATCH("")
373 }

409 //**********************************************************************************
410 // function to solve fluid equations - fractional step 1: compute fractional momentum
411
412 void SolveStep1()
413 {
414 KRATOS_TRY
415
416 // PREREQUISITES
417
418 // variables for node based data handling
419 ModelPart::NodesContainerType &rNodes = mr_model_part.Nodes();
420 int n_nodes = rNodes.size();
421 // storage of nodal values in local variables
422 CalcVectorType rhs;
423 rhs.resize(n_nodes);
424
425 // read velocity and pressure data from Kratos
426 mr_matrix_container.FillVectorFromDatabase(VELOCITY, mvel_n1, rNodes);
427 mr_matrix_container.FillOldVectorFromDatabase(VELOCITY, mvel_n, rNodes);
428 mr_matrix_container.FillVectorFromDatabase(BODY_FORCE, mBodyForce, rNodes);
429 mr_matrix_container.FillScalarFromDatabase(VISCOSITY, mViscosity, rNodes);
430 mr_matrix_container.FillScalarFromDatabase(PRESSURE, mPn1, rNodes);
431 mr_matrix_container.FillOldScalarFromDatabase(PRESSURE, mPn, rNodes);
432
433 mr_matrix_container.FillScalarFromDatabase(DISTANCE, mdistances, mr_model_part.Nodes());
434 mr_matrix_container.FillScalarFromDatabase(POROSITY, mEps, mr_model_part.Nodes());
435 mr_matrix_container.FillScalarFromDatabase(LIN_DARCY_COEF, mA, mr_model_part.Nodes());
436 mr_matrix_container.FillScalarFromDatabase(NONLIN_DARCY_COEF, mB, mr_model_part.Nodes());
437 mr_matrix_container.FillVectorFromDatabase(STRUCTURE_VELOCITY, mStrVel, rNodes);
438
439 // read time step size from Kratos
440 ProcessInfo &CurrentProcessInfo = mr_model_part.GetProcessInfo();
441 double delta_t = CurrentProcessInfo[DELTA_TIME];
442
443 // compute intrinsic time
444 double time_inv_avg = 1.0 / mdelta_t_avg;
445
446 double stabdt_pressure_factor = mstabdt_pressure_factor;
447 double stabdt_convection_factor = mstabdt_convection_factor;
448 double tau2_factor = mtau2_factor;
449
450#pragma omp parallel for firstprivate(time_inv_avg, stabdt_pressure_factor, stabdt_convection_factor, tau2_factor)
451 for (int i_node = 0; i_node < n_nodes; i_node++)
452 {
453 double &h_avg_i = mHavg[i_node];
454
455 array_1d<double, TDim> &a_i = mvel_n1[i_node];
456 const double nu_i = mViscosity[i_node];
457 const double eps_i = mEps[i_node];
458 const double lindarcy_i = mA[i_node];
459 const double nonlindarcy_i = mB[i_node];
460
461 double vel_norm = 0.0;
462 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
463 {
464 vel_norm += a_i[l_comp] * a_i[l_comp];
465 }
466 vel_norm = sqrt(vel_norm);
467
468 const array_1d<double, TDim> &str_v_i = mStrVel[i_node];
469 array_1d<double, TDim> rel_vel_i;
470
471 double rel_vel_norm = 0.0;
472 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
473 {
474 rel_vel_i[l_comp] = a_i[l_comp] - str_v_i[l_comp];
475 rel_vel_norm += rel_vel_i[l_comp] * rel_vel_i[l_comp];
476 }
477 rel_vel_norm = sqrt(rel_vel_norm);
478
479 double porosity_coefficient = ComputePorosityCoefficient(rel_vel_norm, eps_i, lindarcy_i, nonlindarcy_i);
480
481 vel_norm /= eps_i;
482
483 double tau = 1.0 / (2.0 * vel_norm / h_avg_i + stabdt_pressure_factor * time_inv_avg + (4.0 * nu_i) / (h_avg_i * h_avg_i) + porosity_coefficient);
484 double tau_conv = 1.0 / (2.0 * vel_norm / h_avg_i + stabdt_convection_factor * time_inv_avg + (4.0 * nu_i) / (h_avg_i * h_avg_i) + porosity_coefficient);
485 mTauPressure[i_node] = tau;
486 mTauConvection[i_node] = tau_conv;
487
488 mTau2[i_node] = (nu_i + h_avg_i * vel_norm * 0.5) * tau2_factor;
489 }
490
491 // calculating the convective projection
492 IndexPartition<unsigned int>(n_nodes).for_each([&](unsigned int i_node){
493 array_1d<double, TDim> &pi_i = mPi[i_node];
494
495 // setting to zero
496 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
497 pi_i[l_comp] = 0.0;
498
499 array_1d<double, TDim> a_i = mvel_n1[i_node];
500 const array_1d<double, TDim> &U_i = mvel_n1[i_node];
501 const double &eps_i = mEps[i_node];
502
503 a_i /= eps_i;
504
505 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
506 {
507 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
508 array_1d<double, TDim> a_j = mvel_n1[j_neighbour];
509 const array_1d<double, TDim> &U_j = mvel_n1[j_neighbour];
510 const double &eps_j = mEps[j_neighbour];
511
512 a_j /= eps_j;
513
514 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
515
516 edge_ij.Add_ConvectiveContribution(pi_i, a_i, U_i, a_j, U_j);
517 }
518
519 const double m_inv = mr_matrix_container.GetInvertedMass()[i_node];
520
521 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
522 pi_i[l_comp] *= m_inv;
523 });
524
525 mr_matrix_container.AssignVectorToVector(mvel_n, mWork); // mWork = mvel_n
526 // first step of Runge Kutta
527 mr_matrix_container.AssignVectorToVector(mvel_n, mvel_n1); // mvel_n1 = mvel_n
528
529 mr_matrix_container.SetToZero(rhs);
530 CalculateRHS(mvel_n1, mPn, mvel_n1, rhs, mdiag_stiffness);
531 Add_Effective_Inverse_Multiply(mWork, mWork, delta_t / 6.0, mr_matrix_container.GetLumpedMass(), mdiag_stiffness, rhs);
532 Add_Effective_Inverse_Multiply(mvel_n1, mvel_n, 0.5 * delta_t, mr_matrix_container.GetLumpedMass(), mdiag_stiffness, rhs);
533 ApplyVelocityBC(mvel_n1);
534
535 // second step
536 mr_matrix_container.SetToZero(rhs);
537 CalculateRHS(mvel_n1, mPn, mvel_n1, rhs, mdiag_stiffness);
538 Add_Effective_Inverse_Multiply(mWork, mWork, delta_t / 3.0, mr_matrix_container.GetLumpedMass(), mdiag_stiffness, rhs);
539 Add_Effective_Inverse_Multiply(mvel_n1, mvel_n, 0.5 * delta_t, mr_matrix_container.GetLumpedMass(), mdiag_stiffness, rhs);
540 ApplyVelocityBC(mvel_n1);
541
542 // third step
543 mr_matrix_container.SetToZero(rhs);
544 CalculateRHS(mvel_n1, mPn, mvel_n1, rhs, mdiag_stiffness);
545 Add_Effective_Inverse_Multiply(mWork, mWork, delta_t / 3.0, mr_matrix_container.GetLumpedMass(), mdiag_stiffness, rhs);
546 Add_Effective_Inverse_Multiply(mvel_n1, mvel_n, delta_t, mr_matrix_container.GetLumpedMass(), mdiag_stiffness, rhs);
547 ApplyVelocityBC(mvel_n1);
548
549 // fourth step
550 mr_matrix_container.SetToZero(rhs);
551 CalculateRHS(mvel_n1, mPn, mvel_n1, rhs, mdiag_stiffness);
552 Add_Effective_Inverse_Multiply(mWork, mWork, delta_t / 6.0, mr_matrix_container.GetLumpedMass(), mdiag_stiffness, rhs);
553 // compute right-hand side
554 mr_matrix_container.AssignVectorToVector(mWork, mvel_n1);
555 ApplyVelocityBC(mvel_n1);
556
557 KRATOS_CATCH("")
558 }

561 // function to calculate right-hand side of fractional momentum equation
562
563 void CalculateRHS(
564 const CalcVectorType &vel,
565 const ValuesVectorType &pressure,
566 const CalcVectorType &convective_velocity,
567 CalcVectorType &rhs,
568 ValuesVectorType &diag_stiffness)
569 {
570 KRATOS_TRY
571
572 int n_nodes = vel.size();
573
574 // calculating the RHS
575 array_1d<double, TDim> stab_low;
576 array_1d<double, TDim> stab_high;
577
578 double inverse_rho = 1.0 / mRho;
579#pragma omp parallel for private(stab_low, stab_high)
580 for (int i_node = 0; i_node < n_nodes; i_node++)
581 {
582 double dist = mdistances[i_node];
583 if (dist <= 0.0) // node is inside domain ---- if outside do nothing
584 {
585 const double nu_i = mViscosity[i_node];
586 const double nu_j = nu_i;
587 array_1d<double, TDim> &rhs_i = rhs[i_node];
588 const array_1d<double, TDim> &f_i = mBodyForce[i_node];
589 array_1d<double, TDim> a_i = convective_velocity[i_node];
590 const array_1d<double, TDim> &U_i = vel[i_node];
591 const array_1d<double, TDim> &pi_i = mPi[i_node];
592 const double &p_i = pressure[i_node];
593 const double &eps_i = mEps[i_node];
594 const double lindarcy_i = mA[i_node];
595 const double nonlindarcy_i = mB[i_node];
596
597 const array_1d<double, TDim> &str_v_i = mStrVel[i_node];
598 array_1d<double, TDim> rel_vel_i;
599
600 double rel_vel_norm = 0.0;
601 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
602 {
603 rel_vel_i[l_comp] = U_i[l_comp] - str_v_i[l_comp];
604 rel_vel_norm += rel_vel_i[l_comp] * rel_vel_i[l_comp];
605 }
606 rel_vel_norm = sqrt(rel_vel_norm);
607
608 double edge_tau = mTauConvection[i_node];
609
610 a_i /= eps_i;
611
612 // initializing with the external forces (e.g. gravity)
613 double &m_i = mr_matrix_container.GetLumpedMass()[i_node];
614 for (unsigned int comp = 0; comp < TDim; comp++)
615 rhs_i[comp] = m_i * eps_i * f_i[comp];
616
617 // applying the effect of the porosity
618 double porosity_coefficient = ComputePorosityCoefficient(rel_vel_norm, eps_i, lindarcy_i, nonlindarcy_i);
619 diag_stiffness[i_node] = m_i * porosity_coefficient;
620
621 // /**************************************************rel_vel_modifications_b*/
622 for (unsigned int comp = 0; comp < TDim; comp++)
623 {
624 rhs_i[comp] += m_i * porosity_coefficient * str_v_i[comp];
625 }
626 // /*************************************************rel_vel_modifications_e*/
627
628 // convective term
629 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
630 {
631 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
632 array_1d<double, TDim> a_j = convective_velocity[j_neighbour];
633 const array_1d<double, TDim> &U_j = vel[j_neighbour];
634 const array_1d<double, TDim> &pi_j = mPi[j_neighbour];
635 const double &p_j = pressure[j_neighbour];
636 const double &eps_j = mEps[j_neighbour];
637
638 a_j /= eps_j;
639
640 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
641
642 edge_ij.Sub_ConvectiveContribution(rhs_i, a_i, U_i, a_j, U_j);
643
644 // take care! we miss including a B.C. for the external pressure
645 edge_ij.Sub_grad_p(rhs_i, p_i * inverse_rho * eps_i, p_j * inverse_rho * eps_i);
646
647 edge_ij.Sub_ViscousContribution(rhs_i, U_i, nu_i, U_j, nu_j);
648
649 // add stabilization
650 edge_ij.CalculateConvectionStabilization_LOW(stab_low, a_i, U_i, a_j, U_j);
651 edge_ij.CalculateConvectionStabilization_HIGH(stab_high, a_i, pi_i, a_j, pi_j);
652 edge_ij.Sub_StabContribution(rhs_i, edge_tau, 1.0, stab_low, stab_high);
653 }
654 }
655 }
656
657 // apply wall resistance
658 if (mWallLawIsActive == true)
659 ComputeWallResistance(vel, diag_stiffness);
660 ModelPart::NodesContainerType &rNodes = mr_model_part.Nodes();
661 mr_matrix_container.WriteVectorToDatabase(VELOCITY, mvel_n1, rNodes);
662
663 KRATOS_CATCH("")
664 }

667 // function to solve fluid equations - fractional step 2: calculate pressure
668
669 void SolveStep2(typename TLinearSolver::Pointer pLinearSolver)
670 {
671 KRATOS_TRY
672
673 typedef Node<3> PointType;
674 typedef GlobalPointersVector<PointType> PointVector;
675 typedef PointVector::iterator PointIterator;
676
677 // reset is visited flag
678 for (ModelPart::NodesContainerType::iterator inode = mr_model_part.NodesBegin();
679 inode != mr_model_part.NodesEnd();
680 inode++)
681 {
682 inode->GetValue(IS_VISITED) = 0.0;
683 }
684 // Re-generate a container with LAYER 0 and LAYER 1 after convection of the free surface
685 std::vector<PointVector> layers(2);
686
687 // detect the nodes inside the fluid surface LAYER_0
688 for (ModelPart::NodesContainerType::iterator inode = mr_model_part.NodesBegin();
689 inode != mr_model_part.NodesEnd();
690 inode++)
691 {
692 if (inode->FastGetSolutionStepValue(DISTANCE) < 0.0) // candidates are only the ones inside the fluid domain
693 {
694 GlobalPointersVector<Node<3>> &neighb_nodes = inode->GetValue(NEIGHBOUR_NODES);
695 for (GlobalPointersVector<Node<3>>::iterator i = neighb_nodes.begin(); i != neighb_nodes.end(); i++)
696 {
697 if (i->FastGetSolutionStepValue(DISTANCE) >= 0.0) // add the node as free surface if one of its neighb is outside
698 {
699 if (inode->GetValue(IS_VISITED) == 0.0)
700 {
701 layers[0].push_back(*(inode.base()));
702 inode->GetValue(IS_VISITED) = 1.0;
703 }
704 }
705 }
706 }
707 else
708 inode->FastGetSolutionStepValue(PRESSURE) = 0.0;
709 }
710
711 // fill layer 1 by neighbour relationships
712 for (PointIterator iii = (layers[0]).begin(); iii != (layers[0]).end(); iii++)
713 {
714 GlobalPointersVector<Node<3>> &neighb_nodes = iii->GetValue(NEIGHBOUR_NODES);
715 for (GlobalPointersVector<Node<3>>::iterator jjj = neighb_nodes.begin(); jjj != neighb_nodes.end(); jjj++) // destination = origin1 + value * Minv*origin
716 {
717
718 if (jjj->FastGetSolutionStepValue(DISTANCE) >= 0 &&
719 jjj->GetValue(IS_VISITED) == 0.0)
720 {
721 layers[1].push_back(Node<3>::WeakPointer(*jjj.base()));
722 jjj->GetValue(IS_VISITED) = 2.0;
723 }
724 }
725 }
726
727 // on the first layer outside the pressure is set to a value such that on the free surface the pressure is approx 0
728 for (PointIterator iii = layers[1].begin(); iii != layers[1].end(); iii++)
729 {
730 // get the node
731 unsigned int i_node = iii->FastGetSolutionStepValue(AUX_INDEX);
732
733 array_1d<double, TDim> grad_d;
734 for (unsigned int comp = 0; comp < TDim; comp++)
735 grad_d[comp] = 0.0;
736
737 double dist_i = mdistances[i_node];
738
739 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
740 {
741 // get global index of neighbouring node j
742 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
743
744 const double &dist_j = mdistances[j_neighbour];
745
746 // projection of pressure gradients
747 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
748
749 edge_ij.Add_grad_p(grad_d, dist_i, dist_j);
750 }
751
752 const double &m_inv = mr_matrix_container.GetInvertedMass()[i_node];
753 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
754 grad_d[l_comp] *= m_inv;
755
756 double norm_grad = norm_2(grad_d);
757
758 if (norm_grad < 100.0)
759 {
760 grad_d /= norm_grad; // this is the direction of the gradient of the distances
761
762 grad_d *= dist_i; // this is the vector with the distance of node_i from the closest point on the free surface
763
764 double pestimate = 0.0;
765
766 const array_1d<double, 3> &r_press_proj = iii->FastGetSolutionStepValue(PRESS_PROJ);
767 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
768 pestimate += r_press_proj[l_comp] * grad_d[l_comp];
769
770 iii->FastGetSolutionStepValue(PRESSURE) = pestimate;
771 }
772 else
773 {
774 std::cout << "attention gradient of distance much greater than 1 on node:" << i_node << std::endl;
775 double avg_number = 0.0;
776
777 double pavg = 0.0;
778
779 GlobalPointersVector<Node<3>> &neighb_nodes = iii->GetValue(NEIGHBOUR_NODES);
780 for (GlobalPointersVector<Node<3>>::iterator i = neighb_nodes.begin(); i != neighb_nodes.end(); i++)
781 {
782 if (i->GetValue(IS_VISITED) == 1.0)
783 {
784 pavg += i->FastGetSolutionStepValue(PRESSURE);
785 avg_number += 1.0;
786 }
787 }
788
789 KRATOS_ERROR_IF(avg_number == 0) << "can not happen that the extrapolation node has no neighbours" << std::endl;
790
791 iii->FastGetSolutionStepValue(PRESSURE) = pavg / avg_number;
792 }
793 }
794
795 // PREREQUISITES
796
797 // allocate memory for variables
798 ModelPart::NodesContainerType &rNodes = mr_model_part.Nodes();
799 int n_nodes = rNodes.size();
800 // unknown and right-hand side vector
801 TSystemVectorType dp, rhs;
802 dp.resize(n_nodes, false);
803 rhs.resize(n_nodes, false);
804 array_1d<double, TDim> dU_i, dU_j, work_array;
805 // read time step size from Kratos
806 ProcessInfo &CurrentProcessInfo = mr_model_part.GetProcessInfo();
807 double delta_t = CurrentProcessInfo[DELTA_TIME];
808
809 mr_matrix_container.FillOldScalarFromDatabase(PRESSURE, mPn, mr_model_part.Nodes());
810 mr_matrix_container.FillScalarFromDatabase(PRESSURE, mPn1, mr_model_part.Nodes());
811 mr_matrix_container.FillVectorFromDatabase(PRESS_PROJ, mXi, rNodes);
812 mr_matrix_container.FillVectorFromDatabase(VELOCITY, mvel_n1, rNodes);
813
814 // loop over all nodes
815 IndexPartition<unsigned int>(n_nodes).for_each([&](unsigned int i_node){
816
817 double &rhs_i = rhs[i_node];
818 rhs_i = 0.0;
819 const double &p_i = mPn1[i_node];
820 const double &p_old_i = mPn[i_node];
821 const array_1d<double, TDim> &U_i_curr = mvel_n1[i_node];
822
823 array_1d<double, TDim> &xi_i = mXi[i_node];
824
825 double l_ii = 0.0;
826
827 // loop over all neighbours
828 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
829 {
830 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
831 const double &p_j = mPn1[j_neighbour];
832 const double &p_old_j = mPn[j_neighbour];
833 const array_1d<double, TDim> &U_j_curr = mvel_n1[j_neighbour];
834 const array_1d<double, TDim> &xi_j = mXi[j_neighbour];
835
836 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
837
838#ifdef SYMM_PRESS
839 double edge_tau = 0.25 * (mTauPressure[i_node] + mTauPressure[j_neighbour]);
840#else
841 double edge_tau = 0.5 * mTauPressure[i_node];
842#endif
843
844 if (edge_tau < delta_t)
845 edge_tau = delta_t;
846
847 // compute laplacian operator
848 double sum_l_ikjk;
849 edge_ij.CalculateScalarLaplacian(sum_l_ikjk);
850
851 double sum_l_ikjk_onlydt = sum_l_ikjk * (delta_t);
852 sum_l_ikjk *= (delta_t + edge_tau);
853
854 // assemble right-hand side
855 // pressure contribution
856 rhs_i -= sum_l_ikjk * (p_j - p_i);
857 rhs_i += sum_l_ikjk_onlydt * (p_old_j - p_old_i);
858
859 // calculating the divergence of the fract vel
860 edge_ij.Sub_D_v(rhs_i, U_i_curr * mRho, U_j_curr * mRho);
861
862 // high order stabilizing term
863 double temp = 0.0;
864
865 edge_ij.Add_div_v(temp, xi_i, xi_j);
866 rhs_i += edge_tau * temp;
867
868 // assemble laplacian matrix
869 mL(i_node, j_neighbour) = sum_l_ikjk;
870 l_ii -= sum_l_ikjk;
871 }
872
873 mL(i_node, i_node) = l_ii;
874 });
875
876 if (muse_mass_correction == true)
877 {
878 IndexPartition<unsigned int>(n_nodes).for_each([&](unsigned int i_node){
879 double &rhs_i = rhs[i_node];
880 rhs_i -= mdiv_error[i_node];
881 });
882 }
883
884 // find the max diagonal term
885 double max_diag = 0.0;
886 for (int i_node = 0; i_node < n_nodes; i_node++)
887 {
888 double L_diag = mL(i_node, i_node);
889 if (std::abs(L_diag) > std::abs(max_diag))
890 max_diag = L_diag;
891 }
892 if (max_diag < 1e15)
893 max_diag = 1e15;
894
895 // respect pressure boundary conditions by penalization
896 for (unsigned int i_pressure = 0; i_pressure < mPressureOutletList.size(); i_pressure++)
897 {
898 unsigned int i_node = mPressureOutletList[i_pressure];
899 mL(i_node, i_node) = max_diag;
900 rhs[i_node] = 0.0;
901 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
902 {
903 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
904 mL(i_node, j_neighbour) = 0.0;
905 }
906 }
907
908 // modification for level_set
909 IndexPartition<unsigned int>(n_nodes).for_each([&](unsigned int i_node){
910 if (mdistances[i_node] >= 0)
911 {
912 mL(i_node, i_node) = max_diag;
913 rhs[i_node] = 0.0;
914 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
915 {
916 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
917 mL(i_node, j_neighbour) = 0.0;
918 }
919 }
920 else
921 {
922 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
923 {
924 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
925 if (mdistances[j_neighbour] >= 0)
926 mL(i_node, j_neighbour) = 0.0;
927 }
928 }
929 });
930
931 // compute row scaling factors
932 TSystemVectorType scaling_factors(n_nodes);
933 double *Lvalues = mL.value_data().begin();
934 SizeType *Lrow_indices = mL.index1_data().begin();
935 SizeType *Lcol_indices = mL.index2_data().begin();
936
937 IndexPartition<unsigned int>(mL.size1()).for_each([&](unsigned int k){
938 double t = 0.0;
939 SizeType col_begin = Lrow_indices[k];
940 SizeType col_end = Lrow_indices[k + 1];
941
942 for (SizeType j = col_begin; j < col_end; j++){
943 if (static_cast<unsigned int>(Lcol_indices[j]) == k)
944 {
945 t = fabs(Lvalues[j]);
946 break;
947 }
948 }
949 scaling_factors[k] = 1.0 / sqrt(t);
950 });
951
952 IndexPartition<unsigned int>(mL.size1()).for_each([&](unsigned int k){
953 SizeType col_begin = Lrow_indices[k];
954 SizeType col_end = Lrow_indices[k + 1];
955 double k_factor = scaling_factors[k];
956
957 rhs[k] *= k_factor;
958
959 for (SizeType j = col_begin; j < col_end; j++)
960 {
961 Lvalues[j] *= scaling_factors[Lcol_indices[j]] * k_factor;
962 }
963 });
964
965 // set starting vector for iterative solvers
966 IndexPartition<unsigned int>(n_nodes).for_each([&](unsigned int i_node){
967 dp[i_node] = 0.0;
968 });
969
970 pLinearSolver->Solve(mL, dp, rhs);
971
972 // update pressure
973 IndexPartition<unsigned int>(n_nodes).for_each([&](unsigned int i_node){
974 mPn1[i_node] += dp[i_node] * scaling_factors[i_node];
975 });
976
977 // write pressure and density to Kratos
978 mr_matrix_container.WriteScalarToDatabase(PRESSURE, mPn1, rNodes);
979
980 // compute pressure proj for the next step
981
982#pragma omp parallel for private(work_array)
983 for (int i_node = 0; i_node < n_nodes; i_node++)
984 {
985 array_1d<double, TDim> &xi_i = mXi[i_node];
986 for (unsigned int comp = 0; comp < TDim; comp++)
987 xi_i[comp] = 0.0;
988
989 double dist = mdistances[i_node];
990 if (dist <= 0.0) // node is inside domain ---- if outside do nothing
991 {
992
993 const double &p_i = mPn1[i_node];
994
995 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
996 {
997 // get global index of neighbouring node j
998 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
999
1000 const double &p_j = mPn1[j_neighbour];
1001
1002 // projection of pressure gradients
1003 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
1004
1005 edge_ij.Add_grad_p(xi_i, p_i, p_j);
1006 }
1007
1008 const double &m_inv = mr_matrix_container.GetInvertedMass()[i_node];
1009 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
1010 xi_i[l_comp] *= m_inv;
1011 }
1012 }
1013
1014 mr_matrix_container.WriteVectorToDatabase(PRESS_PROJ, mXi, rNodes);
1015
1016 KRATOS_CATCH("")
1017 }

1020 // function to solve fluid equations - fractional step 3: correct fractional momentum
1021
1022 void SolveStep3()
1023 {
1024 KRATOS_TRY
1025 // get number of nodes
1026 ModelPart::NodesContainerType &rNodes = mr_model_part.Nodes();
1027
1028 int n_nodes = rNodes.size();
1029
1030 // define work array
1031 array_1d<double, TDim> correction;
1032 // read time step size from Kratos
1033 ProcessInfo &CurrentProcessInfo = mr_model_part.GetProcessInfo();
1034 double delta_t = CurrentProcessInfo[DELTA_TIME];
1035
1036 double factor = 0.5;
1037 if (massume_constant_dp == true)
1038 factor = 1.0;
1039
1040 // compute end of step momentum
1041 double rho_inv = 1.0 / mRho;
1042#pragma omp parallel for private(correction) firstprivate(delta_t, rho_inv, factor)
1043 for (int i_node = 0; i_node < n_nodes; i_node++)
1044 {
1045 double dist = mdistances[i_node];
1046 if (dist < 0.0) // node is inside domain ---- if outside do nothing
1047 {
1048 array_1d<double, TDim> &U_i_curr = mvel_n1[i_node];
1049 double delta_p_i = (mPn1[i_node] - mPn[i_node]) * rho_inv * factor;
1050
1051 // setting to zero
1052 for (unsigned int l_comp = 0; l_comp < TDim; l_comp++)
1053 correction[l_comp] = 0.0;
1054
1055 // compute edge contributions dt*M^(-1)Gp
1056 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
1057 {
1058 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
1059 double delta_p_j = (mPn1[j_neighbour] - mPn[j_neighbour]) * rho_inv * factor;
1060
1061 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
1062
1063 edge_ij.Sub_grad_p(correction, delta_p_i, delta_p_j);
1064 }
1065 // compute prefactor
1066 const double m = mr_matrix_container.GetLumpedMass()[i_node];
1067 const double &d = mdiag_stiffness[i_node];
1068
1069 // correct fractional momentum
1070 for (unsigned int comp = 0; comp < TDim; comp++)
1071 {
1072 U_i_curr[comp] += delta_t / (m + delta_t * d) * correction[comp];
1073 }
1074 }
1075 }
1076
1077 ApplyVelocityBC(mvel_n1);
1078
1079 // write velocity of time step n+1 to Kratos
1080 mr_matrix_container.WriteVectorToDatabase(VELOCITY, mvel_n1, rNodes);
1081
1082 // calculate the error on the divergence
1083 if (muse_mass_correction == true)
1084 {
1085#pragma omp parallel for private(correction) firstprivate(delta_t, rho_inv)
1086 for (int i_node = 0; i_node < n_nodes; i_node++)
1087 {
1088 const double dist = mdistances[i_node];
1089 double &div_i_err = mdiv_error[i_node];
1090 div_i_err = 0.0;
1091 if (dist < 0.0) // node is inside domain ---- if outside do nothing
1092 {
1093 const array_1d<double, TDim> &U_i_curr = mvel_n1[i_node];
1094
1095 // compute edge contributions dt*M^(-1)Gp
1096 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
1097 {
1098 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
1099 array_1d<double, TDim> &U_j_curr = mvel_n1[j_neighbour];
1100
1101 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
1102
1103 edge_ij.Add_D_v(div_i_err, U_i_curr * mRho, U_j_curr * mRho);
1104 }
1105 }
1106 }
1107 }
1108
1109 KRATOS_CATCH("")

1113
1114 void ApplyVelocityBC(CalcVectorType &VelArray)
1115 {
1116 KRATOS_TRY
1117
1118 if (mWallLawIsActive == false)
1119 {
1120 // apply conditions on corner edges
1121 int edge_size = medge_nodes_direction.size();
1122#pragma omp parallel for firstprivate(edge_size)
1123 for (int i = 0; i < edge_size; i++)
1124 {
1125 int i_node = medge_nodes[i];
1126 const array_1d<double, TDim> &direction = medge_nodes_direction[i];
1127 double dist = mdistances[i_node];
1128
1129 if (dist <= 0.0)
1130 {
1131 array_1d<double, TDim> &U_i = VelArray[i_node];
1132 double temp = 0.0;
1133 for (unsigned int comp = 0; comp < TDim; comp++)
1134 temp += U_i[comp] * direction[comp];
1135
1136 for (unsigned int comp = 0; comp < TDim; comp++)
1137 U_i[comp] = direction[comp] * temp;
1138 }
1139 }
1140
1141 // apply conditions on corners
1142 int corner_size = mcorner_nodes.size();
1143 for (int i = 0; i < corner_size; i++)
1144 {
1145 int i_node = mcorner_nodes[i];
1146
1147 array_1d<double, TDim> &U_i = VelArray[i_node];
1148 for (unsigned int comp = 0; comp < TDim; comp++)
1149 U_i[comp] = 0.0;
1150 }
1151 }
1152
1153 // slip condition
1154 int slip_size = mSlipBoundaryList.size();
1155#pragma omp parallel for firstprivate(slip_size)
1156 for (int i_slip = 0; i_slip < slip_size; i_slip++)
1157 {
1158 unsigned int i_node = mSlipBoundaryList[i_slip];
1159 double dist = mdistances[i_node];
1160 if (dist <= 0.0)
1161 {
1162 array_1d<double, TDim> &U_i = VelArray[i_node];
1163 array_1d<double, TDim> &an_i = mSlipNormal[i_node];
1164 double projection_length = 0.0;
1165 double normalization = 0.0;
1166 for (unsigned int comp = 0; comp < TDim; comp++)
1167 {
1168 projection_length += U_i[comp] * an_i[comp];
1169 normalization += an_i[comp] * an_i[comp];
1170 }
1171 projection_length /= normalization;
1172 // tangential momentum as difference between original and normal momentum
1173 for (unsigned int comp = 0; comp < TDim; comp++)
1174 U_i[comp] -= projection_length * an_i[comp];
1175 }
1176 }
1177
1178 // fixed condition
1179 int fixed_size = mFixedVelocities.size();
1180#pragma omp parallel for firstprivate(fixed_size)
1181 for (int i_velocity = 0; i_velocity < fixed_size; i_velocity++)
1182 {
1183 unsigned int i_node = mFixedVelocities[i_velocity];
1184 double dist = mdistances[i_node];
1185 if (dist <= 0.0)
1186 {
1187 const array_1d<double, TDim> &u_i_fix = mFixedVelocitiesValues[i_velocity];
1188 array_1d<double, TDim> &u_i = VelArray[i_node];
1189
1190 for (unsigned int comp = 0; comp < TDim; comp++)
1191 u_i[comp] = u_i_fix[comp];
1192 }
1193 }

1195 KRATOS_CATCH("")
1196 }

2399 void CalculateRHS_convection(
2400 const ValuesVectorType &mphi,
2401 const CalcVectorType &convective_velocity,
2402 ValuesVectorType &rhs,
2403 ValuesVectorType &active_nodes)
2404 {
2405 KRATOS_TRY
2406
2407 int n_nodes = mphi.size();
2408
2409 // calculating the RHS
2410 double stab_low;
2411 double stab_high;
2412 array_1d<double, TDim> a_i;
2413 array_1d<double, TDim> a_j;
2414#pragma omp parallel for private(stab_low, stab_high, a_i, a_j)
2415 for (int i_node = 0; i_node < n_nodes; i_node++)
2416 {
2417 double &rhs_i = rhs[i_node];
2418 const double &h_i = mHavg[i_node];
2419 const double &phi_i = mphi[i_node];
2420 noalias(a_i) = convective_velocity[i_node];
2421 a_i /= mEps[i_node];
2422
2423 const array_1d<double, TDim> &proj_i = mPiConvection[i_node];
2424 double pi_i = proj_i[0] * a_i[0];
2425 for (unsigned int l_comp = 1; l_comp < TDim; l_comp++)
2426 pi_i += proj_i[l_comp] * a_i[l_comp];
2427
2428 rhs_i = 0.0;
2429
2430 if (active_nodes[i_node] != 0.0)
2431 {
2432 const double &beta = mBeta[i_node];
2433
2434 double norm_a = a_i[0] * a_i[0];
2435 for (unsigned int l_comp = 1; l_comp < TDim; l_comp++)
2436 norm_a += a_i[l_comp] * a_i[l_comp];
2437 norm_a = sqrt(norm_a);
2438
2439 // loop to all the edges surrounding node I
2440 for (unsigned int csr_index = mr_matrix_container.GetRowStartIndex()[i_node]; csr_index != mr_matrix_container.GetRowStartIndex()[i_node + 1]; csr_index++)
2441 {
2442 unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
2443
2444 if (active_nodes[j_neighbour] != 0.0)
2445 {
2446 const double &phi_j = mphi[j_neighbour];
2447 noalias(a_j) = convective_velocity[j_neighbour];
2448 a_j /= mEps[j_neighbour];
2449
2450 const array_1d<double, TDim> &proj_j = mPiConvection[j_neighbour];
2451 double pi_j = proj_j[0] * a_i[0];
2452 for (unsigned int l_comp = 1; l_comp < TDim; l_comp++)
2453 pi_j += proj_j[l_comp] * a_i[l_comp];
2454
2455 CSR_Tuple &edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
2456
2457 // convection operator
2458 edge_ij.Sub_ConvectiveContribution(rhs_i, a_i, phi_i, a_j, phi_j);
2459
2460 // calculate stabilization part
2461 edge_ij.CalculateConvectionStabilization_LOW(stab_low, a_i, phi_i, a_j, phi_j);
2462
2463 double edge_tau = mTauConvection[i_node];
2464
2465 edge_ij.CalculateConvectionStabilization_HIGH(stab_high, a_i, pi_i, a_j, pi_j);
2466
2467 edge_ij.Sub_StabContribution(rhs_i, edge_tau, 1.0, stab_low, stab_high);
2468
2469 double coeff = 0.5 * mshock_coeff; //=0.7*0.5;
2470 double laplacian_ij = 0.0;
2471 edge_ij.CalculateScalarLaplacian(laplacian_ij);
2472 double capturing = laplacian_ij * (phi_j - phi_i);
2473
2474 double aaa = 0.0;
2475 for (unsigned int k_comp = 0; k_comp < TDim; k_comp++)
2476 for (unsigned int m_comp = 0; m_comp < TDim; m_comp++)
2477 aaa += a_i[k_comp] * a_i[m_comp] * edge_ij.LaplacianIJ(k_comp, m_comp);
2478
2479 if (norm_a > 1e-10)
2480 {
2481 aaa /= (norm_a * norm_a);
2482 double capturing2 = aaa * (phi_j - phi_i);
2483
2484 if (fabs(capturing) > fabs(capturing2))
2485 rhs_i -= coeff * (capturing - capturing2) * beta * norm_a * h_i;
2486 }
2487 }
2488 }
2489 }
2490 }

2697 void ComputeWallResistance(
2698 const CalcVectorType &vel,
2699 ValuesVectorType &diag_stiffness
2700 )
2701 {
2702 // parameters:
2703 double k = 0.41;
2704 double B = 5.1;
2705 double toll = 1e-6;
2706 double ym = mY_wall;
2707 double y_plus_incercept = 10.9931899;
2708 unsigned int itmax = 100;
2709
2710 KRATOS_ERROR_IF(mViscosity[0] == 0) << "it is not possible to use the wall law with 0 viscosity" << std::endl;
2711
2712 // slip condition
2713 int slip_size = mSlipBoundaryList.size();
2714#pragma omp parallel for firstprivate(slip_size, B, toll, ym, y_plus_incercept, itmax)
2715 for (int i_slip = 0; i_slip < slip_size; i_slip++)
2716 {
2717 unsigned int i_node = mSlipBoundaryList[i_slip];
2718 double dist = mdistances[i_node];
2719 const double nu = mViscosity[i_node];
2720 if (dist <= 0.0)
2721 {
2722 // array_1d<double, TDim>& rhs_i = rhs[i_node];
2723 const array_1d<double, TDim> &U_i = vel[i_node];
2724 const array_1d<double, TDim> &an_i = mSlipNormal[i_node];
2725
2726 // compute the modulus of the velocity
2727 double mod_vel = 0.0;
2728 double area = 0.0;
2729 for (unsigned int comp = 0; comp < TDim; comp++)
2730 {
2731 mod_vel += U_i[comp] * U_i[comp];
2732 area += an_i[comp] * an_i[comp];
2733 }
2734 mod_vel = sqrt(mod_vel);
2735 area = sqrt(area);
2736 diag_stiffness[i_node] += area * mod_vel / pow(1.0 / k * log(100.00) + B, 2); /* * mWallReductionFactor[ i_node ];*/
2737 // now compute the skin friction
2738 double mod_uthaw = sqrt(mod_vel * nu / ym);
2739 const double y_plus = ym * mod_uthaw / nu;
2740
2741 if (y_plus > y_plus_incercept)
2742 {
2743 // begin cicle to calculate the real u_thaw's module:
2744 unsigned int it = 0;
2745 double dx = 1e10;
2746
2747 while (fabs(dx) > toll * mod_uthaw && it < itmax)
2748 {
2749 double a = 1.0 / k;
2750 double temp = a * log(ym * mod_uthaw / nu) + B;
2751 double y = mod_uthaw * (temp)-mod_vel;
2752 double y1 = temp + a;
2753 dx = y / y1;
2754 mod_uthaw -= dx;
2755 it = it + 1;
2756 }
2757
2758 if (it == itmax)
2759 std::cout << "attention max number of iterations exceeded in wall law computation" << std::endl;
2760 }
2761 }
2762 else
2763 diag_stiffness[i_node] += 0.0;
2764 }

2800 }

2802 // set to zero the gradients

2825 aux += grad_vy[comp] * grad_vy[comp];
2826 aux += grad_vz[comp] * grad_vz[comp];
2827 }
2828 aux *= 0.5;

2830 aux = 0.0;
2831 double turbulent_viscosity = Cs * h * h * sqrt(aux) ;
2832

2834 }
