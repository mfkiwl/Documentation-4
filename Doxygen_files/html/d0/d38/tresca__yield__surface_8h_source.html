157 static void CalculateDamageParameter(
158 ConstitutiveLaw::Parameters& rValues,
159 double& rAParameter,
160 const double CharacteristicLength
161 )
162 {
163 const Properties& r_material_properties = rValues.GetMaterialProperties();
164
165 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
166 const double young_modulus = r_material_properties[YOUNG_MODULUS];
167 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
168 const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
169 const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
170 const double n = yield_compression / yield_tension;
171
172 if (r_material_properties[SOFTENING_TYPE] == static_cast<int>(SofteningType::Exponential)) {
173 rAParameter = 1.00 / (fracture_energy * n * n * young_modulus / (CharacteristicLength * std::pow(yield_compression, 2)) - 0.5);
174 KRATOS_ERROR_IF(rAParameter < 0.0) << "Fracture energy is too low, increase FRACTURE_ENERGY..." << std::endl;
175 } else { // linear
176 rAParameter = -std::pow(yield_compression, 2) / (2.0 * young_modulus * fracture_energy * n * n / CharacteristicLength);
177 }
178 }

210 static void CalculateYieldSurfaceDerivative(
211 const array_1d<double, VoigtSize>& rPredictiveStressVector,
212 const array_1d<double, VoigtSize>& rDeviator,
213 const double J2,
214 array_1d<double, VoigtSize>& rFFlux,
215 ConstitutiveLaw::Parameters& rValues
216 )
217 {
218 array_1d<double, VoigtSize> second_vector, third_vector;
219
220 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateSecondVector(rDeviator, J2, second_vector);
221 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateThirdVector(rDeviator, J2, third_vector);
222
223 double J3, lode_angle;
224 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(rDeviator, J3);
225 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
226
227 const double checker = std::abs(lode_angle * 180.0 / Globals::Pi);
228
229 double c2, c3;
230 if (std::abs(checker) < 29.0) { // the lode_angle cannot be greater than pi/6
231 c2 = 2.0 * (std::cos(lode_angle) + std::sin(lode_angle) * std::tan(3.0 * lode_angle));
232 c3 = std::sqrt(3.0) * std::sin(lode_angle) / (J2 * std::cos(3.0 * lode_angle));
233 } else {
234 c2 = std::sqrt(3.0);
235 c3 = 0.0;
236 }
237
238 noalias(rFFlux) = c2 * second_vector + c3 * third_vector;
239 }

245 static int Check(const Properties& rMaterialProperties)
246 {
247 if (!rMaterialProperties.Has(YIELD_STRESS)) {
248 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
249 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
250
251 const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
252 const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
253
254 KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
255 KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
256 } else {
257 const double yield_stress = rMaterialProperties[YIELD_STRESS];
258
259 KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
260 }
261 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRACTURE_ENERGY)) << "FRACTURE_ENERGY is not a defined value" << std::endl;
262 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "YOUNG_MODULUS is not a defined value" << std::endl;
263
264 return TPlasticPotentialType::Check(rMaterialProperties);
265 }
