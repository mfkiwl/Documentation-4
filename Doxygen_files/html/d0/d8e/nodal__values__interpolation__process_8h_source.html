375  template<class TEntity>
376  void CalculateData(
377  NodeType::Pointer pNode,
378  const typename TEntity::Pointer& pEntity,
379  const Vector& rShapeFunctions
380  )
381  {
382  // The nodal data (non-historical) of each node of the original mesh
383  GeometryType& r_geometry = pEntity->GetGeometry();
384  const SizeType number_of_nodes = r_geometry.size();
385 
386  // Now we interpolate the values of each node
387  double aux_coeff = 0.0;
388  for (IndexType i = 0; i < number_of_nodes; ++i) {
389  aux_coeff += rShapeFunctions[i];
390  }
391  for (auto& r_variable_name : mListVariables) {
392  if (KratosComponents<Variable<double>>::Has(r_variable_name)) {
393  const Variable<double>& r_variable = KratosComponents<Variable<double>>::Get(r_variable_name);
394  if (std::abs(aux_coeff) > std::numeric_limits<double>::epsilon()) {
395  aux_coeff = 1.0/aux_coeff;
396  double aux_value = 0.0;
397  for (IndexType i = 0; i < number_of_nodes; ++i) {
398  if (r_geometry[i].Has(r_variable)) {
399  aux_value += rShapeFunctions[i] * r_geometry[i].GetValue(r_variable);
400  }
401  }
402  pNode->SetValue(r_variable, aux_coeff * aux_value);
403  }
404  } else if (KratosComponents<Variable<array_1d<double, 3>>>::Has(r_variable_name)) {
405  const Variable<array_1d<double, 3>>& r_variable = KratosComponents<Variable<array_1d<double, 3>>>::Get(r_variable_name);
406  if (std::abs(aux_coeff) > std::numeric_limits<double>::epsilon()) {
407  aux_coeff = 1.0/aux_coeff;
408  array_1d<double, 3> aux_value = ZeroVector(3);
409  for (IndexType i = 0; i < number_of_nodes; ++i) {
410  if (r_geometry[i].Has(r_variable)) {
411  aux_value += rShapeFunctions[i] * r_geometry[i].GetValue(r_variable);
412  }
413  }
414  pNode->SetValue(r_variable, aux_coeff * aux_value);
415  }
416  } else if (KratosComponents<Variable<Vector>>::Has(r_variable_name)) {
417  const Variable<Vector>& r_variable = KratosComponents<Variable<Vector>>::Get(r_variable_name);
418  if (std::abs(aux_coeff) > std::numeric_limits<double>::epsilon()) {
419  aux_coeff = 1.0/aux_coeff;
420  Vector aux_value = ZeroVector(r_geometry[0].GetValue(r_variable).size());
421  for (IndexType i = 0; i < number_of_nodes; ++i) {
422  if (r_geometry[i].Has(r_variable)) {
423  aux_value += rShapeFunctions[i] * r_geometry[i].GetValue(r_variable);
424  }
425  }
426  pNode->SetValue(r_variable, aux_coeff * aux_value);
427  }
428  } else if (KratosComponents<Variable<Matrix>>::Has(r_variable_name)) {
429  const Variable<Matrix>& r_variable = KratosComponents<Variable<Matrix>>::Get(r_variable_name);
430  if (std::abs(aux_coeff) > std::numeric_limits<double>::epsilon()) {
431  aux_coeff = 1.0/aux_coeff;
432  Matrix aux_value = ZeroMatrix(r_geometry[0].GetValue(r_variable).size1(), r_geometry[0].GetValue(r_variable).size2());
433  for (IndexType i = 0; i < number_of_nodes; ++i) {
434  if (r_geometry[i].Has(r_variable)) {
435  aux_value += rShapeFunctions[i] * r_geometry[i].GetValue(r_variable);
436  }
437  }
438  pNode->SetValue(r_variable, aux_coeff * aux_value);
439  }
440  }
441  }

455  const Vector& rShapeFunctions,
456  const IndexType Step
457  )

462  step_data[j] = 0;
463 
464  // The nodal data (historical) of each node of the original mesh
465  GeometryType& geom = pEntity->GetGeometry();

467  for (IndexType i = 0; i < number_of_nodes; ++i) {
468  const double* nodal_data = geom[i].SolutionStepData().Data(Step);
469  // Now we interpolate the values of each node

471  step_data[j] += rShapeFunctions[i] * nodal_data[j];
