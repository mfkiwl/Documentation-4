53 static void CalculateRightHandSideDerivative(TElementType& rElement,
54 const Vector& rRHS,
55 const array_1d_component_type& rDesignVariable,
56 Node<3>& rNode,
57 const double& rPertubationSize,
58 Vector& rOutput,
59 const ProcessInfo& rCurrentProcessInfo)
60 {
61 KRATOS_TRY;
62
63 if( rDesignVariable == SHAPE_SENSITIVITY_X || rDesignVariable == SHAPE_SENSITIVITY_Y || rDesignVariable == SHAPE_SENSITIVITY_Z )
64 {
65 const IndexType coord_dir =
66 FiniteDifferenceUtility::GetCoordinateDirection(rDesignVariable);
67
68 // define working variables
69 Vector RHS_perturbed;
70
71 if (rOutput.size() != rRHS.size())
72 rOutput.resize(rRHS.size(), false);
73
74 // perturb the design variable
75 rNode.GetInitialPosition()[coord_dir] += rPertubationSize;
76 rNode.Coordinates()[coord_dir] += rPertubationSize;
77
78 // compute LHS after perturbation
79 rElement.CalculateRightHandSide(RHS_perturbed, rCurrentProcessInfo);
80
81 // compute derivative of RHS w.r.t. design variable with finite differences
82 noalias(rOutput) = (RHS_perturbed - rRHS) / rPertubationSize;
83
84 // unperturb the design variable
85 rNode.GetInitialPosition()[coord_dir] -= rPertubationSize;
86 rNode.Coordinates()[coord_dir] -= rPertubationSize;
87 }
88 else
89 {
90 KRATOS_WARNING("FiniteDifferenceUtility") << "Unsupported nodal design variable: " << rDesignVariable << std::endl;
91 if ( (rOutput.size() != 0) )
92 rOutput.resize(0,false);
93 }
94
95 KRATOS_CATCH("");
96 }
