309 std::tuple<std::vector<TDataType>, std::vector<TDataType>> CalculateConvergenceNorms(
310 const ModelPart& rModelPart,
311 const DofsArrayType& rDofSet,
312 const TSystemVectorType& rDx)
313 {
314 // Initialize
315 std::vector<int> dofs_count(mVariableSize, 0);
316 std::vector<TDataType> solution_norms_vector(mVariableSize, 0.0);
317 std::vector<TDataType> increase_norms_vector(mVariableSize, 0.0);
318
319 // Accumulate the norm values
320 GetNormValues(rModelPart, rDofSet, rDx, dofs_count, solution_norms_vector, increase_norms_vector);
321
322 // Synchronize the norm values
323 const auto& r_data_comm = rModelPart.GetCommunicator().GetDataCommunicator();
324 auto global_solution_norms_vector = r_data_comm.SumAll(solution_norms_vector);
325 auto global_increase_norms_vector = r_data_comm.SumAll(increase_norms_vector);
326 auto global_dofs_count = r_data_comm.SumAll(dofs_count);
327
328 // Check division by zero in global solution norms
329 const double zero_tol = 1.0e-12;
330 for(int i = 0; i < mVariableSize; i++) {
331 if (global_solution_norms_vector[i] < zero_tol) {
332 global_solution_norms_vector[i] = 1.0;
333 }
334 }
335
336 // Calculate the norm values
337 std::vector<TDataType> var_ratio(mVariableSize, 0.0);
338 std::vector<TDataType> var_abs(mVariableSize, 0.0);
339 for(int i = 0; i < mVariableSize; i++) {
340 var_ratio[i] = std::sqrt(global_increase_norms_vector[i] / global_solution_norms_vector[i]);
341 var_abs[i] = std::sqrt(global_increase_norms_vector[i]) / static_cast<TDataType>(global_dofs_count[i]);
342 }
343
344 // Output the ratio and absolute norms as a tuple
345 return std::make_tuple(var_ratio, var_abs);
346 }

385 bool CheckConvergence(
386 const std::tuple<std::vector<TDataType>,
387 std::vector<TDataType>>& rConvergenceNorms)
388 {
389 bool is_converged = true;
390 const auto& var_ratio = std::get<0>(rConvergenceNorms);
391 const auto& var_abs = std::get<1>(rConvergenceNorms);
392
393 for (int i = 0; i < mVariableSize; i++) {
394 const auto r_var_data = mVariableDataVector[i];
395 const int key_map = mLocalKeyMap[r_var_data->Key()];
396 is_converged &= var_ratio[key_map] <= mRatioToleranceVector[key_map] || var_abs[key_map] <= mAbsToleranceVector[key_map];
397 }
398
399 // Note that this check ensures that all the convergence variables fulfil either the relative or the absolute criterion
400 if (is_converged) {
401 KRATOS_INFO_IF("", this->GetEchoLevel() > 0) << "*** CONVERGENCE IS ACHIEVED ***" << std::endl;
402 return true;
403 } else {
404 return false;
405 }

459 virtual void GetNormValues(
460 const ModelPart& rModelPart,
461 const DofsArrayType& rDofSet,
462 const TSystemVectorType& rDx,
463 std::vector<int>& rDofsCount,
464 std::vector<TDataType>& rSolutionNormsVector,
465 std::vector<TDataType>& rIncreaseNormsVector)
466 {
467 int n_dofs = rDofSet.size();
468
469 // Loop over Dofs
470#pragma omp parallel
471 {
472 // Local thread variables
473 int dof_id;
474 TDataType dof_dx;
475 TDataType dof_value;
476
477 // Local reduction variables
478 std::vector<TDataType> var_solution_norm_reduction(mVariableSize);
479 std::vector<TDataType> var_correction_norm_reduction(mVariableSize);
480 std::vector<int> dofs_counter_reduction(mVariableSize);
481 for (int i = 0; i < mVariableSize; i++) {
482 var_solution_norm_reduction[i] = 0.0;
483 var_correction_norm_reduction[i] = 0.0;
484 dofs_counter_reduction[i] = 0;
485 }
486
487#pragma omp for
488 for (int i = 0; i < n_dofs; i++) {
489 auto it_dof = rDofSet.begin() + i;
490 if (it_dof->IsFree()) {
491 dof_id = it_dof->EquationId();
492 dof_value = it_dof->GetSolutionStepValue(0);
493 dof_dx = TSparseSpace::GetValue(rDx, dof_id);
494
495 const auto &r_current_variable = it_dof->GetVariable();
496 int var_local_key = mLocalKeyMap[r_current_variable.IsComponent() ? r_current_variable.GetSourceVariable().Key() : r_current_variable.Key()];
497
498 var_solution_norm_reduction[var_local_key] += dof_value * dof_value;
499 var_correction_norm_reduction[var_local_key] += dof_dx * dof_dx;
500 dofs_counter_reduction[var_local_key]++;
501 }
502 }
503
504#pragma omp critical
505 {
506 for (int i = 0; i < mVariableSize; i++) {
507 rDofsCount[i] += dofs_counter_reduction[i];
508 rSolutionNormsVector[i] += var_solution_norm_reduction[i];
509 rIncreaseNormsVector[i] += var_correction_norm_reduction[i];
510 }
511 }
512 }

520 static ConvergenceVariableListType GenerateConvergenceVariableListFromParameters(Kratos::Parameters ThisParameters)
521 {
522 // Iterate over variables
523 ConvergenceVariableListType aux_list;
524 if (!ThisParameters.Has("convergence_variables_list")) return aux_list;
525 Kratos::Parameters convergence_variables_list = ThisParameters["convergence_variables_list"];
526 for (auto param : convergence_variables_list) {
527 if (param.Has("variable")) {
528 const std::string& r_variable_name = param["variable"].GetString();
529
530 // Variable pointer
531 const VariableData* p_variable = KratosComponents<Variable<double>>::Has(r_variable_name) ? dynamic_cast<const VariableData*>(&KratosComponents<Variable<double>>::Get(r_variable_name)) : dynamic_cast<const VariableData*>(&KratosComponents<Variable<array_1d<double, 3>>>::Get(r_variable_name));
532
533 // Tolerances
534 const double rel_tol = param.Has("relative_tolerance") ? param["relative_tolerance"].GetDouble() : 1.0e-4;
535 const double abs_tol = param.Has("absolute_tolerance") ? param["absolute_tolerance"].GetDouble() : 1.0e-9;
536
537 // Push back list
538 aux_list.push_back(std::make_tuple(p_variable, rel_tol, abs_tol));
539 }
540 }
541
542 return aux_list;
543 }
