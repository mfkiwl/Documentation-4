129 void ExecuteInitialize() override
130 {
131
132 KRATOS_TRY;
133
134 //Defining necessary variables
135 const Variable<double>& var = KratosComponents<Variable<double>>::Get(mVariableName);
136 const int nnodes = mrModelPart.GetMesh(0).Nodes().size();
137 BoundedMatrix<double, 3, 3> RotationMatrix;
138
139 // Computing the rotation matrix accoding with the introduced points by the user
140 this->CalculateRotationMatrix(RotationMatrix);
141 array_1d<double, 3> newCoordinate;
142 array_1d<double, 3> auxiliar_vector;
143 array_1d<double, 3> reference_vector;
144
145 // Gravity direction for computing the hydrostatic pressure
146 int direction;
147
148 if (mGravityDirection == "X")
149 direction = 0;
150 else if (mGravityDirection == "Y")
151 direction = 1;
152 else
153 direction = 2;
154
155 // Computing the reference vector (coordinates)
156 reference_vector = prod(RotationMatrix, mX0);
157
158 if (nnodes != 0)
159 {
160 ModelPart::NodesContainerType::iterator it_begin = mrModelPart.GetMesh(0).NodesBegin();
161
162 if (mDrain == true)
163 {
164 double coefficient_effectiveness = 1.0 - mEffectivenessDrain;
165 double aux_drain = coefficient_effectiveness * ((mWaterLevel - mReferenceCoordinate) - mHeightDrain) * ((mBaseDam - mDistanceDrain) / mBaseDam) + mHeightDrain;
166
167#pragma omp parallel for
168 for (int i = 0; i < nnodes; i++)
169 {
170 ModelPart::NodesContainerType::iterator it = it_begin + i;
171
172 auxiliar_vector.resize(3, false);
173 const array_1d<double,3>& r_coordinates = it->Coordinates();
174 noalias(auxiliar_vector) = r_coordinates;
175
176 // Computing the new coordinates
177 newCoordinate = prod(RotationMatrix, auxiliar_vector);
178
179 // We compute the first part of the uplift law
180 mUpliftPressure = (mSpecific * ((mWaterLevel - aux_drain) - (r_coordinates[direction]))) * (1.0 - ((1.0 / (mDistanceDrain)) * (fabs((newCoordinate(0)) - reference_vector(0))))) + mSpecific * aux_drain;
181
182 // If uplift pressure is greater than the limit we compute the second part and we update the value
183 if (mUpliftPressure <= mSpecific * aux_drain)
184 {
185 mUpliftPressure = (mSpecific * ((mReferenceCoordinate + aux_drain) - (r_coordinates[direction]))) * (1.0 - ((1.0 / (mBaseDam - mDistanceDrain)) * (fabs((newCoordinate(0)) - (reference_vector(0) + mDistanceDrain)))));
186 }
187
188 if (mUpliftPressure < 0.0)
189 {
190 it->FastGetSolutionStepValue(var) = 0.0;
191 }
192 else
193 {
194 it->FastGetSolutionStepValue(var) = mUpliftPressure;
195 }
196 }
197 }
198 else
199 {
200#pragma omp parallel for
201 for (int i = 0; i < nnodes; i++)
202 {
203 ModelPart::NodesContainerType::iterator it = it_begin + i;
204
205 auxiliar_vector.resize(3, false);
206 const array_1d<double,3>& r_coordinates = it->Coordinates();
207 noalias(auxiliar_vector) = r_coordinates;
208
209 newCoordinate = prod(RotationMatrix, auxiliar_vector);
210
211 mUpliftPressure = (mSpecific * (mWaterLevel - (r_coordinates[direction]))) * (1.0 - ((1.0 / mBaseDam) * (fabs(newCoordinate(0) - reference_vector(0)))));
212
213 if (mUpliftPressure < 0.0)
214 {
215 it->FastGetSolutionStepValue(var) = 0.0;
216 }
217 else
218 {
219 it->FastGetSolutionStepValue(var) = mUpliftPressure;
220 }
221 }
222 }
223 }
224
225 KRATOS_CATCH("");
226 }

339 void CalculateRotationMatrix(BoundedMatrix<double, 3, 3> &rRotationMatrix)
340 {
341 KRATOS_TRY;
342
343 //Unitary vector in uplift direction
344 array_1d<double, 3> V_uplift;
345 V_uplift = (mX1 - mX0);
346 mBaseDam = norm_2(V_uplift);
347 double inv_norm_uplift = 1.0 / norm_2(V_uplift);
348 V_uplift[0] *= inv_norm_uplift;
349 V_uplift[1] *= inv_norm_uplift;
350 V_uplift[2] *= inv_norm_uplift;
351
352 //Unitary vector in longitudinal direction
353 array_1d<double, 3> V_longitudinal;
354 V_longitudinal = (mX2 - mX0);
355 double inv_norm_longitudinal = 1.0 / norm_2(V_longitudinal);
356 V_longitudinal[0] *= inv_norm_longitudinal;
357 V_longitudinal[1] *= inv_norm_longitudinal;
358 V_longitudinal[2] *= inv_norm_longitudinal;
359
360 //Unitary vector in local z direction
361 array_1d<double, 3> V_normal;
362 MathUtils<double>::CrossProduct(V_normal, V_uplift, V_longitudinal);
363
364 //Rotation Matrix
365 rRotationMatrix(0, 0) = V_uplift[0];
366 rRotationMatrix(0, 1) = V_uplift[1];
367 rRotationMatrix(0, 2) = V_uplift[2];
368
369 rRotationMatrix(1, 0) = V_longitudinal[0];
370 rRotationMatrix(1, 1) = V_longitudinal[1];
371 rRotationMatrix(1, 2) = V_longitudinal[2];
372
373 rRotationMatrix(2, 0) = V_normal[0];
374 rRotationMatrix(2, 1) = V_normal[1];
375 rRotationMatrix(2, 2) = V_normal[2];
376
377 KRATOS_CATCH("")
