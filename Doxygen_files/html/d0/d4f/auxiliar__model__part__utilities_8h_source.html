299 void GetScalarData(

301 const DataLocation DataLoc,
302 TContainerType& data) const
303 {
304 KRATOS_TRY
305
306 switch (DataLoc)
307 {
308 case (DataLocation::NodeHistorical):{
309 data.resize(mrModelPart.NumberOfNodes());
310
311 auto inodebegin = mrModelPart.NodesBegin();
312
313 IndexPartition<IndexType>(mrModelPart.NumberOfNodes()).for_each([&](IndexType Index){

336 GetScalarDataFromContainer(mrModelPart.Conditions(), rVariable, data);

340 data.resize(1);
341 data[0] = mrModelPart[rVariable];
342 break;

344 case (DataLocation::ProcessInfo):{
345 data.resize(1);
346 data[0] = mrModelPart.GetProcessInfo()[rVariable];
347 break;

359 template<class TContainerType, class TVarType>
360 void GetVectorData(
361 const Variable<TVarType>& rVariable,
362 const DataLocation DataLoc,
363 TContainerType& data) const
364 {
365 KRATOS_TRY
366
367 switch (DataLoc)
368 {
369 case (DataLocation::NodeHistorical):{
370 unsigned int TSize = mrModelPart.NumberOfNodes() > 0 ? mrModelPart.NodesBegin()->FastGetSolutionStepValue(rVariable).size() : 0;
371
372 TSize = mrModelPart.GetCommunicator().GetDataCommunicator().MaxAll(TSize);
373 data.resize(mrModelPart.NumberOfNodes()*TSize);
374
375 auto inodebegin = mrModelPart.NodesBegin();
376
377 IndexPartition<IndexType>(mrModelPart.NumberOfNodes()).for_each([&](IndexType Index){
378 auto inode = inodebegin + Index;
379
380 const auto& r_val = inode->FastGetSolutionStepValue(rVariable);
381 for(std::size_t dim = 0 ; dim < TSize ; dim++){
382 data[(Index*TSize) + dim] = r_val[dim];
383 }
384 });
385
386 break;
387 }
388 case (DataLocation::NodeNonHistorical):{
389 unsigned int TSize = mrModelPart.NumberOfNodes() > 0 ? mrModelPart.NodesBegin()->GetValue(rVariable).size() : 0;
390
391 TSize = mrModelPart.GetCommunicator().GetDataCommunicator().MaxAll(TSize);
392
393 data.resize(mrModelPart.NumberOfNodes()*TSize);
394
395 GetVectorDataFromContainer(mrModelPart.Nodes(), TSize, rVariable, data);
396 break;
397 }
398 case (DataLocation::Element):{
