303 void PerformSolutionStep (
304 SparseMatrixType& rA,
305 VectorType& rX,
306 VectorType& rB
307 ) override
308 {
309 // Auxiliary size
310 const SizeType lm_active_size = mLMActiveIndices.size();
311 const SizeType lm_inactive_size = mLMInactiveIndices.size();
312 const SizeType total_disp_size = mOtherIndices.size() + mMasterIndices.size() + mSlaveInactiveIndices.size() + mSlaveActiveIndices.size();
313
314 // Get the u and lm residuals
315 GetUPart (rB, mResidualDisp);
316
317 // Solve u block
318 if (mDisp.size() != total_disp_size)
319 mDisp.resize(total_disp_size, false);
320 mpSolverDispBlock->Solve (mKDispModified, mDisp, mResidualDisp);
321
322 // Write back solution
323 SetUPart(rX, mDisp);
324
325 // Solve LM
326 if (lm_active_size > 0) {
327 // Now we compute the residual of the LM
328 GetLMAPart (rB, mResidualLMActive);
329
330 // LM = D⁻1*rLM
331 if (mLMActive.size() != lm_active_size)
332 mLMActive.resize(lm_active_size, false);
333 TSparseSpaceType::Mult (mKLMAModified, mResidualLMActive, mLMActive);
334
335 // Write back solution
336 SetLMAPart(rX, mLMActive);
337 }
338
339 if (lm_inactive_size > 0) {
340 // Now we compute the residual of the LM
341 GetLMIPart (rB, mResidualLMInactive);
342
343 // LM = D⁻1*rLM
344 if (mLMInactive.size() != lm_inactive_size)
345 mLMInactive.resize(lm_inactive_size, false);
346 TSparseSpaceType::Mult (mKLMIModified, mResidualLMInactive, mLMInactive);
347
348 // Write back solution
349 SetLMIPart(rX, mLMInactive);
350 }
