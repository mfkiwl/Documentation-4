159 bool PostCriteria(
160 ModelPart& rModelPart,
161 DofsArrayType& rDofSet,
162 const TSystemMatrixType& A,
163 const TSystemVectorType& Dx,
164 const TSystemVectorType& b
165 ) override
166 {
167 const TDataType approx_zero_tolerance = std::numeric_limits<TDataType>::epsilon();
168 const SizeType size_Dx = Dx.size();
169 if (size_Dx != 0) { //if we are solving for something
170 SizeType size_solution;
171 TDataType final_correction_norm = CalculateFinalCorrectionNorm(size_solution, rDofSet, Dx);
172
173 TDataType ratio = 0.0;
174
175 CalculateReferenceNorm(rDofSet);
176 if (mReferenceDispNorm < approx_zero_tolerance) {
177 KRATOS_WARNING("DisplacementCriteria") << "NaN norm is detected. Setting reference to convergence criteria" << std::endl;
178 mReferenceDispNorm = final_correction_norm;
179 }
180
181 if(final_correction_norm < approx_zero_tolerance) {
182 ratio = 0.0;
183 } else {
184 ratio = final_correction_norm/mReferenceDispNorm;
185 }
186
187 const TDataType float_size_solution = static_cast<TDataType>(size_solution);
188
189 const TDataType absolute_norm = (final_correction_norm/std::sqrt(float_size_solution));
190
191 KRATOS_INFO_IF("DISPLACEMENT CRITERION", this->GetEchoLevel() > 0 && rModelPart.GetCommunicator().MyPID() == 0) << " :: [ Obtained ratio = " << ratio << "; Expected ratio = " << mRatioTolerance << "; Absolute norm = " << absolute_norm << "; Expected norm = " << mAlwaysConvergedNorm << "]" << std::endl;
192
193 rModelPart.GetProcessInfo()[CONVERGENCE_RATIO] = ratio;
194 rModelPart.GetProcessInfo()[RESIDUAL_NORM] = absolute_norm;
195
196 if ( ratio <= mRatioTolerance || absolute_norm<mAlwaysConvergedNorm ) { // || (final_correction_norm/x.size())<=1e-7)
197 KRATOS_INFO_IF("DISPLACEMENT CRITERION", this->GetEchoLevel() > 0 && rModelPart.GetCommunicator().MyPID() == 0) << "Convergence is achieved" << std::endl;
198 return true;
199 } else {
200 return false;
201 }
202 } else { //in this case all the displacements are imposed!
203 return true;
204 }
205 }

393 void CalculateReferenceNorm(DofsArrayType& rDofSet)
394 {
395 TDataType reference_disp_norm = TDataType();
396 TDataType dof_value;
397
398 #pragma omp parallel for reduction(+:reference_disp_norm)
399 for (int i = 0; i < static_cast<int>(rDofSet.size()); i++) {
400 auto it_dof = rDofSet.begin() + i;
401
402 if(it_dof->IsFree()) {
403 dof_value = it_dof->GetSolutionStepValue();
404 reference_disp_norm += dof_value * dof_value;
405 }
406 }
407 mReferenceDispNorm = std::sqrt(reference_disp_norm);
408 }
409
417 TDataType CalculateFinalCorrectionNorm(
418 SizeType& rDofNum,
419 DofsArrayType& rDofSet,
420 const TSystemVectorType& Dx
421 )
422 {
423 // Initialize
424 TDataType final_correction_norm = TDataType();
425 SizeType dof_num = 0;
426
427 // Loop over Dofs
428 #pragma omp parallel for reduction(+:final_correction_norm,dof_num)
429 for (int i = 0; i < static_cast<int>(rDofSet.size()); i++) {
430 auto it_dof = rDofSet.begin() + i;
431
432 IndexType dof_id;
433 TDataType variation_dof_value;
434
435 if (it_dof->IsFree()) {
436 dof_id = it_dof->EquationId();
437 variation_dof_value = Dx[dof_id];
438 final_correction_norm += std::pow(variation_dof_value, 2);
439 dof_num++;
440 }
441 }
442
443 rDofNum = dof_num;
444 return std::sqrt(final_correction_norm);
445 }
