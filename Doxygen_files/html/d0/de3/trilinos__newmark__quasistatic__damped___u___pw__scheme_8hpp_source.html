142 virtual void InitializeDofImporter(
143 DofsArrayType& rDofSet,
144 TSystemVectorType& Dx
145 )
146 {
147 int system_size = TSparseSpace::Size(Dx);
148 int number_of_dofs = rDofSet.size();
149 std::vector< int > index_array(number_of_dofs);
150
151 // Filling the array with the global ids
152 unsigned int counter = 0;
153 for(typename DofsArrayType::iterator i_dof = rDofSet.begin() ; i_dof != rDofSet.end() ; ++i_dof)
154 {
155 int id = i_dof->EquationId();
156 if( id < system_size )
157 {
158 index_array[counter] = id;
159 counter += 1;
160 }
161 }
162
163 std::sort(index_array.begin(),index_array.end());
164 std::vector<int>::iterator NewEnd = std::unique(index_array.begin(),index_array.end());
165 index_array.resize(NewEnd-index_array.begin());
166
167 int check_size = -1;
168 int tot_update_dofs = index_array.size();
169 Dx.Comm().SumAll(&tot_update_dofs,&check_size,1);
170 if ( (check_size < system_size) && (Dx.Comm().MyPID() == 0) )
171 {
172 std::stringstream Msg;
173 Msg << "Dof count is not correct. There are less dofs then expected." << std::endl;
174 Msg << "Expected number of active dofs = " << system_size << " dofs found = " << check_size << std::endl;
175 KRATOS_ERROR << Msg.str()
176 << std::endl;
177 }
178
179 // Defining a map as needed
180 Epetra_Map dof_update_map(-1,index_array.size(), &(*(index_array.begin())),0,Dx.Comm() );
181
182 // Defining the importer class
183 Kratos::shared_ptr<Epetra_Import> pDofImporter = Kratos::make_shared<Epetra_Import>(dof_update_map,Dx.Map());
184 mpDofImporter.swap(pDofImporter);
185
186 mImporterIsInitialized = true;
187 }
