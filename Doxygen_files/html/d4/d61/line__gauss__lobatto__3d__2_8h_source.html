379 jacobian( 1, 0 ) = ( this->GetPoint( 1 ).Y() - this->GetPoint( 0 ).Y() ) * 0.5;

771 static Matrix CalculateShapeFunctionsIntegrationPointsValues( typename BaseType::IntegrationMethod ThisMethod )
772 {
773 const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
774 const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[static_cast<int>(ThisMethod)];
775 int integration_points_number = IntegrationPoints.size();
776 Matrix N( integration_points_number, 2 );
777
778 for ( int it_gp = 0; it_gp < integration_points_number; it_gp++ )
779 {
780 double e = IntegrationPoints[it_gp].X();
781 N( it_gp, 0 ) = 0.5 * ( 1 - e );
782 N( it_gp, 1 ) = 0.5 * ( 1 + e );
783 }
784
785 return N;
786 }
