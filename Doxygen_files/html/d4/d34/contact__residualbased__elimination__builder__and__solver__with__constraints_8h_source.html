277  void SetUpDofSetWithConstraints(
278  typename TSchemeType::Pointer pScheme,
279  ModelPart& rModelPart
280  )
281  {
282  KRATOS_TRY;
283 
284  // We are going to enforce the existence of constraints for LM for each displacement dof
285  if (rModelPart.NodesBegin()->SolutionStepsDataHas(VECTOR_LAGRANGE_MULTIPLIER)) {
286  // Reorder constrains
287  IndexType constraint_id = 1;
288  for (auto& constrain : rModelPart.MasterSlaveConstraints()) {
289  constrain.SetId(constraint_id);
290  ++constraint_id;
291  }
292 
293  // Auxiliary dofs lists
294  DofsVectorType dof_list, second_dof_list; // NOTE: The second dof list is only used on constraints to include master/slave relations
295 
296  // Contributions to the system
297  LocalSystemMatrixType transformation_matrix = LocalSystemMatrixType(0, 0);
298  LocalSystemVectorType constant_vector = LocalSystemVectorType(0);
299 
300  // Reference constraint
301  const auto& r_clone_constraint = KratosComponents<MasterSlaveConstraint>::Get("LinearMasterSlaveConstraint");
302 
303  #pragma omp parallel firstprivate(transformation_matrix, constant_vector, dof_list, second_dof_list)
304  {
305  // Current process info
306  ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
307 
308  // A buffer to store auxiliary constraints
309  ConstraintContainerType constraints_buffer;
310 
311  // Gets the array of constraints from the modeler
312  auto& r_constraints_array = rModelPart.MasterSlaveConstraints();
313  const int number_of_constraints = static_cast<int>(r_constraints_array.size());
314  #pragma omp for schedule(guided, 512)
315  for (int i = 0; i < number_of_constraints; ++i) {
316  auto it_const = r_constraints_array.begin() + i;
317 
318  // Gets list of Dof involved on every element
319  it_const->GetDofList(dof_list, second_dof_list, r_current_process_info);
320  it_const->CalculateLocalSystem(transformation_matrix, constant_vector, r_current_process_info);
321 
322  DofPointerVectorType slave_dofs, master_dofs;
323  bool create_lm_constraint = false;
324 
325  // We check if we have SLAVE nodes in the master dofs
326  bool slave_nodes_master_dof = false;
327  // Master DoFs
328  for (auto& p_dof : second_dof_list) {
329  if (IsDisplacementDof(*p_dof)) {
330  const IndexType node_id = p_dof->Id();
331  auto pnode = rModelPart.pGetNode(node_id);
332  if (pnode->Is(SLAVE)) { // The nodes computing contact are the slave nodes
333  slave_nodes_master_dof = true;
334  break;
335  }
336  }
337  }
338 
339  // Slave DoFs
340  for (auto& p_dof : dof_list) {
341  if (IsDisplacementDof(*p_dof)) {
342  const IndexType node_id = p_dof->Id();
343  const auto& r_variable = p_dof->GetVariable();
344  auto pnode = rModelPart.pGetNode(node_id);
345  if (pnode->IsNot(INTERFACE) || slave_nodes_master_dof) { // Nodes from the contact interface cannot be slave DoFs
346  if (r_variable == DISPLACEMENT_X) {
347  slave_dofs.push_back(pnode->pGetDof(VECTOR_LAGRANGE_MULTIPLIER_X));
348  } else if (r_variable == DISPLACEMENT_Y) {
349  slave_dofs.push_back(pnode->pGetDof(VECTOR_LAGRANGE_MULTIPLIER_Y));
350  } else if (r_variable == DISPLACEMENT_Z) {
351  slave_dofs.push_back(pnode->pGetDof(VECTOR_LAGRANGE_MULTIPLIER_Z));
352  }
353  } else { // We remove it
354  it_const->Set(TO_ERASE);
355  }
356  }
357  }
358  // Master DoFs
359  if (slave_nodes_master_dof) { // The nodes computing contact are the slave nodes
360  for (auto& p_dof : second_dof_list) {
361  if (IsDisplacementDof(*p_dof)) {
362  const IndexType node_id = p_dof->Id();
363  const auto& r_variable = p_dof->GetVariable();
364  auto pnode = rModelPart.pGetNode(node_id);
365  if (r_variable == DISPLACEMENT_X) {
366  master_dofs.push_back(pnode->pGetDof(VECTOR_LAGRANGE_MULTIPLIER_X));
367  } else if (r_variable == DISPLACEMENT_Y) {
368  master_dofs.push_back(pnode->pGetDof(VECTOR_LAGRANGE_MULTIPLIER_Y));
369  } else if (r_variable == DISPLACEMENT_Z) {
370  master_dofs.push_back(pnode->pGetDof(VECTOR_LAGRANGE_MULTIPLIER_Z));
371  }
372  }
373  }
374  }
375 
376  // We check if we create constraints
377  if ((slave_dofs.size() == dof_list.size()) &&
378  (master_dofs.size() == second_dof_list.size())) {
379  create_lm_constraint = true;
380  }
381 
382  // We create the new constraint
383  if (create_lm_constraint) {
384  auto p_constraint = r_clone_constraint.Create(constraint_id + i + 1, master_dofs, slave_dofs, transformation_matrix, constant_vector);
385  (constraints_buffer).insert((constraints_buffer).begin(), p_constraint);
386  }
387  }
388 
389  // We transfer
390  #pragma omp critical
391  {
392  rModelPart.AddMasterSlaveConstraints(constraints_buffer.begin(),constraints_buffer.end());
393  }
394  }
395  }
396 
397  // We remove the marked constraints
398  rModelPart.RemoveMasterSlaveConstraintsFromAllLevels(TO_ERASE);
399 
400  KRATOS_INFO_IF("ContactResidualBasedEliminationBuilderAndSolverWithConstraints", (this->GetEchoLevel() > 0)) <<
401  "Model part after creating new constraints" << rModelPart << std::endl;
402 
403  // Calling base SetUpDofSetWithConstraints
404  BaseType::SetUpDofSetWithConstraints(pScheme, rModelPart);
405 
406  KRATOS_CATCH("");
407  }

480  void BaseSetUpSystem(ModelPart& rModelPart)
481  {
486  // We create a set of dofs of the displacement slave dofs with LM associated
487  std::unordered_map<IndexType, IndexSetType> set_nodes_with_lm_associated;
488  if (rModelPart.HasSubModelPart("Contact"))
489  set_nodes_with_lm_associated.reserve(rModelPart.GetSubModelPart("Contact").NumberOfNodes());
490  // Allocating auxiliary parameters
491  IndexType node_id;
492  // We start the dof loop
493  for (auto& i_dof : BaseType::mDofSet) {
494  node_id = i_dof.Id();
495  if (IsLMDof(i_dof))
496  set_nodes_with_lm_associated.insert({node_id, IndexSetType({})});
497  }
498 
499  // Auxiliary keys
500  const IndexType key_lm_x = VECTOR_LAGRANGE_MULTIPLIER_X.Key();
501  const IndexType key_lm_y = VECTOR_LAGRANGE_MULTIPLIER_Y.Key();
502  const IndexType key_lm_z = VECTOR_LAGRANGE_MULTIPLIER_Z.Key();
503 
504  // We see which LM block
505  for (auto& i_dof : BaseType::mDofSet) {
506  node_id = i_dof.Id();
507  auto it = set_nodes_with_lm_associated.find(node_id);
508  if ( it != set_nodes_with_lm_associated.end()) {
509  if (i_dof.IsFixed()) {
510  const auto& r_variable = i_dof.GetVariable();
511  auto& aux_set = (it->second);
512  if (r_variable == DISPLACEMENT_X) {
513  aux_set.insert(key_lm_x);
514  } else if (r_variable == DISPLACEMENT_Y) {
515  aux_set.insert(key_lm_y);
516  } else if (r_variable == DISPLACEMENT_Z) {
517  aux_set.insert(key_lm_z);
518  }
519  }
520  }
521  }
522 
523  // We do now the loop over the dofs
524  for (auto& i_dof : BaseType::mDofSet) {
525  if (i_dof.IsFree()) {
526  node_id = i_dof.Id();
527  auto it = set_nodes_with_lm_associated.find(node_id);
528  if (it != set_nodes_with_lm_associated.end()) {
529  auto& aux_set = it->second;
530  if (aux_set.find((i_dof.GetVariable()).Key()) != aux_set.end()) {
531  i_dof.FixDof();
532  }
533  }
534  }
535  }
536 
537  BaseType::SetUpSystem(rModelPart);
538  }
