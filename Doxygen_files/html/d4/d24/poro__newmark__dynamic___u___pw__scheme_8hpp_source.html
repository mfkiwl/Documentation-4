59  void Predict(
60  ModelPart& r_model_part,
61  DofsArrayType& rDofSet,
62  TSystemMatrixType& A,
63  TSystemVectorType& Dx,
64  TSystemVectorType& b) override
65  {
66  // Predict Displacements on free nodes and update Acceleration, Velocity and DtPressure
67 
68  array_1d<double,3> DeltaDisplacement;
69  double DeltaPressure;
70 
71  const int NNodes = static_cast<int>(r_model_part.Nodes().size());
72  ModelPart::NodesContainerType::iterator node_begin = r_model_part.NodesBegin();
73 
74  #pragma omp parallel for private(DeltaDisplacement,DeltaPressure)
75  for(int i = 0; i < NNodes; i++)
76  {
77  ModelPart::NodesContainerType::iterator itNode = node_begin + i;
78 
79  array_1d<double,3>& CurrentDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT);
80  array_1d<double,3>& CurrentAcceleration = itNode->FastGetSolutionStepValue(ACCELERATION);
81  array_1d<double,3>& CurrentVelocity = itNode->FastGetSolutionStepValue(VELOCITY);
82 
83  const array_1d<double,3>& PreviousDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT, 1);
84  const array_1d<double,3>& PreviousAcceleration = itNode->FastGetSolutionStepValue(ACCELERATION, 1);
85  const array_1d<double,3>& PreviousVelocity = itNode->FastGetSolutionStepValue(VELOCITY, 1);
86 
87  if (itNode -> IsFixed(ACCELERATION_X))
88  {
89  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime * PreviousVelocity[0] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[0] + mBeta * CurrentAcceleration[0] );
90  }
91  else if (itNode -> IsFixed(VELOCITY_X))
92  {
93  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[0]-PreviousVelocity[0])+PreviousVelocity[0]);
94  }
95  else if (itNode -> IsFixed(DISPLACEMENT_X) == false)
96  {
97  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime * PreviousVelocity[0] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[0];
98  }
99 
100  if (itNode -> IsFixed(ACCELERATION_Y))
101  {
102  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime * PreviousVelocity[1] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[1] + mBeta * CurrentAcceleration[1] );
103  }
104  else if (itNode -> IsFixed(VELOCITY_Y))
105  {
106  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[1]-PreviousVelocity[1])+PreviousVelocity[1]);
107  }
108  else if (itNode -> IsFixed(DISPLACEMENT_Y) == false)
109  {
110  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime * PreviousVelocity[1] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[1];
111  }
112 
113  // For 3D cases
114  if (itNode -> HasDofFor(DISPLACEMENT_Z))
115  {
116  if (itNode -> IsFixed(ACCELERATION_Z))
117  {
118  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime * PreviousVelocity[2] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[2] + mBeta * CurrentAcceleration[2] );
119  }
120  else if (itNode -> IsFixed(VELOCITY_Z))
121  {
122  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[2]-PreviousVelocity[2])+PreviousVelocity[2]);
123  }
124  else if (itNode -> IsFixed(DISPLACEMENT_Z) == false)
125  {
126  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime * PreviousVelocity[2] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[2];
127  }
128  }
129 
130  noalias(DeltaDisplacement) = CurrentDisplacement - PreviousDisplacement;
131 
132  noalias(CurrentAcceleration) = 1.0/(mBeta*mDeltaTime*mDeltaTime)*(DeltaDisplacement - mDeltaTime*PreviousVelocity - (0.5-mBeta)*mDeltaTime*mDeltaTime*PreviousAcceleration);
133  noalias(CurrentVelocity) = PreviousVelocity + (1.0-mGamma)*mDeltaTime*PreviousAcceleration + mGamma*mDeltaTime*CurrentAcceleration;
134 
135  double& CurrentDtPressure = itNode->FastGetSolutionStepValue(DT_WATER_PRESSURE);
136  DeltaPressure = itNode->FastGetSolutionStepValue(WATER_PRESSURE) - itNode->FastGetSolutionStepValue(WATER_PRESSURE, 1);
137  const double& PreviousDtPressure = itNode->FastGetSolutionStepValue(DT_WATER_PRESSURE, 1);
138 
139  CurrentDtPressure = 1.0/(mTheta*mDeltaTime)*(DeltaPressure - (1.0-mTheta)*mDeltaTime*PreviousDtPressure);
140  }
141  }

240  void AddDynamicsToLHS(LocalSystemMatrixType& LHS_Contribution,LocalSystemMatrixType& M,LocalSystemMatrixType& C,const ProcessInfo& CurrentProcessInfo)
241  {
242  // adding mass contribution
243  if (M.size1() != 0)
244  noalias(LHS_Contribution) += 1.0/(mBeta*mDeltaTime*mDeltaTime)*M;
245 
246  // adding damping contribution
247  if (C.size1() != 0)
248  noalias(LHS_Contribution) += mGamma/(mBeta*mDeltaTime)*C;
249  }

253  void AddDynamicsToRHS(Element& rCurrentElement,LocalSystemVectorType& RHS_Contribution,
254  LocalSystemMatrixType& M,LocalSystemMatrixType& C,const ProcessInfo& CurrentProcessInfo)
255  {
256  int thread = OpenMPUtils::ThisThread();
257 
258  //adding inertia contribution
259  if (M.size1() != 0)
260  {
261  rCurrentElement.GetSecondDerivativesVector(mAccelerationVector[thread], 0);
262 
263  noalias(RHS_Contribution) -= prod(M, mAccelerationVector[thread]);
264  }
265 
266  //adding damping contribution
267  if (C.size1() != 0)
268  {
269  rCurrentElement.GetFirstDerivativesVector(mVelocityVector[thread], 0);
270 
271  noalias(RHS_Contribution) -= prod(C, mVelocityVector[thread]);
272  }
273  }
