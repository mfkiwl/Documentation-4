144 {
145 ptr_iterator sorted_part_end;

147 if(mData.size() - mSortedPartSize >= mMaxBufferSize) {
148 Sort();
149 sorted_part_end = mData.end();

153 ptr_iterator i(std::lower_bound(mData.begin(), sorted_part_end, Key, CompareKey()));
