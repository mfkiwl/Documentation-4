270 template<class TVarType>
271 void InterpolateMeshVariableToSkinSpecialization(
272 const Variable<TVarType> &rMeshVariable,
273 const Variable<TVarType> &rSkinVariable,
274 const std::string &rInterfaceSide = "positive")
275 {
276 // Check requested variables
277 KRATOS_ERROR_IF((mrModelPart.NodesBegin())->SolutionStepsDataHas(rMeshVariable) == false)
278 << "Mesh model part solution step data missing variable: " << rMeshVariable << std::endl;
279 KRATOS_ERROR_IF((mrSkinModelPart.NodesBegin())->SolutionStepsDataHas(rSkinVariable) == false)
280 << "Generated skin model part solution step data missing variable: " << rSkinVariable << std::endl;
281
282 // Check that the mesh model part has elements
283 KRATOS_ERROR_IF(mrModelPart.NumberOfElements() == 0) << "Mesh model part has no elements.";
284
285 // Loop the edge intersection nodes to set their values
286 unsigned int i_edge;
287 Element::Pointer p_elem;
288 #pragma omp parallel for private (i_edge, p_elem)
289 for (int i_node = 0; i_node < static_cast<int>(mrSkinModelPart.NumberOfNodes()); ++i_node) {
290 // Get the current node
291 auto it_node = mrSkinModelPart.NodesBegin() + i_node;
292 Node<3>::Pointer p_node = &(*it_node);
293
294 // Search for the current node in the intersected edges map
295 const auto i_node_info = mEdgeNodesMap.find(p_node);
296 if (i_node_info != mEdgeNodesMap.end()){
297 // Get the cut node info from the map tuple iterator
298 std::tie(p_elem, i_edge) = std::get<1>(*i_node_info);
299
300 // Set the modified shape functions for the parent element
301 const auto p_elem_geom = p_elem->pGetGeometry();
302 const auto elem_dist = this->SetDistancesVector(*p_elem);
303 const auto p_mod_sh_func = pCreateModifiedShapeFunctions(p_elem_geom, elem_dist);
304
305 // Get interface modified shape function values
306 const auto edge_sh_func = this->GetModifiedShapeFunctionsValuesOnEdge(
307 p_mod_sh_func,
308 rInterfaceSide);
309
310 // Compute the interpolation
311 const auto edge_N = row(edge_sh_func, i_edge);
312 const auto &r_elem_geom = p_elem->GetGeometry();
313 auto &r_value = it_node->FastGetSolutionStepValue(rSkinVariable);
314 r_value = rSkinVariable.Zero();
315 for (unsigned int i_elem_node = 0; i_elem_node < r_elem_geom.PointsNumber(); ++i_elem_node) {
316 r_value += edge_N[i_elem_node] * r_elem_geom[i_elem_node].FastGetSolutionStepValue(rMeshVariable);
317 }
318 } else{
319 KRATOS_ERROR << "Intersected edge node " << it_node->Id() << " not found in intersected edges nodes map" << std::endl;
320 }
321 }
322 };
