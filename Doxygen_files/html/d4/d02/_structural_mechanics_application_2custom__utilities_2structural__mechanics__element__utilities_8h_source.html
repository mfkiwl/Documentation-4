70template<class TVectorType, class TMatrixType>
71void ComputeEquivalentF(
72 const Element& rElement,
73 const TVectorType& rStrainTensor,
74 TMatrixType& rF
75 )
76{
77 const auto& r_geometry = rElement.GetGeometry();
78 const SizeType dimension = r_geometry.WorkingSpaceDimension();
79
80 if(dimension == 2) {
81 rF(0,0) = 1.0+rStrainTensor(0);
82 rF(0,1) = 0.5*rStrainTensor(2);
83 rF(1,0) = 0.5*rStrainTensor(2);
84 rF(1,1) = 1.0+rStrainTensor(1);
85 } else {
86 rF(0,0) = 1.0+rStrainTensor(0);
87 rF(0,1) = 0.5*rStrainTensor(3);
88 rF(0,2) = 0.5*rStrainTensor(5);
89 rF(1,0) = 0.5*rStrainTensor(3);
90 rF(1,1) = 1.0+rStrainTensor(1);
91 rF(1,2) = 0.5*rStrainTensor(4);
92 rF(2,0) = 0.5*rStrainTensor(5);
93 rF(2,1) = 0.5*rStrainTensor(4);
94 rF(2,2) = 1.0+rStrainTensor(2);
95 }
96}

106template<class TMatrixType1, class TMatrixType2>
107void CalculateB(
108 const GeometricalObject& rElement,
109 const TMatrixType1& rDN_DX,
110 TMatrixType2& rB
111 )
112{
113 const auto& r_geometry = rElement.GetGeometry();
114 const SizeType number_of_nodes = r_geometry.PointsNumber();
115 const SizeType dimension = rDN_DX.size2();
116
117 rB.clear();
118
119 if(dimension == 2) {
120 for ( IndexType i = 0; i < number_of_nodes; ++i ) {
121 const IndexType initial_index = i*2;
122 rB(0, initial_index ) = rDN_DX(i, 0);
123 rB(1, initial_index + 1) = rDN_DX(i, 1);
124 rB(2, initial_index ) = rDN_DX(i, 1);
125 rB(2, initial_index + 1) = rDN_DX(i, 0);
126 }
127 } else if(dimension == 3) {
128 for ( IndexType i = 0; i < number_of_nodes; ++i ) {
129 const IndexType initial_index = i*3;
130 rB(0, initial_index ) = rDN_DX(i, 0);
131 rB(1, initial_index + 1) = rDN_DX(i, 1);
132 rB(2, initial_index + 2) = rDN_DX(i, 2);
133 rB(3, initial_index ) = rDN_DX(i, 1);
134 rB(3, initial_index + 1) = rDN_DX(i, 0);
135 rB(4, initial_index + 1) = rDN_DX(i, 2);
136 rB(4, initial_index + 2) = rDN_DX(i, 1);
137 rB(5, initial_index ) = rDN_DX(i, 2);
138 rB(5, initial_index + 2) = rDN_DX(i, 0);
139 }
140 }
141}
