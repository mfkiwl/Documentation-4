183 // Adding sub value item
184 template<typename TItemType, class... TArgumentsList>
185 RegistryItem& AddItem(
186 std::string const& ItemName,
187 TArgumentsList&&... Arguments)
188 {
189 KRATOS_ERROR_IF(this->HasItem(ItemName))
190 << "The RegistryItem '" << this->Name() << "' already has an item with name "
191 << ItemName << "." << std::endl;
192
193 using ValueType = typename std::conditional<std::is_same<TItemType, RegistryItem>::value, SubRegistryItemFunctor, SubValueItemFunctor<TItemType>>::type;
194
195 auto insert_result = GetSubRegistryItemMap().emplace(
196 std::make_pair(
197 ItemName,
198 ValueType::Create(ItemName, std::forward<TArgumentsList>(Arguments)...)
199 )
200 );
201
202 KRATOS_ERROR_IF_NOT(insert_result.second)
203 << "Error in inserting '" << ItemName
204 << "' in registry item with name '" << this->Name() << "'." << std::endl;
205
206 return *insert_result.first->second;
207 }
