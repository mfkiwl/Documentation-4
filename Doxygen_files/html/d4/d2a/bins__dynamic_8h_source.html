242  void CalculateBoundingBox()
243  {
244  for(SizeType i = 0 ; i < Dimension ; i++)
245  {
246  mMinPoint[i] = (**mPointBegin)[i];
247  mMaxPoint[i] = (**mPointBegin)[i];
248  }
249  for(IteratorType Point = mPointBegin ; Point != mPointEnd ; Point++)
250  for(SizeType i = 0 ; i < Dimension ; i++)
251  {
252  if( (**Point)[i] < mMinPoint[i] ) mMinPoint[i] = (**Point)[i];
253  if( (**Point)[i] > mMaxPoint[i] ) mMaxPoint[i] = (**Point)[i];
254  }
255  }

266  void CalculateCellSize(std::size_t ApproximatedSize)
267  {
268  std::size_t average_number_of_cells = static_cast<std::size_t>(std::pow(static_cast<double>(ApproximatedSize), 1.00 / Dimension));
269 
270  std::array<double, 3> lengths;
271  double average_length = 0.00;
272 
273  for (int i = 0; i < Dimension; i++) {
274  lengths[i] = mMaxPoint[i] - mMinPoint[i];
275  average_length += lengths[i];
276  }
277  average_length *= 1.00 / 3.00;
278 
279  if (average_length < std::numeric_limits<double>::epsilon()) {
280  for(int i = 0; i < Dimension; i++) {
281  mN[i] = 1;
282  }
283  return;
284  }
285 
286  for (int i = 0; i < Dimension; i++) {
287  mN[i] = static_cast<std::size_t>(lengths[i] / average_length * (double)average_number_of_cells) + 1;
288 
289  if (mN[i] > 1) {
290  mCellSize[i] = lengths[i] / mN[i];
291  } else {
292  mCellSize[i] = average_length;
293  }
294 
295  mInvCellSize[i] = 1.00 / mCellSize[i];
296  }
297  }
