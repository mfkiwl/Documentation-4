234 void ProjectToFineBasis(
235 const TSystemVectorType& rRomUnkowns,
236 const ModelPart& rModelPart,

238 {
239 const auto& r_dof_set = BaseType::GetDofSet();

242 const auto& r_node = rModelPart.GetNode(r_dof.Id());
243 const Matrix& r_rom_nodal_basis = r_node.GetValue(ROM_BASIS);
244 const Matrix::size_type row_id = mMapPhi.at(r_dof.GetVariable().Key());
245 rDx[r_dof.EquationId()] = inner_prod(row(r_rom_nodal_basis, row_id), rRomUnkowns);

423 void InitializeHROMWeights(ModelPart& rModelPart)
424 {
425 KRATOS_TRY
426
427 using ElementQueue = moodycamel::ConcurrentQueue<Element::Pointer>;
428 using ConditionQueue = moodycamel::ConcurrentQueue<Condition::Pointer>;
429
430 // Inspecting elements
431 ElementQueue element_queue;
432 block_for_each(rModelPart.Elements().GetContainer(),
433 [&](Element::Pointer p_element)
434 {
435 if (p_element->Has(HROM_WEIGHT)) {
436 element_queue.enqueue(std::move(p_element));
437 } else {
438 p_element->SetValue(HROM_WEIGHT, 1.0);
439 }
440 });
441
442 // Inspecting conditions
443 ConditionQueue condition_queue;
444 block_for_each(rModelPart.Conditions().GetContainer(),
445 [&](Condition::Pointer p_condition)
446 {
447 if (p_condition->Has(HROM_WEIGHT)) {
448 condition_queue.enqueue(std::move(p_condition));
449 } else {
450 p_condition->SetValue(HROM_WEIGHT, 1.0);
451 }
452 });
453
454 // Dequeueing elements
455 std::size_t err_id;
456 mSelectedElements.reserve(element_queue.size_approx());
457 Element::Pointer p_element;
458 while ( (err_id = element_queue.try_dequeue(p_element)) != 0) {
459 mSelectedElements.push_back(std::move(p_element));
460 }
461
462 // Dequeueing conditions
463 mSelectedConditions.reserve(condition_queue.size_approx());
464 Condition::Pointer p_condition;
465 while ( (err_id = condition_queue.try_dequeue(p_condition)) != 0) {
466 mSelectedConditions.push_back(std::move(p_condition));
467 }
468
469 // Wrap-up
470 mHromSimulation = !(mSelectedElements.empty() && mSelectedConditions.empty());
471 mHromWeightsInitialized = true;
472
473 KRATOS_CATCH("")
474 }

476 static DofQueue ExtractDofSet(
477 typename TSchemeType::Pointer pScheme,
478 ModelPart& rModelPart)
479 {
480 KRATOS_TRY
481
482 DofQueue dof_queue;
483
484 // Emulates ConcurrentQueue::enqueue_bulk adding move semantics to avoid atomic ops
485 const auto enqueue_bulk_move = [](DofQueue& r_queue, DofsVectorType& r_dof_list) {
486 for(auto& p_dof: r_dof_list) {
487 r_queue.enqueue(std::move(p_dof));
488 }
489 r_dof_list.clear();
490 };
491
492 // Inspecting elements
493 DofsVectorType tls_dof_list; // Preallocation
494 block_for_each(rModelPart.Elements(), tls_dof_list,
495 [&](const Element& r_element, DofsVectorType& r_dof_list)
496 {
497 pScheme->GetDofList(r_element, r_dof_list, rModelPart.GetProcessInfo());
498 enqueue_bulk_move(dof_queue, r_dof_list);
499 });
500
501 // Inspecting conditions
502 block_for_each(rModelPart.Conditions(), tls_dof_list,
503 [&](const Condition& r_condition, DofsVectorType& r_dof_list)
504 {
505 pScheme->GetDofList(r_condition, r_dof_list, rModelPart.GetProcessInfo());
506 enqueue_bulk_move(dof_queue, r_dof_list);
507 });
508
509 // Inspecting master-slave constraints
510 std::pair<DofsVectorType, DofsVectorType> tls_ms_dof_lists; // Preallocation
511 block_for_each(rModelPart.MasterSlaveConstraints(), tls_ms_dof_lists,
512 [&](const MasterSlaveConstraint& r_constraint, std::pair<DofsVectorType, DofsVectorType>& r_dof_lists)
513 {
514 r_constraint.GetDofList(r_dof_lists.first, r_dof_lists.second, rModelPart.GetProcessInfo());
515
516 enqueue_bulk_move(dof_queue, r_dof_lists.first);
517 enqueue_bulk_move(dof_queue, r_dof_lists.second);
518 });
519
520 return dof_queue;
521
522 KRATOS_CATCH("")
523 }

525 static DofsArrayType SortAndRemoveDuplicateDofs(DofQueue& rDofQueue)
526 {
527 KRATOS_TRY
528
529 DofsArrayType dof_array;
530 dof_array.reserve(rDofQueue.size_approx());
531 DofType::Pointer p_dof;
532 std::size_t err_id;
533 while ( (err_id = rDofQueue.try_dequeue(p_dof)) != 0) {

538
539 return dof_array;

622 virtual void BuildROM(
623 typename TSchemeType::Pointer pScheme,
624 ModelPart &rModelPart,
625 RomSystemMatrixType &rA,
626 RomSystemVectorType &rb)
627 {
628 KRATOS_TRY
629
630 // Define a dense matrix to hold the reduced problem
631 rA = ZeroMatrix(GetNumberOfROMModes(), GetNumberOfROMModes());
632 rb = ZeroVector(GetNumberOfROMModes());
633
634 // Build the system matrix by looping over elements and conditions and assembling to A
635 KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
636
637 // Get ProcessInfo from main model part
638 const auto& r_current_process_info = rModelPart.GetProcessInfo();
639
640
641 // Assemble all entities
642 const auto assembling_timer = BuiltinTimer();
643
644 using SystemSumReducer = CombinedReduction<NonTrivialSumReduction<RomSystemMatrixType>, NonTrivialSumReduction<RomSystemVectorType>>;
645 AssemblyTLS assembly_tls_container(GetNumberOfROMModes());
646
647 auto& elements = mHromSimulation ? mSelectedElements : rModelPart.Elements();
648 if(!elements.empty())
649 {
650 std::tie(rA, rb) =
651 block_for_each<SystemSumReducer>(elements, assembly_tls_container,
652 [&](Element& r_element, AssemblyTLS& r_thread_prealloc)
653 {
654 return CalculateLocalContribution(r_element, r_thread_prealloc, *pScheme, r_current_process_info);
655 });
656 }
657
658 auto& conditions = mHromSimulation ? mSelectedConditions : rModelPart.Conditions();
659 if(!conditions.empty())
660 {
661 RomSystemMatrixType Aconditions;
662 RomSystemVectorType bconditions;
663
664 std::tie(Aconditions, bconditions) =
665 block_for_each<SystemSumReducer>(conditions, assembly_tls_container,
666 [&](Condition& r_condition, AssemblyTLS& r_thread_prealloc)
667 {
668 return CalculateLocalContribution(r_condition, r_thread_prealloc, *pScheme, r_current_process_info);
669 });
670
671 rA += Aconditions;
672 rb += bconditions;
673 }
674
675 KRATOS_INFO_IF("ROMBuilderAndSolver", (this->GetEchoLevel() > 0)) << "Build time: " << assembling_timer.ElapsedSeconds() << std::endl;
676 KRATOS_INFO_IF("ROMBuilderAndSolver", (this->GetEchoLevel() > 2)) << "Finished parallel building" << std::endl;
677
678 KRATOS_CATCH("")
679 }

684 virtual void SolveROM(
685 ModelPart &rModelPart,
686 RomSystemMatrixType &rA,
687 RomSystemVectorType &rb,
688 TSystemVectorType &rDx)
689 {
690 KRATOS_TRY
691
692 RomSystemVectorType dxrom(GetNumberOfROMModes());
693
694 const auto solving_timer = BuiltinTimer();
695 MathUtils<double>::Solve(rA, dxrom, rb);
696 // KRATOS_WATCH(dxrom)
697 KRATOS_INFO_IF("ROMBuilderAndSolver", (this->GetEchoLevel() > 0)) << "Solve reduced system time: " << solving_timer.ElapsedSeconds() << std::endl;
698
699 // Save the ROM solution increment in the root modelpart database
700 auto& r_root_mp = rModelPart.GetRootModelPart();
701 noalias(r_root_mp.GetValue(ROM_SOLUTION_INCREMENT)) += dxrom;
702
703 // project reduced solution back to full order model
704 const auto backward_projection_timer = BuiltinTimer();
705 ProjectToFineBasis(dxrom, rModelPart, rDx);
706 KRATOS_INFO_IF("ROMBuilderAndSolver", (this->GetEchoLevel() > 0)) << "Project to fine basis time: " << backward_projection_timer.ElapsedSeconds() << std::endl;
707
708 KRATOS_CATCH("")
709 }

736 template<typename TEntity>
737 std::tuple<LocalSystemMatrixType, LocalSystemVectorType> CalculateLocalContribution(
738 TEntity& rEntity,
739 AssemblyTLS& rPreAlloc,
740 TSchemeType& rScheme,
741 const ProcessInfo& rCurrentProcessInfo)
742 {
743 if (rEntity.IsDefined(ACTIVE) && rEntity.IsNot(ACTIVE))
744 {
745 rPreAlloc.romA = ZeroMatrix(GetNumberOfROMModes(), GetNumberOfROMModes());
746 rPreAlloc.romB = ZeroVector(GetNumberOfROMModes());
747 return std::tie(rPreAlloc.romA, rPreAlloc.romB);
748 }
749
750 // Calculate elemental contribution
751 rScheme.CalculateSystemContributions(rEntity, rPreAlloc.lhs, rPreAlloc.rhs, rPreAlloc.eq_id, rCurrentProcessInfo);
752 rEntity.GetDofList(rPreAlloc.dofs, rCurrentProcessInfo);
753
754 const SizeType ndofs = rPreAlloc.dofs.size();
755 ResizeIfNeeded(rPreAlloc.phiE, ndofs, GetNumberOfROMModes());
756 ResizeIfNeeded(rPreAlloc.aux, ndofs, GetNumberOfROMModes());
757
758 const auto &r_geom = rEntity.GetGeometry();
759 RomAuxiliaryUtilities::GetPhiElemental(rPreAlloc.phiE, rPreAlloc.dofs, r_geom, mMapPhi);
760
761 const double h_rom_weight = mHromSimulation ? rEntity.GetValue(HROM_WEIGHT) : 1.0;
762
763 noalias(rPreAlloc.aux) = prod(rPreAlloc.lhs, rPreAlloc.phiE);
764 noalias(rPreAlloc.romA) = prod(trans(rPreAlloc.phiE), rPreAlloc.aux) * h_rom_weight;
765 noalias(rPreAlloc.romB) = prod(trans(rPreAlloc.phiE), rPreAlloc.rhs) * h_rom_weight;
766
767 return std::tie(rPreAlloc.romA, rPreAlloc.romB);
768 }
