140 void AddDampingToLHS(
141 LocalSystemMatrixType& LHS_Contribution,
142 LocalSystemMatrixType& D,
143 const ProcessInfo& CurrentProcessInfo)
144 {
145 // Adding damping contribution
146 if (D.size1() != 0) // if D matrix declared
147 {
148 noalias(LHS_Contribution) += D * (1.0 - mAlpha.f) * mNewmark.c1;
149 }
150 }

154 void AddDampingToRHS(
155 Element& rCurrentElement,
156 LocalSystemVectorType& RHS_Contribution,
157 LocalSystemMatrixType& D,
158 const ProcessInfo& CurrentProcessInfo)
159 {
160 int thread = OpenMPUtils::ThisThread();
161
162 // Adding damping contribution
163 if (D.size1() != 0)
164 {
165 rCurrentElement.GetFirstDerivativesVector(mVector.v[thread], 0);
166
167 noalias(RHS_Contribution) -= prod(D, mVector.v[thread]);
168 }
169 }
