369  // Loop over Dofs
370  if (rModelPart.NumberOfMasterSlaveConstraints() > 0) {
371  #pragma omp parallel for firstprivate(residual_dof_value) reduction(+:residual_solution_norm, dof_num)
372  for (int i = 0; i < number_of_dof; i++) {
373  auto it_dof = it_dof_begin + i;
374 
375  const IndexType dof_id = it_dof->EquationId();
376 
377  if (mActiveDofs[dof_id] == 1) {
378  residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
379  residual_solution_norm += std::pow(residual_dof_value, 2);
380  dof_num++;
381  }
382  }
383  } else {
384  #pragma omp parallel for firstprivate(residual_dof_value) reduction(+:residual_solution_norm, dof_num)
385  for (int i = 0; i < number_of_dof; i++) {
386  auto it_dof = it_dof_begin + i;
387 
388  if (!it_dof->IsFixed()) {
389  const IndexType dof_id = it_dof->EquationId();
390  residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
391  residual_solution_norm += std::pow(residual_dof_value, 2);
392  dof_num++;
393  }
394  }
395  }
