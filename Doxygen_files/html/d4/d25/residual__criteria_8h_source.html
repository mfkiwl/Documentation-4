167 bool PostCriteria(
168 ModelPart& rModelPart,
169 DofsArrayType& rDofSet,
170 const TSystemMatrixType& rA,
171 const TSystemVectorType& rDx,
172 const TSystemVectorType& rb
173 ) override
174 {
175 const SizeType size_b = TSparseSpace::Size(rb);
176 if (size_b != 0) { //if we are solving for something
177
178 SizeType size_residual;
179 CalculateResidualNorm(rModelPart, mCurrentResidualNorm, size_residual, rDofSet, rb);
180
181 TDataType ratio = 0.0;
182 if(mInitialResidualNorm < std::numeric_limits<TDataType>::epsilon()) {
183 ratio = 0.0;
184 } else {
185 ratio = mCurrentResidualNorm/mInitialResidualNorm;
186 }
187
188 const TDataType float_size_residual = static_cast<TDataType>(size_residual);
189 const TDataType absolute_norm = (mCurrentResidualNorm/float_size_residual);
190
191 KRATOS_INFO_IF("RESIDUAL CRITERION", this->GetEchoLevel() > 1 && rModelPart.GetCommunicator().MyPID() == 0) << " :: [ Initial residual norm = " << mInitialResidualNorm << "; Current residual norm = " << mCurrentResidualNorm << "]" << std::endl;
192 KRATOS_INFO_IF("RESIDUAL CRITERION", this->GetEchoLevel() > 0 && rModelPart.GetCommunicator().MyPID() == 0) << " :: [ Obtained ratio = " << ratio << "; Expected ratio = " << mRatioTolerance << "; Absolute norm = " << absolute_norm << "; Expected norm = " << mAlwaysConvergedNorm << "]" << std::endl;
193
194 rModelPart.GetProcessInfo()[CONVERGENCE_RATIO] = ratio;
195 rModelPart.GetProcessInfo()[RESIDUAL_NORM] = absolute_norm;
196
197 if (ratio <= mRatioTolerance || absolute_norm < mAlwaysConvergedNorm) {
198 KRATOS_INFO_IF("RESIDUAL CRITERION", this->GetEchoLevel() > 0 && rModelPart.GetCommunicator().MyPID() == 0) << "Convergence is achieved" << std::endl;
199 return true;
200 } else {
201 return false;
202 }
203 } else {
204 return true;
205 }

353 virtual void CalculateResidualNorm(
354 ModelPart& rModelPart,
355 TDataType& rResidualSolutionNorm,
356 SizeType& rDofNum,
357 DofsArrayType& rDofSet,
358 const TSystemVectorType& rb
359 )
360 {
361 // Initialize
362 TDataType residual_solution_norm = TDataType();
363 SizeType dof_num = 0;
364
365 // Auxiliar values
366 TDataType residual_dof_value = 0.0;
367 const auto it_dof_begin = rDofSet.begin();
368 const int number_of_dof = static_cast<int>(rDofSet.size());
369
370 // Loop over Dofs
371 if (rModelPart.NumberOfMasterSlaveConstraints() > 0) {
372 #pragma omp parallel for firstprivate(residual_dof_value) reduction(+:residual_solution_norm, dof_num)
373 for (int i = 0; i < number_of_dof; i++) {
374 auto it_dof = it_dof_begin + i;
375
376 const IndexType dof_id = it_dof->EquationId();
377
378 if (mActiveDofs[dof_id] == 1) {
379 residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
380 residual_solution_norm += std::pow(residual_dof_value, 2);
381 dof_num++;
382 }
383 }
384 } else {
385 #pragma omp parallel for firstprivate(residual_dof_value) reduction(+:residual_solution_norm, dof_num)
386 for (int i = 0; i < number_of_dof; i++) {
387 auto it_dof = it_dof_begin + i;
388
389 if (!it_dof->IsFixed()) {
390 const IndexType dof_id = it_dof->EquationId();
391 residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
392 residual_solution_norm += std::pow(residual_dof_value, 2);
393 dof_num++;
394 }
395 }
396 }
397
398 rDofNum = dof_num;
399 rResidualSolutionNorm = std::sqrt(residual_solution_norm);
400 }
