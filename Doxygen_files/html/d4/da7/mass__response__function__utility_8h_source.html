113 double CalculateValue()
114 {
115 KRATOS_TRY;
116
117 // Variables
118 double total_mass = 0.0;
119 const std::size_t domain_size = mrModelPart.GetProcessInfo()[DOMAIN_SIZE];
120
121 // Incremental computation of total mass
122 for (auto& elem_i : mrModelPart.Elements()){
123 const bool element_is_active = elem_i.IsDefined(ACTIVE) ? elem_i.Is(ACTIVE) : true;
124 if(element_is_active)
125 total_mass += TotalStructuralMassProcess::CalculateElementMass(elem_i, domain_size);
126 }
127
128 return total_mass;

130 KRATOS_CATCH("");

134 void CalculateGradient()
135 {
136 KRATOS_TRY;
137
138 // Formula computed in general notation:
139 // \frac{dm_{total}}{dx}
140
141 // First gradients are initialized
142 VariableUtils().SetHistoricalVariableToZero(SHAPE_SENSITIVITY, mrModelPart.Nodes());
143
144 const std::size_t domain_size = mrModelPart.GetProcessInfo()[DOMAIN_SIZE];
145
146 // Start process to identify element neighbors for every node
147 FindNodalNeighboursProcess neighorFinder(mrModelPart);
148 neighorFinder.Execute();
149
150 for(auto& node_i : mrModelPart.Nodes())
151 {
152 // Get all neighbor elements of current node
153 GlobalPointersVector<Element >& ng_elem = node_i.GetValue(NEIGHBOUR_ELEMENTS);
154
155 // Compute total mass of all neighbor elements before finite differencing
156 double mass_before_fd = 0.0;
157 for(std::size_t i = 0; i < ng_elem.size(); i++)
158 {
159 Element& ng_elem_i = ng_elem[i];
160 const bool element_is_active = ng_elem_i.IsDefined(ACTIVE) ? ng_elem_i.Is(ACTIVE) : true;
161
162 // Compute mass according to element dimension
163 if(element_is_active)
164 mass_before_fd += TotalStructuralMassProcess::CalculateElementMass(ng_elem_i, domain_size);
165 }
166
167 // Compute sensitivities using finite differencing in the three spatial direction
168 array_3d gradient(3, 0.0);
169
170 // Apply pertubation in X-direction and recompute total mass of all neighbor elements
171 double mass_after_fd = 0.0;
172 node_i.X() += mDelta;
173 node_i.X0() += mDelta;
174 for(std::size_t i = 0; i < ng_elem.size(); i++)
175 {
176 Element& ng_elem_i = ng_elem[i];
177 const bool element_is_active = ng_elem_i.IsDefined(ACTIVE) ? ng_elem_i.Is(ACTIVE) : true;
178
179 // Compute mass according to element dimension
180 if(element_is_active)
181 mass_after_fd += TotalStructuralMassProcess::CalculateElementMass(ng_elem_i, domain_size);
182 }
183 gradient[0] = (mass_after_fd - mass_before_fd) / mDelta;
184 node_i.X() -= mDelta;
185 node_i.X0() -= mDelta;
186
187 // Apply pertubation in Y-direction and recompute total mass of all neighbor elements
188 mass_after_fd = 0.0;
189 node_i.Y() += mDelta;
190 node_i.Y0() += mDelta;
191 for(std::size_t i = 0; i < ng_elem.size(); i++)
192 {
193 Element& ng_elem_i = ng_elem[i];
194 const bool element_is_active = ng_elem_i.IsDefined(ACTIVE) ? ng_elem_i.Is(ACTIVE) : true;
195
196 // Compute mass according to element dimension
197 if(element_is_active)
198 mass_after_fd += TotalStructuralMassProcess::CalculateElementMass(ng_elem_i, domain_size);
199 }
200 gradient[1] = (mass_after_fd - mass_before_fd) / mDelta;
201 node_i.Y() -= mDelta;
202 node_i.Y0() -= mDelta;
203
204 // Apply pertubation in Z-direction and recompute total mass of all neighbor elements
205 mass_after_fd = 0.0;
206 node_i.Z() += mDelta;
207 node_i.Z0() += mDelta;
208 for(std::size_t i = 0; i < ng_elem.size(); i++)
209 {
210 Element& ng_elem_i = ng_elem[i];
211 const bool element_is_active = ng_elem_i.IsDefined(ACTIVE) ? ng_elem_i.Is(ACTIVE) : true;
212
213 // Compute mass according to element dimension
214 if(element_is_active)
215 mass_after_fd += TotalStructuralMassProcess::CalculateElementMass(ng_elem_i, domain_size);
216 }
217 gradient[2] = (mass_after_fd - mass_before_fd) / mDelta;
218 node_i.Z() -= mDelta;
219 node_i.Z0() -= mDelta;
220
221 // Compute sensitivity
222 noalias(node_i.FastGetSolutionStepValue(SHAPE_SENSITIVITY)) = gradient;
223
224 }
225
226 KRATOS_CATCH("");
227 }
