411 template <typename TDataType>
412 void CalculateAdjointFieldOnIntegrationPoints(const Variable<TDataType>& rVariable, std::vector< TDataType >& rOutput, const ProcessInfo& rCurrentProcessInfo)
413 {
414 KRATOS_TRY;
415
416 KRATOS_WARNING_IF("CalculateAdjointFieldOnIntegrationPoints", OpenMPUtils::IsInParallel() != 0)
417 << "The call of this non shared-memory-parallelized function within a parallel section should be avoided for efficiency reasons!" << std::endl;
418
419 const SizeType num_nodes = mpPrimalElement->GetGeometry().PointsNumber();
420 const SizeType dimension = mpPrimalElement->GetGeometry().WorkingSpaceDimension();
421 const SizeType num_dofs_per_node = (mHasRotationDofs) ? 2 * dimension : dimension;
422 const SizeType num_dofs = num_nodes * num_dofs_per_node;
423 Vector initial_state_variables;
424 initial_state_variables.resize(num_dofs);
425
426 Vector particular_solution = ZeroVector(num_dofs);
427 if(this->Has(ADJOINT_PARTICULAR_DISPLACEMENT)) {
428 particular_solution = this->GetValue(ADJOINT_PARTICULAR_DISPLACEMENT);
429 }
430
431 // Build vector of variables containing the DOF-variables of the primal problem
432 std::vector<Variable<double>*> primal_solution_variable_list;
433 (mHasRotationDofs) ? primal_solution_variable_list = {&DISPLACEMENT_X, &DISPLACEMENT_Y, &DISPLACEMENT_Z, &ROTATION_X, &ROTATION_Y, &ROTATION_Z} :
434 primal_solution_variable_list = {&DISPLACEMENT_X, &DISPLACEMENT_Y, &DISPLACEMENT_Z};
435
436 // Build vector of variables containing the DOF-variables of the adjoint problem
437 std::vector<Variable<double>*> adjoint_solution_variable_list;
438 (mHasRotationDofs) ? adjoint_solution_variable_list = {&ADJOINT_DISPLACEMENT_X, &ADJOINT_DISPLACEMENT_Y, &ADJOINT_DISPLACEMENT_Z, &ADJOINT_ROTATION_X, &ADJOINT_ROTATION_Y, &ADJOINT_ROTATION_Z} :
439 adjoint_solution_variable_list = {&ADJOINT_DISPLACEMENT_X, &ADJOINT_DISPLACEMENT_Y, &ADJOINT_DISPLACEMENT_Z};
440
441 for (IndexType i = 0; i < num_nodes; ++i) {
442 const IndexType index = i * num_dofs_per_node;
443 for(IndexType j = 0; j < primal_solution_variable_list.size(); ++j) {
444 initial_state_variables[index + j] = mpPrimalElement->GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]);
445 mpPrimalElement->GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]) =
446 this->GetGeometry()[i].FastGetSolutionStepValue(*adjoint_solution_variable_list[j]) +
447 particular_solution[index + j];
448 }
449 }
450
451 mpPrimalElement->CalculateOnIntegrationPoints(rVariable, rOutput, rCurrentProcessInfo);
452
453 for (IndexType i = 0; i < num_nodes; ++i) {
454 const IndexType index = i * num_dofs_per_node;
455 for(IndexType j = 0; j < primal_solution_variable_list.size(); ++j) {
456 mpPrimalElement->GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]) = initial_state_variables[index + j];
457 }
458 }
459
460 KRATOS_CATCH("")
461 }
