262

265 #pragma omp parallel for schedule(guided,512)
266 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
267 auto it_node = (it_node_begin + i);
268 it_node->SetValue(NODAL_MASS, 0.0);

276 auto it_node = (it_node_begin + i);

279 r_fractional_acceleration = ZeroVector(3);

283 #pragma omp parallel for schedule(guided,512)
284 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {

293 Double3DArray& r_current_residual_moment = it_node->FastGetSolutionStepValue(MOMENT_RESIDUAL);

386 template <typename TObjectType>
387 void UpdateAccelerationStage(
388 NodeIterator itCurrentNode,
389 const IndexType FieldPosition,
390 const ArrayVarType& rAccelerationVariable,
391 const ArrayVarType& rVelocityVariable,
392 const TObjectType& rDampingVariable,
393 const ArrayVarType& rResidualVariable,
394 const TObjectType& rIntertiaVariable,
395 const DoubleVarType& rFixVariable1,
396 const DoubleVarType& rFixVariable2,
397 const DoubleVarType& rFixVariable3,
398 const SizeType DomainSize = 3
399 )
400 {
401 Double3DArray& r_acceleration = itCurrentNode->FastGetSolutionStepValue(rAccelerationVariable);
402 Double3DArray& r_current_velocity = itCurrentNode->FastGetSolutionStepValue(rVelocityVariable);
403 const Double3DArray& r_current_residual = itCurrentNode->FastGetSolutionStepValue(rResidualVariable);
404
405 if (std::is_same<TObjectType,DoubleVarType>::value) {
406 const double& r_nodal_inertia = itCurrentNode->GetValue(NODAL_MASS);
407 const double& r_nodal_damping = itCurrentNode->GetValue(NODAL_DISPLACEMENT_DAMPING);
408
409 // Solution of the explicit equation:
410 if (r_nodal_inertia > numerical_limit)
411 {
412 noalias(r_acceleration) = (r_current_residual - r_nodal_damping * r_current_velocity) / r_nodal_inertia;
413 }
414 else
415 noalias(r_acceleration) = ZeroVector(3);
416 }
417 else if (std::is_same<TObjectType,ArrayVarType>::value) {
418 const Double3DArray& r_nodal_inertia = itCurrentNode->GetValue(NODAL_INERTIA);
419 const Double3DArray& r_nodal_rotational_damping = itCurrentNode->GetValue(NODAL_ROTATION_DAMPING);
420
421 const IndexType initial_k = DomainSize == 3 ? 0 : 2; // We do this because in 2D only the rotation Z is needed, then we start with 2, instead of 0
422 for (IndexType kk = initial_k; kk < 3; ++kk) {
423 if (r_nodal_inertia[kk] > numerical_limit)
424 r_acceleration[kk] = (r_current_residual[kk] - r_nodal_rotational_damping[kk] * r_current_velocity[kk]) / r_nodal_inertia[kk];
425 else
426 r_acceleration[kk] = 0.0;
427 }
428 }
429 else KRATOS_ERROR << "cannot handle input variable in \"explicit_multi_stage_kim_scheme.hpp\"" << std::endl;
430
431 std::array<bool, 3> fix_field = {false, false, false};
432 fix_field[0] = (itCurrentNode->GetDof(rFixVariable1, FieldPosition).IsFixed());
433 fix_field[1] = (itCurrentNode->GetDof(rFixVariable2, FieldPosition + 1).IsFixed());
434 if (DomainSize == 3)
435 fix_field[2] = (itCurrentNode->GetDof(rFixVariable3, FieldPosition + 2).IsFixed());
436
437 for (IndexType j = 0; j < DomainSize; j++) {
438 if (fix_field[j]) {
439 r_acceleration[j] = 0.0;
440 }
441 }
442 }

450 void UpdateDegreesOfFreedomStage1(
451 NodeIterator itCurrentNode,
452 const ArrayVarType& rVelocityVariable,
453 const ArrayVarType& rDisplacementVariable,
454 const ArrayVarType& rAccelerationVariable,
455 const SizeType DomainSize = 3
456 )
457 {
458 Double3DArray& r_current_velocity = itCurrentNode->FastGetSolutionStepValue(rVelocityVariable);
459 Double3DArray& r_current_displacement = itCurrentNode->FastGetSolutionStepValue(rDisplacementVariable);
460
461 const Double3DArray& r_previous_displacement = itCurrentNode->FastGetSolutionStepValue(rDisplacementVariable, 1);
462 const Double3DArray& r_previous_velocity = itCurrentNode->FastGetSolutionStepValue(rVelocityVariable, 1);
463 const Double3DArray& r_previous_acceleration = itCurrentNode->FastGetSolutionStepValue(rAccelerationVariable, 1);
464
465 for (IndexType j = 0; j < DomainSize; j++) {
466
467 r_current_displacement[j] = r_previous_displacement[j] + mTime.MidStep * r_previous_velocity[j];
468 r_current_displacement[j] += 0.50 * mTime.MidStep * mTime.MidStep * r_previous_acceleration[j];
469
470 r_current_velocity[j] = r_previous_velocity[j] + mTime.MidStep * r_previous_acceleration[j];
471 } // for DomainSize
472 }

474 void UpdateDegreesOfFreedomStage2(
475 NodeIterator itCurrentNode,
476 const ArrayVarType& rVelocityVariable,
477 const ArrayVarType& rDisplacementVariable,
478 const ArrayVarType& rAccelerationVariable,
479 const ArrayVarType& rFractionalAccelerationVariable,
480 const SizeType DomainSize = 3
481 )
482 {
483 Double3DArray& r_current_velocity = itCurrentNode->FastGetSolutionStepValue(rVelocityVariable);
484 Double3DArray& r_current_displacement = itCurrentNode->FastGetSolutionStepValue(rDisplacementVariable);
485 Double3DArray& r_fractional_acceleration = itCurrentNode->FastGetSolutionStepValue(rFractionalAccelerationVariable);
486
487 const Double3DArray& r_previous_displacement = itCurrentNode->FastGetSolutionStepValue(rDisplacementVariable, 1);
488 const Double3DArray& r_previous_velocity = itCurrentNode->FastGetSolutionStepValue(rVelocityVariable, 1);
489 const Double3DArray& r_previous_acceleration = itCurrentNode->FastGetSolutionStepValue(rAccelerationVariable, 1);
490
491
492 for (IndexType j = 0; j < DomainSize; j++) {
493
494 r_current_displacement[j] = r_previous_displacement[j] + mTime.Delta * r_previous_velocity[j];
495 r_current_displacement[j] += 0.50 * mTime.Delta * mTime.Delta * r_previous_acceleration[j] * ((3.0*mDeltaTime.Fraction-1.0)/(3.0*mDeltaTime.Fraction));
496 r_current_displacement[j] += 0.50 * mTime.Delta * mTime.Delta * r_fractional_acceleration[j] * (1.0/(3.0*mDeltaTime.Fraction));
497
498
499 r_current_velocity[j] = r_previous_velocity[j];
500 r_current_velocity[j] += mTime.Delta * r_previous_acceleration[j] * ((2.0*mDeltaTime.Fraction-1.0)/(2.0*mDeltaTime.Fraction));
501 r_current_velocity[j] += mTime.Delta * r_fractional_acceleration[j] * ((1.0)/(2.0*mDeltaTime.Fraction));
502 } // for DomainSize
503 }

544 virtual void SchemeCustomInitialization(
545 ModelPart& rModelPart,
546 const SizeType DomainSize = 3
547 )
548 {
549 KRATOS_TRY
550
551 // The array containing the nodes
552 NodesArrayType& r_nodes = rModelPart.Nodes();
553
554 // The fisrt node interator
555 const auto it_node_begin = rModelPart.NodesBegin();
556
557 // Auxiliary zero array
558 const Double3DArray zero_array = ZeroVector(3);
559
560 // If we consider the rotation DoF
561 const bool has_dof_for_rot_z = it_node_begin->HasDofFor(ROTATION_Z);
562
563 // Getting dof position
564 const IndexType disppos = it_node_begin->GetDofPosition(DISPLACEMENT_X);
565 const IndexType rotppos = has_dof_for_rot_z ? it_node_begin->GetDofPosition(ROTATION_X) : 0;
566
567 #pragma omp parallel for schedule(guided,512)
568 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
569 // Current step information "N+1" (before step update).
570 auto it_node = it_node_begin + i;
571
572 const double nodal_mass = it_node->GetValue(NODAL_MASS);
573 const Double3DArray& r_current_residual = it_node->FastGetSolutionStepValue(FORCE_RESIDUAL);
574 Double3DArray& r_current_acceleration = it_node->FastGetSolutionStepValue(ACCELERATION);
575
576 // Solution of the explicit equation:
577 if (nodal_mass > numerical_limit) {
578 r_current_acceleration = r_current_residual / nodal_mass;
579 } else {
580 r_current_acceleration = zero_array;
581 }
582
583 std::array<bool, 3> fix_displacements = {false, false, false};
584
585 fix_displacements[0] = (it_node->GetDof(DISPLACEMENT_X, disppos).IsFixed());
586 fix_displacements[1] = (it_node->GetDof(DISPLACEMENT_Y, disppos + 1).IsFixed());
587 if (DomainSize == 3)
588 fix_displacements[2] = (it_node->GetDof(DISPLACEMENT_Z, disppos + 2).IsFixed());
589
590 for (IndexType j = 0; j < DomainSize; j++) {
591 if (fix_displacements[j]) {
592 r_current_acceleration[j] = 0.0;
593 }
594 } // for DomainSize
595
596
598 if (has_dof_for_rot_z) {
599 const Double3DArray& nodal_inertia = it_node->GetValue(NODAL_INERTIA);
600 const Double3DArray& r_current_residual_moment = it_node->FastGetSolutionStepValue(MOMENT_RESIDUAL);
601 Double3DArray& r_current_angular_acceleration = it_node->FastGetSolutionStepValue(ANGULAR_ACCELERATION);
602
603 const IndexType initial_k = DomainSize == 3 ? 0 : 2; // We do this because in 2D only the rotation Z is needed, then we start with 2, instead of 0
604 for (IndexType kk = initial_k; kk < 3; ++kk) {
605 if (nodal_inertia[kk] > numerical_limit) {
606 r_current_angular_acceleration[kk] = r_current_residual_moment[kk] / nodal_inertia[kk];
607 } else {
608 r_current_angular_acceleration[kk] = 0.0;
609 }
610 }
611
612 std::array<bool, 3> fix_rotation = {false, false, false};
613 if (DomainSize == 3) {
614 fix_rotation[0] = (it_node->GetDof(ROTATION_X, rotppos).IsFixed());
615 fix_rotation[1] = (it_node->GetDof(ROTATION_Y, rotppos + 1).IsFixed());
616 }
617 fix_rotation[2] = (it_node->GetDof(ROTATION_Z, rotppos + 2).IsFixed());
618
619 for (IndexType j = initial_k; j < 3; j++) {
620 if (fix_rotation[j]) {
621 r_current_angular_acceleration[j] = 0.0;
622 }
623 } // trans DoF
624 } // Rot DoF
625 } // for node parallel
626 KRATOS_CATCH("")
627 }

700 void Predict(
701 ModelPart& rModelPart,
702 DofsArrayType& rDofSet,
703 TSystemMatrixType& A,
704 TSystemVectorType& Dx,
705 TSystemVectorType& b
706 ) override
707 {
708 KRATOS_TRY;
709 // The current process info
710 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
711
712 // The array of nodes
713 NodesArrayType& r_nodes = rModelPart.Nodes();
714
716 const SizeType dim = r_current_process_info[DOMAIN_SIZE];
717
718 // Step Update
719 // The first step is time = initial_time ( 0.0) + delta time
720 mTime.Delta = r_current_process_info[DELTA_TIME];
721 mTime.MidStep = mTime.Delta * mDeltaTime.Fraction;
722
723 // The iterator of the first node
724 const auto it_node_begin = rModelPart.NodesBegin();
725 const bool has_dof_for_rot_z = it_node_begin->HasDofFor(ROTATION_Z);
726
727
728 // Getting dof position
729 const IndexType disppos = it_node_begin->GetDofPosition(DISPLACEMENT_X);
730 const IndexType rotppos = has_dof_for_rot_z ? it_node_begin->GetDofPosition(ROTATION_X) : 0;
731
732 // _____________________________________________________________________
733 // ______________________________ STAGE 1 ______________________________
734 // _____________________________________________________________________
735
736 #pragma omp parallel for schedule(guided,512)
737 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
738 // Current step information "N+1" (before step update).
739 this->UpdateDegreesOfFreedomStage1(it_node_begin + i,
740 VELOCITY,DISPLACEMENT,ACCELERATION,dim);
741 } // for Node parallel
742
743
744 if (has_dof_for_rot_z){
745 #pragma omp parallel for schedule(guided,512)
746 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
747 // Current step information "N+1" (before step update).
748 this->UpdateDegreesOfFreedomStage1(it_node_begin + i,
749 ANGULAR_VELOCITY,ROTATION,ANGULAR_ACCELERATION,dim);
750 } // for Node parallel
751 }
752
753
754 InitializeResidual(rModelPart);
755 CalculateAndAddRHS(rModelPart);
756
757 #pragma omp parallel for schedule(guided,512)
758 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
759 // Current step information "N+1" (before step update).
760 this->UpdateAccelerationStage(
761 it_node_begin + i, disppos,FRACTIONAL_ACCELERATION,
762 VELOCITY,NODAL_DISPLACEMENT_DAMPING,FORCE_RESIDUAL,NODAL_MASS,
763 DISPLACEMENT_X,DISPLACEMENT_Y,DISPLACEMENT_Z,dim);
764 } // for Node parallel
765
766 if (has_dof_for_rot_z){
767 #pragma omp parallel for schedule(guided,512)
768 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
769 // Current step information "N+1" (before step update).
770 this->UpdateAccelerationStage(
771 it_node_begin + i, rotppos,FRACTIONAL_ANGULAR_ACCELERATION,
772 ANGULAR_VELOCITY,NODAL_ROTATION_DAMPING,MOMENT_RESIDUAL,
773 NODAL_INERTIA,ROTATION_X,ROTATION_Y,ROTATION_Z,dim);
774 } // for Node parallel
775 }
776
777 // _____________________________________________________________________
778 // ______________________________ STAGE 2 ______________________________
779 // _____________________________________________________________________
780
781 #pragma omp parallel for schedule(guided,512)
782 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
783 // Current step information "N+1" (before step update).
784 this->UpdateDegreesOfFreedomStage2(it_node_begin + i,
785 VELOCITY,DISPLACEMENT,ACCELERATION,FRACTIONAL_ACCELERATION,dim);
786 } // for Node parallel
787
788 if (has_dof_for_rot_z){
789 #pragma omp parallel for schedule(guided,512)
790 for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
791 // Current step information "N+1" (before step update).
792 this->UpdateDegreesOfFreedomStage2(it_node_begin + i,
793 ANGULAR_VELOCITY,ROTATION,ANGULAR_ACCELERATION,
794 FRACTIONAL_ANGULAR_ACCELERATION,dim);
795 } // for Node parallel
796 }
797
798 InitializeResidual(rModelPart);
799 CalculateAndAddRHS(rModelPart);
800
801 KRATOS_CATCH("")
802 }
