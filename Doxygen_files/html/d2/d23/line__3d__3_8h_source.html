744  static Matrix CalculateShapeFunctionsIntegrationPointsValues( typename BaseType::IntegrationMethod ThisMethod )
745  {
746  const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
747  const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[static_cast<int>(ThisMethod)];
748  int integration_points_number = IntegrationPoints.size();
749  Matrix N( integration_points_number, 3 );
750 
751  for ( int it_gp = 0; it_gp < integration_points_number; it_gp++ )
752  {
753  double e = IntegrationPoints[it_gp].X();
754  N( it_gp, 0 ) = 0.5 * ( e - 1 ) * e;
755  N( it_gp, 2 ) = 1.0 - e * e;
756  N( it_gp, 1 ) = 0.5 * ( 1 + e ) * e;
