408 Vector& DeterminantOfJacobian( Vector& rResult, IntegrationMethod ThisMethod ) const override
409 {
410 const std::size_t number_of_integration_points = this->IntegrationPointsNumber( ThisMethod );
411 if( rResult.size() != number_of_integration_points)
412 rResult.resize( number_of_integration_points, false );
413
414 Matrix J(3, 1);
415 for (std::size_t pnt = 0; pnt < number_of_integration_points; ++pnt) {
416 this->Jacobian( J, pnt, ThisMethod);
417 rResult[pnt] = std::sqrt(std::pow(J(0,0), 2) + std::pow(J(1,0), 2) + std::pow(J(2,0), 2));
418 }

744 static Matrix CalculateShapeFunctionsIntegrationPointsValues( typename BaseType::IntegrationMethod ThisMethod )
745 {
746 const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
747 const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[static_cast<int>(ThisMethod)];
748 int integration_points_number = IntegrationPoints.size();
749 Matrix N( integration_points_number, 3 );
750
751 for ( int it_gp = 0; it_gp < integration_points_number; it_gp++ )
752 {
753 double e = IntegrationPoints[it_gp].X();
754 N( it_gp, 0 ) = 0.5 * ( e - 1 ) * e;
755 N( it_gp, 2 ) = 1.0 - e * e;
756 N( it_gp, 1 ) = 0.5 * ( 1 + e ) * e;
757 }
758
759 return N;
760 }

762 static ShapeFunctionsGradientsType CalculateShapeFunctionsIntegrationPointsLocalGradients(
763 typename BaseType::IntegrationMethod ThisMethod )
764 {
765 const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
766 const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[static_cast<int>(ThisMethod)];
767 ShapeFunctionsGradientsType DN_De( IntegrationPoints.size() );
768 std::fill( DN_De.begin(), DN_De.end(), Matrix( 3, 1 ) );
769
770 for ( unsigned int it_gp = 0; it_gp < IntegrationPoints.size(); it_gp++ )
771 {
772 Matrix aux_mat = ZeroMatrix(3,1);
773 const double e = IntegrationPoints[it_gp].X();
774 aux_mat(0,0) = e - 0.5;
775 aux_mat(2,0) = -2.0 * e;
776 aux_mat(1,0) = e + 0.5;
777 DN_De[it_gp] = aux_mat;
778 }
779
780 return DN_De;
781 }
