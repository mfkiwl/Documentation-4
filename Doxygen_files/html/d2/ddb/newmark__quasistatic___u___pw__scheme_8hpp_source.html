280  void FinalizeSolutionStep(
281  ModelPart& rModelPart,
282  TSystemMatrixType& A,
283  TSystemVectorType& Dx,
284  TSystemVectorType& b) override
285  {
286  KRATOS_TRY
287 
288  if (rModelPart.GetProcessInfo()[NODAL_SMOOTHING]) {
289  unsigned int Dim = rModelPart.GetProcessInfo()[DOMAIN_SIZE];
290 
291  SizeType StressTensorSize = STRESS_TENSOR_SIZE_2D;
292  if (Dim == N_DIM_3D) StressTensorSize = STRESS_TENSOR_SIZE_3D;
293 
294  // Clear nodal variables
295  block_for_each(rModelPart.Nodes(), [&StressTensorSize](Node<3>& rNode) {
296  rNode.FastGetSolutionStepValue(NODAL_AREA) = 0.0;
297  Matrix& rNodalStress = rNode.FastGetSolutionStepValue(NODAL_CAUCHY_STRESS_TENSOR);
298  if (rNodalStress.size1() != StressTensorSize)
299  rNodalStress.resize(StressTensorSize,StressTensorSize,false);
300  noalias(rNodalStress) = ZeroMatrix(StressTensorSize, StressTensorSize);
301  rNode.FastGetSolutionStepValue(NODAL_DAMAGE_VARIABLE) = 0.0;
302  rNode.FastGetSolutionStepValue(NODAL_JOINT_AREA) = 0.0;
303  rNode.FastGetSolutionStepValue(NODAL_JOINT_WIDTH) = 0.0;
304  rNode.FastGetSolutionStepValue(NODAL_JOINT_DAMAGE) = 0.0;
305  });
306 
307  FinalizeSolutionStepActiveEntities(rModelPart,A,Dx,b);
308 
309  // Compute smoothed nodal variables
310  block_for_each(rModelPart.Nodes(), [&](Node<3>& rNode) {
311  const double& NodalArea = rNode.FastGetSolutionStepValue(NODAL_AREA);
312  if (NodalArea > 1.0e-20) {
313  const double InvNodalArea = 1.0/NodalArea;
314  Matrix& rNodalStress = rNode.FastGetSolutionStepValue(NODAL_CAUCHY_STRESS_TENSOR);
315  for(unsigned int i = 0; i < rNodalStress.size1(); ++i) {
316  for(unsigned int j = 0; j < rNodalStress.size2(); ++j) {
317  rNodalStress(i,j) *= InvNodalArea;
318  }
319  }
320  rNode.FastGetSolutionStepValue(NODAL_DAMAGE_VARIABLE) *= InvNodalArea;
321  }
322 
323  const double& NodalJointArea = rNode.FastGetSolutionStepValue(NODAL_JOINT_AREA);
324  if (NodalJointArea > 1.0e-20) {
325  const double InvNodalJointArea = 1.0/NodalJointArea;
326  rNode.FastGetSolutionStepValue(NODAL_JOINT_WIDTH) *= InvNodalJointArea;
327  rNode.FastGetSolutionStepValue(NODAL_JOINT_DAMAGE) *= InvNodalJointArea;
328  }
329  });
330 
331  } else {
332  FinalizeSolutionStepActiveEntities(rModelPart,A,Dx,b);
333  }
334 
335  KRATOS_CATCH("")
336  }

500  for (typename DofsArrayType::iterator itDof = DofsBegin; itDof != DofsEnd; ++itDof) {
501  if (itDof->IsFree())
502  itDof->GetSolutionStepValue() += TSparseSpace::GetValue(Dx, itDof->EquationId());

508  KRATOS_CATCH( "" )

521  virtual inline void SetTimeFactors(ModelPart& rModelPart)
522  {
523  KRATOS_TRY
524 
525  mDeltaTime = rModelPart.GetProcessInfo()[DELTA_TIME];
526  rModelPart.GetProcessInfo()[VELOCITY_COEFFICIENT] = mGamma/(mBeta*mDeltaTime);

532 //----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
533  virtual inline void UpdateVariablesDerivatives(ModelPart& rModelPart)
534  {
535  KRATOS_TRY
536 
537  //Update Acceleration, Velocity and DtPressure
538  block_for_each(rModelPart.Nodes(), [&](Node<3>& rNode) {
539  noalias(rNode.FastGetSolutionStepValue(ACCELERATION)) = (( rNode.FastGetSolutionStepValue(DISPLACEMENT)
540  - rNode.FastGetSolutionStepValue(DISPLACEMENT, 1))
541  - mDeltaTime * rNode.FastGetSolutionStepValue(VELOCITY, 1)
542  - (0.5-mBeta) * mDeltaTime * mDeltaTime
543  * rNode.FastGetSolutionStepValue(ACCELERATION, 1) )
544  / (mBeta*mDeltaTime*mDeltaTime);
545 
546  noalias(rNode.FastGetSolutionStepValue(VELOCITY)) = rNode.FastGetSolutionStepValue(VELOCITY, 1)
547  + (1.0-mGamma)*mDeltaTime
548  * rNode.FastGetSolutionStepValue(ACCELERATION, 1)
549  + mGamma * mDeltaTime
550  * rNode.FastGetSolutionStepValue(ACCELERATION);
551 
552  const double DeltaPressure = rNode.FastGetSolutionStepValue(WATER_PRESSURE)
553  - rNode.FastGetSolutionStepValue(WATER_PRESSURE, 1);
554 
555  rNode.FastGetSolutionStepValue(DT_WATER_PRESSURE) = ( DeltaPressure
556  - (1.0-mTheta) * mDeltaTime
557  * rNode.FastGetSolutionStepValue(DT_WATER_PRESSURE, 1))
558  / (mTheta*mDeltaTime);
559  });
560 
561  KRATOS_CATCH( "" )
562  }
