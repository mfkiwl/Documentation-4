173 static void IntegrateStressVector(
174 BoundedArrayType& rPredictiveStressVector,
175 Vector& rStrainVector,
176 double& rUniaxialStress,
177 double& rThreshold,
178 double& rPlasticDenominator,
179 BoundedArrayType& rYieldSurfaceDerivative,
180 BoundedArrayType& rDerivativePlasticPotential,
181 double& rPlasticDissipation,
182 BoundedArrayType& rPlasticStrainIncrement,
183 Matrix& rConstitutiveMatrix,
184 Vector& rPlasticStrain,
185 ConstitutiveLaw::Parameters& rValues,
186 const double CharacteristicLength,
187 Vector& rBackStressVector,
188 const Vector& rPreviousStressVector
189 )
190 {
191 // Material properties
192 const Properties& r_material_properties = rValues.GetMaterialProperties();
193
194 // Defining some variables
195 bool is_converged = false;
196 IndexType iteration = 0, max_iter = r_material_properties.Has(MAX_NUMBER_NL_CL_ITERATIONS) ? r_material_properties.GetValue(MAX_NUMBER_NL_CL_ITERATIONS) : 100;
197 BoundedArrayType delta_sigma;
198 double plastic_consistency_factor_increment, threshold_indicator;
199 BoundedArrayType kin_hard_stress_vector;
200 Matrix tangent_tensor = ZeroMatrix(6,6);
201
202 // Backward Euler
203 while (is_converged == false && iteration <= max_iter) {
204 threshold_indicator = rUniaxialStress - rThreshold;
205 plastic_consistency_factor_increment = threshold_indicator * rPlasticDenominator;
206 noalias(rPlasticStrainIncrement) = plastic_consistency_factor_increment * rDerivativePlasticPotential;
207 noalias(rPlasticStrain) += rPlasticStrainIncrement;
208 noalias(delta_sigma) = prod(rConstitutiveMatrix, rPlasticStrainIncrement);
209 noalias(rPredictiveStressVector) -= delta_sigma;
210 CalculateBackStress(rPredictiveStressVector, rValues, rPreviousStressVector,
211 rPlasticStrainIncrement, rBackStressVector);
212 noalias(kin_hard_stress_vector) = rPredictiveStressVector - rBackStressVector;
213 threshold_indicator = CalculatePlasticParameters(kin_hard_stress_vector, rStrainVector, rUniaxialStress, rThreshold,
214 rPlasticDenominator, rYieldSurfaceDerivative, rDerivativePlasticPotential, rPlasticDissipation, rPlasticStrainIncrement,
215 rConstitutiveMatrix, rValues, CharacteristicLength, rPlasticStrain, rBackStressVector);
216
217
218 if (std::abs(threshold_indicator) <= std::abs(1.0e-4 * rThreshold)) { // Has converged
219 is_converged = true;
220 } else {
221 iteration++;
222 }
223 }
224 CalculateTangentMatrix(tangent_tensor, rConstitutiveMatrix, rYieldSurfaceDerivative, rDerivativePlasticPotential, rPlasticDenominator);
225 noalias(rConstitutiveMatrix) = tangent_tensor;
226 KRATOS_WARNING_IF("GenericConstitutiveLawIntegratorKinematicPlasticity", iteration > max_iter) << "Maximum number of iterations in plasticity loop reached..." << std::endl;
227 }

329 static void CalculateBackStress(
330 BoundedArrayType& rPredictiveStressVector,
331 ConstitutiveLaw::Parameters& rValues,
332 const Vector& rPreviousStressVector,
333 const Vector& rPlasticStrainIncrement,
334 Vector& rBackStressVector
335 )
336 {
337 const Vector& r_kinematic_parameters = rValues.GetMaterialProperties()[KINEMATIC_PLASTICITY_PARAMETERS];
338 const unsigned int kinematic_hardening_type = rValues.GetMaterialProperties()[KINEMATIC_HARDENING_TYPE];
339
340 switch (static_cast<KinematicHardeningType>(kinematic_hardening_type))
341 {
342 double pDot, denominator, dot_product_dp;
343 case KinematicHardeningType::LinearKinematicHardening:
344 KRATOS_ERROR_IF(r_kinematic_parameters.size() == 0) << "Kinematic Parameters not defined..." << std::endl;
345 rBackStressVector += 2.0 / 3.0 * r_kinematic_parameters[0] * rPlasticStrainIncrement;
346 break;
347
348 case KinematicHardeningType::ArmstrongFrederickKinematicHardening:
349 KRATOS_ERROR_IF(r_kinematic_parameters.size() < 2) << "Kinematic Parameters not defined..." << std::endl;
350 dot_product_dp = 0.0;
351 for (IndexType i = 0; i < rPlasticStrainIncrement.size(); ++i) {
352 dot_product_dp += rPlasticStrainIncrement[i] * rPlasticStrainIncrement[i];
353 }
354 pDot = std::sqrt(2.0 / 3.0 * dot_product_dp);
355 denominator = 1.0 + (r_kinematic_parameters[1] * pDot);
356 rBackStressVector = (rBackStressVector + ((2.0 / 3.0 * r_kinematic_parameters[0]) * rPlasticStrainIncrement)) / denominator;
357 break;
358
359 case KinematicHardeningType::AraujoVoyiadjisKinematicHardening:
360 KRATOS_ERROR_IF(r_kinematic_parameters.size() != 3) << "Kinematic Parameters not defined..." << std::endl;
361 dot_product_dp = 0.0;
362 for (IndexType i = 0; i < rPlasticStrainIncrement.size(); ++i) {
363 dot_product_dp += rPlasticStrainIncrement[i] * rPlasticStrainIncrement[i];
364 }
365 pDot = std::sqrt(2.0 / 3.0 * dot_product_dp);
366 denominator = 1.0 + (r_kinematic_parameters[1] * pDot);
367 if (pDot > tolerance) {
368 rBackStressVector = (rBackStressVector + ((2.0 / 3.0 * r_kinematic_parameters[0]) * rPlasticStrainIncrement)) / denominator;
369 } else {
370 const Vector& r_delta_stress = rPredictiveStressVector - rPreviousStressVector;
371 rBackStressVector = (rBackStressVector + ((2.0 / 3.0 * r_kinematic_parameters[0]) * rPlasticStrainIncrement) +
372 r_kinematic_parameters[2] * r_delta_stress) / denominator;
373 }
374 break;
375 default:
376 KRATOS_ERROR << " The Kinematic hardening type of plasticity is not set or wrong..." << kinematic_hardening_type << std::endl;
377 break;
378 }
379 }

645 static void CalculatePlasticDenominator(
646 const BoundedArrayType& rFFlux,
647 const BoundedArrayType& rGFlux,
648 const Matrix& rConstitutiveMatrix,
649 double& rHardeningParameter,
650 double& rPlasticDenominator,
651 const Vector& rBackStressVector,
652 ConstitutiveLaw::Parameters& rValues
653 )
654 {
655 const Vector& r_kinematic_parameters = rValues.GetMaterialProperties()[KINEMATIC_PLASTICITY_PARAMETERS];
656 const int kinematic_hardening_type = rValues.GetMaterialProperties()[KINEMATIC_HARDENING_TYPE];
657
658 const BoundedArrayType delta_vector = prod(rGFlux, rConstitutiveMatrix);
659 double A1 = 0.0;
660 for (IndexType i = 0; i < VoigtSize; ++i) {
661 A1 += rFFlux[i] * delta_vector[i];
662 }
663 if (r_kinematic_parameters.size() == 3) {
664 A1 *= (1.0 - r_kinematic_parameters[2]);
665 } // Araujo case with 3 params
666
667 double dot_fflux_gflux = 0.0, A2;
668 for (IndexType i = 0; i < VoigtSize; ++i) {
669 dot_fflux_gflux += rFFlux[i] * rGFlux[i];
670 }
671 const double two_thirds = 2.0 / 3.0;
672 double dot_fflux_backstress = 0.0, dot_gflux_gflux = 0.0;
673 switch (static_cast<KinematicHardeningType>(kinematic_hardening_type))
674 {
675 case KinematicHardeningType::LinearKinematicHardening:
676 A2 = two_thirds * r_kinematic_parameters[0] * dot_fflux_gflux;
677 break;
678
679 case KinematicHardeningType::ArmstrongFrederickKinematicHardening:
680 A2 = two_thirds * r_kinematic_parameters[0] * dot_fflux_gflux;
681 for (IndexType i = 0; i < VoigtSize; ++i) {
682 dot_fflux_backstress += rFFlux[i] * rBackStressVector[i];
683 }
684 for (IndexType i = 0; i < VoigtSize; ++i) {
685 dot_gflux_gflux += rGFlux[i] * rGFlux[i];
686 }
687 A2 -= r_kinematic_parameters[1] * dot_fflux_backstress * std::sqrt(two_thirds * dot_gflux_gflux);
688 break;
689
690 case KinematicHardeningType::AraujoVoyiadjisKinematicHardening:
691 A2 = two_thirds * r_kinematic_parameters[0] * dot_fflux_gflux;
692 for (IndexType i = 0; i < VoigtSize; ++i) {
693 dot_fflux_backstress += rFFlux[i] * rBackStressVector[i];
694 }
695 for (IndexType i = 0; i < VoigtSize; ++i) {
696 dot_gflux_gflux += rGFlux[i] * rGFlux[i];
697 }
698 A2 -= r_kinematic_parameters[1] * dot_fflux_backstress * std::sqrt(two_thirds * dot_gflux_gflux);
699 break;
700
701 default:
702 KRATOS_ERROR << " The Kinematic hardening type of plasticity is not set or wrong..." << kinematic_hardening_type << std::endl;
703 break;
704 }
705
706 const double A3 = rHardeningParameter;
707 rPlasticDenominator = 1.0 / (A1 + A2 + A3);
708
709 if (r_kinematic_parameters.size() == 3) {
710 rPlasticDenominator *= (1.0 - r_kinematic_parameters[2]);
711 } // Araujo case with 3 params
712 }

718 static int Check(const Properties& rMaterialProperties)
719 {
720 // Checking is defined
721 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "HARDENING_CURVE is not a defined value" << std::endl;
722 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(HARDENING_CURVE)) << "HARDENING_CURVE is not a defined value" << std::endl;
723 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRACTURE_ENERGY)) << "FRACTURE_ENERGY is not a defined value" << std::endl;
724
725 // Checking curves
726 const int curve_type = rMaterialProperties[HARDENING_CURVE];
727 if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::InitialHardeningExponentialSoftening) {
728 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS)) << "MAXIMUM_STRESS is not a defined value" << std::endl;
729 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS_POSITION)) << "MAXIMUM_STRESS_POSITION is not a defined value" << std::endl;
730 } else if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::CurveFittingHardening) {
731 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(CURVE_FITTING_PARAMETERS)) << "CURVE_FITTING_PARAMETERS is not a defined value" << std::endl;
732 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(PLASTIC_STRAIN_INDICATORS)) << "PLASTIC_STRAIN_INDICATORS is not a defined value" << std::endl;
733 }
734
735 if (!rMaterialProperties.Has(YIELD_STRESS)) {
736 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
737 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
738
739 const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
740 const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
741
742 KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
743 KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
744 } else {
745 const double yield_stress = rMaterialProperties[YIELD_STRESS];
746
747 KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
748 }
749
750 return TYieldSurfaceType::Check(rMaterialProperties);
751 }
