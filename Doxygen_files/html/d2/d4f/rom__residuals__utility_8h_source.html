39 RomResidualsUtility(
40 ModelPart& rModelPart,
41 Parameters ThisParameters,
42 BaseSchemeType::Pointer pScheme
43 ): mrModelPart(rModelPart), mpScheme(pScheme){
44 // Validate default parameters
45 Parameters default_parameters = Parameters(R"(
46 {
47 "nodal_unknowns" : [],
48 "number_of_rom_dofs" : 10,
49 "petrov_galerkin_number_of_rom_dofs" : 10
50 })" );
51
52 ThisParameters.ValidateAndAssignDefaults(default_parameters);
53
54 mNodalVariablesNames = ThisParameters["nodal_unknowns"].GetStringArray();
55
56 mNodalDofs = mNodalVariablesNames.size();
57 mRomDofs = ThisParameters["number_of_rom_dofs"].GetInt();
58 mPetrovGalerkinRomDofs = ThisParameters["petrov_galerkin_number_of_rom_dofs"].GetInt();
59
60 // Setting up mapping: VARIABLE_KEY --> CORRECT_ROW_IN_BASIS
61 for(int k=0; k<mNodalDofs; k++){
62 if(KratosComponents<Variable<double>>::Has(mNodalVariablesNames[k]))
63 {
64 const auto& var = KratosComponents<Variable<double>>::Get(mNodalVariablesNames[k]);
65 MapPhi[var.Key()] = k;
66 }
67 else
68 KRATOS_ERROR << "variable \""<< mNodalVariablesNames[k] << "\" not valid" << std::endl;
69
70 }
71 }

75 Matrix GetProjectedResidualsOntoPhi()
76 {
77 // Getting the number of elements and conditions from the model
78 const int n_elements = static_cast<int>(mrModelPart.Elements().size());
79 const int n_conditions = static_cast<int>(mrModelPart.Conditions().size());
80
81 const auto& r_current_process_info = mrModelPart.GetProcessInfo();
82 const auto el_begin = mrModelPart.ElementsBegin();
83 const auto cond_begin = mrModelPart.ConditionsBegin();
84
85 //contributions to the system
86 Matrix lhs_contribution = ZeroMatrix(0, 0);
87 Vector rhs_contribution = ZeroVector(0);
88
89 //vector containing the localization in the system of the different terms
90 Element::EquationIdVectorType equation_id;
91 Matrix matrix_residuals( (n_elements + n_conditions), mRomDofs); // Matrix of reduced residuals.
92 Matrix phi_elemental;
93
94 //dofs container initialization
95 Element::DofsVectorType elem_dofs;
96 Condition::DofsVectorType cond_dofs;
97 #pragma omp parallel firstprivate(n_elements, n_conditions, lhs_contribution, rhs_contribution, equation_id, phi_elemental, el_begin, cond_begin, elem_dofs, cond_dofs)
98 {
99 #pragma omp for nowait
100 for (int k = 0; k < n_elements; k++){
101 const auto it_el = el_begin + k;
102 //detect if the element is active or not. If the user did not make any choice the element is active by default
103 bool element_is_active = true;
104 if ((it_el)->IsDefined(ACTIVE))
105 element_is_active = (it_el)->Is(ACTIVE);
106 if (element_is_active){
107 //calculate elemental contribution
108 mpScheme->CalculateSystemContributions(*it_el, lhs_contribution, rhs_contribution, equation_id, r_current_process_info);
109 it_el->GetDofList(elem_dofs, r_current_process_info);
110 //assemble the elemental contribution - here is where the ROM acts
111 //compute the elemental reduction matrix phi_elemental
112 const auto& r_geom = it_el->GetGeometry();
113 if(phi_elemental.size1() != elem_dofs.size() || phi_elemental.size2() != mRomDofs)
114 phi_elemental.resize(elem_dofs.size(), mRomDofs,false);
115 RomAuxiliaryUtilities::GetPhiElemental(phi_elemental, elem_dofs, r_geom, MapPhi);
116 noalias(row(matrix_residuals, k)) = prod(trans(phi_elemental), rhs_contribution); // The size of the residual will vary only when using more ROM modes, one row per condition
117 }
118
119 }
120
121 #pragma omp for nowait
122 for (int k = 0; k < n_conditions; k++){
123 ModelPart::ConditionsContainerType::iterator it = cond_begin + k;
124 //detect if the condition is active or not. If the user did not make any choice the condition is active by default
125 bool condition_is_active = true;
126 if ((it)->IsDefined(ACTIVE))
127 condition_is_active = (it)->Is(ACTIVE);
128 if (condition_is_active){
129 it->GetDofList(cond_dofs, r_current_process_info);
130 //calculate elemental contribution
131 mpScheme->CalculateSystemContributions(*it, lhs_contribution, rhs_contribution, equation_id, r_current_process_info);
132 //assemble the elemental contribution - here is where the ROM acts
133 //compute the elemental reduction matrix phi_elemental
134 const auto& r_geom = it->GetGeometry();
135 if(phi_elemental.size1() != cond_dofs.size() || phi_elemental.size2() != mRomDofs)
136 phi_elemental.resize(cond_dofs.size(), mRomDofs,false);
137 RomAuxiliaryUtilities::GetPhiElemental(phi_elemental, cond_dofs, r_geom, MapPhi);
138 noalias(row(matrix_residuals, k+n_elements)) = prod(trans(phi_elemental), rhs_contribution); // The size of the residual will vary only when using more ROM modes, one row per condition
139 }
140 }
141 }
142 return matrix_residuals;
143 }

211 Matrix GetProjectedGlobalLHS()
212 {
213 const int n_elements = static_cast<int>(mrModelPart.Elements().size());
214 const int n_conditions = static_cast<int>(mrModelPart.Conditions().size());
215 const auto& n_nodes = mrModelPart.NumberOfNodes();
216
217 const auto& r_current_process_info = mrModelPart.GetProcessInfo();
218
219 const int system_size = n_nodes*mNodalDofs;
220
221 const auto el_begin = mrModelPart.ElementsBegin();
222 const auto cond_begin = mrModelPart.ConditionsBegin();
223
224 //contributions to the system
225 Matrix lhs_contribution = ZeroMatrix(0,0);
226
227 //vector containing the localization in the system of the different terms
228 Element::EquationIdVectorType equation_id;
229 Matrix a_phi = ZeroMatrix(system_size, mRomDofs);
230
231 //dofs container initialization
232 Element::DofsVectorType elem_dofs;
233 Condition::DofsVectorType cond_dofs;
234
235 Matrix phi_elemental;
236 Matrix temp_a_phi = ZeroMatrix(system_size,mRomDofs);
237 Matrix aux;
238
239 #pragma omp parallel firstprivate(n_elements, n_conditions, lhs_contribution, equation_id, el_begin, cond_begin, elem_dofs, cond_dofs)
240 {
241
242 #pragma omp for nowait
243 for (int k = 0; k < static_cast<int>(n_elements); k++) {
244 const auto it_el = el_begin + k;
245
246 // Detect if the element is active or not. If the user did not make any choice the element is active by default
247 const bool element_is_active = it_el->IsDefined(ACTIVE) ? it_el->Is(ACTIVE) : true;
248
249 // Calculate elemental contribution
250 if (element_is_active){
251 mpScheme->CalculateLHSContribution(*it_el, lhs_contribution, equation_id, r_current_process_info);
252 it_el->GetDofList(elem_dofs, r_current_process_info);
253 const auto &r_geom = it_el->GetGeometry();
254 if(phi_elemental.size1() != elem_dofs.size() || phi_elemental.size2() != mRomDofs) {
255 phi_elemental.resize(elem_dofs.size(), mRomDofs,false);
256 }
257 if(aux.size1() != elem_dofs.size() || aux.size2() != mRomDofs) {
258 aux.resize(elem_dofs.size(), mRomDofs,false);
259 }
260 RomAuxiliaryUtilities::GetPhiElemental(phi_elemental, elem_dofs, r_geom, MapPhi);
261 noalias(aux) = prod(lhs_contribution, phi_elemental);
262 for(int d = 0; d < static_cast<int>(elem_dofs.size()); ++d){
263 if (elem_dofs[d]->IsFixed()==false){
264 row(temp_a_phi,elem_dofs[d]->EquationId()) += row(aux,d);// Add contributions to global system for free dofs.
265 }
266 }
267 }
268 }
269
270 #pragma omp for nowait
271 for (int k = 0; k < static_cast<int>(n_conditions); k++){
272 const auto it = cond_begin + k;
273
274 // Detect if the element is active or not. If the user did not make any choice the condition is active by default
275 const bool condition_is_active = it->IsDefined(ACTIVE) ? it->Is(ACTIVE) : true;
276
277 // Calculate condition contribution
278 if (condition_is_active) {
279 it->GetDofList(cond_dofs, r_current_process_info);
280 mpScheme->CalculateLHSContribution(*it, lhs_contribution, equation_id, r_current_process_info);
281 const auto &r_geom = it->GetGeometry();
282 if(phi_elemental.size1() != cond_dofs.size() || phi_elemental.size2() != mRomDofs) {
283 phi_elemental.resize(cond_dofs.size(), mRomDofs,false);
284 }
285 if(aux.size1() != cond_dofs.size() || aux.size2() != mRomDofs) {
286 aux.resize(cond_dofs.size(), mRomDofs,false);
287 }
288 RomAuxiliaryUtilities::GetPhiElemental(phi_elemental, cond_dofs, r_geom, MapPhi);
289 noalias(aux) = prod(lhs_contribution, phi_elemental);
290 for(int d = 0; d < static_cast<int>(cond_dofs.size()); ++d){
291 if (cond_dofs[d]->IsFixed()==false){
292 row(temp_a_phi,cond_dofs[d]->EquationId()) += row(aux,d);
293 }
294 }
295 }
296 }
297
298 #pragma omp critical
299 {
300 noalias(a_phi) += temp_a_phi;
301 }
302
303 }
304 return a_phi;
305 }
