181 bool LineSearches( TSystemVectorType& X_old,
182 TSystemVectorType& Delta_p,
183 TSystemVectorType& mDx,
184 TSystemVectorType& mb,
185 TSystemMatrixType& mA
186 )
187 {
188
189 KRATOS_TRY
190 double seta = 0.00;
191 double so = 0.00;
192 unsigned int ils = 0;
193 unsigned int ico = 0;
194 unsigned iteration_number = 1;
195 Vector prodr(mMaxLineSearchIterations+1,false);
196 Vector lseta(mMaxLineSearchIterations+2,false);
197 lseta(1) = 1.00;
198 prodr(0) = 1.00;
199 meta = 1.00;
200
201 typename TSchemeType::Pointer pScheme = GetScheme();
202 typename TBuilderAndSolverType::Pointer pBuilderAndSolver = GetBuilderAndSolver();
203 DofsArrayType& rDofSet = pBuilderAndSolver->GetDofSet();
204
205
206 //TSystemMatrixType& mA = *(mpA);
207 //TSystemVectorType& mDx = *(mpDx);
208 //TSystemVectorType& mb = *(mpb);
209 //TSystemVectorType& X_old = *(pX_old);
210 //TSystemVectorType& Delta_p = *(pDelta_p);
211
212 TSparseSpace::SetToZero(mb);
213 pBuilderAndSolver->BuildRHS(pScheme,rmodel_part,mb);
214 so = TSparseSpace::Dot(Delta_p,mb); // mb inicial
215
216 KRATOS_WATCH(so);
217
218 while (iteration_number <= mMaxLineSearchIterations)
219 {
220
221 pScheme->Update(rmodel_part,rDofSet,mA,mDx,mb);
222 //if(BaseType::MoveMeshFlag() == true) BaseType::MoveMesh();
223 pScheme->FinalizeNonLinIteration(rmodel_part,mA,mDx,mb);
224
225 std::cout<<"Line_Search_Iteration_Number:"<<(iteration_number)<<std::endl;
226 TSparseSpace::SetToZero(mb);
227 pBuilderAndSolver->BuildRHS(pScheme,rmodel_part,mb); //mb para calculat eta
228
229 TSparseSpace::Copy(Delta_p, mDx);
230
231 seta = TSparseSpace::Dot(Delta_p,mb); //Delta_p = Const
232 //KRATOS_WATCH(mb);
233 //KRATOS_WATCH(Delta_p);
234 KRATOS_WATCH(seta);
235
236 // Posible restriccion en el denominador
237 ils = iteration_number -1;
238 prodr(ils+1) = (seta/so);
239 KRATOS_WATCH(prodr(ils+1))
240 if (fabs(prodr(ils+1)) < mtolls)
241 {
242 return true;
243 break;
244 }
245 else
246 {
247 //Search devuelve lseta(ils+2)
248 SetDatabaseToValue(rDofSet, X_old);
249 //if(BaseType::MoveMeshFlag() == true) BaseType::MoveMesh();
250 Search(ils,prodr,lseta,ico);
251 std::cout<<"Eta:"<< lseta(ils+2)<<std::endl;
252 meta = lseta(ils+2);
254 TSparseSpace::Assign(mDx,meta, mDx); //mDx =lseta(iteration_number+1)*mDx
256 }
257
258 if (ico==2)
259 {
260 //ilfail = 2;
261 std::cout<<"********************************"<<std::endl;
262 std::cout<<"******Line Search Trouble*******"<<std::endl;
263 std::cout<<"********************************"<<std::endl;
264 return false;
265 break;
266 }
267
268 if (iteration_number>= mMaxLineSearchIterations)
269 {
270 this->MaxIterationsExceeded();
271 return false;
272 break;
273 }
274
275
276 iteration_number++;
277 }
278 KRATOS_CATCH("")
279 }

318//realiza Line-Search local para encontrar la longitud del paso en eta(ils+2)
319//eta: historia de longitudes de pasos en la iteracion de L-S (eta(1)=0.,eta(2)=1.)
320//prodr: productos r
321//amp: factor de amplificacion maximo para el paso de carga en caso de extrapolacion
322//etmxa y etmna: longitudes de pasos maximos y minimos permitidos
323//ico: 1 si en el paso anterior ha sido necesario usar etmxa o etmna, sino es 0
324// al salir ico es 1 si en esta iteracion se ha usado etmxa o etmna y 2 si se han
325//usado por segunda vez.
326
327//obtener ineg=No del S-L actual que este mas cerca del origen y que su prodr sea negativo
328//tambien busca el S-L maximo en etmaxp
329// si no hay existe un eta con prodr<0 ineg=999
330
331 void Search(unsigned int& ils,Vector& prodr, Vector& eta, unsigned int& ico)
332 {
333 KRATOS_TRY
334
335 double etaint = 0.00;
336 double etaneg = 1e5;
337 double etmxt = 0.00;
338 double etmaxp = 0.00;
339 double etaalt = 0.00;
340 double etaext = 0.00;
341 unsigned int ipos = 0;
342 unsigned int ineg = 999;
343
344 for (unsigned int i=0; i<=ils+1; i++)
345 {
346 if (eta(i)>etmaxp)
347 {
348 etmaxp=eta(i);
349 }
350 if (prodr(i)<0 && eta(i)<=etaneg)
351 {
352 etaneg=eta(i);
353 ineg=i;
354 }
355 }
356
357 //Decide si interpola o extrapola
358 if (ineg !=999)
359 {
360 //Interpola
361 //entontrar ipos=No del S-L con prodr positivo mas cercano a ineg pero con S-L mas pequeo
362 ipos = 0;
363 for( unsigned int i=0; i<=ils+1; i++)
364 {
365 if (prodr(i)>=0 && eta(i)<=eta(ineg) && eta(i)>=eta(ipos))
366 {
367 ipos=i;
368 }
369 }
370 //interpola para encontrar S-L etaint
371 etaint=prodr(ineg)*eta(ipos)-prodr(ipos)*eta(ineg);
372 if ((prodr(ineg)-prodr(ipos))==0)
373 {
374 ico=2;
375 return;
376 }
377 else
378 {
379 etaint=etaint/(prodr(ineg)-prodr(ipos));
380 }
381 //alternativamente encuentra etaalt para asegurar un cambio razonable
382 etaalt = eta(ipos) + 0.2*(eta(ineg)-eta(ipos));
383 //toma el maximo entre etaint y etaalt
384 etaint=std::max(etaint,etaalt);
385 //la longitud de paso minimo
386 if (etaint<metmna)
387 {
388 etaint=metmna;
389 if (ico==1)
390 {
391 ico=2;
392 }
393 else if (ico==0)
394 {
395 ico=1;
396 }
397 }
398
399 eta(ils+2)=etaint;
400 }
401
402 else if (ineg==999)
403 {
404
405 //extrapola
406 //asigna temporalmente la longitud de paso maxima
407 etmxt = mamp*etmaxp;
408 if(etmxt>metmxa)
409 {
410 etmxt=metmxa;
411 }
412 //extrapola la longitud de paso entre el actual y el anterior
413 etaext=prodr(ils+1)*eta(ils)-prodr(ils)*eta(ils+1);
414 if ((prodr(ils+1)-prodr(ils))==0)
415 {
416 ico=2;
417 return;
418 }
419 else
420 {
421 etaext=etaext/(prodr(ils+1)-prodr(ils));
422 }
423
424 eta(ils+2)=etaext;
425 //se acepta eta si esta dentro de los limites
426 if (etaext<=0 || etaext> etmxt)
427 {
428 eta(ils+2) = etmxt;
429 }
430 if (eta(ils+2)==metmxa && ico==1)
431 {
432 ico=2;
433 return;
434 }
435 if (eta(ils+2)==metmxa)
436 {
437 ico=1;
438 }
439 }
440 KRATOS_CATCH("")
441 }
