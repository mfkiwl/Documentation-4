69 static inline void CalculateGeometryData(
70 const GeometryType& rGeometry,
71 BoundedMatrix<double,4,3>& rDN_DX,
72 array_1d<double,4>& rN,
73 double& rVolume
74 )
75 {
76 const double x10 = rGeometry[1].X() - rGeometry[0].X();
77 const double y10 = rGeometry[1].Y() - rGeometry[0].Y();
78 const double z10 = rGeometry[1].Z() - rGeometry[0].Z();
79
80 const double x20 = rGeometry[2].X() - rGeometry[0].X();
81 const double y20 = rGeometry[2].Y() - rGeometry[0].Y();
82 const double z20 = rGeometry[2].Z() - rGeometry[0].Z();
83
84 const double x30 = rGeometry[3].X() - rGeometry[0].X();
85 const double y30 = rGeometry[3].Y() - rGeometry[0].Y();
86 const double z30 = rGeometry[3].Z() - rGeometry[0].Z();
87
88 const double detJ = x10 * y20 * z30 - x10 * y30 * z20 + y10 * z20 * x30 - y10 * x20 * z30 + z10 * x20 * y30 - z10 * y20 * x30;
89
90 rDN_DX(0,0) = -y20 * z30 + y30 * z20 + y10 * z30 - z10 * y30 - y10 * z20 + z10 * y20;
91 rDN_DX(0,1) = -z20 * x30 + x20 * z30 - x10 * z30 + z10 * x30 + x10 * z20 - z10 * x20;
92 rDN_DX(0,2) = -x20 * y30 + y20 * x30 + x10 * y30 - y10 * x30 - x10 * y20 + y10 * x20;
93 rDN_DX(1,0) = y20 * z30 - y30 * z20;
94 rDN_DX(1,1) = z20 * x30 - x20 * z30;
95 rDN_DX(1,2) = x20 * y30 - y20 * x30;
96 rDN_DX(2,0) = -y10 * z30 + z10 * y30;
97 rDN_DX(2,1) = x10 * z30 - z10 * x30;
98 rDN_DX(2,2) = -x10 * y30 + y10 * x30;
99 rDN_DX(3,0) = y10 * z20 - z10 * y20;
100 rDN_DX(3,1) = -x10 * z20 + z10 * x20;
101 rDN_DX(3,2) = x10 * y20 - y10 * x20;
102
103 rDN_DX /= detJ;
104
105 rN[0] = 0.25;
106 rN[1] = 0.25;
107 rN[2] = 0.25;
108 rN[3] = 0.25;
109
110 rVolume = detJ*0.1666666666666666666667;
111 }

144 static inline void CalculateGeometryData(
145 const GeometryType& rGeometry,
146 BoundedMatrix<double,3,2>& DN_DX,
147 array_1d<double,3>& N,
148 double& rArea
149 )
150 {
151 const double x10 = rGeometry[1].X() - rGeometry[0].X();
152 const double y10 = rGeometry[1].Y() - rGeometry[0].Y();
153
154 const double x20 = rGeometry[2].X() - rGeometry[0].X();
155 const double y20 = rGeometry[2].Y() - rGeometry[0].Y();
156
157 //Jacobian is calculated:
158 // |dx/dxi dx/deta| |x1-x0 x2-x0|
159 //J=| |= | |
160 // |dy/dxi dy/deta| |y1-y0 y2-y0|
161
162
163 double detJ = x10 * y20-y10 * x20;
164
165 DN_DX(0,0) = -y20 + y10;
166 DN_DX(0,1) = x20 - x10;
167 DN_DX(1,0) = y20 ;
168 DN_DX(1,1) = -x20 ;
169 DN_DX(2,0) = -y10 ;
170 DN_DX(2,1) = x10 ;
171
172 DN_DX /= detJ;
173 N[0] = 0.333333333333333;
174 N[1] = 0.333333333333333;
175 N[2] = 0.333333333333333;
176
177 rArea = 0.5*detJ;
178 }

206 const GeometryType& rGeometry,
207 double& hmin,
208 double& hmax
209 )
210 {
211 const double x10 = rGeometry[1].X() - rGeometry[0].X();
212 const double y10 = rGeometry[1].Y() - rGeometry[0].Y();
213
214 const double x20 = rGeometry[2].X() - rGeometry[0].X();
215 const double y20 = rGeometry[2].Y() - rGeometry[0].Y();
216
217 double l = std::pow(x20, 2) + std::pow(y20, 2);
218 hmax = l;
219 hmin = l;
220
221 if(l>hmax) hmax = l;
222 else if(l<hmin) hmin = l;
223
224 l = (x20-x10)*(x20-x10) + (y20-y10)*(y20-y10);
225 if(l>hmax) hmax = l;
226 else if(l<hmin) hmin = l;
227
228 hmax = std::sqrt(hmax);
229 hmin = std::sqrt(hmin);
230 }

265 template<std::size_t TSize>
266 static void CalculateTetrahedraDistances(
267 const GeometryType& rGeometry, array_1d<double, TSize>& rDistances)
268 {
269 // Calculating the intersection points
270 array_1d<Point, 4> intersection_points;
271 int number_of_intersection_points = CalculateTetrahedraIntersectionPoints(rGeometry, rDistances, intersection_points);
272
273 if(number_of_intersection_points == 0) {
274 KRATOS_WARNING("CalculateTetrahedraDistances") << "WARNING:: The intersection with interface hasn't found!" << std::endl << "The distances are: " << rDistances << std::endl;
275 } else if(number_of_intersection_points == 1) {
276 // There is one point with zero distance. The distance of the nodes are their distance to this point
277 array_1d<double,3> temp;
278 // Loop over nodes to calculate their distance to the zero distance node.
279 for(unsigned int i_node = 0; i_node < rGeometry.size() ; ++i_node) {
280 noalias(temp) = intersection_points[0] - rGeometry[i_node];
281 rDistances[i_node] = norm_2(temp);
282 }
283 } else if(number_of_intersection_points == 2) {
284 // Loop over nodes to calculate their distance to the zero distance line.
285 for(unsigned int i_node = 0; i_node < rGeometry.size() ; ++i_node) {
286 rDistances[i_node] = PointDistanceToLineSegment3D(intersection_points[0], intersection_points[1], rGeometry[i_node]);
287 }
288 } else if(number_of_intersection_points == 3) {
289 // Loop over nodes to calculate their distance to the zero distance triangle.
290 for(unsigned int i_node = 0; i_node < rGeometry.size() ; ++i_node) {
291 rDistances[i_node] = PointDistanceToTriangle3D(intersection_points[0], intersection_points[1], intersection_points[2], rGeometry[i_node]);
292// rDistances[i_node] = std::abs(rGeometry[i_node].Z()); // TODO: To be removed. Pooyan.
293 }
294
295 } else if(number_of_intersection_points == 4) {
296 // Loop over nodes to calculate their distance to the each zero distance triangle.
297 for(unsigned int i_node = 0; i_node < rGeometry.size() ; ++i_node) {
298 // Here I'm taking in account the order of edges where I'm looking for intersection
299 double d1 = PointDistanceToTriangle3D(intersection_points[0], intersection_points[1], intersection_points[3], rGeometry[i_node]);
300 double d2 = PointDistanceToTriangle3D(intersection_points[0], intersection_points[3], intersection_points[2], rGeometry[i_node]);
301
302 rDistances[i_node] = (d1 > d2) ? d2 : d1;
303 }
304 }

347 template<std::size_t TSize>
348 static void CalculateExactDistancesToPlane(
349 const GeometryType& rThisGeometry,
350 array_1d<double, TSize>& rDistances
351 )
352 {
353 array_1d<Point, TSize> intersection_points;
354 int number_of_intersection_points = CalculateTetrahedraIntersectionPoints(rThisGeometry, rDistances, intersection_points);
355
356 if(number_of_intersection_points == 0) {
357 KRATOS_WARNING("GeometryUtilities") << "Warning: The intersection with interface hasn't found! The distances are" << rDistances << std::endl;
358 } else {
359 BoundedMatrix<double,TSize,TSize-1> DN_DX;
360 array_1d<double, TSize> N;
361 double volume;
362 GeometryUtils::CalculateGeometryData(rThisGeometry, DN_DX, N, volume);
363 array_1d<double, TSize-1> distance_gradient = prod(trans(DN_DX), rDistances);
364 double distance_gradient_norm = norm_2(distance_gradient);
365 if (distance_gradient_norm < 1e-15) distance_gradient_norm = 1e-15; //avoid division by zero
366 distance_gradient /= distance_gradient_norm;
367 // We use the first intersection point as reference
368 const auto &ref_point = intersection_points[0].Coordinates();
369 for (unsigned int i = 0; i < TSize; i++) {
370 double d = 0.0;
371 const auto &i_coords = rThisGeometry[i].Coordinates();
372 for (unsigned int Dim = 0; Dim < TSize - 1; Dim++) {
373 d += (i_coords[Dim] - ref_point[Dim]) * distance_gradient[Dim];
374 }
375 d = std::abs(d);
376 rDistances[i] = std::min(std::abs(rDistances[i]), d);
377 }
378 }
379 }

388 template<std::size_t TSize1, std::size_t TSize2>
389 static int CalculateTetrahedraIntersectionPoints(
390 const GeometryType& rGeometry,
391 array_1d<double, TSize1>& rDistances,
392 array_1d<Point, TSize2>& rIntersectionPoints
393 )
394 {
395 const double epsilon = 1e-15; //1.00e-9;
396
397 int number_of_intersection_points = 0;
398 for(unsigned int i = 0 ; i < TSize1 ; i++) {
399 if(std::abs(rDistances[i]) < epsilon) {
400 noalias(rIntersectionPoints[number_of_intersection_points].Coordinates()) = rGeometry[i].Coordinates();
401
402 number_of_intersection_points++;
403 continue;
404 }
405 for(unsigned int j = i + 1 ; j < TSize1 ; j++) {
406 if(std::abs(rDistances[j]) < epsilon)
407 continue; // we will add it to the intersections by the i index to be unique
408
409 if(rDistances[i] * rDistances[j] < 0.00) { // The interface passes through the edge
410 const double delta_d = std::abs(rDistances[i]) + std::abs(rDistances[j]); // we know that both distances are greater than epsilon.
411
412 const double di = std::abs(rDistances[i]) / delta_d;
413 const double dj = std::abs(rDistances[j]) / delta_d;
414
415 noalias(rIntersectionPoints[number_of_intersection_points].Coordinates()) = dj * rGeometry[i].Coordinates();
416 noalias(rIntersectionPoints[number_of_intersection_points].Coordinates()) += di * rGeometry[j].Coordinates();
417
418 number_of_intersection_points++;
419 }
420 }
421 }
422
423 return number_of_intersection_points;
424 }

433 static double PointDistanceToLineSegment3D(
434 Point const& rLinePoint1,
435 Point const& rLinePoint2,
436 Point const& rToPoint
437 )
438 {
439 const double epsilon = 1e-15; //1.00e-9;
440
441 const array_1d<double,3> v1 = rLinePoint2 - rLinePoint1;
442 const array_1d<double,3> v2 = rLinePoint1 - rToPoint;
443 array_1d<double,3> v3;
444
445 const double square_distance = inner_prod(v1,v1);
446
447 if(square_distance < epsilon) // near zero length line
448 return norm_2(v2); // we return the distance to the first point of line
449
450 const double t = - inner_prod(v1,v2) / square_distance;
451
452 if(t < 0.0) { // it is before point 1
453 // We return the distance to point 1
454 noalias(v3) = rLinePoint1 - rToPoint;
455
456 return norm_2(v3);
457 }
458
459 if(t > 1.00) { // it is after point 2
460 // We return the distance to point 2
461 noalias(v3) = rLinePoint2 - rToPoint;
462
463 return norm_2(v3);
464 }
465
466 // The projection point is between point 1 and 2 of the line segment
467 noalias(v3) = rLinePoint1 * (1.0 - t) + rLinePoint2 * t;
468
469 return norm_2(v3 - rToPoint);
470
471 }

483 static double PointDistanceToTriangle3D(
484 Point const& TrianglePoint1,
485 Point const& TrianglePoint2,
486 Point const& TrianglePoint3,
487 Point const& ToPoint
488 )
489 {
490 const array_1d<double, 3> e0 = TrianglePoint2 - TrianglePoint1;
491 const array_1d<double, 3> e1 = TrianglePoint3 - TrianglePoint1;
492 const array_1d<double, 3> dd = TrianglePoint1 - ToPoint;
493
494 const double a = inner_prod(e0, e0);
495 const double b = inner_prod(e0, e1);
496 const double c = inner_prod(e1, e1);
497 const double d = inner_prod(e0, dd);
498 const double e = inner_prod(e1, dd);
499 const double f = inner_prod(dd, dd);
500
501 const double det = a*c-b*b;
502 double s = b*e-c*d;
503 double t = b*d-a*e;
504
505 double square_distance = 0.00;
506
507 if ( s + t <= det ) {
508 if ( s < 0.0 ) {
509 if ( t < 0.0 ) { // region 4
510 if (d < 0) {
511 t = 0;
512 if (-d >= a) {
513 s = 1;
514 square_distance = a + 2*d + f;
515 } else {
516 s = -d/a;
517 square_distance = d*s + f;
518 }
519 } else {
520 s = 0;
521 if (e >= 0) {
522 t = 0;
523 square_distance = f;
524 } else {
525 if (-e >= c) {
526 t = 1;
527 square_distance = c + 2*e + f;
528 } else {
529 t = -e/c;
530 square_distance = e*t + f;
531 }
532 }
533 }
534 } else { // region 3
535 s = 0.0;
536 if(e >= 0.0) {
537 t = 0.0;
538 square_distance = f;
539 } else {
540 if (-e >= c) {
541 t = 1.00;
542 square_distance = c + 2*e +f;
543 } else {
544 t = -e/c;
545 square_distance = e*t + f;
546 }
547 }
548
549 }
550 } else if ( t < 0.00 ) { // region 5
551 t = 0;
552 if (d >= 0) {
553 s = 0;
554 square_distance = f;
555 } else {
556 if (-d >= a) {
557 s = 1;
558 square_distance = a + 2.0 * d + f;
559 } else {
560 s = -d / a;
561 square_distance = d * s + f;
562 }
563 }
564 } else { // region 0
565 double inv_det = 1.0 / det;
566 s *= inv_det;
567 t *= inv_det;
568 square_distance = s*(a*s + b*t + 2*d) + t*(b*s + c*t + 2*e) + f;
569 }
570 } else {
571 if ( s < 0.00 ) {
572 // Region 2
573 const double temp0 = b + d;
574 const double temp1 = c + e;
575 if (temp1 > temp0) { // Minimum on edge s+t=1
576 const double numer = temp1 - temp0;
577 const double denom = a - 2*b + c;
578 if(numer >= denom) {
579 s = 1.0;
580 t = 0.0;
581 square_distance = a + 2*d + f;
582 } else {
583 s = numer/denom;
584 t = 1.0 - s;
585 square_distance = s*(a*s + b*t + 2*d) + t*(b*s + c*t + 2*e) + f;
586 }
587 } else { // Minimum on edge s=0
588 s = 0.0;
589 if(temp1 <= 0.0) {
590 t = 1;
591 square_distance = c + 2*e + f;
592 } else {
593 if(e >= 0.0) {
594 t = 0.0;
595 square_distance = f;
596 } else {
597 t = -e/c;
598 square_distance = e*t + f;
599 }
600 }
601 }
602 } else if ( t < 0.0 ) {
603 // Region 6
604 double temp0 = b + e;
605 double temp1 = a + d;
606 if (temp1 > temp0) {
607 double numer = temp1 - temp0;
608 double denom = a - 2*b + c;
609 if(numer >= denom) {
610 s = 0.0;
611 t = 1.0;
612 square_distance = c + 2*e + f;
613 } else {
614 t = numer/denom;
615 s = 1.0 - t;
616 square_distance = s*(a*s + b*t + 2*d) + t*(b*s + c*t + 2*e) + f;
617 }
618 } else {
619 t = 0.0;
620 if(temp1 <= 0.0) {
621 s = 1;
622 square_distance = a + 2*d + f;
623 } else {
624 if(d >= 0.0) {
625 s = 0.0;
626 square_distance = f;
627 } else {
628 s = -d/a;
629 square_distance = d*s + f;
630 }
631 }
632 }
633 } else {
634 // Region 1
635 double numer = c + e - b - d;
636
637 if (numer <= 0.0) {
638 s = 0.0;
639 t = 1.0;
640 square_distance = c + 2.0 * e + f;
641 } else {
642 double denom = a - 2.0 * b + c;
643 if (numer >= denom) {
644 s = 1.0;
645 t = 0.0;
646 square_distance = a + 2.0 * d + f;
647 } else {
648 s = numer / denom;
649 t = 1.0 - s;
650 square_distance = s*(a*s + b*t + 2*d) + t*(b*s + c*t + 2*e) + f;
651 }
652 }
653 }
654 }
655
656 if(square_distance < 0.0)
657 return 0.0; // avoiding -0 case!!
658
659 return std::sqrt(square_distance);
660 }
