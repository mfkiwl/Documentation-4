293  }

347  template<std::size_t TSize>
348  static void CalculateExactDistancesToPlane(
349  const GeometryType& rThisGeometry,
350  array_1d<double, TSize>& rDistances
351  )
352  {
353  array_1d<Point, TSize> intersection_points;
354  int number_of_intersection_points = CalculateTetrahedraIntersectionPoints(rThisGeometry, rDistances, intersection_points);
355 
356  if(number_of_intersection_points == 0) {
357  KRATOS_WARNING("GeometryUtilities") << "Warning: The intersection with interface hasn't found! The distances are" << rDistances << std::endl;
358  } else {
359  BoundedMatrix<double,TSize,TSize-1> DN_DX;
360  array_1d<double, TSize> N;
361  double volume;
362  GeometryUtils::CalculateGeometryData(rThisGeometry, DN_DX, N, volume);
363  array_1d<double, TSize-1> distance_gradient = prod(trans(DN_DX), rDistances);
364  double distance_gradient_norm = norm_2(distance_gradient);
365  if (distance_gradient_norm < 1e-15) distance_gradient_norm = 1e-15; //avoid division by zero
366  distance_gradient /= distance_gradient_norm;
367  // We use the first intersection point as reference
368  const auto &ref_point = intersection_points[0].Coordinates();
369  for (unsigned int i = 0; i < TSize; i++) {
370  double d = 0.0;
371  const auto &i_coords = rThisGeometry[i].Coordinates();
372  for (unsigned int Dim = 0; Dim < TSize - 1; Dim++) {
373  d += (i_coords[Dim] - ref_point[Dim]) * distance_gradient[Dim];
374  }
375  d = std::abs(d);
376  rDistances[i] = std::min(std::abs(rDistances[i]), d);
377  }
378  }
379  }

388  template<std::size_t TSize1, std::size_t TSize2>
389  static int CalculateTetrahedraIntersectionPoints(
390  const GeometryType& rGeometry,
391  array_1d<double, TSize1>& rDistances,
392  array_1d<Point, TSize2>& rIntersectionPoints
393  )
394  {
395  const double epsilon = 1e-15; //1.00e-9;
396 
397  int number_of_intersection_points = 0;
398  for(unsigned int i = 0 ; i < TSize1 ; i++) {
399  if(std::abs(rDistances[i]) < epsilon) {
400  noalias(rIntersectionPoints[number_of_intersection_points].Coordinates()) = rGeometry[i].Coordinates();
401 
402  number_of_intersection_points++;
403  continue;
404  }
405  for(unsigned int j = i + 1 ; j < TSize1 ; j++) {
406  if(std::abs(rDistances[j]) < epsilon)
407  continue; // we will add it to the intersections by the i index to be unique
408 
409  if(rDistances[i] * rDistances[j] < 0.00) { // The interface passes through the edge
410  const double delta_d = std::abs(rDistances[i]) + std::abs(rDistances[j]); // we know that both distances are greater than epsilon.
411 
412  const double di = std::abs(rDistances[i]) / delta_d;
413  const double dj = std::abs(rDistances[j]) / delta_d;
414 
415  noalias(rIntersectionPoints[number_of_intersection_points].Coordinates()) = dj * rGeometry[i].Coordinates();
416  noalias(rIntersectionPoints[number_of_intersection_points].Coordinates()) += di * rGeometry[j].Coordinates();
417 
418  number_of_intersection_points++;
419  }
420  }
421  }
422 
423  return number_of_intersection_points;
424  }

433  static double PointDistanceToLineSegment3D(
434  Point const& rLinePoint1,
435  Point const& rLinePoint2,
436  Point const& rToPoint
437  )
438  {
439  const double epsilon = 1e-15; //1.00e-9;
440 
441  const array_1d<double,3> v1 = rLinePoint2 - rLinePoint1;
442  const array_1d<double,3> v2 = rLinePoint1 - rToPoint;
443  array_1d<double,3> v3;
444 
445  const double square_distance = inner_prod(v1,v1);
446 
447  if(square_distance < epsilon) // near zero length line
448  return norm_2(v2); // we return the distance to the first point of line
449 
450  const double t = - inner_prod(v1,v2) / square_distance;
451 
452  if(t < 0.0) { // it is before point 1
453  // We return the distance to point 1
454  noalias(v3) = rLinePoint1 - rToPoint;
455 
456  return norm_2(v3);
457  }
458 
459  if(t > 1.00) { // it is after point 2
460  // We return the distance to point 2
461  noalias(v3) = rLinePoint2 - rToPoint;
462 
463  return norm_2(v3);
464  }
465 
466  // The projection point is between point 1 and 2 of the line segment
467  noalias(v3) = rLinePoint1 * (1.0 - t) + rLinePoint2 * t;
468 
469  return norm_2(v3 - rToPoint);
470 
471  }

483  static double PointDistanceToTriangle3D(
484  Point const& TrianglePoint1,
485  Point const& TrianglePoint2,
486  Point const& TrianglePoint3,
487  Point const& ToPoint
488  )
489  {
490  const array_1d<double, 3> e0 = TrianglePoint2 - TrianglePoint1;
491  const array_1d<double, 3> e1 = TrianglePoint3 - TrianglePoint1;
492  const array_1d<double, 3> dd = TrianglePoint1 - ToPoint;
493 
494  const double a = inner_prod(e0, e0);
495  const double b = inner_prod(e0, e1);
496  const double c = inner_prod(e1, e1);
497  const double d = inner_prod(e0, dd);
498  const double e = inner_prod(e1, dd);
499  const double f = inner_prod(dd, dd);
500 
501  const double det = a*c-b*b;
502  double s = b*e-c*d;
503  double t = b*d-a*e;
504 
505  double square_distance = 0.00;
506 
507  if ( s + t <= det ) {
508  if ( s < 0.0 ) {
509  if ( t < 0.0 ) { // region 4
510  if (d < 0) {
511  t = 0;
512  if (-d >= a) {
513  s = 1;
514  square_distance = a + 2*d + f;
515  } else {
516  s = -d/a;
517  square_distance = d*s + f;
518  }
519  } else {
520  s = 0;
521  if (e >= 0) {
522  t = 0;
523  square_distance = f;
524  } else {
525  if (-e >= c) {
526  t = 1;
527  square_distance = c + 2*e + f;
528  } else {
529  t = -e/c;
530  square_distance = e*t + f;
531  }
532  }
533  }
534  } else { // region 3
535  s = 0.0;
536  if(e >= 0.0) {
537  t = 0.0;
538  square_distance = f;
539  } else {
540  if (-e >= c) {
541  t = 1.00;
542  square_distance = c + 2*e +f;
543  } else {
544  t = -e/c;
545  square_distance = e*t + f;
546  }
547  }
548 
549  }
550  } else if ( t < 0.00 ) { // region 5
551  t = 0;
552  if (d >= 0) {
553  s = 0;
554  square_distance = f;
555  } else {
556  if (-d >= a) {
557  s = 1;
558  square_distance = a + 2.0 * d + f;
559  } else {
560  s = -d / a;
561  square_distance = d * s + f;
562  }
563  }
564  } else { // region 0
565  double inv_det = 1.0 / det;
566  s *= inv_det;
567  t *= inv_det;
568  square_distance = s*(a*s + b*t + 2*d) + t*(b*s + c*t + 2*e) + f;
569  }
570  } else {
571  if ( s < 0.00 ) {
572  // Region 2
573  const double temp0 = b + d;
574  const double temp1 = c + e;
575  if (temp1 > temp0) { // Minimum on edge s+t=1
576  const double numer = temp1 - temp0;
577  const double denom = a - 2*b + c;
578  if(numer >= denom) {
579  s = 1.0;
580  t = 0.0;
581  square_distance = a + 2*d + f;
582  } else {
583  s = numer/denom;
584  t = 1.0 - s;
585  square_distance = s*(a*s + b*t + 2*d) + t*(b*s + c*t + 2*e) + f;
586  }
587  } else { // Minimum on edge s=0
588  s = 0.0;
589  if(temp1 <= 0.0) {
590  t = 1;
591  square_distance = c + 2*e + f;
592  } else {
593  if(e >= 0.0) {
594  t = 0.0;
595  square_distance = f;
596  } else {
597  t = -e/c;
598  square_distance = e*t + f;
599  }
600  }
601  }
602  } else if ( t < 0.0 ) {
603  // Region 6
604  double temp0 = b + e;
605  double temp1 = a + d;
606  if (temp1 > temp0) {
607  double numer = temp1 - temp0;
608  double denom = a - 2*b + c;
609  if(numer >= denom) {
610  s = 0.0;
611  t = 1.0;
612  square_distance = c + 2*e + f;
613  } else {
614  t = numer/denom;
615  s = 1.0 - t;
616  square_distance = s*(a*s + b*t + 2*d) + t*(b*s + c*t + 2*e) + f;
617  }
618  } else {
619  t = 0.0;
620  if(temp1 <= 0.0) {
621  s = 1;
622  square_distance = a + 2*d + f;
623  } else {
624  if(d >= 0.0) {
625  s = 0.0;
626  square_distance = f;
627  } else {
628  s = -d/a;
629  square_distance = d*s + f;
630  }
631  }
632  }
633  } else {
634  // Region 1
635  double numer = c + e - b - d;
636 
637  if (numer <= 0.0) {
638  s = 0.0;
639  t = 1.0;
640  square_distance = c + 2.0 * e + f;
641  } else {

643  if (numer >= denom) {
