97 void FinalizeSolutionStep() override
98 {
99 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "Max Piping Iterations: " << mPipingIterations << std::endl;
100
101 bool grow = true;
102
103 // get piping elements
104 std::vector<Element*> PipeElements = GetPipingElements();
105 unsigned int n_el = PipeElements.size(); // number of piping elements
106
107 // get initially open pipe elements
108 unsigned int openPipeElements = this->InitialiseNumActivePipeElements(PipeElements);
109
110 if (PipeElements.size()==0)
111 {
112 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "No Pipe Elements -> Finalizing Solution " << std::endl;
113 GeoMechanicsNewtonRaphsonStrategy<TSparseSpace, TDenseSpace, TLinearSolver>::FinalizeSolutionStep();
114 return;
115 }
116 // calculate max pipe height and pipe increment
117 double amax = CalculateMaxPipeHeight(PipeElements);
118
119 // continue this loop, while the pipe is growing in length
120 while (grow && (openPipeElements < n_el))
121 {
122 // todo: JDN (20220817) : grow not used.
123 // bool Equilibrium = false;
124 bool converged = true;
125
126 // get tip element and activate
127 Element* tip_element = PipeElements.at(openPipeElements);
128 openPipeElements += 1;
129 tip_element->SetValue(PIPE_ACTIVE, true);
130
131 // Get all open pipe elements
132 std::function<bool(Element*)> filter = [](Element* i) {return i->Has(PIPE_ACTIVE) && i->GetValue(PIPE_ACTIVE); };
133 filtered_elements OpenPipeElements = PipeElements | boost::adaptors::filtered(filter);
134 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "Number of Open Pipe Elements: " << boost::size(OpenPipeElements) << std::endl;
135
136 // non-lin picard iteration, for deepening the pipe
137 //Todo JDN (20220817):: Deal with Equilibrium redundancy
138 //Equilibrium = check_pipe_equilibrium(OpenPipeElements, amax, mPipingIterations);
139 check_pipe_equilibrium(OpenPipeElements, amax, mPipingIterations);
140
141 // check if pipe should grow in length
142 std::tie(grow, openPipeElements) = check_status_tip_element(openPipeElements, n_el, amax, PipeElements);
143
144 // if n open elements is lower than total pipe elements, save pipe height current growing iteration or reset to previous
145 // iteration in case the pipe should not grow.
146 if (openPipeElements < n_el)
147 {
148 save_or_reset_pipe_heights(OpenPipeElements, grow);
149 }
150 // recalculate ground water flow
151 converged = Recalculate();
152
153 // error check
154 KRATOS_ERROR_IF_NOT(converged) << "Groundwater flow calculation failed to converge." << std::endl;
155
156 }

180 std::vector<Element*> GetPipingElements() {
181 ModelPart& CurrentModelPart = this->GetModelPart();
182 std::vector<Element*> PipeElements;
183 double PipeElementStartX;
184
185 for (Element& element : CurrentModelPart.Elements())
186 {
187 if (element.GetProperties().Has(PIPE_START_ELEMENT))
188 {
189 PipeElements.push_back(&element);
190
191 long unsigned int startElement = element.GetProperties()[PIPE_START_ELEMENT];
192
193 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << element.Id() << " " << startElement << std::endl;
194
195 if (element.Id() == startElement)
196 {
197 PipeElementStartX = element.GetGeometry().GetPoint(0)[0];
198 }
199 }
200 }
201
202 if (PipeElements.size() == 0)
203 {
204 return PipeElements;
205 }
206
207 // Get Maximum X Value in Pipe
208 auto rightPipe = std::max_element(PipeElements.begin(), PipeElements.end(), [](const Element* a, const Element* b)
209 {
210 return a->GetGeometry().GetPoint(0)[0] < b->GetGeometry().GetPoint(0)[0];
211 });
212
213 // Get Minimum X Value in Pipe
214 auto leftPipe = std::min_element(PipeElements.begin(), PipeElements.end(), [](const Element* a, const Element* b)
215 {
216 return a->GetGeometry().GetPoint(0)[0] < b->GetGeometry().GetPoint(0)[0];
217 });
218
219 double minX = leftPipe[0]->GetGeometry().GetPoint(0)[0];
220 double maxX = rightPipe[0]->GetGeometry().GetPoint(0)[0];
221
222 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << minX << " " << maxX << " " << PipeElementStartX << std::endl;
223
224 if ((minX != PipeElementStartX) && (maxX != PipeElementStartX))
225 {
226 KRATOS_ERROR << "Unable to determine pipe direction (multiple directions possible) - Check PIPE_START_ELEMENT" << std::endl;
227 }
228
229 if (minX == PipeElementStartX)
230 {
231 // Pipe Left -> Right
232 sort(PipeElements.begin(), PipeElements.end(), [](const Element* lhs, const Element* rhs) {
233 return lhs->GetGeometry().GetPoint(0)[0] < rhs->GetGeometry().GetPoint(0)[0];
234 });
235 }
236 else
237 {
238 // Pipe Right -> Left
239 sort(PipeElements.begin(), PipeElements.end(), [](const Element* lhs, const Element* rhs) {
240 return lhs->GetGeometry().GetPoint(0)[0] > rhs->GetGeometry().GetPoint(0)[0];
241 });
242 }
243
244 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "Number of Pipe Elements: " << PipeElements.size() << std::endl;
245 for (const Element* pipeElement : PipeElements)
246 {
247 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "PipeElementIDs (in order): " << pipeElement->Id() << std::endl;
248 }
249
250 return PipeElements;
251 }

293 double CalculateParticleDiameter(const PropertiesType& Prop)
294 {
295 double diameter;
296
297 if (Prop[PIPE_MODIFIED_D])
298 diameter = 2.08e-4 * pow((Prop[PIPE_D_70] / 2.08e-4), 0.4);

309 double CalculateMaxPipeHeight(std::vector<Element*> pipe_elements)
310 {
311 double max_diameter = 0;
312 double height_factor = 100;
313
314 // loop over all elements
315 for (Element* pipe_element : pipe_elements)
316 {
317 // calculate pipe particle diameter of pipe element
318 PropertiesType prop = pipe_element->GetProperties();
319 double particle_diameter = this->CalculateParticleDiameter(prop);
320
321 // get maximum pipe particle diameter of all pipe elements
322 if (particle_diameter > max_diameter)
323 {
324 max_diameter = particle_diameter;
325 }
326 }
327
328 // max pipe height is maximum pipe particle diameter * a constant
329 return max_diameter * height_factor;
330 }

362 bool check_pipe_equilibrium(filtered_elements open_pipe_elements, double amax, unsigned int mPipingIterations)
363 {
364 bool equilibrium = false;
365 bool converged = true;
366 unsigned int PipeIter = 0;
367 // todo: JDN (20220817) : grow not used.
368 // bool grow = true;
369
370 // calculate max pipe height and pipe increment
371 double da = CalculatePipeHeightIncrement(amax, mPipingIterations);
372
373 while (PipeIter < mPipingIterations && !equilibrium && converged)
374 {
375 // set equilibirum on true
376 equilibrium = true;
377
378 // perform a flow calculation and stop growing if the calculation doesnt converge
379 converged = Recalculate();
380
381 // todo: JDN (20220817) : grow not used.
382 //if (!converged)
383 //{
384 // grow = false;
385 //}
386
387 if (converged)
388 {
389 // Update depth of open piping Elements
390 equilibrium = true;
391 for (auto OpenPipeElement : open_pipe_elements)
392 {
393 auto pElement = static_cast<SteadyStatePwPipingElement<2, 4>*>(OpenPipeElement);
394
395 // get open pipe element geometry and properties
396 auto& Geom = OpenPipeElement->GetGeometry();
397 auto& prop = OpenPipeElement->GetProperties();
398
399 // calculate equilibrium pipe height and get current pipe height
400 double eq_height = pElement->CalculateEquilibriumPipeHeight(prop, Geom, OpenPipeElement->GetValue(PIPE_ELEMENT_LENGTH));
401 double current_height = OpenPipeElement->GetValue(PIPE_HEIGHT);
402
403 // set erosion on true if current pipe height is greater than the equilibirum height
404 if (current_height > eq_height)
405 {
406 OpenPipeElement->SetValue(PIPE_EROSION, true);
407 }
408
409 // check this if statement, I dont understand the check for pipe erosion
410 if (((!OpenPipeElement->GetValue(PIPE_EROSION) || (current_height > eq_height)) && current_height < amax))
411 {
412 OpenPipeElement->SetValue(PIPE_HEIGHT, OpenPipeElement->GetValue(PIPE_HEIGHT) + da);
413 equilibrium = false;
414 }
415
416 // check if equilibrium height and current pipe heights are diverging, stop picard iterations if this is the case and set pipe height on zero
417 if (!OpenPipeElement->GetValue(PIPE_EROSION) && (PipeIter > 1) && ((eq_height - current_height) > OpenPipeElement->GetValue(DIFF_PIPE_HEIGHT)))
418 {
419 equilibrium = true;
420 OpenPipeElement->SetValue(PIPE_HEIGHT, small_pipe_height);
421 }
422
423 // calculate difference between equilibrium height and current pipe height
424 OpenPipeElement->SetValue(DIFF_PIPE_HEIGHT, eq_height - current_height);
425
426 }
427 // increment piping iteration number
428 PipeIter += 1;
429 }
430 }
431 return equilibrium;
432 }

442 std::tuple<bool, int> check_status_tip_element(unsigned int n_open_elements, unsigned int n_elements, double max_pipe_height, std::vector<Element*> PipeElements)
443 {
444 bool grow = true;
445 // check status of tip element, stop growing if pipe_height is zero or greater than maximum pipe height or if all elements are open
446 if (n_open_elements < n_elements)
447 {
448 Element* tip_element = PipeElements.at(n_open_elements - 1);
449 double pipe_height = tip_element->GetValue(PIPE_HEIGHT);
450
451 if ((pipe_height > max_pipe_height + std::numeric_limits<double>::epsilon()) || (pipe_height < pipe_height_accuracy))
452 {
453 // stable element found; pipe length does not increase during current time step
454 grow = false;
455 tip_element->SetValue(PIPE_EROSION, false);
456 tip_element->SetValue(PIPE_ACTIVE, false);
457 n_open_elements -= 1;
458
459 KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "Number of Open Pipe Elements: " << n_open_elements << std::endl;
460 }
461 }
462 else
463 {
464 grow = false;
465 }
466 return std::make_tuple(grow, n_open_elements);
467 }

486 }
