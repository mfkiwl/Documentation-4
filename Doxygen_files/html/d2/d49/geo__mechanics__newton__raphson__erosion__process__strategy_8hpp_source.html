180  std::vector<Element*> GetPipingElements() {
181  ModelPart& CurrentModelPart = this->GetModelPart();
182  std::vector<Element*> PipeElements;
183  double PipeElementStartX;
184 
185  for (Element& element : CurrentModelPart.Elements())
186  {
187  if (element.GetProperties().Has(PIPE_START_ELEMENT))
188  {
189  PipeElements.push_back(&element);
190 
191  long unsigned int startElement = element.GetProperties()[PIPE_START_ELEMENT];
192 
193  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << element.Id() << " " << startElement << std::endl;
194 
195  if (element.Id() == startElement)
196  {
197  PipeElementStartX = element.GetGeometry().GetPoint(0)[0];
198  }
199  }
200  }
201 
202  if (PipeElements.size() == 0)
203  {
204  return PipeElements;
205  }
206 
207  // Get Maximum X Value in Pipe
208  auto rightPipe = std::max_element(PipeElements.begin(), PipeElements.end(), [](const Element* a, const Element* b)
209  {
210  return a->GetGeometry().GetPoint(0)[0] < b->GetGeometry().GetPoint(0)[0];
211  });
212 
213  // Get Minimum X Value in Pipe
214  auto leftPipe = std::min_element(PipeElements.begin(), PipeElements.end(), [](const Element* a, const Element* b)
215  {
216  return a->GetGeometry().GetPoint(0)[0] < b->GetGeometry().GetPoint(0)[0];
217  });
218 
219  double minX = leftPipe[0]->GetGeometry().GetPoint(0)[0];
220  double maxX = rightPipe[0]->GetGeometry().GetPoint(0)[0];
221 
222  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << minX << " " << maxX << " " << PipeElementStartX << std::endl;
223 
224  if ((minX != PipeElementStartX) && (maxX != PipeElementStartX))
225  {
226  KRATOS_ERROR << "Unable to determine pipe direction (multiple directions possible) - Check PIPE_START_ELEMENT" << std::endl;
227  }
228 
229  if (minX == PipeElementStartX)
230  {
231  // Pipe Left -> Right
232  sort(PipeElements.begin(), PipeElements.end(), [](const Element* lhs, const Element* rhs) {
233  return lhs->GetGeometry().GetPoint(0)[0] < rhs->GetGeometry().GetPoint(0)[0];
234  });
235  }
236  else
237  {
238  // Pipe Right -> Left
239  sort(PipeElements.begin(), PipeElements.end(), [](const Element* lhs, const Element* rhs) {
240  return lhs->GetGeometry().GetPoint(0)[0] > rhs->GetGeometry().GetPoint(0)[0];
241  });
242  }
243 
244  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "Number of Pipe Elements: " << PipeElements.size() << std::endl;
245  for (const Element* pipeElement : PipeElements)
246  {
247  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "PipeElementIDs (in order): " << pipeElement->Id() << std::endl;
248  }
249 
250  return PipeElements;
251  }

398 
399  // calculate equilibrium pipe height and get current pipe height

401  double current_height = OpenPipeElement->GetValue(PIPE_HEIGHT);
402 
403  // set erosion on true if current pipe height is greater than the equilibirum height
404  if (current_height > eq_height)
405  {
406  OpenPipeElement->SetValue(PIPE_EROSION, true);
407  }
408 
409  // check this if statement, I dont understand the check for pipe erosion

412  OpenPipeElement->SetValue(PIPE_HEIGHT, OpenPipeElement->GetValue(PIPE_HEIGHT) + da);
413  equilibrium = false;
414  }
415 
416  // check if equilibrium height and current pipe heights are diverging, stop picard iterations if this is the case and set pipe height on zero
417  if (!OpenPipeElement->GetValue(PIPE_EROSION) && (PipeIter > 1) && ((eq_height - current_height) > OpenPipeElement->GetValue(DIFF_PIPE_HEIGHT)))
418  {

442  std::tuple<bool, int> check_status_tip_element(unsigned int n_open_elements, unsigned int n_elements, double max_pipe_height, std::vector<Element*> PipeElements)
443  {
444  bool grow = true;
445  // check status of tip element, stop growing if pipe_height is zero or greater than maximum pipe height or if all elements are open
446  if (n_open_elements < n_elements)
447  {
448  Element* tip_element = PipeElements.at(n_open_elements - 1);
449  double pipe_height = tip_element->GetValue(PIPE_HEIGHT);
450 
451  if ((pipe_height > max_pipe_height + std::numeric_limits<double>::epsilon()) || (pipe_height < pipe_height_accuracy))
452  {
453  // stable element found; pipe length does not increase during current time step
454  grow = false;
455  tip_element->SetValue(PIPE_EROSION, false);
456  tip_element->SetValue(PIPE_ACTIVE, false);
457  n_open_elements -= 1;
458 
459  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0) << "Number of Open Pipe Elements: " << n_open_elements << std::endl;
460  }
461  }

464  grow = false;
