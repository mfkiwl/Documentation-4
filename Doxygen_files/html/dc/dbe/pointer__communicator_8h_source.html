220 template< class TFunctorType >
221 void Update(
222 TFunctorType& rUserFunctor,
223 GlobalPointersUnorderedMap< TPointerDataType, typename ResultsProxy<TPointerDataType, TFunctorType >::TSendType >& rNonLocalData)
224 {
225 //sendrecv data
226 for(auto color : mColors)
227 {
228 if(color >= 0) //-1 would imply no communication
229 {
230 auto& gps_to_be_sent = mNonLocalPointers[color];
231
232 auto recv_global_pointers = mrDataCommunicator.SendRecv(gps_to_be_sent, color, color );
233
234 std::vector< typename ResultsProxy<TPointerDataType, TFunctorType >::TSendType > locally_gathered_data; //this is local but needs to be sent to the remote node
235 for(auto& gp : recv_global_pointers.GetContainer())
236 locally_gathered_data.push_back( rUserFunctor(gp) );
237
238 auto remote_data = mrDataCommunicator.SendRecv(locally_gathered_data, color, color );
239
240 for(unsigned int i=0; i<remote_data.size(); ++i)
241 rNonLocalData[gps_to_be_sent(i)] = remote_data[i];
242 }
243 }

312 template< class TIteratorType >
313 void AddPointers( TIteratorType begin, TIteratorType end)
314 {
315 if(mrDataCommunicator.IsDistributed())
316 {
317 const int current_rank = mrDataCommunicator.Rank();
318 for(auto it = begin; it != end; ++it)
319 {
320 auto& gp = *it;
321 if(gp.GetRank() != current_rank)
322 {
323 mNonLocalPointers[gp.GetRank()].push_back(gp);
324 }
325 }
326
327 //ensure that no repeated info is stored
328 for(auto& non_local : mNonLocalPointers)
329 non_local.second.Unique();
330 }
331 }

333 void ComputeCommunicationPlan()
334 {
335 std::vector<int> send_list;
336 send_list.reserve( mNonLocalPointers.size() );
337 for(auto& it : mNonLocalPointers)
338 send_list.push_back( it.first );
339
340 std::sort(send_list.begin(), send_list.end());
341 mColors = MPIColoringUtilities::ComputeCommunicationScheduling(send_list, mrDataCommunicator);
342 }
