116 double CalculateValue()
117 {
118 KRATOS_TRY;
119
120 const ProcessInfo &CurrentProcessInfo = mrModelPart.GetProcessInfo();
121 double strain_energy = 0.0;
122
123 // Sum all elemental strain energy values calculated as: W_e = u_e^T K_e u_e
124 for (auto& elem_i : mrModelPart.Elements())
125 {
126 const bool element_is_active = elem_i.IsDefined(ACTIVE) ? elem_i.Is(ACTIVE) : true;
127 if(element_is_active)
128 {
129 Matrix LHS;
130 Vector RHS;
131 Vector u;
132
133 // Get state solution relevant for energy calculation
134 const auto& rConstElemRef = elem_i;
135 rConstElemRef.GetValuesVector(u,0);
136
137 elem_i.CalculateLocalSystem(LHS,RHS,CurrentProcessInfo);
138
139 // Compute strain energy
140 strain_energy += 0.5 * inner_prod(u,prod(LHS,u));
141 }
142 }
143
144 return strain_energy;
145
146 KRATOS_CATCH("");
147 }

150 void CalculateGradient()
151 {
152 KRATOS_TRY;
153
154 // Formula computed in general notation:
155 // \frac{dF}{dx} = \frac{\partial F}{\partial x}
156 // - \lambda^T \frac{\partial R_S}{\partial x}
157
158 // Adjoint variable:
159 // \lambda = \frac{1}{2} u^T
160
161 // Correspondingly in specific notation for given response function
162 // \frac{dF}{dx} = \frac{1}{2} u^T \cdot \frac{\partial f_{ext}}{\partial x}
163 // + \frac{1}{2} u^T \cdot ( \frac{\partial f_{ext}}{\partial x} - \frac{\partial K}{\partial x} )
164
165 // First gradients are initialized
166 VariableUtils().SetHistoricalVariableToZero(SHAPE_SENSITIVITY, mrModelPart.Nodes());
167
168
169 // Gradient calculation
170 // 1st step: Calculate partial derivative of response function w.r.t. node coordinates
171 // 2nd step: Calculate adjoint field
172 // 3rd step: Calculate partial derivative of state equation w.r.t. node coordinates and multiply with adjoint field
173
174 // Semi analytic sensitivities
175 CalculateResponseDerivativePartByFiniteDifferencing();
176 CalculateAdjointField();
177 CalculateStateDerivativePartByFiniteDifferencing();
178
179 KRATOS_CATCH("");
180 }

245 // --------------------------------------------------------------------------
246 void CalculateStateDerivativePartByFiniteDifferencing()
247 {
248 KRATOS_TRY;
249
250 // Working variables
251 const ProcessInfo &CurrentProcessInfo = mrModelPart.GetProcessInfo();
252
253 // Computation of: \frac{1}{2} u^T \cdot ( - \frac{\partial K}{\partial x} )
254 for (auto& elem_i : mrModelPart.Elements())
255 {
256 const bool element_is_active = elem_i.IsDefined(ACTIVE) ? elem_i.Is(ACTIVE) : true;
257 if(element_is_active)
258 {
259 Vector u;
260 Vector lambda;
261 Vector RHS;
262
263 // Get state solution
264 const auto& rConstElemRef = elem_i;
265 rConstElemRef.GetValuesVector(u,0);
266
267 // Get adjoint variables (Corresponds to 1/2*u)
268 lambda = 0.5*u;
269
270 // Semi-analytic computation of partial derivative of state equation w.r.t. node coordinates
271 elem_i.CalculateRightHandSide(RHS, CurrentProcessInfo);
272 for (auto& node_i : elem_i.GetGeometry())
273 {
274 array_3d gradient_contribution(3, 0.0);
275 Vector derived_RHS = Vector(0);
276
277 // x-direction
278 FiniteDifferenceUtility::CalculateRightHandSideDerivative(elem_i, RHS, SHAPE_SENSITIVITY_X, node_i, mDelta, derived_RHS, CurrentProcessInfo);
279 gradient_contribution[0] = inner_prod(lambda, derived_RHS);
280
281 // y-direction
282 FiniteDifferenceUtility::CalculateRightHandSideDerivative(elem_i, RHS, SHAPE_SENSITIVITY_Y, node_i, mDelta, derived_RHS, CurrentProcessInfo);
283 gradient_contribution[1] = inner_prod(lambda, derived_RHS);
284
285 // z-direction
286 FiniteDifferenceUtility::CalculateRightHandSideDerivative(elem_i, RHS, SHAPE_SENSITIVITY_Z, node_i, mDelta, derived_RHS, CurrentProcessInfo);
287 gradient_contribution[2] = inner_prod(lambda, derived_RHS);
288
289 // Assemble sensitivity to node
290 noalias(node_i.FastGetSolutionStepValue(SHAPE_SENSITIVITY)) += gradient_contribution;
291 }
292 }
293 }
294
295 // Computation of \frac{1}{2} u^T \cdot ( \frac{\partial f_{ext}}{\partial x} )
296 for (auto& cond_i : mrModelPart.Conditions())
297 {
298 //detect if the condition is active or not. If the user did not make any choice the element
299 //is active by default
300 const bool condition_is_active = cond_i.IsDefined(ACTIVE) ? cond_i.Is(ACTIVE) : true;
301 if (condition_is_active)
302 {
303 Vector u;
304 Vector lambda;
305 Vector RHS;
306
307 // Get adjoint variables (Corresponds to 1/2*u)
308 const auto& rConstCondRef = cond_i;
309 rConstCondRef.GetValuesVector(u,0);
310 lambda = 0.5*u;
311
312 // Semi-analytic computation of partial derivative of force vector w.r.t. node coordinates
313 cond_i.CalculateRightHandSide(RHS, CurrentProcessInfo);
314 for (auto& node_i : cond_i.GetGeometry())
315 {
316 array_3d gradient_contribution(3, 0.0);
317 Vector perturbed_RHS = Vector(0);
318
319 // Pertubation, gradient analysis and recovery of x
320 node_i.X0() += mDelta;
321 cond_i.CalculateRightHandSide(perturbed_RHS, CurrentProcessInfo);
322 gradient_contribution[0] = inner_prod(lambda, (perturbed_RHS - RHS) / mDelta);
323 node_i.X0() -= mDelta;
324
325 // Reset pertubed vector
326 perturbed_RHS = Vector(0);
327
328 // Pertubation, gradient analysis and recovery of y
329 node_i.Y0() += mDelta;
330 cond_i.CalculateRightHandSide(perturbed_RHS, CurrentProcessInfo);
331 gradient_contribution[1] = inner_prod(lambda, (perturbed_RHS - RHS) / mDelta);
332 node_i.Y0() -= mDelta;
333
334 // Reset pertubed vector
335 perturbed_RHS = Vector(0);
336
337 // Pertubation, gradient analysis and recovery of z
338 node_i.Z0() += mDelta;
339 cond_i.CalculateRightHandSide(perturbed_RHS, CurrentProcessInfo);
340 gradient_contribution[2] = inner_prod(lambda, (perturbed_RHS - RHS) / mDelta);
341 node_i.Z0() -= mDelta;
342
343 // Assemble shape gradient to node
344 noalias(node_i.FastGetSolutionStepValue(SHAPE_SENSITIVITY)) += gradient_contribution;
345 }
346 }
347 }
348
349 KRATOS_CATCH("");

353 void CalculateResponseDerivativePartByFiniteDifferencing()
354 {
355 KRATOS_TRY;
356
357 // Working variables
358 const ProcessInfo &CurrentProcessInfo = mrModelPart.GetProcessInfo();
359
360 // Computation of \frac{1}{2} u^T \cdot ( \frac{\partial f_{ext}}{\partial x} )
361 for (auto& cond_i : mrModelPart.Conditions())
362 {
363 //detect if the condition is active or not. If the user did not make any choice the element
364 //is active by default
365 const bool condition_is_active = cond_i.IsDefined(ACTIVE) ? cond_i.Is(ACTIVE) : true;
366 if (condition_is_active)
367 {
368 Vector u;
369 Vector RHS;
370
371 // Get state solution
372 const auto& rConstCondRef = cond_i;
373 rConstCondRef.GetValuesVector(u,0);
374
375 // Perform finite differencing of RHS vector
376 cond_i.CalculateRightHandSide(RHS, CurrentProcessInfo);
377 for (auto& node_i : cond_i.GetGeometry())
378 {
379 array_3d gradient_contribution(3, 0.0);
380 Vector perturbed_RHS = Vector(0);
381
382 // Pertubation, gradient analysis and recovery of x
383 node_i.X0() += mDelta;
384 cond_i.CalculateRightHandSide(perturbed_RHS, CurrentProcessInfo);
385 gradient_contribution[0] = inner_prod(0.5*u, (perturbed_RHS - RHS) / mDelta);
386 node_i.X0() -= mDelta;
387
388 // Reset pertubed vector
389 perturbed_RHS = Vector(0);
390
391 // Pertubation, gradient analysis and recovery of y
392 node_i.Y0() += mDelta;
393 cond_i.CalculateRightHandSide(perturbed_RHS, CurrentProcessInfo);
394 gradient_contribution[1] = inner_prod(0.5*u, (perturbed_RHS - RHS) / mDelta);
395 node_i.Y0() -= mDelta;

402 cond_i.CalculateRightHandSide(perturbed_RHS, CurrentProcessInfo);
403 gradient_contribution[2] = inner_prod(0.5*u, (perturbed_RHS - RHS) / mDelta);
