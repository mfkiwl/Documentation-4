184 void Build(
185 typename TSchemeType::Pointer pScheme,
186 ModelPart& rModelPart,
187 TSystemMatrixType& rA,
188 TSystemVectorType& rb
189 ) override
190 {
191 KRATOS_TRY
192
193 KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
194
195 // Getting the elements from the model
196 ElementsArrayType& r_elements_array = rModelPart.Elements();
197
198 // Getting the array of the conditions
199 ConditionsArrayType& r_conditions_array = rModelPart.Conditions();
200
201 // Getting the elements from the model
202 const int nelements = static_cast<int>(r_elements_array.size());
203
204 // Getting the array of the conditions
205 const int nconditions = static_cast<int>(r_conditions_array.size());
206
207 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
208 const auto it_elem_begin = r_elements_array.begin();
209 const auto it_cond_begin = r_conditions_array.begin();
210
211 //contributions to the system
212 LocalSystemMatrixType LHS_Contribution = LocalSystemMatrixType(0, 0);
213 LocalSystemVectorType RHS_Contribution = LocalSystemVectorType(0);
214
215 // Vector containing the localization in the system of the different terms
216 EquationIdVectorType equation_id;
217
218 // Assemble all elements
219 const auto timer = BuiltinTimer();
220
221 #pragma omp parallel firstprivate(LHS_Contribution, RHS_Contribution, equation_id )
222 {
223 #pragma omp for schedule(guided, 512) nowait
224 for (int k = 0; k < nelements; ++k) {
225 auto it_elem = it_elem_begin + k;
226
227 // Detect if the element is active or not. If the user did not make any choice the element is active by default
228 bool element_is_active = true;
229 if (it_elem->IsDefined(ACTIVE))
230 element_is_active = it_elem->Is(ACTIVE);
231
232 if (element_is_active) {
233 // Calculate elemental contribution
234 pScheme->CalculateSystemContributions(*it_elem, LHS_Contribution, RHS_Contribution, equation_id, r_current_process_info);
235
236 // Assemble the elemental contribution
237#ifdef USE_LOCKS_IN_ASSEMBLY
238 Assemble(rA, rb, LHS_Contribution, RHS_Contribution, equation_id, mLockArray);
239#else
240 Assemble(rA, rb, LHS_Contribution, RHS_Contribution, equation_id);
241#endif
242 }
243 }
244
245 #pragma omp for schedule(guided, 512)
246 for (int k = 0; k < nconditions; ++k) {
247 auto it_cond = it_cond_begin + k;
248
249 // Detect if the element is active or not. If the user did not make any choice the element is active by default
250 bool condition_is_active = true;
251 if (it_cond->IsDefined(ACTIVE))
252 condition_is_active = it_cond->Is(ACTIVE);

254 if (condition_is_active) {

260#else
261 Assemble(rA, rb, LHS_Contribution, RHS_Contribution, equation_id);
262#endif
263 }
264 }
265 }
266 KRATOS_INFO_IF("ResidualBasedEliminationBuilderAndSolver", this->GetEchoLevel() >=1) << "System build time: " << timer.ElapsedSeconds() << std::endl;
267
268
269 KRATOS_INFO_IF("ResidualBasedEliminationBuilderAndSolver", this->GetEchoLevel() > 2) << "Finished building" << std::endl;

678 void SetUpDofSet(
679 typename TSchemeType::Pointer pScheme,
680 ModelPart& rModelPart
681 ) override
682 {
683 KRATOS_TRY;
684
685 KRATOS_INFO_IF("ResidualBasedEliminationBuilderAndSolver", this->GetEchoLevel() > 1 && rModelPart.GetCommunicator().MyPID() == 0) << "Setting up the dofs" << std::endl;
686
687 // Gets the array of elements from the modeler
688 ElementsArrayType& r_elements_array = rModelPart.Elements();
689 const int nelements = static_cast<int>(r_elements_array.size());
690
691 DofsVectorType elemental_dof_list;
692
693 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
694
695 SizeType nthreads = ParallelUtilities::GetNumThreads();
696
697 typedef std::unordered_set < NodeType::DofType::Pointer, DofPointerHasher> set_type;
698
699 std::vector<set_type> dofs_aux_list(nthreads);
700
701 for (int i = 0; i < static_cast<int>(nthreads); ++i) {
702 dofs_aux_list[i].reserve(nelements);
703 }
704
705 IndexPartition<std::size_t>(nelements).for_each(elemental_dof_list, [&](std::size_t Index, DofsVectorType& tls_elemental_dof_list){
706 auto it_elem = r_elements_array.begin() + Index;
707 const IndexType this_thread_id = OpenMPUtils::ThisThread();
708
709 // Gets list of Dof involved on every element
710 pScheme->GetDofList(*it_elem, tls_elemental_dof_list, r_current_process_info);
711
712 dofs_aux_list[this_thread_id].insert(tls_elemental_dof_list.begin(), tls_elemental_dof_list.end());
713 });
714
715 ConditionsArrayType& r_conditions_array = rModelPart.Conditions();
716 const int nconditions = static_cast<int>(r_conditions_array.size());
717
718 IndexPartition<std::size_t>(nconditions).for_each(elemental_dof_list, [&](std::size_t Index, DofsVectorType& tls_elemental_dof_list){
719 auto it_cond = r_conditions_array.begin() + Index;
720 const IndexType this_thread_id = OpenMPUtils::ThisThread();
721
722 // Gets list of Dof involved on every element
723 pScheme->GetDofList(*it_cond, tls_elemental_dof_list, r_current_process_info);
724 dofs_aux_list[this_thread_id].insert(tls_elemental_dof_list.begin(), tls_elemental_dof_list.end());
725 });
726
727 // Here we do a reduction in a tree so to have everything on thread 0
728 SizeType old_max = nthreads;
729 SizeType new_max = ceil(0.5*static_cast<double>(old_max));
730 while (new_max >= 1 && new_max != old_max) {
731 IndexPartition<std::size_t>(new_max).for_each([&](std::size_t Index){
732 if (Index + new_max < old_max) {
733 dofs_aux_list[Index].insert(dofs_aux_list[Index + new_max].begin(), dofs_aux_list[Index + new_max].end());
734 dofs_aux_list[Index + new_max].clear();
735 }
736 });
737
738 old_max = new_max;
739 new_max = ceil(0.5*static_cast<double>(old_max));
740 }
741
742 DofsArrayType dof_temp;
743 BaseType::mDofSet = DofsArrayType();
744
745 dof_temp.reserve(dofs_aux_list[0].size());
746 for (auto it = dofs_aux_list[0].begin(); it != dofs_aux_list[0].end(); ++it) {
747 dof_temp.push_back(*it);
748 }
749 dof_temp.Sort();
750
751 BaseType::mDofSet = dof_temp;
752
753 // Throws an exception if there are no Degrees of freedom involved in the analysis
754 KRATOS_ERROR_IF(BaseType::mDofSet.size() == 0) << "No degrees of freedom!" << std::endl;
755
756 BaseType::mDofSetIsInitialized = true;
757
758 KRATOS_INFO_IF("ResidualBasedEliminationBuilderAndSolver", this->GetEchoLevel() > 2 && rModelPart.GetCommunicator().MyPID() == 0) << "Finished setting up the dofs" << std::endl;
759
760#ifdef USE_LOCKS_IN_ASSEMBLY
761 if (mLockArray.size() != 0) {
762 for (int i = 0; i < static_cast<int>(mLockArray.size()); i++)
763 omp_destroy_lock(&mLockArray[i]);
764 }
765
766 mLockArray.resize(BaseType::mDofSet.size());
767
768 for (int i = 0; i < static_cast<int>(mLockArray.size()); i++)
769 omp_init_lock(&mLockArray[i]);
770#endif
771
772 // If reactions are to be calculated, we check if all the dofs have reactions defined
773 // This is tobe done only in debug mode
774#ifdef KRATOS_DEBUG
775 if(BaseType::GetCalculateReactionsFlag()) {
776 for(auto dof_iterator = BaseType::mDofSet.begin(); dof_iterator != BaseType::mDofSet.end(); ++dof_iterator) {
777 KRATOS_ERROR_IF_NOT(dof_iterator->HasReaction()) << "Reaction variable not set for the following : " << std::endl
778 << "Node : " << dof_iterator->Id() << std::endl
779 << "Dof : " << (*dof_iterator) << std::endl << "Not possible to calculate reactions." << std::endl;
780 }
781 }
782#endif
783
784 KRATOS_CATCH("");

1110 virtual void ConstructMatrixStructure(
1111 typename TSchemeType::Pointer pScheme,
1112 TSystemMatrixType& rA,
1113 ModelPart& rModelPart
1114 )
1115 {
1116 // Filling with zero the matrix (creating the structure)
1117 Timer::Start("MatrixStructure");
1118
1119 const SizeType equation_size = BaseType::mEquationSystemSize;
1120
1121 std::vector<std::unordered_set<IndexType> > indices(equation_size);
1122
1123 block_for_each(indices, [](std::unordered_set<IndexType>& rIndices){
1124 rIndices.reserve(40);
1125 });
1126
1127 Element::EquationIdVectorType ids(3, 0);
1128
1129 #pragma omp parallel firstprivate(ids)
1130 {
1131 // The process info
1132 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
1133
1134 // We repeat the same declaration for each thead
1135 std::vector<std::unordered_set<IndexType> > temp_indexes(equation_size);
1136
1137 #pragma omp for
1138 for (int index = 0; index < static_cast<int>(equation_size); ++index)
1139 temp_indexes[index].reserve(30);
1140
1141 // Getting the size of the array of elements from the model
1142 const int number_of_elements = static_cast<int>(rModelPart.Elements().size());
1143
1144 // Element initial iterator
1145 const auto it_elem_begin = rModelPart.ElementsBegin();
1146
1147 // We iterate over the elements
1148 #pragma omp for schedule(guided, 512) nowait
1149 for (int i_elem = 0; i_elem<number_of_elements; ++i_elem) {
1150 auto it_elem = it_elem_begin + i_elem;
1151 pScheme->EquationId( *it_elem, ids, r_current_process_info);
1152
1153 for (auto& id_i : ids) {
1154 if (id_i < BaseType::mEquationSystemSize) {
1155 auto& row_indices = temp_indexes[id_i];
1156 for (auto& id_j : ids)
1157 if (id_j < BaseType::mEquationSystemSize)
1158 row_indices.insert(id_j);
1159 }
1160 }
1161 }
1162
1163 // Getting the size of the array of the conditions
1164 const int number_of_conditions = static_cast<int>(rModelPart.Conditions().size());
1165
1166 // Condition initial iterator
1167 const auto it_cond_begin = rModelPart.ConditionsBegin();
1168
1169 // We iterate over the conditions
1170 #pragma omp for schedule(guided, 512) nowait
1171 for (int i_cond = 0; i_cond<number_of_conditions; ++i_cond) {
1172 auto it_cond = it_cond_begin + i_cond;
1173 pScheme->EquationId( *it_cond, ids, r_current_process_info);
1174
1175 for (auto& id_i : ids) {
1176 if (id_i < BaseType::mEquationSystemSize) {
1177 auto& row_indices = temp_indexes[id_i];
1178 for (auto& id_j : ids)
1179 if (id_j < BaseType::mEquationSystemSize)
1180 row_indices.insert(id_j);
1181 }
1182 }
1183 }
1184
1185 // Merging all the temporal indexes
1186 #pragma omp critical
1187 {
1188 for (int i = 0; i < static_cast<int>(temp_indexes.size()); ++i) {
1189 indices[i].insert(temp_indexes[i].begin(), temp_indexes[i].end());
1190 }
1191 }
1192 }
1193
1194 // Count the row sizes
1195 SizeType nnz = 0;
1196 for (IndexType i = 0; i < indices.size(); ++i)
1197 nnz += indices[i].size();
1198
1199 rA = TSystemMatrixType(indices.size(), indices.size(), nnz);
1200
1201 double* Avalues = rA.value_data().begin();
1202 std::size_t* Arow_indices = rA.index1_data().begin();
1203 std::size_t* Acol_indices = rA.index2_data().begin();
1204
1205 // Filling the index1 vector - DO NOT MAKE PARALLEL THE FOLLOWING LOOP!
1206 Arow_indices[0] = 0;
1207 for (IndexType i = 0; i < rA.size1(); ++i)
1208 Arow_indices[i + 1] = Arow_indices[i] + indices[i].size();
1209
1210 IndexPartition<std::size_t>(rA.size1()).for_each([&](std::size_t Index){
1211 const IndexType row_begin = Arow_indices[Index];
1212 const IndexType row_end = Arow_indices[Index + 1];
1213 IndexType k = row_begin;
1214 for (auto it = indices[Index].begin(); it != indices[Index].end(); ++it) {
1215 Acol_indices[k] = *it;
1216 Avalues[k] = 0.0;
1217 ++k;
1218 }
1219
1220 std::sort(&Acol_indices[row_begin], &Acol_indices[row_end]);
1221 });
1222
1223 rA.set_filled(indices.size() + 1, nnz);
1224
1225 Timer::Stop("MatrixStructure");
1226 }

1259 inline void AssembleRowContributionFreeDofs(
1260 TSystemMatrixType& rA,
1261 const Matrix& rALocal,
1262 const IndexType i,
1263 const IndexType i_local,
1264 const Element::EquationIdVectorType& EquationId
1265 )
1266 {
1267 double* values_vector = rA.value_data().begin();
1268 IndexType* index1_vector = rA.index1_data().begin();
1269 IndexType* index2_vector = rA.index2_data().begin();
1270
1271 const IndexType left_limit = index1_vector[i];
1272
1273 // Find the first entry
1274 // We iterate over the equation ids until we find the first equation id to be considered
1275 // We count in which component we find an ID
1276 IndexType last_pos = 0;
1277 IndexType last_found = 0;
1278 IndexType counter = 0;
1279 for(IndexType j=0; j < EquationId.size(); ++j) {
1280 ++counter;
1281 const IndexType j_global = EquationId[j];
1282 if (j_global < BaseType::mEquationSystemSize) {
1283 last_pos = ForwardFind(j_global,left_limit,index2_vector);
1284 last_found = j_global;
1285 break;
1286 }
1287 }
1288
1289 // If the counter is equal to the size of the EquationID vector that means that only one dof will be considered, if the number is greater means that all the dofs are fixed. If the number is below means that at we have several dofs free to be considered
1290 if (counter <= EquationId.size()) {
1291#ifndef USE_LOCKS_IN_ASSEMBLY
1292 double& r_a = values_vector[last_pos];
1293 const double& v_a = rALocal(i_local,counter - 1);
1294 AtomicAdd(r_a, v_a);
1295#else
1296 values_vector[last_pos] += rALocal(i_local,counter - 1);
1297#endif
1298 // Now find all of the other entries
1299 IndexType pos = 0;
1300 for(IndexType j = counter; j < EquationId.size(); ++j) {
1301 IndexType id_to_find = EquationId[j];
1302 if (id_to_find < BaseType::mEquationSystemSize) {
1303 if(id_to_find > last_found)
1304 pos = ForwardFind(id_to_find,last_pos+1,index2_vector);
1305 else if(id_to_find < last_found)
1306 pos = BackwardFind(id_to_find,last_pos-1,index2_vector);
1307 else
1308 pos = last_pos;
1309
1310#ifndef USE_LOCKS_IN_ASSEMBLY
1311 double& r = values_vector[pos];
1312 const double& v = rALocal(i_local,j);
1313 AtomicAdd(r, v);
1314#else
1315 values_vector[pos] += rALocal(i_local,j);
1316#endif
1317 last_found = id_to_find;
1318 last_pos = pos;
1319 }
1320 }
1321 }
