261 typedef void (AssignVectorVariableToConditionsProcess::*AssignmentMethodPointer) (ModelPart::ConditionType&, const Variable<array_1d<double,3> >&, const array_1d<double,3>&);
262
263 AssignmentMethodPointer AssignmentMethod = this->GetAssignmentMethod<AssignmentMethodPointer>();
264
265 const int nconditions = this->mrModelPart.GetMesh().Conditions().size();
266
267 if(nconditions != 0)
268 {
269 ModelPart::ConditionsContainerType::iterator it_begin = this->mrModelPart.GetMesh().ConditionsBegin();
270
271 #pragma omp parallel for
272 for(int i = 0; i<nconditions; i++)
273 {
274 ModelPart::ConditionsContainerType::iterator it = it_begin + i;
275
276 (this->*AssignmentMethod)(*it, rVariable, rvector_value);
277 }
