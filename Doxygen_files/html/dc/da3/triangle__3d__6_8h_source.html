505 bool IsInside(
506 const CoordinatesArrayType& rPoint,
507 CoordinatesArrayType& rResult,
508 const double Tolerance = std::numeric_limits<double>::epsilon()
509 )const override
510 {
511 this->PointLocalCoordinates( rResult, rPoint );
512
513 if ( (rResult[0] >= (0.0-Tolerance)) && (rResult[0] <= (1.0+Tolerance)) )
514 {
515 if ( (rResult[1] >= (0.0-Tolerance)) && (rResult[1] <= (1.0+Tolerance)) )
516 {
517 if ( (rResult[0] + rResult[1]) <= (1.0+Tolerance) )
518 {
519 return true;
520 }
521 }
522 }
523
524 return false;
525 }

543 double ShapeFunctionValue( IndexType ShapeFunctionIndex,
544 const CoordinatesArrayType& rPoint ) const override
545 {
546
547 double thirdCoord = 1 - rPoint[0] - rPoint[1];
548
549 switch ( ShapeFunctionIndex )
550 {
551 case 0:
552 return( thirdCoord*( 2*thirdCoord - 1 ) );
553 case 1:
554 return( rPoint[0]*( 2*rPoint[0] - 1 ) );
555 case 2:
556 return( rPoint[1]*( 2*rPoint[1] - 1 ) );
557 case 3:
558 return( 4*thirdCoord*rPoint[0] );
559 case 4:
560 return( 4*rPoint[0]*rPoint[1] );
561 case 5:
562 return( 4*rPoint[1]*thirdCoord );
563
564 default:
565 KRATOS_ERROR << "Wrong index of shape function!" << *this << std::endl;
566 }
567
568 return 0;
569 }

711 Matrix& ShapeFunctionsLocalGradients( Matrix& rResult,
712 const CoordinatesArrayType& rPoint ) const override
713 {
714 rResult.resize( 6, 2,false );
715 double thirdCoord = 1 - rPoint[0] - rPoint[1];
716 double thirdCoord_DX = -1;
717 double thirdCoord_DY = -1;
718
719 noalias( rResult ) = ZeroMatrix( 6, 2 );
720 rResult( 0, 0 ) = ( 4 * thirdCoord - 1 ) * thirdCoord_DX;
721 rResult( 0, 1 ) = ( 4 * thirdCoord - 1 ) * thirdCoord_DY;
722 rResult( 1, 0 ) = 4 * rPoint[0] - 1;
723 rResult( 1, 1 ) = 0;
724 rResult( 2, 0 ) = 0;
725 rResult( 2, 1 ) = 4 * rPoint[1] - 1;
726 rResult( 3, 0 ) = 4 * thirdCoord_DX * rPoint[0] + 4 * thirdCoord;
727 rResult( 3, 1 ) = 4 * thirdCoord_DY * rPoint[0];
728 rResult( 4, 0 ) = 4 * rPoint[1];
729 rResult( 4, 1 ) = 4 * rPoint[0];
730 rResult( 5, 0 ) = 4 * rPoint[1] * thirdCoord_DX;
731 rResult( 5, 1 ) = 4 * rPoint[1] * thirdCoord_DY + 4 * thirdCoord;
732 return rResult;

929 static Matrix CalculateShapeFunctionsIntegrationPointsValues(
930 typename BaseType::IntegrationMethod ThisMethod )
931 {
932 IntegrationPointsContainerType all_integration_points =
933 AllIntegrationPoints();
934 IntegrationPointsArrayType integration_points = all_integration_points[static_cast<int>(ThisMethod)];
935 //number of integration points
936 const int integration_points_number = integration_points.size();
937 //number of nodes in current geometry
938 const int points_number = 6;
939 //setting up return matrix
940 Matrix shape_function_values( integration_points_number, points_number );
941 //loop over all integration points
942
943 for ( int pnt = 0; pnt < integration_points_number; pnt++ )
944 {
945 double thirdCoord = 1 - integration_points[pnt].X() - integration_points[pnt].Y();
946
947 shape_function_values( pnt, 0 ) = thirdCoord * ( 2 * thirdCoord - 1 ) ;
948 shape_function_values( pnt, 1 ) = integration_points[pnt].X() * ( 2 * integration_points[pnt].X() - 1 ) ;
949 shape_function_values( pnt, 2 ) = integration_points[pnt].Y() * ( 2 * integration_points[pnt].Y() - 1 ) ;
950 shape_function_values( pnt, 3 ) = 4 * thirdCoord * integration_points[pnt].X();
951 shape_function_values( pnt, 4 ) = 4 * integration_points[pnt].X() * integration_points[pnt].Y();
952 shape_function_values( pnt, 5 ) = 4 * integration_points[pnt].Y() * thirdCoord;
953
954 }
955
956 return shape_function_values;
957 }
