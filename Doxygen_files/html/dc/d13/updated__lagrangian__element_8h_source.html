757 void GetOutwardsUnitNormalForTwoPoints(array_1d<double, 3> &NormalVector,
758 unsigned int idA,
759 unsigned int idB,
760 unsigned int otherId)
761 {
762 GeometryType &rGeom = this->GetGeometry();
763 double deltaX = rGeom[idB].X() - rGeom[idA].X();
764 double deltaY = rGeom[idB].Y() - rGeom[idA].Y();
765 double elementSize = sqrt(deltaX * deltaX + deltaY * deltaY);
766 if (fabs(deltaX) > fabs(deltaY))
767 { //to avoid division by zero or small numbers
768 NormalVector[0] = -deltaY / deltaX;
769 NormalVector[1] = 1.0;
770 double normNormal = NormalVector[0] * NormalVector[0] + NormalVector[1] * NormalVector[1];
771 NormalVector *= 1.0 / sqrt(normNormal);
772 }
773 else
774 {
775 NormalVector[0] = 1.0;
776 NormalVector[1] = -deltaX / deltaY;
777 double normNormal = NormalVector[0] * NormalVector[0] + NormalVector[1] * NormalVector[1];
778 NormalVector *= 1.0 / sqrt(normNormal);
779 }
780 //to determine if the computed normal outwards or inwards
781 const array_1d<double, 3> MeanPoint = (rGeom[idB].Coordinates() + rGeom[idA].Coordinates()) * 0.5;
782 const array_1d<double, 3> DistanceA = rGeom[otherId].Coordinates() - (MeanPoint + NormalVector * elementSize);
783 const array_1d<double, 3> DistanceB = rGeom[otherId].Coordinates() - (MeanPoint - NormalVector * elementSize);
784 const double normA = DistanceA[0] * DistanceA[0] + DistanceA[1] * DistanceA[1];
785 const double normB = DistanceB[0] * DistanceB[0] + DistanceB[1] * DistanceB[1];
786 if (normB > normA)
787 {
788 NormalVector *= -1.0;
789 }
790 }

845 double deltaZ = rGeom[idB].Z() - rGeom[idA].Z();

859 unsigned int idA,
860 unsigned int idB,
861 unsigned int idC,
862 unsigned int otherId,
863 double &surfaceArea)
864 {
865 GeometryType &rGeom = this->GetGeometry();
866 const array_1d<double, 3> TangentXi = rGeom[idB].Coordinates() - rGeom[idA].Coordinates();
867 const array_1d<double, 3> TangentEta = rGeom[idC].Coordinates() - rGeom[idA].Coordinates();
868
869 MathUtils<double>::CrossProduct(NormalVector, TangentXi, TangentEta);
870 double normNormal = NormalVector[0] * NormalVector[0] + NormalVector[1] * NormalVector[1] + NormalVector[2] * NormalVector[2];
871 NormalVector *= 1.0 / sqrt(normNormal);
872
873 //to determine if the computed normal outwards or inwards
874 double deltaX = rGeom[idB].X() - rGeom[idA].X();
875 double deltaY = rGeom[idB].Y() - rGeom[idA].Y();
876 double deltaZ = rGeom[idB].Z() - rGeom[idA].Z();
877
878 const double a = MathUtils<double>::Norm3(rGeom.GetPoint(idA) - rGeom.GetPoint(idB));
879 const double b = MathUtils<double>::Norm3(rGeom.GetPoint(idB) - rGeom.GetPoint(idC));
880 const double c = MathUtils<double>::Norm3(rGeom.GetPoint(idC) - rGeom.GetPoint(idA));
881
882 const double s = (a + b + c) / 2.0;
883
884 surfaceArea= std::sqrt(s * (s - a) * (s - b) * (s - c));
885
886 double elementSize = sqrt(deltaX * deltaX + deltaY * deltaY + deltaZ * deltaZ); // this is just to have an idea of the size of the element
887 const array_1d<double, 3> MeanPoint = (rGeom[idC].Coordinates() + rGeom[idB].Coordinates() + rGeom[idA].Coordinates()) / 3.0;
888 const array_1d<double, 3> DistanceA = rGeom[otherId].Coordinates() - (MeanPoint + NormalVector * elementSize);
889 const array_1d<double, 3> DistanceB = rGeom[otherId].Coordinates() - (MeanPoint - NormalVector * elementSize);
890 const double normA = DistanceA[0] * DistanceA[0] + DistanceA[1] * DistanceA[1] + DistanceA[2] * DistanceA[2];
891 const double normB = DistanceB[0] * DistanceB[0] + DistanceB[1] * DistanceB[1] + DistanceB[2] * DistanceB[2];
892 if (normB > normA)
893 {
894 NormalVector *= -1.0;
