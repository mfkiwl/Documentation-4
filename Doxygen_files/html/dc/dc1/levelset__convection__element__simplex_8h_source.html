113 void CalculateLocalSystem(MatrixType& rLeftHandSideMatrix, VectorType& rRightHandSideVector, const ProcessInfo& rCurrentProcessInfo) override
114 {
115 KRATOS_TRY
116
117 if (rLeftHandSideMatrix.size1() != TNumNodes)
118 rLeftHandSideMatrix.resize(TNumNodes, TNumNodes, false); //false says not to preserve existing storage!!
119
120 if (rRightHandSideVector.size() != TNumNodes)
121 rRightHandSideVector.resize(TNumNodes, false); //false says not to preserve existing storage!!
122
123 const double delta_t = rCurrentProcessInfo[DELTA_TIME];
124 const double dt_inv = 1.0 / delta_t;
125 const double theta = rCurrentProcessInfo.Has(TIME_INTEGRATION_THETA) ? rCurrentProcessInfo[TIME_INTEGRATION_THETA] : 0.5;
126
127 const ConvectionDiffusionSettings::Pointer& my_settings = rCurrentProcessInfo.GetValue(CONVECTION_DIFFUSION_SETTINGS);
128 const Variable<double>& rUnknownVar = my_settings->GetUnknownVariable();
129 const Variable<array_1d<double, 3 > >& rConvVar = my_settings->GetConvectionVariable();
130 const double dyn_st_beta = rCurrentProcessInfo[DYNAMIC_TAU];
131
132
133 //getting data for the given geometry
134 BoundedMatrix<double, TNumNodes, TDim > DN_DX;
135 array_1d<double, TNumNodes > N;
136 double Volume;
137 GeometryUtils::CalculateGeometryData(GetGeometry(), DN_DX, N, Volume);
138 double h = ComputeH(DN_DX, Volume);
139
140
141 //here we get all the variables we will need
142 array_1d<double,TNumNodes> phi, phi_old;
143 array_1d< array_1d<double,3 >, TNumNodes> v, vold;
144
145 for (unsigned int i = 0; i < TNumNodes; i++)
146 {
147 phi[i] = GetGeometry()[i].FastGetSolutionStepValue(rUnknownVar);
148 phi_old[i] = GetGeometry()[i].FastGetSolutionStepValue(rUnknownVar,1);
149// dphi_dt[i] = dt_inv*(phi[i] - phi_old [i];
150
151 v[i] = GetGeometry()[i].FastGetSolutionStepValue(rConvVar);
152 vold[i] = GetGeometry()[i].FastGetSolutionStepValue(rConvVar,1);
153 }
154 array_1d<double,TDim> grad_phi_halfstep = prod(trans(DN_DX), 0.5*(phi+phi_old));
155 const double norm_grad = norm_2(grad_phi_halfstep);
156
157 //here we use a term beta which takes into account a reaction term of the type "beta*div_v"
158
159
160 //compute the divergence of v
161 double div_v = 0.0;
162 for (unsigned int i = 0; i < TNumNodes; i++)
163 for(unsigned int k=0; k<TDim; k++)
164 div_v += 0.5*DN_DX(i,k)*(v[i][k] + vold[i][k]);
165
166 double beta = 0.0; //1.0;
167
168// unsigned int nneg=0;
169// for(unsigned int i=0; i<TNumNodes; i++) if(phi[i] < 0.0) nneg++;
170// if(nneg > 0) beta = 1.0; //beta = 0.1;
171
172 BoundedMatrix<double,TNumNodes, TNumNodes> aux1 = ZeroMatrix(TNumNodes, TNumNodes); //terms multiplying dphi/dt
173 BoundedMatrix<double,TNumNodes, TNumNodes> aux2 = ZeroMatrix(TNumNodes, TNumNodes); //terms multiplying phi
174 BoundedMatrix<double,TNumNodes, TDim> tmp;
175
176
177 BoundedMatrix<double,TNumNodes, TNumNodes> Ncontainer;
178 GetShapeFunctionsOnGauss(Ncontainer);
179 for(unsigned int igauss=0; igauss<TDim+1; igauss++)
180 {
181 noalias(N) = row(Ncontainer,igauss);
182
183 //obtain the velocity in the middle of the tiem step
184 array_1d<double, TDim > vel_gauss=ZeroVector(TDim);
185 for (unsigned int i = 0; i < TNumNodes; i++)
186 {
187 for(unsigned int k=0; k<TDim; k++)
188 vel_gauss[k] += 0.5*N[i]*(v[i][k]+vold[i][k]);
189 }
190 const double norm_vel = norm_2(vel_gauss);
191 array_1d<double, TNumNodes > a_dot_grad = prod(DN_DX, vel_gauss);
192
193 const double tau_denom = std::max(dyn_st_beta *dt_inv + 2.0 * norm_vel / h + std::abs(/*beta**/div_v), 1e-2); //the term std::abs(div_v) is added following Pablo Becker's suggestion
194 const double tau = 1.0 / (tau_denom);
195
196 //terms multiplying dphi/dt (aux1)
197 noalias(aux1) += (1.0+tau*beta*div_v)*outer_prod(N, N);
198 noalias(aux1) += tau*outer_prod(a_dot_grad, N);
199
200 //terms which multiply the gradient of phi
201 noalias(aux2) += (1.0+tau*beta*div_v)*outer_prod(N, a_dot_grad);
202 noalias(aux2) += tau*outer_prod(a_dot_grad, a_dot_grad);
203
204 //cross-wind term
205 if(norm_grad > 1e-3 && norm_vel > 1e-9)
206 {
207 const double C = rCurrentProcessInfo.GetValue(CROSS_WIND_STABILIZATION_FACTOR);
208 const double time_derivative = dt_inv*(inner_prod(N,phi)-inner_prod(N,phi_old));
209 const double res = -time_derivative -inner_prod(vel_gauss, grad_phi_halfstep);
210
211 const double disc_capturing_coeff = 0.5*C*h*fabs(res/norm_grad);
212 BoundedMatrix<double,TDim,TDim> D = disc_capturing_coeff*( IdentityMatrix(TDim));
213 const double norm_vel_squared = norm_vel*norm_vel;
214 D += (std::max( disc_capturing_coeff - tau*norm_vel_squared , 0.0) - disc_capturing_coeff)/(norm_vel_squared) * outer_prod(vel_gauss,vel_gauss);
215
216 noalias(tmp) = prod(DN_DX,D);
217 noalias(aux2) += prod(tmp,trans(DN_DX));
218 }
219 }
220
221 //adding the second and third term in the formulation
222 noalias(rLeftHandSideMatrix) = (dt_inv + theta*beta*div_v)*aux1;
223 noalias(rRightHandSideVector) = (dt_inv - (1.0 - theta)*beta*div_v)*prod(aux1,phi_old);
224
225 //terms in aux2
226 noalias(rLeftHandSideMatrix) += theta*aux2;
227 noalias(rRightHandSideVector) -= (1.0 - theta)*prod(aux2,phi_old);
228
229 //take out the dirichlet part to finish computing the residual
230 noalias(rRightHandSideVector) -= prod(rLeftHandSideMatrix, phi);
231
232 rRightHandSideVector *= Volume/static_cast<double>(TNumNodes);
233 rLeftHandSideMatrix *= Volume/static_cast<double>(TNumNodes);
234
235 KRATOS_CATCH("Error in Levelset Element")
236 }

347 double ComputeH(BoundedMatrix<double,TNumNodes, TDim>& DN_DX, const double Volume)
348 {
349 double h=0.0;
350 for(unsigned int i=0; i<TNumNodes; i++)
351 {
352 double h_inv = 0.0;
353 for(unsigned int k=0; k<TDim; k++)
354 {
355 h_inv += DN_DX(i,k)*DN_DX(i,k);
356 }
357 h = std::max(h, 1.0 / h_inv);
358 }
359 h = std::sqrt(h);
360 return h;
361 }
