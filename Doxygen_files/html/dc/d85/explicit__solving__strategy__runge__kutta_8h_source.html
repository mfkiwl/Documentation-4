257 void SolveWithLumpedMassMatrix() override
258 {
259 KRATOS_TRY
260
261 // Get the required data from the explicit builder and solver
262 auto& r_explicit_bs = BaseType::GetExplicitBuilder();
263 auto& r_dof_set = r_explicit_bs.GetDofSet();
264 const unsigned int dof_size = r_explicit_bs.GetEquationSystemSize();
265 const auto& r_lumped_mass_vector = r_explicit_bs.GetLumpedMassMatrixVector();
266
267 // Set the auxiliary RK vectors
268 LocalSystemVectorType u_n(dof_size); // TODO: THIS IS INEFICCIENT. CREATE A UNORDERED_SET WITH THE IDOF AND VALUE AS ENTRIES. THIS HAS TO BE OPTIONAL
269 LocalSystemMatrixType rk_K(dof_size, TButcherTableau::SubstepCount());
270
271 // Perform the RK update
272 const double dt = BaseType::GetDeltaTime();
273 KRATOS_ERROR_IF(dt < 1.0e-12) << "ProcessInfo DELTA_TIME is close to zero." << std::endl;
274
275 // Set the previous step solution in the current buffer position
276 // Note that we set the 0 position of the buffer to be equal to the values in step n (not n+1)
277 // Additionally, we save in an auxiliary vector the value of the fixed DOFs, which is also taken from the previous time step
278 IndexPartition<int>(r_dof_set.size()).for_each(
279 [&](int i_dof){
280 auto it_dof = r_dof_set.begin() + i_dof;
281 double& r_u_0 = it_dof->GetSolutionStepValue(0);
282 const double& r_u_1 = it_dof->GetSolutionStepValue(1);
283 if (it_dof->IsFixed()) {
284 u_n(i_dof) = r_u_0;
285 }
286 r_u_0 = r_u_1;
287 }
288 );
289
290 // Calculate the RK intermediate sub steps
291 for(unsigned int i=1; i<TButcherTableau::SubstepCount(); ++i) {
292 PerformRungeKuttaIntermediateSubStep(i, u_n, rk_K);
293 }
294 PerformRungeKuttaLastSubStep(rk_K);
295
296 // Do the final solution update
297 const auto& weights = mButcherTableau.GetWeights();
298 IndexPartition<int>(r_dof_set.size()).for_each(
299 [&](int i_dof){
300 auto it_dof = r_dof_set.begin() + i_dof;
301 // Do the DOF update
302 double& r_u = it_dof->GetSolutionStepValue(0);
303 const double& r_u_old = it_dof->GetSolutionStepValue(1);
304 if (!it_dof->IsFixed()) {
305 const double mass = r_lumped_mass_vector(i_dof);
306 const MatrixRow<LocalSystemMatrixType> substeps_k = row(rk_K, i_dof);
307 r_u = r_u_old + (dt / mass) * inner_prod(weights, substeps_k);
308 } else {
309 r_u = u_n(i_dof);
310 }
311 }
312 );
313
314 KRATOS_CATCH("");
315 }

349 virtual void PerformRungeKuttaIntermediateSubStep(
350 const IndexType SubStepIndex,
351 const LocalSystemVectorType& rFixedDofsValues,
352 LocalSystemMatrixType& rIntermediateStepResidualVectors)
353 {
354 KRATOS_TRY
355
356 // Get the required data from the explicit builder and solver
357 auto& r_explicit_bs = BaseType::GetExplicitBuilder();
358 auto& r_dof_set = r_explicit_bs.GetDofSet();
359 const auto& r_lumped_mass_vector = r_explicit_bs.GetLumpedMassMatrixVector();
360
361 // Get model part and information
362 const double dt = BaseType::GetDeltaTime();
363 KRATOS_ERROR_IF(dt < 1.0e-12) << "ProcessInfo DELTA_TIME is close to zero." << std::endl;
364 auto& r_model_part = BaseType::GetModelPart();
365 auto& r_process_info = r_model_part.GetProcessInfo();
366
367 // Fetch this substeps's values from tableau
368 const double integration_theta = mButcherTableau.GetIntegrationTheta(SubStepIndex);
369 const auto alphas = mButcherTableau.GetMatrixRow(SubStepIndex); // Runge kutta matrix row
370
371 // Set the RUNGE_KUTTA_STEP value. This has to be done prior to the InitializeRungeKuttaStep()
372 r_process_info.GetValue(RUNGE_KUTTA_STEP) = SubStepIndex;
373 r_process_info.GetValue(TIME_INTEGRATION_THETA) = integration_theta;
374
375 // Perform the intermidate sub step update
376 InitializeRungeKuttaIntermediateSubStep();
377 r_explicit_bs.BuildRHS(r_model_part);
378
379 IndexPartition<int>(r_dof_set.size()).for_each(
380 [&](int i_dof){
381 auto it_dof = r_dof_set.begin() + i_dof;
382 // Save current value in the corresponding vector
383 const double& r_res = it_dof->GetSolutionStepReactionValue();
384 rIntermediateStepResidualVectors(i_dof, SubStepIndex-1) = r_res;
385 // Do the DOF update
386 double& r_u = it_dof->GetSolutionStepValue(0);
387 const double& r_u_old = it_dof->GetSolutionStepValue(1);
388 if (!it_dof->IsFixed()) {
389 const double mass = r_lumped_mass_vector(i_dof);
390 const auto k = row(rIntermediateStepResidualVectors, i_dof);
391 r_u = r_u_old + (dt / mass) * std::inner_product(alphas.begin, alphas.end, k.begin(), 0.0);
392 /* ^~~~~~~~~~~~~~~~~~
393 * Using std::inner_product instead of boost's inner_prod because it allows us to
394 * chose a begin and, more importantly, an end.
395 *
396 * This is useful because alpha_ij = 0 for j >= i, hence the tail end of this scalar product
397 * can be ignored by setting the past-the-end iterator at j=i
398 */
399 } else {
400 const double delta_u = rFixedDofsValues(i_dof) - r_u_old;
401 r_u = r_u_old + integration_theta * delta_u;
402 }
403 }
404 );
405
406 FinalizeRungeKuttaIntermediateSubStep();
407
408 KRATOS_CATCH("SubstepIndex = " + std::to_string(SubStepIndex));
409 }

416 virtual void PerformRungeKuttaLastSubStep(LocalSystemMatrixType& rLastStepResidualVector)
417 {
418 KRATOS_TRY
419
420 // Get the required data from the explicit builder and solver
421 auto& r_explicit_bs = BaseType::GetExplicitBuilder();
422 auto& r_dof_set = r_explicit_bs.GetDofSet();
423
424 // Get model part
425 auto& r_model_part = BaseType::GetModelPart();
426 auto& r_process_info = r_model_part.GetProcessInfo();
427 constexpr unsigned int substep_index = TButcherTableau::SubstepCount();
428
429 // Set the RUNGE_KUTTA_STEP value. This has to be done prior to the InitializeRungeKuttaStep()
430 r_process_info.GetValue(RUNGE_KUTTA_STEP) = TButcherTableau::SubstepCount();
431 r_process_info.GetValue(TIME_INTEGRATION_THETA) = mButcherTableau.GetIntegrationTheta(substep_index);
432
433 // Perform the last sub step residual calculation
434 InitializeRungeKuttaLastSubStep();
435 r_explicit_bs.BuildRHS(r_model_part);
436
437 IndexPartition<int>(r_dof_set.size()).for_each(
438 [&](int i_dof){
439 const auto it_dof = r_dof_set.begin() + i_dof;
440 // Save current value in the corresponding vector
441 const double& r_res = it_dof->GetSolutionStepReactionValue();
442 rLastStepResidualVector(i_dof, substep_index - 1) = r_res;
443 }
444 );
445
446 FinalizeRungeKuttaLastSubStep();
447
448 KRATOS_CATCH("");
449 }
