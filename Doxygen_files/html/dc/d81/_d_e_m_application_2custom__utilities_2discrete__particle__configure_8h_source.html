164 static inline void GetBoxCenter(double box_center[3], const double min_point[3], const double max_point[3])
165 {
166 for (unsigned int i = 0; i < 3; ++i){
167 box_center[i] = 0.5 * (min_point[i] + max_point[i]);
168
169 if (min_point[i] > max_point[i]){ // the box is broken by the boundary in this dimension: ] x * [ (outside), ] x [ * (inside)
170 const double& min = mDomainMin[i];
171 const double& max = mDomainMax[i];
172 box_center[i] += 0.5 * (max - min); // The center of the box and of its complementary are always half a domain period apart: |..]..x..[...*.| (our bet: we assume we should ADD to go from x to *)
173 if (box_center[i] > max){ // we made a mistake, we should have gone the opposite way
174 box_center[i] -= max - min;
175 }
176 }
177 }

214 static inline bool Intersection(const PointerType& rObj_1, const PointerType& rObj_2)
215 {
216 double rObj_2_to_rObj_1[3];
217 PointType& point_1 = rObj_1->GetGeometry()[0];
218 const double coors_1[3] = {point_1[0], point_1[1], point_1[2]};
219 PointType& point_2 = rObj_2->GetGeometry()[0];
220 const double coors_2[3] = {point_2[0], point_2[1], point_2[2]};

227 SphericParticle* p_particle2 = static_cast<SphericParticle*>(&*rObj_2);
228 const double radius_sum = p_particle1->GetSearchRadius() + p_particle2->GetSearchRadius();
229 const bool intersect = floatle(distance_2, std::pow(radius_sum, 2));

240 static inline bool IntersectionBox(const PointerType& rObject, const PointType& rLowPoint, const PointType& rHighPoint)
241 {
242 const array_1d<double, 3>& center_of_particle = rObject->GetGeometry()[0];
243
244 SphericParticle* p_particle = static_cast<SphericParticle*>(&*rObject);
245 const double& radius = p_particle->GetSearchRadius();
246
247 bool intersect;
248
249 if (mDomainIsPeriodic){
250 double expanded_box_min[3];
251 double expanded_box_max[3];
252
253 for (unsigned int i = 0; i < 3; ++i){
254 expanded_box_min[i] = rLowPoint[i] - radius;
255 expanded_box_max[i] = rHighPoint[i] + radius;
256 }
257
258 double box_center[3];
259 GetBoxCenter(box_center, expanded_box_min, expanded_box_max);
260 //double box_center[3] = {0.5 * (rLowPoint[0] + rHighPoint[0]), 0.5 * (rLowPoint[1] + rHighPoint[1]), 0.5 * (rLowPoint[2] + rHighPoint[2])};
261 double representative_center_of_particle[3] = {center_of_particle[0],
262 center_of_particle[1],
263 center_of_particle[2]};
264 TransformToClosestPeriodicCoordinates(box_center, representative_center_of_particle);
265
266 for (unsigned int i = 0; i < 3; ++i){
267 const bool is_broken = rLowPoint[i] > rHighPoint[i];
268
269 if (is_broken){ // i.e., we have | ] [ | in this direction
270 intersect = floatge(expanded_box_min[i], representative_center_of_particle[i])
271 && floatle(expanded_box_max[i], representative_center_of_particle[i]);
272 }
273
274 else { // i.e., we have | [ ] | in this direction
275 intersect = floatle(expanded_box_min[i], representative_center_of_particle[i])
276 && floatge(expanded_box_max[i], representative_center_of_particle[i]);
277 }
278 }
279 }
280
281 else {
282 for (unsigned int i = 0; i < 3; ++i){
283 intersect = floatle(rLowPoint[i] - radius, center_of_particle[i])
284 && floatge(rHighPoint[i] + radius, center_of_particle[i]);
285 }
286 }
287
288 return intersect;
289 }

389 static inline void PeriodicSubstract(const double a[3], const double b[3], double c[3])
390 {
391 for (unsigned int i = 0; i < 3; ++i){
392 c[i] = a[i] - b[i];
393 }
394
395 if (mDomainIsPeriodic){ // Periods have been set (the domain is periodic)
396 for (unsigned int i = 0; i < 3; ++i){
397 if (std::fabs(c[i]) > 0.5 * mDomainPeriods[i]){ // the objects are closer through the boundary
398 c[i] -= GetSign(c[i]) * mDomainPeriods[i];
399 }
400 }
401 }
