160 template<typename TPointType>
161 void SearchInRadius(TPointType const& ThePoint, double Radius, std::vector<ResultType>& rResults) {
162 std::unordered_set<GeometricalObject*> results;
163
164 array_1d< std::size_t, 3 > min_position;
165 array_1d< std::size_t, 3 > max_position;
166
167 for(int i = 0; i < 3; i++ ) {
168 min_position[i] = CalculatePosition(ThePoint[i] - Radius, i);
169 max_position[i] = CalculatePosition(ThePoint[i] + Radius, i) + 1;
170 }
171 for(std::size_t k = min_position[2] ; k < max_position[2] ; k++){
172 for(std::size_t j = min_position[1] ; j < max_position[1] ; j++){
173 for(std::size_t i = min_position[0] ; i < max_position[0] ; i++){
174 auto& cell = GetCell(i,j,k);
175 SearchInRadiusInCell(cell, ThePoint, Radius, results);
176 }
177 }
178 }
179
180 rResults.clear();
181 for(auto p_object : results){
182 rResults.push_back(ResultType(p_object));
183 }

191 template<typename TPointType>
192 ResultType SearchNearestInRadius(TPointType const& ThePoint, double Radius) {
193 ResultType current_result;
194 current_result.SetDistance(std::numeric_limits<double>::max());
195
196 double radius_increment = *std::max_element(mCellSizes.begin(), mCellSizes.end());
197
198 array_1d< std::size_t, 3 > min_position;
199 array_1d< std::size_t, 3 > max_position;
200
201 for(double current_radius = 0 ; current_radius < Radius + radius_increment ; current_radius += radius_increment){
202 current_radius = (current_radius > Radius) ? Radius : current_radius;
203 for(int i = 0; i < 3; i++ ) {
204 min_position[i] = CalculatePosition(ThePoint[i] - current_radius, i);
205 max_position[i] = CalculatePosition(ThePoint[i] + current_radius, i) + 1;
206 }
207
208 for(std::size_t k = min_position[2] ; k < max_position[2] ; k++){
209 for(std::size_t j = min_position[1] ; j < max_position[1] ; j++){
210 for(std::size_t i = min_position[0] ; i < max_position[0] ; i++){
211 auto& cell = GetCell(i,j,k);
212 SearchNearestInCell(cell, ThePoint, current_result, current_radius);
213 }
214 }
215 }
216 bool all_cells_are_covered = (min_position[0] == 0) && (min_position[1] == 0) && (min_position[2] == 0);
217 all_cells_are_covered &= (max_position[0] == mNumberOfCells[0] - 1) && (max_position[1] == mNumberOfCells[1] - 1) && (max_position[2] == mNumberOfCells[2] - 1);
218
219 if(all_cells_are_covered || current_result.IsObjectFound()){
220 break;
221 }
222 }
223 return current_result;
224 }

319 void CalculateCellSize(std::size_t NumberOfCells) {
320 std::size_t avarage_number_of_cells = static_cast<std::size_t>(std::pow(static_cast<double>(NumberOfCells), 1.00 / Dimension));
321 std::array<double, 3> lengths;
322 double avarage_length = 0.00;
323 for (int i = 0; i < Dimension; i++) {
324 lengths[i] = mBoundingBox.GetMaxPoint()[i] - mBoundingBox.GetMinPoint()[i];
325 avarage_length += lengths[i];
326 }
327 avarage_length *= 1.00 / 3.00;
328
329 if (avarage_length < std::numeric_limits<double>::epsilon()) {
330 mNumberOfCells = ScalarVector(3, 1);
331 return;
332 }
333
334 for (int i = 0; i < Dimension; i++) {
335 mNumberOfCells[i] = static_cast<std::size_t>(lengths[i] / avarage_length * avarage_number_of_cells) + 1;
336 if (mNumberOfCells[i] > 1)
337 mCellSizes[i] = lengths[i] / mNumberOfCells[i];
338 else
339 mCellSizes[i] = avarage_length;
340
341 mInverseOfCellSize[i] = 1.00 / mCellSizes[i];
342 }

347 template<typename TIteratorType>
348 void AddObjectsToCells(TIteratorType GeometricalObjectsBegin, TIteratorType GeometricalObjectsEnd) {
349 for(auto i_geometrical_object = GeometricalObjectsBegin ; i_geometrical_object != GeometricalObjectsEnd ; i_geometrical_object++){
350 array_1d<std::size_t, 3> min_position(3,0);
351 array_1d<std::size_t, 3> max_position(3,0);
352 CalculateMinMaxPositions(i_geometrical_object->GetGeometry(), min_position, max_position);
353 for(std::size_t k = min_position[2] ; k < max_position[2] ; k++){
354 for(std::size_t j = min_position[1] ; j < max_position[1] ; j++){
355 for(std::size_t i = min_position[0] ; i < max_position[0] ; i++){
356 auto cell_bounding_box = GetCellBoundingBox(i,j,k);
357 if(IsIntersected(i_geometrical_object->GetGeometry(), cell_bounding_box, Tolerance)){
358 GetCell(i,j,k).push_back(&(*i_geometrical_object));
359 }
360 }
361 }
362 }

367 template<typename TGeometryType>
368 void CalculateMinMaxPositions(TGeometryType const& TheGeometry, array_1d< std::size_t, 3 >& MinPosition, array_1d< std::size_t, 3 >& MaxPosition){
369 if(TheGeometry.empty())
370 return;
371
372 BoundingBox<Point> bounding_box(TheGeometry.begin(), TheGeometry.end());
373
374 for(int i = 0; i < 3; i++ ) {
375 MinPosition[ i ] = CalculatePosition( bounding_box.GetMinPoint()[i], i );
376 MaxPosition[ i ] = CalculatePosition( bounding_box.GetMaxPoint()[i], i ) + 1;

381 std::size_t CalculatePosition( double Coordinate, int ThisDimension ) const {
382 auto distance = Coordinate - mBoundingBox.GetMinPoint()[ ThisDimension ];
383 distance = ( distance < 0.00 ) ? 0.00 : distance;
384 std::size_t position =
385 static_cast< std::size_t >( distance * mInverseOfCellSize[ ThisDimension ] );
386 std::size_t result= ( position > mNumberOfCells[ ThisDimension ] - 1 )
387 ? mNumberOfCells[ ThisDimension ] - 1
388 : position;
389 return result;
390 }

394 static inline bool IsIntersected(TGeometryType& TheGeometry, BoundingBox<Point> Box, const double tolerance)
395 {
396 Point rLowPointTolerance;
397 Point rHighPointTolerance;
398
399 for(std::size_t i = 0; i<3; i++)
400 {
401 rLowPointTolerance[i] = Box.GetMinPoint()[i] - tolerance;
402 rHighPointTolerance[i] = Box.GetMaxPoint()[i] + tolerance;
403 }
