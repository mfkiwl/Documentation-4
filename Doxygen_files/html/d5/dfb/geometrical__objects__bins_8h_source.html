319  void CalculateCellSize(std::size_t NumberOfCells) {
320  std::size_t avarage_number_of_cells = static_cast<std::size_t>(std::pow(static_cast<double>(NumberOfCells), 1.00 / Dimension));
321  std::array<double, 3> lengths;
322  double avarage_length = 0.00;
323  for (int i = 0; i < Dimension; i++) {
324  lengths[i] = mBoundingBox.GetMaxPoint()[i] - mBoundingBox.GetMinPoint()[i];
325  avarage_length += lengths[i];
326  }
327  avarage_length *= 1.00 / 3.00;
328 
329  if (avarage_length < std::numeric_limits<double>::epsilon()) {
330  mNumberOfCells = ScalarVector(3, 1);
331  return;
332  }
333 
334  for (int i = 0; i < Dimension; i++) {
335  mNumberOfCells[i] = static_cast<std::size_t>(lengths[i] / avarage_length * avarage_number_of_cells) + 1;
336  if (mNumberOfCells[i] > 1)
337  mCellSizes[i] = lengths[i] / mNumberOfCells[i];
338  else
339  mCellSizes[i] = avarage_length;
340 
341  mInverseOfCellSize[i] = 1.00 / mCellSizes[i];
342  }
