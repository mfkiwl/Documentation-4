114 static void CalculatePlasticPotentialDerivative(
115 const array_1d<double, VoigtSize>& rPredictiveStressVector,
116 const array_1d<double, VoigtSize>& rDeviator,
117 const double J2,
118 array_1d<double, VoigtSize>& rGFlux,
119 ConstitutiveLaw::Parameters& rValues
120 )
121 {
122 array_1d<double, VoigtSize> first_vector, second_vector, third_vector;
123 const Properties& r_material_properties = rValues.GetMaterialProperties();
124 const double dilatancy = r_material_properties[DILATANCY_ANGLE] * Globals::Pi / 180.0;
125
126 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateFirstVector(first_vector);
127 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateSecondVector(rDeviator, J2, second_vector);
128 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateThirdVector(rDeviator, J2, third_vector);
129
130 double J3, lode_angle;
131 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(rDeviator, J3);
132 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
133
134 double c1, c3, c2;
135 double checker = std::abs(lode_angle * 180.0 / Globals::Pi);
136
137 if (std::abs(checker) < 29.0) {
138 c1 = std::sin(dilatancy);
139 c3 = (std::sqrt(3.0) * std::sin(lode_angle) + std::sin(dilatancy) * std::cos(lode_angle)) /
140 (2.0 * J2 * std::cos(3.0 * lode_angle));
141 c2 = 0.5 * std::cos(lode_angle)*(1.0 + std::tan(lode_angle) * std::sin(3.0 * lode_angle) +
142 std::sin(dilatancy) * (std::tan(3.0 * lode_angle) - std::tan(lode_angle)) / std::sqrt(3.0));
143 } else { // smoothing with drucker-praguer
144 c1 = 3.0 * (2.0 * std::sin(dilatancy) / (std::sqrt(3.0) * (3.0 - std::sin(dilatancy))));
145 c2 = 1.0;
146 c3 = 0.0;
147 }
148
149 noalias(rGFlux) = c1 * first_vector + c2 * second_vector + c3 * third_vector;
150 }
