170 double GetConditionNumber(
171 SparseMatrixType& rInputMatrix,
172 LinearSolverType::Pointer pEigenSolverMax,
173 LinearSolverType::Pointer pEigenSolverMin
174 )
175 {
176 // The eigen system
177 DenseVectorType eigen_values;
178 DenseMatrixType eigen_vectors;
179
180 const SizeType size_matrix = rInputMatrix.size1();
181
182 SparseMatrixType identity_matrix(size_matrix, size_matrix);
183 for (IndexType i = 0; i < size_matrix; ++i)
184 identity_matrix.push_back(i, i, 1.0);
185
186 pEigenSolverMax->Solve(rInputMatrix, identity_matrix, eigen_values, eigen_vectors);
187 const double max_lambda = eigen_values[0];
188
189 pEigenSolverMin->Solve(rInputMatrix, identity_matrix, eigen_values, eigen_vectors);
190 const double min_lambda = eigen_values[0];
191
192 KRATOS_ERROR_IF(min_lambda < std::numeric_limits<double>::epsilon()) << "ERROR:: NOT POSSIBLE TO COMPUTE CONDITION NUMBER. ZERO EIGENVALUE" << std::endl;
193
194 const double condition_number = std::abs(max_lambda)/std::abs(min_lambda);
195
196 return condition_number;
197 }
