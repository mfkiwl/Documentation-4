159 static void IntegrateStressVector(
160 array_1d<double, VoigtSize>& rPredictiveStressVector,
161 Vector& rStrainVector,
162 double& rUniaxialStress,
163 double& rThreshold,
164 double& rPlasticDenominator,
165 array_1d<double, VoigtSize>& rFflux,
166 array_1d<double, VoigtSize>& rGflux,
167 double& rPlasticDissipation,
168 array_1d<double, VoigtSize>& rPlasticStrainIncrement,
169 Matrix& rConstitutiveMatrix,
170 Vector& rPlasticStrain,
171 ConstitutiveLaw::Parameters& rValues,
172 const double CharacteristicLength
173 )
174 {
175 // Material properties
176 const Properties& r_material_properties = rValues.GetMaterialProperties();
177
178 bool is_converged = false;
179 IndexType iteration = 0, max_iter = r_material_properties.Has(MAX_NUMBER_NL_CL_ITERATIONS)
180 ? r_material_properties.GetValue(MAX_NUMBER_NL_CL_ITERATIONS) : 100;
181 array_1d<double, VoigtSize> delta_sigma;
182 double plastic_consistency_factor_increment;
183 double F = rUniaxialStress - rThreshold;
184 Matrix tangent_tensor = ZeroMatrix(6,6);
185
186 // Backward Euler
187 while (is_converged == false && iteration <= max_iter) {
188 plastic_consistency_factor_increment = F * rPlasticDenominator;
189 if (plastic_consistency_factor_increment < 0.0) plastic_consistency_factor_increment = 0.0; // NOTE: It could be useful, maybe
190 noalias(rPlasticStrainIncrement) = plastic_consistency_factor_increment * rGflux;
191 noalias(rPlasticStrain) += rPlasticStrainIncrement;
192 noalias(delta_sigma) = prod(rConstitutiveMatrix, rPlasticStrainIncrement);
193
194 noalias(rPredictiveStressVector) -= delta_sigma;
195
196 F = CalculatePlasticParameters(rPredictiveStressVector, rStrainVector, rUniaxialStress, rThreshold,
197 rPlasticDenominator, rFflux, rGflux, rPlasticDissipation, rPlasticStrainIncrement,
198 rConstitutiveMatrix, rValues, CharacteristicLength, rPlasticStrain);
199
200 if (F <= std::abs(1.0e-4 * rThreshold)) { // Has converged
201 is_converged = true;
202 } else {
203 iteration++;
204 }
205 }
206 CalculateTangentMatrix(tangent_tensor, rConstitutiveMatrix, rFflux, rGflux, rPlasticDenominator);
207 noalias(rConstitutiveMatrix) = tangent_tensor;
208 KRATOS_WARNING_IF("GenericConstitutiveLawIntegratorPlasticity", iteration > max_iter) << "Maximum number of iterations in plasticity loop reached..." << std::endl;
209 }

326 static void CalculateIndicatorsFactors(
327 const array_1d<double, VoigtSize>& rPredictiveStressVector,
328 double& rTensileIndicatorFactor,
329 double& rCompressionIndicatorFactor
330 )
331 {
332 // We do an initial check
333 if (norm_2(rPredictiveStressVector) < 1.0e-8) {
334 rTensileIndicatorFactor = 1.0;
335 rCompressionIndicatorFactor = 0.0;
336 return;
337 }
338
339 // We proceed as usual
340 array_1d<double, Dimension> principal_stresses = ZeroVector(Dimension);
341 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculatePrincipalStresses(principal_stresses, rPredictiveStressVector);
342
343 double suma = 0.0, sumb = 0.0, sumc = 0.0;
344 double aux_sa;
345
346 for (IndexType i = 0; i < Dimension; ++i) {
347 aux_sa = std::abs(principal_stresses[i]);
348 suma += aux_sa;
349 sumb += 0.5 * (principal_stresses[i] + aux_sa);
350 sumc += 0.5 * (-principal_stresses[i] + aux_sa);
351 }
352
353 if (std::abs(suma) > tolerance) {
354 rTensileIndicatorFactor = sumb / suma;
355 rCompressionIndicatorFactor = sumc / suma;
356 } else {
357 rTensileIndicatorFactor = sumb;
358 rCompressionIndicatorFactor = sumc;
359 }
360
361 // Final check
362 if ((std::abs(rTensileIndicatorFactor) + std::abs(rCompressionIndicatorFactor)) < tolerance) {
363 rTensileIndicatorFactor = 0.0;
364 rCompressionIndicatorFactor = 0.0;
365 return;
366 }
367 }

380 static void CalculatePlasticDissipation(
381 const array_1d<double, VoigtSize>& rPredictiveStressVector,
382 const double TensileIndicatorFactor,
383 const double CompressionIndicatorFactor,
384 const Vector& PlasticStrainInc,
385 double& rPlasticDissipation,
386 array_1d<double, VoigtSize>& rHCapa,
387 ConstitutiveLaw::Parameters& rValues,
388 const double CharacteristicLength
389 )
390 {
391 const Properties& r_material_properties = rValues.GetMaterialProperties();
392
393 const double young_modulus = r_material_properties[YOUNG_MODULUS];
394 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
395 const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
396 const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
397 const double n = yield_compression / yield_tension;
398 const double fracture_energy_tension = r_material_properties[FRACTURE_ENERGY]; // Frac energy in tension
399 const double fracture_energy_compression = r_material_properties[FRACTURE_ENERGY] * std::pow(n, 2); // Frac energy in compression
400
401 const double characteristic_fracture_energy_tension = fracture_energy_tension / CharacteristicLength;
402 const double characteristic_fracture_energy_compression = fracture_energy_compression / CharacteristicLength;
403
404 const double hlim = 2.0 * young_modulus * fracture_energy_compression / (std::pow(yield_compression, 2));
405 KRATOS_ERROR_IF(CharacteristicLength > hlim) << "The Fracture Energy is to low: " << characteristic_fracture_energy_compression << std::endl;
406
407 double constant0 = 0.0, constant1 = 0.0, dplastic_dissipation = 0.0;
408 if (characteristic_fracture_energy_tension > 0.000001) {
409 constant0 = TensileIndicatorFactor / characteristic_fracture_energy_tension;
410 constant1 = CompressionIndicatorFactor / characteristic_fracture_energy_compression;
411 }
412 const double constant = constant0 + constant1;
413
414 for (IndexType i = 0; i < VoigtSize; ++i) {
415 rHCapa[i] = constant * rPredictiveStressVector[i];
416 dplastic_dissipation += rHCapa[i] * PlasticStrainInc[i];
417 }
418
419 if (dplastic_dissipation < 0.0 || dplastic_dissipation > 1.0)
420 dplastic_dissipation = 0.0;
421
422 rPlasticDissipation += dplastic_dissipation;
423 if (rPlasticDissipation >= 0.9999)
424 rPlasticDissipation = 0.9999;
425 else if (rPlasticDissipation < 0.0)
426 rPlasticDissipation = 0.0;
427
428 // We add a check
429 KRATOS_DEBUG_ERROR_IF(std::isnan(rPlasticDissipation)) << "rPlasticDissipation is nan" << std::endl;
430 }

443 static void CalculateEquivalentStressThreshold(
444 const double PlasticDissipation,
445 const double TensileIndicatorFactor,
446 const double CompressionIndicatorFactor,
447 double& rEquivalentStressThreshold,
448 double& rSlope,
449 ConstitutiveLaw::Parameters& rValues,
450 const double EquivalentPlasticStrain,
451 const double CharacteristicLength
452 )
453 {
454 const Properties& r_material_properties = rValues.GetMaterialProperties();
455 const int curve_type = r_material_properties[HARDENING_CURVE];
456 BoundedVector<double, 2> slopes, eq_thresholds;
457
458 for (IndexType i = 0; i < 2; ++i) { // i:0 Tension ; i:1 compression
459 switch (static_cast<HardeningCurveType>(curve_type))
460 {
461 case HardeningCurveType::LinearSoftening:
462 CalculateEquivalentStressThresholdHardeningCurveLinearSoftening(
463 PlasticDissipation, TensileIndicatorFactor,
464 CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
465 rValues);
466 break;
467
468 case HardeningCurveType::ExponentialSoftening:
469 CalculateEquivalentStressThresholdHardeningCurveExponentialSoftening(
470 PlasticDissipation, TensileIndicatorFactor,
471 CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
472 rValues, CharacteristicLength);
473 break;
474
475 case HardeningCurveType::InitialHardeningExponentialSoftening:
476 CalculateEquivalentStressThresholdHardeningCurveInitialHardeningExponentialSoftening(
477 PlasticDissipation, TensileIndicatorFactor,
478 CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
479 rValues);
480 break;
481
482 case HardeningCurveType::PerfectPlasticity:
483 CalculateEquivalentStressThresholdHardeningCurvePerfectPlasticity(
484 PlasticDissipation, TensileIndicatorFactor,
485 CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
486 rValues);
487 break;
488
489 case HardeningCurveType::CurveFittingHardening: // Only for VonMises + Tresca!
490 CalculateEquivalentStressThresholdCurveFittingHardening(
491 PlasticDissipation, TensileIndicatorFactor,
492 CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
493 rValues, EquivalentPlasticStrain, CharacteristicLength);
494 break;
495
496 case HardeningCurveType::LinearExponentialSoftening:
497 CalculateEquivalentStressThresholdHardeningCurveLinearExponentialSoftening(
498 PlasticDissipation, TensileIndicatorFactor,
499 CompressionIndicatorFactor, eq_thresholds[i], slopes[i], CharacteristicLength,
500 rValues);
501 break;
502
503 case HardeningCurveType::CurveDefinedByPoints:
504 CalculateEquivalentStressThresholdHardeningCurveDefinedByPoints(
505 PlasticDissipation, TensileIndicatorFactor,
506 CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
507 rValues, CharacteristicLength);
508 break;
509
510 // Add more cases...
511 default:
512 KRATOS_ERROR << " The HARDENING_CURVE of plasticity is not set or wrong..." << curve_type << std::endl;
513 break;
514 }
515 }
516
517 rEquivalentStressThreshold = TensileIndicatorFactor * eq_thresholds[0] + CompressionIndicatorFactor * eq_thresholds[1];
518 rSlope = rEquivalentStressThreshold * ((TensileIndicatorFactor * slopes[0] / eq_thresholds[0]) + (CompressionIndicatorFactor * slopes[1] / eq_thresholds[1]));
519 }

530 static void CalculateEquivalentStressThresholdHardeningCurveLinearSoftening(
531 const double PlasticDissipation,
532 const double TensileIndicatorFactor,
533 const double CompressionIndicatorFactor,
534 double& rEquivalentStressThreshold,
535 double& rSlope,
536 ConstitutiveLaw::Parameters& rValues
537 )
538 {
539 const Properties& r_material_properties = rValues.GetMaterialProperties();
540 const bool has_plastic_dissipation_limit = r_material_properties.Has(PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING);
541 const double plastic_dissipation_limit = has_plastic_dissipation_limit ? r_material_properties[PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING] : 0.99;
542 double initial_threshold;
543 GetInitialUniaxialThreshold(rValues, initial_threshold);
544
545 if (PlasticDissipation <= plastic_dissipation_limit){ //Linear branch
546 rEquivalentStressThreshold = initial_threshold * std::sqrt(1.0 - PlasticDissipation);
547 rSlope = -0.5 * (std::pow(initial_threshold, 2.0) / (rEquivalentStressThreshold));
548 } else { //Exponential branch included to achieve consistent results after full plasticity scenarios
549 rEquivalentStressThreshold = (initial_threshold / std::sqrt(1.0 - plastic_dissipation_limit)) * (1.0 - PlasticDissipation);
550 rSlope = - (initial_threshold / std::sqrt(1.0 - plastic_dissipation_limit));
551 }
552 }

564 static void CalculateEquivalentStressThresholdHardeningCurveExponentialSoftening(
565 const double PlasticDissipation,
566 const double TensileIndicatorFactor,
567 const double CompressionIndicatorFactor,
568 double& rEquivalentStressThreshold,
569 double& rSlope,
570 ConstitutiveLaw::Parameters& rValues,
571 const double CharacteristicLength
572 )
573 {
574 const Properties& r_material_properties = rValues.GetMaterialProperties();
575
576 const double young_modulus = r_material_properties[YOUNG_MODULUS];
577 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
578 const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];

589 rEquivalentStressThreshold = initial_threshold * (1.0 - PlasticDissipation);
590 rSlope = - initial_threshold;

602 static void CalculateEquivalentStressThresholdHardeningCurveInitialHardeningExponentialSoftening(
603 const double PlasticDissipation,
604 const double TensileIndicatorFactor,
605 const double CompressionIndicatorFactor,
606 double& rEquivalentStressThreshold,
607 double& rSlope,
608 ConstitutiveLaw::Parameters& rValues
609 )
610 {
611 const Properties& r_material_properties = rValues.GetMaterialProperties();
612
613 double initial_threshold;
614 GetInitialUniaxialThreshold(rValues, initial_threshold);
615 const double ultimate_stress = r_material_properties[MAXIMUM_STRESS]; // sikpi
616 const double max_stress_position = r_material_properties[MAXIMUM_STRESS_POSITION]; // cappi
617
618 if (PlasticDissipation < 1.0) {
619 const double ro = std::sqrt(1.0 - initial_threshold / ultimate_stress);
620 double alpha = std::log((1.0 - (1.0 - ro) * (1.0 - ro)) / ((3.0 - ro) * (1.0 + ro) * max_stress_position));
621 alpha = std::exp(alpha / (1.0 - max_stress_position));
622 const double phi = std::pow((1.0 - ro), 2.0) + ((3.0 - ro) * (1.0 + ro) * PlasticDissipation * (std::pow(alpha, (1.0 - PlasticDissipation))));
623
624 rEquivalentStressThreshold = ultimate_stress * (2.0 * std::sqrt(phi) - phi);
625 rSlope = ultimate_stress * ((1.0 / std::sqrt(phi)) - 1.0) * (3.0 - ro) * (1.0 + ro) * (std::pow(alpha, (1.0 - PlasticDissipation))) *
626 (1.0 - std::log(alpha) * PlasticDissipation);
627 } else {
628 KRATOS_ERROR << "PlasticDissipation > 1.0 " << PlasticDissipation << std::endl;
629 }
630 }

668 static void CalculateEquivalentStressThresholdCurveFittingHardening(
669 const double PlasticDissipation,
670 const double TensileIndicatorFactor,
671 const double CompressionIndicatorFactor,
672 double& rEquivalentStressThreshold,
673 double& rSlope,
674 ConstitutiveLaw::Parameters& rValues,
675 const double EquivalentPlasticStrain,
676 const double CharacteristicLength
677 )
678 {
679 const Properties& r_material_properties = rValues.GetMaterialProperties();
680 const Vector& curve_fitting_parameters = r_material_properties[CURVE_FITTING_PARAMETERS];
681
682 const bool has_tangency_linear_region = r_material_properties.Has(TANGENCY_REGION2);
683 const bool tangency_linear_region = has_tangency_linear_region ? r_material_properties[TANGENCY_REGION2] : false;
684
685 const Vector& plastic_strain_indicators = r_material_properties[PLASTIC_STRAIN_INDICATORS];
686 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
687 const double volumetric_fracture_energy = fracture_energy / CharacteristicLength;
688
689 const SizeType order_polinomial = curve_fitting_parameters.size();
690 const double plastic_strain_indicator_1 = plastic_strain_indicators[0];
691 const double plastic_strain_indicator_2 = plastic_strain_indicators[1];
692
693 // Compute the initial and the final stresses
694 double stress_indicator_1 = curve_fitting_parameters[0];
695 double dS_dEp = 0.0;
696
697 for (IndexType i = 1; i < order_polinomial; ++i) {
698 stress_indicator_1 += curve_fitting_parameters[i] * (std::pow(plastic_strain_indicator_1, i));
699 dS_dEp += i * curve_fitting_parameters[i] * std::pow(plastic_strain_indicator_1, i - 1);
700 }
701
702 double dKp_dEp = stress_indicator_1 / volumetric_fracture_energy;
703 if (!tangency_linear_region){
704 dS_dEp = 0.0; // initial slope is zero, else the slope is the tangent of the polinomial region.
705 }
706
707 const double stress_indicator_2 = stress_indicator_1 + dS_dEp * (plastic_strain_indicator_2 - plastic_strain_indicator_1);
708
709 // Compute volumetric fracture energies of each region
710 double Gt1 = 0.0;
711 for (IndexType i = 0; i < order_polinomial; ++i) {
712 Gt1 += curve_fitting_parameters[i] * (std::pow(plastic_strain_indicator_1, i + 1)) / (i + 1);
713 }
714 const double Gt2 = (stress_indicator_1 + stress_indicator_2) * (plastic_strain_indicator_2 - plastic_strain_indicator_1) * 0.5;
715 const double Gt3 = volumetric_fracture_energy - Gt2 - Gt1;
716
717 KRATOS_ERROR_IF(Gt3 < 0.0) << "Fracture energy too low in CurveFittingHardening of plasticity..." << std::endl;
718
719 // Compute segment threshold
720 const double segment_threshold = (Gt2 + Gt1) / volumetric_fracture_energy;
721
722 if (PlasticDissipation <= segment_threshold) {
723 const double Eps = EquivalentPlasticStrain;
724
725 if (EquivalentPlasticStrain < plastic_strain_indicator_1) { // Polinomial region
726 double S_Ep = curve_fitting_parameters[0];
727 double dS_dEp = 0.0;
728 for (IndexType i = 1; i < order_polinomial; ++i) {
729 S_Ep += curve_fitting_parameters[i] * std::pow(Eps, i);
730 dS_dEp += i * curve_fitting_parameters[i] * std::pow(Eps, i - 1);
731 }
732 dKp_dEp = S_Ep / volumetric_fracture_energy;
733
734 rEquivalentStressThreshold = S_Ep;
735 rSlope = dS_dEp / dKp_dEp;
736 } else { // Linear region
737 const double S_Ep = stress_indicator_1 + (stress_indicator_2 - stress_indicator_1) / (plastic_strain_indicator_2 - plastic_strain_indicator_1) * (Eps - plastic_strain_indicator_1);
738 double dS_dEp = (stress_indicator_2 - stress_indicator_1) / (plastic_strain_indicator_2 - plastic_strain_indicator_1);
739 dKp_dEp = S_Ep / volumetric_fracture_energy;
740
741 rEquivalentStressThreshold = S_Ep;
742 rSlope = dS_dEp / dKp_dEp;
743 }
744 } else { // Exponential softening
745 const double Eps = EquivalentPlasticStrain;
746 const double alpha = std::pow(stress_indicator_1, 2);
747 const double beta = (std::pow(stress_indicator_2, 2) - alpha) / (plastic_strain_indicator_2 - plastic_strain_indicator_1);
748
749 const double S_Ep = std::sqrt(alpha + beta * (Eps - plastic_strain_indicator_1));
750 const double plastic_dissipation_region_3 = PlasticDissipation - segment_threshold;
751
752 const double beta2 = 1.5 * S_Ep / Gt3;
753 const double alpha2 = std::sqrt((plastic_dissipation_region_3 * 2.0 * beta2 * volumetric_fracture_energy / S_Ep) + 1.0);
754 rEquivalentStressThreshold = S_Ep * alpha2 * (2.0 - alpha2);
755 rSlope = 2.0 * beta2 * volumetric_fracture_energy * (1.0 / alpha2 - 1.0);
756 }
757
758 }

770 static void CalculateEquivalentStressThresholdHardeningCurveLinearExponentialSoftening(
771 const double PlasticDissipation,
772 const double TensileIndicatorFactor,
773 const double CompressionIndicatorFactor,
774 double& rEquivalentStressThreshold,
775 double& rSlope,
776 const double CharacteristicLength,
777 ConstitutiveLaw::Parameters& rValues
778 )
779 {
780 const Properties& r_material_properties = rValues.GetMaterialProperties();
781 const bool has_plastic_dissipation_limit = r_material_properties.Has(PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING);
782
783 const double plastic_dissipation_limit = has_plastic_dissipation_limit ? r_material_properties[PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING] : 0.9;
784 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
785 const double volumetric_fracture_energy = fracture_energy / CharacteristicLength;
786 double initial_threshold;
787 GetInitialUniaxialThreshold(rValues, initial_threshold);
788
789 const double volumetric_fracture_energy_linear_branch = 0.5 * volumetric_fracture_energy * (plastic_dissipation_limit + 1.0);
790
791 if (PlasticDissipation <= plastic_dissipation_limit){ //Linear branch
792 rEquivalentStressThreshold = initial_threshold * std::sqrt(1.0 - PlasticDissipation * volumetric_fracture_energy / volumetric_fracture_energy_linear_branch);
793 rSlope = - 0.5 * initial_threshold * (volumetric_fracture_energy / volumetric_fracture_energy_linear_branch) * std::pow(1.0 - PlasticDissipation * volumetric_fracture_energy / volumetric_fracture_energy_linear_branch, -0.5);
794 } else { //Exponential branch included to achieve consistent results after full plasticity scenarios
795 const double volumetric_fracture_energy_exponential_branch = volumetric_fracture_energy * (1.0 - plastic_dissipation_limit) * std::exp((plastic_dissipation_limit + 1.0) / (std::sqrt(1.0 - std::pow(plastic_dissipation_limit, 2.0))) - 1.0);
796 const double initial_threshold_exponential = initial_threshold * volumetric_fracture_energy_exponential_branch / volumetric_fracture_energy * std::sqrt(1.0 - plastic_dissipation_limit * volumetric_fracture_energy / volumetric_fracture_energy_linear_branch) / (1.0 - plastic_dissipation_limit);
797 rEquivalentStressThreshold = initial_threshold_exponential * (1.0 - PlasticDissipation) * volumetric_fracture_energy / volumetric_fracture_energy_exponential_branch;
798 rSlope = - initial_threshold_exponential * volumetric_fracture_energy / volumetric_fracture_energy_exponential_branch;
799 }
800 }

814 static void CalculateEquivalentStressThresholdHardeningCurveDefinedByPoints(
815 const double PlasticDissipation,
816 const double TensileIndicatorFactor,
817 const double CompressionIndicatorFactor,
818 double& rEquivalentStressThreshold,
819 double& rSlope,
820 ConstitutiveLaw::Parameters& rValues,
821 const double CharacteristicLength
822 )
823 {
824 const Properties& r_material_properties = rValues.GetMaterialProperties();
825 const Vector& equivalent_stress_vector = r_material_properties[EQUIVALENT_STRESS_VECTOR_PLASTICITY_POINT_CURVE];
826 const Vector& plastic_strain_vector = r_material_properties[PLASTIC_STRAIN_VECTOR_PLASTICITY_POINT_CURVE];
827 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
828 const double volumetric_fracture_energy = fracture_energy / CharacteristicLength;
829 const SizeType points_hardening_curve = equivalent_stress_vector.size();
830
831 // Compute volumetric fracture energies of each region
832 double Gt1 = 0.0;
833 for (IndexType i = 1; i < points_hardening_curve; ++i) {
834 Gt1 += 0.5 * (equivalent_stress_vector(i - 1) + equivalent_stress_vector(i)) * (plastic_strain_vector(i) - plastic_strain_vector(i - 1));
835 }
836 const double Gt2 = volumetric_fracture_energy - Gt1;
837
838 KRATOS_ERROR_IF(Gt2 < 0.0) << "Fracture energy too low in CurveDefinedByPoints of plasticity..." << std::endl;
839
840 // Compute segment threshold
841 const double segment_threshold = (Gt1) / volumetric_fracture_energy;
842 if (PlasticDissipation < segment_threshold) {
843 IndexType i = 0;
844 double gf_point_region = 0.0;
845 double plastic_dissipation_previous_point = 0.0;
846 while (PlasticDissipation >= gf_point_region / volumetric_fracture_energy) {
847 i += 1;
848 plastic_dissipation_previous_point = gf_point_region / volumetric_fracture_energy;
849 gf_point_region += 0.5 * (equivalent_stress_vector(i - 1) + equivalent_stress_vector(i)) * (plastic_strain_vector(i) - plastic_strain_vector(i - 1));
850 }
851 const double plastic_dissipation_next_point = gf_point_region / volumetric_fracture_energy;
852
853 //Stress is computed using an equivalent equation to the one used for the linear softening curve, i.e. f(S) = a * (1.0 - b * kp)
854 const double b = (std::pow(equivalent_stress_vector(i), 2.0) - std::pow(equivalent_stress_vector(i - 1), 2.0)) / (plastic_dissipation_previous_point * std::pow(equivalent_stress_vector(i), 2.0) - plastic_dissipation_next_point * std::pow(equivalent_stress_vector(i - 1), 2.0));
855 const double a = equivalent_stress_vector(i - 1) / std::sqrt(1.0 - b * plastic_dissipation_previous_point);
856 rEquivalentStressThreshold = a * std::sqrt(1.0 - b * PlasticDissipation);
857 rSlope = - 0.5 * std::pow(a, 2.0) * b / rEquivalentStressThreshold;
858
859 } else { //Exponential branch included to achieve consistent results after full plasticity scenarios
860 const double b = equivalent_stress_vector(points_hardening_curve - 1) / (1.0 - segment_threshold);
861 const double a = b;
862 rEquivalentStressThreshold = a - b * PlasticDissipation;
863 rSlope = - b;
864 }
865 }

966 static int Check(const Properties& rMaterialProperties)
967 {
968 // Checking is defined
969 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "HARDENING_CURVE is not a defined value" << std::endl;
970 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(HARDENING_CURVE)) << "HARDENING_CURVE is not a defined value" << std::endl;
971 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRACTURE_ENERGY)) << "FRACTURE_ENERGY is not a defined value" << std::endl;
972
973 // Checking curves
974 const int curve_type = rMaterialProperties[HARDENING_CURVE];
975 if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::InitialHardeningExponentialSoftening) {
976 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS)) << "MAXIMUM_STRESS is not a defined value" << std::endl;
977 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS_POSITION)) << "MAXIMUM_STRESS_POSITION is not a defined value" << std::endl;
978 } else if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::CurveFittingHardening) {
979 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(CURVE_FITTING_PARAMETERS)) << "CURVE_FITTING_PARAMETERS is not a defined value" << std::endl;
980 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(PLASTIC_STRAIN_INDICATORS)) << "PLASTIC_STRAIN_INDICATORS is not a defined value" << std::endl;
981 }
982
983 if (!rMaterialProperties.Has(YIELD_STRESS)) {
984 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
985 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
986
987 const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
988 const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
989
990 KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
991 KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
992 } else {
993 const double yield_stress = rMaterialProperties[YIELD_STRESS];
994
995 KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
996 }
997
998 return TYieldSurfaceType::Check(rMaterialProperties);
999 }
