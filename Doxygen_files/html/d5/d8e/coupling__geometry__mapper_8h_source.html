81// CouplingGeometryMapper
82//
83// The mapper always forward maps from the master to the slave.
84// Normally:
85// master = interface origin
86// slave = interface destination
87//
88// However, this can be reversed by setting 'destination_is_slave' = false.
89// This yields:
90// master = interface destination
91// slave = interface origin

159 void Map(
160 const Variable<double>& rOriginVariable,
161 const Variable<double>& rDestinationVariable,
162 Kratos::Flags MappingOptions) override
163 {
164 if (MappingOptions.Is(MapperFlags::USE_TRANSPOSE)) {
165 MappingOptions.Reset(MapperFlags::USE_TRANSPOSE);
166 MappingOptions.Set(MapperFlags::INTERNAL_USE_TRANSPOSE, true);
167 GetInverseMapper()->Map(rDestinationVariable, rOriginVariable, MappingOptions);
168 }
169 else if (MappingOptions.Is(MapperFlags::INTERNAL_USE_TRANSPOSE)) {
170 MapInternalTranspose(rOriginVariable, rDestinationVariable, MappingOptions);
171 }
172 else {
173 MapInternal(rOriginVariable, rDestinationVariable, MappingOptions);
174 }
