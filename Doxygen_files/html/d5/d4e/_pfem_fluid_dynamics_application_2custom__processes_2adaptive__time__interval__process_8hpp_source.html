102 {
103
104 KRATOS_TRY
105 KRATOS_INFO("AdaptiveTimeIntervalProcess") << " Execute() " << std::endl;
106
107 ProcessInfo &rCurrentProcessInfo = mrModelPart.GetProcessInfo();
108
109 const double initialTimeInterval = rCurrentProcessInfo[INITIAL_DELTA_TIME];
110 const double currentTimeInterval = rCurrentProcessInfo[CURRENT_DELTA_TIME];
111 double updatedTime = rCurrentProcessInfo[TIME];
112 double updatedTimeInterval = rCurrentProcessInfo[DELTA_TIME];
113 double deltaTimeToNewMilestone = initialTimeInterval;
114 double minimumTimeInterval = initialTimeInterval * 0.0001;
115 rCurrentProcessInfo.SetValue(PREVIOUS_DELTA_TIME, currentTimeInterval);
116
117 bool milestoneTimeReached = true;
118 bool increaseTimeInterval = true;
119 bool timeIntervalReduced = rCurrentProcessInfo[TIME_INTERVAL_CHANGED];
120
121 unsigned int &stepsWithChangedDt = rCurrentProcessInfo[STEPS_WITH_CHANGED_DT];
122
123 if (stepsWithChangedDt == 2)
124 {
125 if (timeIntervalReduced == false)
126 {
127 stepsWithChangedDt = 0;
128 // std::cout << " 2 steps with the same reduced step " << stepsWithChangedDt << std::endl;
129 }
130 else
131 {
132 stepsWithChangedDt = 1;
133 // std::cout << " 2 steps with the same reduced step but I have reduced the dt again" << stepsWithChangedDt << std::endl;
134 }
135 }
136 timeIntervalReduced = false;
137 rCurrentProcessInfo.SetValue(TIME_INTERVAL_CHANGED, false);
138
139 double tolerance = 0.0001;
140 updatedTime -= initialTimeInterval;
141 unsigned int previousMilestoneStep = updatedTime / initialTimeInterval;
142 deltaTimeToNewMilestone = initialTimeInterval * (previousMilestoneStep + 1) - updatedTime;
143
144 updatedTimeInterval = currentTimeInterval;
145
146 bool badVelocityConvergence = rCurrentProcessInfo[BAD_VELOCITY_CONVERGENCE];
147 bool badPressureConvergence = rCurrentProcessInfo[BAD_PRESSURE_CONVERGENCE];
148
149 if (updatedTimeInterval < 2.0 * minimumTimeInterval && mEchoLevel > 0 && mrModelPart.GetCommunicator().MyPID() == 0)
150 {
151 std::cout << "ATTENTION! time step much smaller than initial time step, I'll not reduce it" << std::endl;
152 }
153 //here the time step is reduced because at the previous step we obtained a bad convergence (in velocity or pressure), see the criterion in two_step_vp_strategy.h
154 if ((badPressureConvergence == true || badVelocityConvergence == true) && updatedTimeInterval > (2.0 * minimumTimeInterval))
155 {
156 updatedTimeInterval *= 0.5;
157 // std::cout<<" reducing time step (bad convergence at the previous step)"<<updatedTimeInterval<<std::endl;
158 rCurrentProcessInfo.SetValue(TIME_INTERVAL_CHANGED, true);
159 timeIntervalReduced = true;
160 }
161
162 //here the time step is changed to allow to reach the milestone time. This is done only if, using the new time step, there would be a small gap with respect to the milestone (writing) time
163 if (deltaTimeToNewMilestone < (1.0 + tolerance) * updatedTimeInterval && deltaTimeToNewMilestone > initialTimeInterval * tolerance)
164 {
165 rCurrentProcessInfo.SetValue(DELTA_TIME, deltaTimeToNewMilestone);
166 if (deltaTimeToNewMilestone < 0.75 * updatedTimeInterval)
167 {
168 timeIntervalReduced = true;
169 rCurrentProcessInfo.SetValue(TIME_INTERVAL_CHANGED, true);
170 }
171 updatedTimeInterval = deltaTimeToNewMilestone;
172 milestoneTimeReached = true;
173 }
174 else
175 {
176 milestoneTimeReached = false;
177 double ratioInitialTime=fabs(updatedTimeInterval-initialTimeInterval*0.5)/(initialTimeInterval*0.5);
178 if(ratioInitialTime<0.05){
179 updatedTimeInterval=initialTimeInterval*0.5;;
180 // std::cout<<"I clean the initial time interval ("<<updatedTimeInterval<<") because not so different from an half of the initial one="<<initialTimeInterval<<std::endl;
181 };
182 rCurrentProcessInfo.SetValue(DELTA_TIME, updatedTimeInterval);
183 }
184
185 //here it is checked node by node and element by element, if with the new time step there could be penetration or element inversion. In these cases, the time step is reduced
186 if (timeIntervalReduced == false)
187 {
188 if (updatedTimeInterval > (2.0 * minimumTimeInterval))
189 {
190
191 const unsigned int dimension = mrModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
192 if (dimension == 2)
193 {
194 // the time step is reduced if during a time step a node moves more than its mean distance with respect to its neighbors
195 CheckNodalCriterionForTimeStepReduction(updatedTimeInterval, increaseTimeInterval, timeIntervalReduced);
196 // if (timeIntervalReduced == false)
197 // {
198 // // the time step is reduced if there can be some element inversion
199 // CheckElementalCriterionForTimeStepReduction(increaseTimeInterval);
200 // }
201 }
202 }
203
204 // if(increaseTimeInterval==true && initialTimeInterval>(1.0+tolerance)*updatedTimeInterval && badPressureConvergence==false && badVelocityConvergence==false ){
205 if (stepsWithChangedDt == 0 && increaseTimeInterval == true && initialTimeInterval > (1.0 + tolerance) * updatedTimeInterval && badVelocityConvergence == false)
206 {
207 if(increaseTimeInterval == true && (stepsWithChangedDt == 0 || stepsWithChangedDt == 2)){
208 IncreaseTimeInterval(updatedTimeInterval, deltaTimeToNewMilestone, tolerance, increaseTimeInterval);
209 }
210 }
211 else
212 {
213 increaseTimeInterval = false;
214 }
215 }
216
217 double newTimeInterval = rCurrentProcessInfo[DELTA_TIME];
218 double milestoneGap = fabs(newTimeInterval - deltaTimeToNewMilestone);
219 if (milestoneGap < 0.49 * newTimeInterval && milestoneTimeReached == false)
220 {
221 // std::cout<<"the milestone is very close, I add "<<milestoneGap<<" to "<<newTimeInterval<<std::endl;
222 newTimeInterval += milestoneGap;
223 rCurrentProcessInfo.SetValue(DELTA_TIME, newTimeInterval);
224 milestoneTimeReached = true;
225 }
226
227 updatedTime += newTimeInterval;
228 rCurrentProcessInfo.SetValue(TIME, updatedTime);
229 rCurrentProcessInfo.SetValue(CURRENT_DELTA_TIME, newTimeInterval);
230
231 /* if(newTimeInterval<(0.49*currentTimeInterval)){ */
232 /* std::cout<<"ATTENTION! new time step is more than 2 times smaller than the previous one"<<std::endl; */
233 /* } */
234
235 if (increaseTimeInterval == false && milestoneTimeReached == true && fabs(newTimeInterval - initialTimeInterval) > tolerance && !(deltaTimeToNewMilestone > newTimeInterval * (1.0 + tolerance)))
236 {
237 rCurrentProcessInfo.SetValue(CURRENT_DELTA_TIME, currentTimeInterval);
238 }
239
240 if (newTimeInterval < initialTimeInterval)
241 {
242 KRATOS_INFO("AdaptiveTimeIntervalProcess") << "current time " << updatedTime << " time step: new " << newTimeInterval << " previous " << currentTimeInterval << " initial " << initialTimeInterval << std::endl;
243 }
244 if (stepsWithChangedDt == 0 && timeIntervalReduced == true)
245 {
246 stepsWithChangedDt += 1;
247 }
248 if (stepsWithChangedDt == 1 && timeIntervalReduced == false)
249 {
250 stepsWithChangedDt += 1;
251 }
252 if ((stepsWithChangedDt == 0 || stepsWithChangedDt == 1) && increaseTimeInterval == true && timeIntervalReduced == false)
253 {
254 // std::cout << " stepsWithChangedDt due to increase time Interval " << std::endl;
255 stepsWithChangedDt += 1;
256 }
257 // std::cout << " stepsWithChangedDt: " << stepsWithChangedDt << std::endl;
258 KRATOS_CATCH("");

261 void CheckNodalCriterionForTimeStepReduction(double updatedTimeInterval,
262 bool &increaseTimeInterval,
263 bool &timeIntervalReduced)
264 {
265
266 ProcessInfo &rCurrentProcessInfo = mrModelPart.GetProcessInfo();
267
268#pragma omp parallel
269 {
270 ModelPart::NodeIterator NodeBegin;
271 ModelPart::NodeIterator NodeEnd;
272 OpenMPUtils::PartitionedIterators(mrModelPart.Nodes(), NodeBegin, NodeEnd);
273 for (ModelPart::NodeIterator itNode = NodeBegin; itNode != NodeEnd; ++itNode)
274 {
275 if (itNode->IsNot(TO_ERASE) && itNode->IsNot(ISOLATED) && itNode->IsNot(SOLID))
276 {
277 const array_1d<double, 3> &Vel = itNode->FastGetSolutionStepValue(VELOCITY);
278 double NormVelNode = 0;
279 for (unsigned int d = 0; d < 3; ++d)
280 {
281 NormVelNode += Vel[d] * Vel[d];
282 }
283 double motionInStep = sqrt(NormVelNode) * updatedTimeInterval;
284 double unsafetyFactor = 0;
285 NodeWeakPtrVectorType &neighb_nodes = itNode->GetValue(NEIGHBOUR_NODES);
286 for (NodeWeakPtrVectorType::iterator nn = neighb_nodes.begin(); nn != neighb_nodes.end(); nn++)
287 {
288 array_1d<double, 3> CoorNeighDifference = itNode->Coordinates() - (nn)->Coordinates();
289 double squaredDistance = 0;
290 for (unsigned int d = 0; d < 3; ++d)
291 {
292 squaredDistance += CoorNeighDifference[d] * CoorNeighDifference[d];
293 }
294 double nodeDistance = sqrt(squaredDistance);
295 double tempUnsafetyFactor = motionInStep / nodeDistance;
296 if (tempUnsafetyFactor > unsafetyFactor)
297 {
298 unsafetyFactor = tempUnsafetyFactor;
299 }
300 }
301
302 if (unsafetyFactor > 0.35)
303 {
304 increaseTimeInterval = false;
305 if (unsafetyFactor > 1.0)
306 {
307 double temporaryTimeInterval = rCurrentProcessInfo[DELTA_TIME];
308 double reducedTimeInterval = 0.5 * updatedTimeInterval;
309 if (reducedTimeInterval < temporaryTimeInterval)
310 {
311 rCurrentProcessInfo.SetValue(DELTA_TIME, reducedTimeInterval);
312 // std::cout<<"reducing time step (nodal criterion)"<<reducedTimeInterval<<std::endl;
313 rCurrentProcessInfo.SetValue(TIME_INTERVAL_CHANGED, true);
314 timeIntervalReduced = true;
315 break;
316 }
317 }
318 }
319 }
320 }
