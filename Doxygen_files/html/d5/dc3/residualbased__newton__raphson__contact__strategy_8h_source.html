445  bool SolveSolutionStep() override
446  {
447  KRATOS_TRY;
448 
449 // bool is_converged = BaseType::SolveSolutionStep(); // FIXME: Requires to separate the non linear iterations
450 
451 // bool is_converged = BaseSolveSolutionStep(); // Direct solution
452  bool is_converged = false;
453 
454  // Getting model part
455  ModelPart& r_model_part = StrategyBaseType::GetModelPart();
456 
457  if (r_model_part.IsNot(INTERACTION)) {
458  // We get the system
459  TSystemMatrixType& A = *BaseType::mpA;
460  TSystemVectorType& Dx = *BaseType::mpDx;
461  TSystemVectorType& b = *BaseType::mpb;
462 
463  // We get the process info
464  ProcessInfo& r_process_info = r_model_part.GetProcessInfo();
465 
466  int inner_iteration = 0;
467  while (!is_converged && inner_iteration < mThisParameters["inner_loop_iterations"].GetInt()) {
468  ++inner_iteration;
469 
470  if (mConvergenceCriteriaEchoLevel > 0 && StrategyBaseType::GetModelPart().GetCommunicator().MyPID() == 0 ) {
471  std::cout << std::endl << BOLDFONT("Simplified semi-smooth strategy. INNER ITERATION: ") << inner_iteration;;
472  }
473 
474  // We solve one loop
475  r_process_info[NL_ITERATION_NUMBER] = 1;
476  r_process_info[INNER_LOOP_ITERATION] = inner_iteration;
477  is_converged = BaseSolveSolutionStep();
478 
479  // We check the convergence
480  BaseType::mpConvergenceCriteria->SetEchoLevel(0);
481  is_converged = BaseType::mpConvergenceCriteria->PostCriteria(r_model_part, BaseType::GetBuilderAndSolver()->GetDofSet(), A, Dx, b);
482  BaseType::mpConvergenceCriteria->SetEchoLevel(mConvergenceCriteriaEchoLevel);
483 
484  if (mConvergenceCriteriaEchoLevel > 0 && StrategyBaseType::GetModelPart().GetCommunicator().MyPID() == 0 ) {
485  if (is_converged) std::cout << BOLDFONT("Simplified semi-smooth strategy. INNER ITERATION: ") << BOLDFONT(FGRN("CONVERGED")) << std::endl;
486  else std::cout << BOLDFONT("Simplified semi-smooth strategy. INNER ITERATION: ") << BOLDFONT(FRED("NOT CONVERGED")) << std::endl;
487  }
488  }
489  } else {
490  // We compute the base loop
491  r_model_part.GetProcessInfo()[INNER_LOOP_ITERATION] = 1;
492  is_converged = BaseSolveSolutionStep();
493  }
494 
495  if (mThisParameters["adaptative_strategy"].GetBool()) {
496  if (!is_converged) {
497  is_converged = AdaptativeStep();
498  }
499  }
500 
501  return is_converged;
502 
503  KRATOS_CATCH("");
504  }

595  bool BaseSolveSolutionStep()
596  {
597  KRATOS_TRY;
598 
599  // Pointers needed in the solution
600  ModelPart& r_model_part = StrategyBaseType::GetModelPart();
601  ProcessInfo& r_process_info = r_model_part.GetProcessInfo();
602  typename TSchemeType::Pointer p_scheme = BaseType::GetScheme();
603  typename TBuilderAndSolverType::Pointer p_builder_and_solver = BaseType::GetBuilderAndSolver();
604  auto& r_dof_set = p_builder_and_solver->GetDofSet();
605 
606  TSystemMatrixType& rA = *BaseType::mpA;
607  TSystemVectorType& rDx = *BaseType::mpDx;
608  TSystemVectorType& rb = *BaseType::mpb;
609 
610  // Initializing the parameters of the Newton-Raphson cicle
611  IndexType iteration_number = 1;
612  r_process_info[NL_ITERATION_NUMBER] = iteration_number;
613 
614  bool is_converged = false;
615  bool residual_is_updated = false;
616  p_scheme->InitializeNonLinIteration(r_model_part, rA, rDx, rb);
617  BaseType::mpConvergenceCriteria->InitializeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
618  is_converged = BaseType::mpConvergenceCriteria->PreCriteria(r_model_part, r_dof_set, rA, rDx, rb);
619 
620  // We do a geometry check before solve the system for first time
621  if (mThisParameters["adaptative_strategy"].GetBool()) {
622  if (CheckGeometryInverted()) {
623  KRATOS_WARNING("Element inverted") << "INVERTED ELEMENT BEFORE FIRST SOLVE" << std::endl;
624  r_process_info[STEP] -= 1; // We revert one step in the case that the geometry is already broken before start the computing
625  return false;
626  }
627  }
628 
629  // Function to perform the building and the solving phase.
630  if (StrategyBaseType::mRebuildLevel > 1 || StrategyBaseType::mStiffnessMatrixIsBuilt == false) {
631  TSparseSpace::SetToZero(rA);
632  TSparseSpace::SetToZero(rDx);
633  TSparseSpace::SetToZero(rb);
634 
635  p_builder_and_solver->BuildAndSolve(p_scheme, r_model_part, rA, rDx, rb);
636  } else {
637  TSparseSpace::SetToZero(rDx); //Dx=0.00;
638  TSparseSpace::SetToZero(rb);
639 
640  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
641  }
642 
643  // Debugging info
644  BaseType::EchoInfo(iteration_number);
645 
646  // Updating the results stored in the database
647  UpdateDatabase(rA, rDx, rb, StrategyBaseType::MoveMeshFlag());
648 
649  // We now check the geometry
650  if (mThisParameters["adaptative_strategy"].GetBool()) {
651  if (CheckGeometryInverted()) {
652  KRATOS_WARNING("Element inverted") << "INVERTED ELEMENT DURING DATABASE UPDATE" << std::endl;
653  r_process_info[STEP] -= 1; // We revert one step in the case that the geometry is already broken before start the computing
654  return false;
655  }
656  }
657 
658  p_scheme->FinalizeNonLinIteration(r_model_part, rA, rDx, rb);
659  BaseType::mpConvergenceCriteria->FinalizeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
660 
661  if (is_converged) {
662  // Initialisation of the convergence criteria
663  BaseType::mpConvergenceCriteria->InitializeSolutionStep(r_model_part, r_dof_set, rA, rDx, rb);
664 
665  if (BaseType::mpConvergenceCriteria->GetActualizeRHSflag()) {
666  TSparseSpace::SetToZero(rb);
667 
668  p_builder_and_solver->BuildRHS(p_scheme, r_model_part, rb);
669  }
670 
671  is_converged = BaseType::mpConvergenceCriteria->PostCriteria(r_model_part, r_dof_set, rA, rDx, rb);
672  }
673 
674  // Iteration Cicle... performed only for NonLinearProblems
675  while (is_converged == false && iteration_number++<BaseType::mMaxIterationNumber) {
676  //setting the number of iteration
677  r_process_info[NL_ITERATION_NUMBER] = iteration_number;
678 
679  p_scheme->InitializeNonLinIteration(r_model_part, rA, rDx, rb);
680  BaseType::mpConvergenceCriteria->InitializeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
681 
682  is_converged = BaseType::mpConvergenceCriteria->PreCriteria(r_model_part, r_dof_set, rA, rDx, rb);
683 
684  //call the linear system solver to find the correction mDx for the
685  //it is not called if there is no system to solve
686  if (SparseSpaceType::Size(rDx) != 0) {
687  if (StrategyBaseType::mRebuildLevel > 1 || StrategyBaseType::mStiffnessMatrixIsBuilt == false ) {
688  if( BaseType::GetKeepSystemConstantDuringIterations() == false) {
689  //A = 0.00;
690  TSparseSpace::SetToZero(rA);
691  TSparseSpace::SetToZero(rDx);
692  TSparseSpace::SetToZero(rb);
693 
694  p_builder_and_solver->BuildAndSolve(p_scheme, r_model_part, rA, rDx, rb);
695  }
696  else {
697  TSparseSpace::SetToZero(rDx);
698  TSparseSpace::SetToZero(rb);
699 
700  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
701  }
702  }
703  else {
704  TSparseSpace::SetToZero(rDx);
705  TSparseSpace::SetToZero(rb);
706 
707  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
708  }
709  } else {
710  KRATOS_WARNING("No DoFs") << "ATTENTION: no free DOFs!! " << std::endl;
711  }
712 
713  // Debugging info
714  BaseType::EchoInfo(iteration_number);
715 
716  // Updating the results stored in the database
717  UpdateDatabase(rA, rDx, rb, StrategyBaseType::MoveMeshFlag());
718 
719  // We now check the geometry
720  if (mThisParameters["adaptative_strategy"].GetBool()) {
721  if (CheckGeometryInverted()) {
722  KRATOS_WARNING("Element inverted") << "INVERTED ELEMENT DURING DATABASE UPDATE" << std::endl;
723  r_process_info[STEP] -= 1; // We revert one step in the case that the geometry is already broken before start the computing
724  return false;
725  }
726  }
727 
728  p_scheme->FinalizeNonLinIteration(r_model_part, rA, rDx, rb);
729  BaseType::mpConvergenceCriteria->FinalizeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
730 
731  residual_is_updated = false;
732 
733  if (is_converged) {
734 
735  if (BaseType::mpConvergenceCriteria->GetActualizeRHSflag()) {
736  TSparseSpace::SetToZero(rb);
737 
738  p_builder_and_solver->BuildRHS(p_scheme, r_model_part, rb);
739  residual_is_updated = true;
740  //std::cout << "mb is calculated" << std::endl;
741  }
742 
743  is_converged = BaseType::mpConvergenceCriteria->PostCriteria(r_model_part, r_dof_set, rA, rDx, rb);
744  }
745  }
746 
747  // Plots a warning if the maximum number of iterations is exceeded
748  if (iteration_number >= BaseType::mMaxIterationNumber && r_model_part.GetCommunicator().MyPID() == 0)
749  MaxIterationsExceeded();
750 
751  // Recalculate residual if needed
752  // (note that some convergence criteria need it to be recalculated)
753  if (residual_is_updated == false) {
754  // NOTE:

776  bool AdaptativeStep()
777  {
778  KRATOS_TRY;
779 
780  bool is_converged = false;
781  // Plots a warning if the maximum number of iterations is exceeded
782  if (mpMyProcesses == nullptr && StrategyBaseType::mEchoLevel > 0)
783  KRATOS_WARNING("No python processes") << "If you have not implemented any method to recalculate BC or loads in function of time, this strategy will be USELESS" << std::endl;
784 
785  if (mpPostProcesses == nullptr && StrategyBaseType::mEchoLevel > 0)
786  KRATOS_WARNING("No python post processes") << "If you don't add the postprocesses and the time step if splitted you won't postprocess that steps" << std::endl;
787 
788  ModelPart& r_model_part = StrategyBaseType::GetModelPart();
789  ProcessInfo& r_process_info = r_model_part.GetProcessInfo();
790 
791  const double original_delta_time = r_process_info[DELTA_TIME]; // We save the delta time to restore later
792 
793  int split_number = 0;
794 
795  // We iterate until we reach the convergence or we split more than desired
796  while (is_converged == false && split_number <= mThisParameters["max_number_splits"].GetInt()) {
797  // Expliting time step as a way to try improve the convergence
798  split_number += 1;
799  double aux_delta_time, current_time;
800  const double aux_time = SplitTimeStep(aux_delta_time, current_time);
801  current_time += aux_delta_time;
802 
803  bool inside_the_split_is_converged = false;
804  IndexType inner_iteration = 0;
805  while (current_time <= aux_time) {
806  inner_iteration += 1;
807  r_process_info[STEP] += 1;
808 
809  if (inner_iteration == 1) {
810  if (StrategyBaseType::MoveMeshFlag())
811  UnMoveMesh();
812 
813  NodesArrayType& r_nodes_array = r_model_part.Nodes();
814  block_for_each(r_nodes_array, [&](NodeType& rNode) {
815  rNode.OverwriteSolutionStepData(1, 0);
816 // rNode.OverwriteSolutionStepData(2, 1);
817  });
818 
819  r_process_info.SetCurrentTime(current_time); // Reduces the time step
820 
821  FinalizeSolutionStep();
822  } else {
823  NodesArrayType& r_nodes_array = r_model_part.Nodes();
824  block_for_each(r_nodes_array, [&](NodeType& rNode) {
825  rNode.CloneSolutionStepData();
826  });
827 
828  r_process_info.CloneSolutionStepInfo();
829  r_process_info.ClearHistory(r_model_part.GetBufferSize());
830  r_process_info.SetAsTimeStepInfo(current_time); // Sets the new time step
831  }
832 
833  // We execute the processes before the non-linear iteration
834  if (mpMyProcesses != nullptr)
835  mpMyProcesses->ExecuteInitializeSolutionStep();
836 
837  if (mpPostProcesses != nullptr)
838  mpPostProcesses->ExecuteInitializeSolutionStep();
839 
840  // In order to initialize again everything
841  BaseType::mInitializeWasPerformed = false;
842  mFinalizeWasPerformed = false;
843 
844  // We repeat the solve with the new DELTA_TIME
845  this->Initialize();
846  this->InitializeSolutionStep();
847  this->Predict();
848  inside_the_split_is_converged = BaseType::SolveSolutionStep();
849  this->FinalizeSolutionStep();
850 
851  // We execute the processes after the non-linear iteration
852  if (mpMyProcesses != nullptr)
853  mpMyProcesses->ExecuteFinalizeSolutionStep();
854 
855  if (mpPostProcesses != nullptr)
856  mpPostProcesses->ExecuteFinalizeSolutionStep();
857 
858  if (mpMyProcesses != nullptr)
859  mpMyProcesses->ExecuteBeforeOutputStep();
860 
861  if (mpPostProcesses != nullptr)
862  mpPostProcesses->PrintOutput();
863 
864  if (mpMyProcesses != nullptr)
865  mpMyProcesses->ExecuteAfterOutputStep();
866 
867  current_time += aux_delta_time;
868  }
869 
870  if (inside_the_split_is_converged)
871  is_converged = true;
872  }
873 
874  // Plots a warning if the maximum number of iterations and splits are exceeded
875  if (is_converged == false)
876  MaxIterationsAndSplitsExceeded();
877 
878  // Restoring original DELTA_TIME
879  r_process_info[DELTA_TIME] = original_delta_time;
880 
881  return is_converged;
882 
883  KRATOS_CATCH("");
884  }
