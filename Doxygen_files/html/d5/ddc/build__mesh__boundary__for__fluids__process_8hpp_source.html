178 bool UniqueSkinSearch(ModelPart &rModelPart)
179 {
180
181 KRATOS_TRY
182
183 if (mEchoLevel > 0)
184 {
185 std::cout << " [ Initial Conditions : " << rModelPart.Conditions().size() << std::endl;
186 }
187
188 if (!rModelPart.Elements().size() || (rModelPart.Is(ACTIVE)))
189 {
190 if (mEchoLevel > 0)
191 {
192 std::cout << " [ Final Conditions : " << rModelPart.Conditions().size() << std::endl;
193 }
194 return true;
195 }
196 const unsigned int dimension = mrModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
197
198 //reset the boundary flag in all nodes and check if a remesh process has been performed
199 bool any_node_to_erase = false;
200 for (ModelPart::NodesContainerType::const_iterator in = rModelPart.NodesBegin(); in != rModelPart.NodesEnd(); in++)
201 {
202
203 NodeWeakPtrVectorType &nNodes = in->GetValue(NEIGHBOUR_NODES);
204 unsigned int neighNodes = nNodes.size();
205 if (neighNodes < (dimension+1))
206 {
207 in->FastGetSolutionStepValue(ISOLATED_NODE) = 1;
208 }
209 else
210 {
211 in->FastGetSolutionStepValue(ISOLATED_NODE) = 0;
212 }
213
214 if (in->Is(ISOLATED)) //to record the information of the previous step
215 {
216 in->FastGetSolutionStepValue(PREVIOUS_FREESURFACE) = 1;
217 }
218 else
219 {
220 in->FastGetSolutionStepValue(PREVIOUS_FREESURFACE) = 0;
221 }
222
223 if (in->Is(FREE_SURFACE)) //to record the information of the previous step
224 {
225 in->FastGetSolutionStepValue(FREESURFACE) = 1;
226 }
227 else
228 {
229 in->FastGetSolutionStepValue(FREESURFACE) = 0;
230 }
231
232 in->Reset(BOUNDARY);
233 in->Reset(FREE_SURFACE);
234
235 if (any_node_to_erase == false)
236 if (in->Is(TO_ERASE))
237 any_node_to_erase = true;
238 }
239 SetBoundaryAndFreeSurface(rModelPart);
240
241 return true;
242
243 KRATOS_CATCH("")
244 }

247 bool SetBoundaryAndFreeSurface(ModelPart &rModelPart)
248 {
249
250 KRATOS_TRY
251
252 //properties to be used in the generation
253 int number_properties = rModelPart.GetParentModelPart().NumberOfProperties();
254 Properties::Pointer properties = rModelPart.GetParentModelPart().pGetProperties(number_properties - 1);
255
256 ModelPart::ElementsContainerType::iterator elements_begin = mrModelPart.ElementsBegin();
257 ModelPart::ElementsContainerType::iterator elements_end = mrModelPart.ElementsEnd();
258
259 for (ModelPart::ElementsContainerType::iterator ie = elements_begin; ie != elements_end; ie++)
260 {
261
262 Geometry<Node<3>> &rElementGeometry = ie->GetGeometry();
263
264 if (rElementGeometry.FacesNumber() >= 3)
265 { //3 or 4
266
267 /*each face is opposite to the corresponding node number so in 2D triangle
268 0 ----- 1 2
269 1 ----- 2 0
270 2 ----- 0 1
271 */
272
273 /*each face is opposite to the corresponding node number so in 3D tetrahedron
274 0 ----- 1 2 3
275 1 ----- 2 0 3
276 2 ----- 0 1 3
277 3 ----- 0 2 1
278 */
279
280 //finding boundaries and creating the "skin"
281 //
282 //********************************************************************
283
284 boost::numeric::ublas::matrix<unsigned int> lpofa; //connectivities of points defining faces
285 boost::numeric::ublas::vector<unsigned int> lnofa; //number of points defining faces
286
287 ElementWeakPtrVectorType &rE = ie->GetValue(NEIGHBOUR_ELEMENTS);
288
289 //get matrix nodes in faces
290 rElementGeometry.NodesInFaces(lpofa);
291 rElementGeometry.NumberNodesInFaces(lnofa);
292
293 //loop on neighbour elements of an element
294 unsigned int iface = 0;
295 for (ElementWeakPtrVectorType::iterator ne = rE.begin(); ne != rE.end(); ne++)
296 {
297
298 unsigned int NumberNodesInFace = lnofa[iface];
299
300 if ((ne)->Id() == ie->Id())
301 {
302
303 //if no neighbour is present => the face is free surface
304 bool freeSurfaceFace = false;
305 for (unsigned int j = 1; j <= NumberNodesInFace; j++)
306 {
307 rElementGeometry[lpofa(j, iface)].Set(BOUNDARY);
308 if (rElementGeometry[lpofa(j, iface)].IsNot(RIGID))
309 {
310 freeSurfaceFace = true;
311 }
312 }
313 if (freeSurfaceFace == true)
314 {
315 for (unsigned int j = 1; j <= NumberNodesInFace; j++)
316 {
317 rElementGeometry[lpofa(j, iface)].Set(FREE_SURFACE);
318 }
319 }
320
321 } //end face condition
322
323 iface += 1;
324
325 } //end loop neighbours
326 }
327 }
328
329 return true;
330
331 KRATOS_CATCH("")
332 }

337 bool CheckAcceptedCondition(ModelPart &rModelPart, Condition &rCondition) override
338 {
339 KRATOS_TRY
340
341 bool node_not_preserved = false;
342 bool condition_not_preserved = false;
343
344 Geometry<Node<3>> &rConditionGeometry = rCondition.GetGeometry();
345
346 for (unsigned int j = 0; j < rConditionGeometry.size(); j++)
347 {
348 if (rConditionGeometry[j].Is(TO_ERASE) || rConditionGeometry[j].Is(TO_REFINE))
349 node_not_preserved = true;
350
351 if (rConditionGeometry[j].Is(ISOLATED) || rConditionGeometry[j].IsNot(BOUNDARY))
352 condition_not_preserved = true;
353 }
354
355 if (rCondition.Is(TO_ERASE))
356 condition_not_preserved = true;
357
358 if (rCondition.Is(BOUNDARY)) //flag for composite condition
359 condition_not_preserved = true;
360
361 if (node_not_preserved == true || condition_not_preserved == true)
362 return false;
363 else
364 return true;
365
366 KRATOS_CATCH("")
367 }

424

451 rConditionGeometry[0].Id() == rElementGeometry[lpofa(3, iface)].Id() ||
