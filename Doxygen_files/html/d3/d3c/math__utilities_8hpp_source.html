80 static inline bool CardanoFormula(double a, double b, double c, double d, Vector& solution)
81 {
82 solution.resize(3,false);
83 noalias(solution)= ZeroVector(3);
84
85 if (a==0)
86 {
87 std::cout<<"This is not a cubic equation: CardanoFormula"<<std::endl;
88
89 return false;
90 }
91
92 double p= (3.0*a*c-b*b)/(3.0*a*a);
93
94 double q= 2.0*b*b*b/(27.0*a*a*a)-b*c/(3.0*a*a)+d/a;
95
96 double discriminante= p*p*p/27.0+q*q/4.0;
97
98 if(discriminante>0)
99 {
100 return false;
101 }
102
103 if(discriminante==0)
104 {
105 if( a == 0 )
106 return false;
107
108 solution(0)= pow(q/2.0, 1.0/3.0)-b/(3*a);
109 solution(1)= pow(q/2.0, 1.0/3.0)-b/(3*a);
110 solution(2)= pow(-4.0*q, 1.0/3.0)-b/(3*a);
111
112 return true;
113 }
114
115 if(p < 0) return false; //in this case the square roots below will be negative. This substitutes with better efficiency lines 107-110
116
117 solution(0) = -sqrt(-4.0/3.0*p)*cos(1.0/3.0*acos(-q/2.0*sqrt(-27.0/(p*p*p)))+ Globals::Pi / 3.0) -b/(3*a);
118 solution(1) = sqrt(-4.0/3.0*p)*cos(1.0/3.0*acos(-q/2.0*sqrt(-27.0/(p*p*p))))-b/(3*a);
119 solution(2) = -sqrt(-4.0/3.0*p)*cos(1.0/3.0*acos(-q/2.0*sqrt(-27.0/(p*p*p)))- Globals::Pi / 3.0) -b/(3*a);

141 static inline Vector EigenValues(const Matrix& A, double tolerance, double zero)
142 {
143 int dim= A.size1();
144
145 Matrix Convergence(2,dim);
146 noalias(Convergence) = ZeroMatrix(2,dim);
147
148 double delta = 0.0;
149
150 double abs = 0.0;
151
152 Vector Result(dim);
153 noalias(Result) = ZeroVector(dim);
154
155 Matrix HelpA(dim,dim);
156 noalias(HelpA) = ZeroMatrix(dim, dim);
157
158 Matrix HelpQ(dim,dim);
159 noalias(HelpQ) = ZeroMatrix(dim, dim);
160
161 Matrix HelpR(dim,dim);
162 noalias(HelpR) = ZeroMatrix(dim, dim);
163
164 HelpA=A;
165
166 bool is_converged=false;
167 unsigned int max_iters = 10;
168 unsigned int iter = 0;
169
170 while(is_converged == false && iter<max_iters )
171 {
172 double shift= HelpA((dim-1),(dim-1));
173 //
174 for(int i=0; i<dim; i++)
175 {
176 HelpA(i,i) = HelpA(i,i)- shift;
177 }
178
179 StructuralMechanicsMathUtilities<double>::QRFactorization(HelpA, HelpQ, HelpR);
180
181 HelpA= ZeroMatrix(dim, dim);
182
183 for(int i=0; i<dim; i++)
184 {
185 HelpA(i,i) += shift;
186 for(int j=0; j< dim; j++)
187 {
188 for(int k=0; k< dim; k++)
189 {
190 HelpA(i,j) += HelpR(i,k)*HelpQ(k,j);
191 }
192 }
193 }
194
195 delta= 0.0;
196
197 abs = 0.0;
198
199 for(int i=0; i<dim; i++)
200 {
201 Convergence(0,i)=Convergence(1,i);
202 Convergence(1,i)=HelpA(i,i);
203 delta+= (Convergence(1,i)-Convergence(0,i))*(Convergence(1,i)-Convergence(0,i));
204 abs+=(Convergence(1,i))*(Convergence(1,i));
205 }
206
207 delta= sqrt(delta);
208
209 abs=sqrt(abs);
210
211 if(abs < zero)
212 abs=1.0;
213
214 if(delta < zero || (delta/abs) < tolerance)
215 is_converged=true;
216
217 iter++;
218 }
219
220
221 for(int i=0; i<dim; i++)
222 {
223 Result(i)= HelpA(i,i);
224
225 if(fabs(Result(i)) <zero)
226 Result(i)=0.0;
227 }
228
229 return Result;
230 }

242 static inline Vector EigenValuesDirectMethod(const Matrix& A)
243 {
244 // Given a real symmetric 3x3 matrix A, compute the eigenvalues
245 int dim= A.size1();
246 Vector Result(dim);
247 noalias(Result) = ZeroVector(dim);
248
249 const double p1 = A(0,1)*A(0,1) + A(0,2)*A(0,2) + A(1,2)*A(1,2);
250 if (p1 == 0)
251 {//A is diagonal.
252 Result[0] = A(0,0);
253 Result[1] = A(1,1);
254 Result[2] = A(2,2);
255 return Result;
256 }
257
258 const double q = (A(0,0) + A(1,1) + A(2,2)) / 3.0;
259 const double p2 = (A(0,0) - q) * (A(0,0) - q) + (A(1,1) - q) * (A(1,1) - q) + (A(2,2) - q) * (A(2,2) - q) + 2.0 * p1;
260 const double p = sqrt(p2 / 6.0);
261
262 Matrix B(3,3);
263 const double inv_p = 1.0/p;
264
265 // B = (1 / p) * (A - q * I) where I is the identity matrix
266 B(0,0) = inv_p * (A(0,0) - q);
267 B(1,1) = inv_p * (A(1,1) - q);
268 B(2,2) = inv_p * (A(2,2) - q);
269 B(0,1) = inv_p * A(0,1);
270 B(1,0) = inv_p * A(1,0);
271 B(0,2) = inv_p * A(0,2);
272 B(2,0) = inv_p * A(2,0);
273 B(1,2) = inv_p * A(1,2);
274 B(2,1) = inv_p * A(2,1);
275
276 //r = det(B) / 2
277 double r = 0.5 * ( B(0,0)*B(1,1)*B(2,2) + B(0,1)*B(1,2)*B(2,0) + B(1,0)*B(2,1)*B(0,2) - B(2,0)*B(1,1)*B(0,2) - B(1,0)*B(0,1)*B(2,2) - B(0,0)*B(2,1)*B(1,2) );
278
279 // In exact arithmetic for a symmetric matrix -1 <= r <= 1
280 // but computation error can leave it slightly outside this range.
281 double phi = 0.0;
282 if (r <= -1) { phi = Globals::Pi / 3.0; }
283 else if (r >= 1) { phi = 0.0; }
284 else { phi = acos(r) / 3.0;}
285
286 // the eigenvalues satisfy eig3 <= eig2 <= eig1
287 Result[0] = q + 2.0 * p * cos(phi);
288 Result[2] = q + 2.0 * p * cos(phi + (2.0 * Globals::Pi / 3.0));
289 Result[1] = 3.0 * q - Result[0] - Result[2]; //% since trace(A) = eig1 + eig2 + eig3
290
291 return Result;
292 }

306 static inline void QRFactorization(const MatrixType& A, MatrixType& Q, MatrixType& R)
307 {
308
309 //QR Factorization with Householder-Algo
310 int dim= A.size1();
311
312 Vector y(dim);
313
314 Vector w(dim);
315
316 R.resize(dim,dim,false);
317
318 noalias(R)=ZeroMatrix(dim,dim);
319
320 Q.resize(dim,dim,false);
321
322 noalias(Q)=ZeroMatrix(dim,dim);
323
324 Matrix Help(A.size1(),A.size2());
325 noalias(Help) = A;
326
327 Matrix unity(dim,dim);
328 noalias(unity) = ZeroMatrix(dim,dim);
329
330 for(int j=0; j<dim; j++)
331 unity(j,j)=1.0;
332
333 std::vector<Matrix> HelpQ(dim-1);
334
335 std::vector<Matrix> HelpR(dim-1);
336
337 for(int i=0; i< dim-1; i++)
338 {
339 HelpQ[i].resize(dim,dim,false);
340 HelpR[i].resize(dim,dim,false);
341 noalias(HelpQ[i])= unity;
342 noalias(HelpR[i])= ZeroMatrix(dim,dim);
343 }
344
345 for(int iteration=0; iteration< dim-1; iteration++)
346 {
347 //Vector y
348 for(int i=iteration; i<dim; i++)
349 y(i)= Help(i,iteration);
350
351
352 //Helpvalue l
353 double normy=0.0;
354
355 for(int i=iteration; i<dim; i++)
356 normy += y(i)*y(i);
357
358 normy= sqrt(normy);
359
360 double l= sqrt((normy*(normy+fabs(y(iteration))))/2);
361
362 double k=0.0;
363
364 if(y[iteration] !=0)
365 k= - y(iteration)/fabs(y(iteration))*normy;
366 else
367 k= -normy;
368
369 for(int i=iteration; i<dim; i++)
370 {
371 double e=0;
372
373 if(i==iteration)
374 e=1;
375
376 w(i)= 1/(2*l)*(y(i)-k*e);
377 }
378
379 for(int i=iteration; i<dim; i++)
380 for(int j=iteration; j<dim; j++)
381 HelpQ[iteration](i,j)= unity(i,j)- 2*w(i)*w(j);
382
383
384 for(int i=iteration; i<dim; i++)
385 for(int j=iteration; j<dim; j++)
386 for(int k=iteration; k<dim; k++)
387 HelpR[iteration](i,j)+= HelpQ[iteration](i,k)*Help(k,j);
388
389 Help= HelpR[iteration];
390
391 }
392
393 //Assembling R
394 for(int k=0; k<dim-1; k++)
395 {
396 for(int i=k; i<dim; i++)
397 for(int j=k; j<dim; j++)
398 R(i,j) =HelpR[k](i,j);
399
400 }
401
402
403 for(int k=1; k<dim-1; k++)
404 {
405 for(int i=0; i<dim; i++)
406 for(int j=0; j<dim; j++)
407 for(int l=0; l<dim; l++)
408 Q(i,j)+= HelpQ[(k-1)](i,l)*HelpQ[k](l,j);
409 noalias(HelpQ[k])=Q;
410 }
411 if(dim-1==1)
412 noalias(Q)=HelpQ[0];
413
414 }

431 static inline void EigenVectors(const MatrixType& A,

434 double zero_tolerance =1e-9,
435 int max_iterations = 10)
436 {
437 Matrix Help= A;
438
439 for(int i=0; i<3; i++)
440 for(int j=0; j<3; j++)
441 Help(i,j)= Help(i,j);
442
443
444 vectors.resize(Help.size1(),Help.size2(),false);
445
446 lambda.resize(Help.size1(),false);
447
448 Matrix HelpDummy(Help.size1(),Help.size2());
449
450 bool is_converged = false;
451
452 Matrix unity(Help.size1(),Help.size2());
453 noalias(unity) = ZeroMatrix(Help.size1(),Help.size2());
454
455 for(unsigned int i=0; i< Help.size1(); i++)
456 unity(i,i)= 1.0;
457
458 Matrix V= unity;
459
460 Matrix VDummy(Help.size1(),Help.size2());
461
462 Matrix Rotation(Help.size1(),Help.size2());
463
464
465 for(int iterations=0; iterations<max_iterations; iterations++)
466 {
467
468 is_converged= true;
469
470 double a= 0.0;
471
472 unsigned int index1= 0;
473
474 unsigned int index2= 1;
475
476 for(unsigned int i=0; i< Help.size1(); i++)
477 {
478 for(unsigned int j=(i+1); j< Help.size2(); j++)
479 {
480 if((fabs(Help(i,j)) > a ) && (fabs(Help(i,j)) > zero_tolerance))
481 {
482 a= fabs(Help(i,j));
483

485 index2= j;

490 }
491
492// KRATOS_WATCH( Help )
493
494 if(is_converged)
495 break;
496
497 //Calculation of Rotationangle
498
499 double gamma= (Help(index2,index2)-Help(index1,index1))/(2*Help(index1,index2));
500
501 double u=1.0;
502
503 if(fabs(gamma) > zero_tolerance && fabs(gamma)< (1/zero_tolerance))
504 {
505 u= gamma/fabs(gamma)*1.0/(fabs(gamma)+sqrt(1.0+gamma*gamma));
506 }
507 else
508 {
509 if (fabs(gamma)>= (1.0/zero_tolerance))
510 u= 0.5/gamma;
511 }
512

514

516

518
519 //Ratotion of the Matrix

521

541

543 Rotation =unity;
544 Rotation(index2,index1)=-s;
545 Rotation(index1,index2)=s;
546 Rotation(index1,index1)=c;
547 Rotation(index2,index2)=c;
548
549// Help.resize(A.size1(),A.size1(),false);
550// noalias(Help)=ZeroMatrix(A.size1(),A.size1());
551
552 noalias(VDummy) = ZeroMatrix(Help.size1(), Help.size2());
553
554 for(unsigned int i=0; i< Help.size1(); i++)
555 {
556 for(unsigned int j=0; j< Help.size1(); j++)
557 {
558 for(unsigned int k=0; k< Help.size1(); k++)
559 {
560 VDummy(i,j) += V(i,k)*Rotation(k,j);

562 }
563 }

1142 static bool Clipping(std::vector<Point* >& clipping_points,std::vector<Point* >& subjected_points, std::vector<Point* >& result_points, double tolerance)
1143 {
1144 result_points= subjected_points;
1145 Vector actual_edge(3);
1146 Vector actual_normal(3);
1147 std::vector<Point* > temp_results;
1148 bool is_visible= false;
1149 for(unsigned int clipp_edge=0; clipp_edge<clipping_points.size(); clipp_edge++)
1150 {
1151 temp_results.clear();
1152 unsigned int index_clipp_2=0;
1153 if(clipp_edge< (clipping_points.size()-1))
1154 index_clipp_2= clipp_edge+1;
1155 //define clipping edge vector
1156 noalias(actual_edge)= *(clipping_points[clipp_edge])-*(clipping_points[index_clipp_2]);
1157 noalias(actual_edge)= actual_edge/sqrt(inner_prod(actual_edge,actual_edge));
1158
1159 //define normal on clipping-edge vector towards visible side
1160 if(clipp_edge< (clipping_points.size()-2))
1161 actual_normal=*(clipping_points[clipp_edge+2])-(*(clipping_points[clipp_edge])+actual_edge*inner_prod((*(clipping_points[clipp_edge+2])-*(clipping_points[clipp_edge])),actual_edge));
1162 else if(clipp_edge< (clipping_points.size()-1))
1163 actual_normal=*(clipping_points[0])-(*(clipping_points[clipp_edge])+actual_edge*inner_prod((*(clipping_points[0])-*(clipping_points[clipp_edge])),actual_edge));
1164 else
1165 actual_normal=*(clipping_points[1])-(*(clipping_points[clipp_edge])+actual_edge*inner_prod((*(clipping_points[1])-*(clipping_points[clipp_edge])),actual_edge));
1166
1167 noalias(actual_normal)=actual_normal/(sqrt(inner_prod(actual_normal,actual_normal)));
1168
1169 //test if the first point is visible or unvisible
1170 if( inner_prod((*(result_points[0])-*(clipping_points[clipp_edge])), actual_normal)> tolerance)
1171 {
1172 is_visible= true;
1173 }
1174 else
1175 {
1176 is_visible= false;
1177 }
1178
1179 for(unsigned int subj_edge=0; subj_edge< result_points.size(); subj_edge++)
1180 {
1181 unsigned int index_subj_2=0;
1182
1183 if(subj_edge< (result_points.size()-1))
1184 index_subj_2= subj_edge+1;
1185
1186 //Test whether the points of the actual subj_edge lay on clipp_edge
1187 if(fabs(inner_prod((*(result_points[subj_edge])-(*(clipping_points[clipp_edge]))), actual_normal))<= tolerance)
1188 {
1189 temp_results.push_back(result_points[subj_edge]);
1190 if( inner_prod((*(result_points[index_subj_2])-*(clipping_points[clipp_edge])), actual_normal)> tolerance)
1191 is_visible= true;
1192 else
1193 is_visible= false;
1194
1195 continue;
1196 }
1197 //Calculate minimal distance between the two points
1198 Vector b(2);
1199 b(0)= -inner_prod((*(result_points[index_subj_2])-*(result_points[subj_edge])),(*(result_points[subj_edge])-*(clipping_points[clipp_edge])));
1200 b(1)= inner_prod((*(clipping_points[index_clipp_2])-*(clipping_points[clipp_edge])),(*(result_points[subj_edge])-*(clipping_points[clipp_edge])));
1201 Matrix A(2,2);
1202 A(0,0)=inner_prod((*(result_points[index_subj_2])-*(result_points[subj_edge])),(*(result_points[index_subj_2])-*(result_points[subj_edge])));
1203 A(0,1)=-inner_prod((*(result_points[index_subj_2])-*(result_points[subj_edge])),(*(clipping_points[index_clipp_2])-*(clipping_points[clipp_edge])));
1204 A(1,0)= A(0,1);
1205 A(1,1)=inner_prod(*(clipping_points[index_clipp_2])-*(clipping_points[clipp_edge]),*(clipping_points[index_clipp_2])-*(clipping_points[clipp_edge]));
1206 Vector coeff(2);
1207 coeff(0)=1.0/A(0,0)*(b(0)-A(0,1)/(A(1,1)-A(0,1)*A(1,0)/A(0,0))*(b(1)-b(0)*A(1,0)/A(0,0)));
1208 coeff(1)=1.0/(A(1,1)-A(0,1)*A(1,0)/A(0,0))*(b(1)-b(0)*A(1,0)/A(0,0));
1209
1210
1211 //TEST on distance to endpoints of the line
1212 Vector dist_vec(3);
1213 noalias(dist_vec)= *(result_points[subj_edge])+coeff(0)*(*(result_points[index_subj_2])-*(result_points[subj_edge]))-(*(clipping_points[clipp_edge])+coeff(1)*(*(clipping_points[index_clipp_2])-*(clipping_points[clipp_edge])));
1214
1215 if( coeff(0) > tolerance && coeff(0) < (1-tolerance)&& (sqrt(inner_prod(dist_vec,dist_vec))< tolerance))
1216 {
1217 if(is_visible)
1218 temp_results.push_back(result_points[subj_edge]);
1219
1220 temp_results.push_back(new Point((*(clipping_points[clipp_edge])+coeff(1)*(*(clipping_points[index_clipp_2])-*(clipping_points[clipp_edge])))));
1221
1222 is_visible= !is_visible;
1223
1224 continue;
1225 }
1226 if(is_visible)
1227 temp_results.push_back(result_points[subj_edge]);
1228 }
1229 result_points=temp_results;
1230 }
1231 if(result_points.size()==0)
1232 return false;
1233 else
1234 return true;
1235 }
