68 void ExecuteInitializeSolutionStep() override
69 {
70 KRATOS_TRY;
71
72 const Variable<double>& var = KratosComponents< Variable<double> >::Get(mVariableName);
73
74 double time;
75 if (mTimeUnitConverter != 0) {
76 time = mrModelPart.GetProcessInfo()[TIME] / mTimeUnitConverter;
77 } else {
78 time = mrModelPart.GetProcessInfo()[TIME];
79 }
80 const double value = mpTable->GetValue(time);
81
82 const int number_nodes = static_cast<int>(mrModelPart.Nodes().size());
83
84 if (number_nodes != 0) {
85 const auto& it_begin = mrModelPart.NodesBegin();
86 #pragma omp parallel for
87 for (int i = 0; i < number_nodes; i++) {
88 auto it_node = it_begin + i;
89
90 double reduction_factor_pressure = 1.0;
91 if (it_node->GetValue(PRESSURE_INITIAL_VOLUME) != 0.0) {
92 reduction_factor_pressure = it_node->GetValue(PRESSURE_INITIAL_VOLUME) / it_node->GetValue(PRESSURE_VOLUME);
93 }
94 it_node->FastGetSolutionStepValue(var) = reduction_factor_pressure * value;
95 }
96 }
97 KRATOS_CATCH("");
