358  NRBaseType::UpdateDatabase(A,aux,b,MoveMesh);
359  TSparseSpace::SetToZero(b);
360  pBuilderAndSolver->BuildRHS(pScheme, BaseType::GetModelPart(), b );
361  double rfDisp;
362  double rfLM;
363  ComputeMixedResidual(b, rfDisp, rfLM);
364 
365  // We compute the parabola
366  double XminDisp = 1e-3;
367  double XmaxDisp = 1.0;
368  double XminLM = 1e-3;
369  double XmaxLM = 1.0;
370 
371  ComputeParabola(XminDisp, XmaxDisp, rfDisp, roDisp, rhDisp);
372  ComputeParabola(XminLM, XmaxLM, rfLM, roLM, rhLM);
373 
374  // Perform final update
375  TSparseSpace::Assign(aux,-(1.0 - XmaxDisp), DxDisp);
376  TSparseSpace::UnaliasedAdd(aux,-(1.0 - XmaxLM), DxLM);

386  void ComputeSplitDx(
387  TSystemVectorType& Dx,
388  TSystemVectorType& DxDisp,
389  TSystemVectorType& DxLM
390  )
391  {
392  // Now we iterate over all the nodes
393  NodesArrayType& r_nodes_array = StrategyBaseType::GetModelPart().Nodes();
394  block_for_each(r_nodes_array, [&](Node<3>& rNode) {
395  for(auto itDoF = rNode.GetDofs().begin() ; itDoF != rNode.GetDofs().end() ; itDoF++) {
396  const int j = (**itDoF).EquationId();
397  const std::size_t CurrVar = (**itDoF).GetVariable().Key();
398 
399  if ((CurrVar == DISPLACEMENT_X) || (CurrVar == DISPLACEMENT_Y) || (CurrVar == DISPLACEMENT_Z)) {
400  DxDisp[j] = Dx[j];
401  DxLM[j] = 0.0;
402  } else { // Corresponding with contact
403  DxDisp[j] = 0.0;
404  DxLM[j] = Dx[j];

407  });

416  void ComputeMixedResidual(
417  TSystemVectorType& b,
418  double& normDisp,
419  double& normLM
420  )
421  {
422  // Now we iterate over all the nodes
423  NodesArrayType& r_nodes_array = StrategyBaseType::GetModelPart().Nodes();
424  block_for_each(r_nodes_array, [&](Node<3>& rNode) {
425  for(auto itDoF = rNode.GetDofs().begin() ; itDoF != rNode.GetDofs().end() ; itDoF++) {
426  const int j = (**itDoF).EquationId();
427  const std::size_t CurrVar = (**itDoF).GetVariable().Key();
428 
429  if ((CurrVar == DISPLACEMENT_X) || (CurrVar == DISPLACEMENT_Y) || (CurrVar == DISPLACEMENT_Z)) {
430  AtomicAdd(normDisp, b[j] * b[j]);
431  } else { // Corresponding with contact
432  AtomicAdd(normLM, b[j] * b[j]);
433  }
434  }
435  });
436 
437  normDisp = std::sqrt(normDisp);
438  normLM = std::sqrt(normLM);
439  }

449  void ComputeParabola(
450  double& Xmax,
451  double& Xmin,
452  const double rf,
453  const double ro,
454  const double rh
455  )
456  {
457  // Compute optimal (limited to the range 0-1)
458  // Parabola is y = a*x^2 + b*x + c -> min/max for
459  // x=0 --> r=ro
460  // x=1/2 --> r=rh
461  // x=1 --> r =
462  // c= ro, b= 4*rh -rf -3*ro, a= 2*rf - 4*rh + 2*ro
463  // max found if a>0 at the position Xmax = (rf/4 - rh)/(rf - 2*rh);
464 
465  const double parabole_a = 2 * rf + 2 * ro - 4 * rh;
466  const double parabole_b = 4 * rh - rf - 3 * ro;
467 
468  if( parabole_a > 0.0) // If parabola has a local minima
469  {
470  Xmax = -0.5 * parabole_b/parabole_a; // -b / 2a
471  if( Xmax > 1.0)
472  Xmax = 1.0;
473  else if(Xmax < -1.0)
474  Xmax = -1.0;
475  }
476  else // Parabola degenerates to either a line or to have a local max. best solution on either extreme
477  {
478  if(rf < ro)
479  Xmax = 1.0;
480  else
481  Xmax = Xmin; // Should be zero, but otherwise it will stagnate
482  }
483  }
