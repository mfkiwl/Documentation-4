806 virtual void CalculateBoundingBox()
807 {
808 PointType Low, High;
809 TConfigure::CalculateBoundingBox(*mObjectsBegin,mMinPoint,mMaxPoint);
810
811#ifdef _OPENMP
812 SizeType number_of_threads = omp_get_max_threads();
813#else
814 SizeType number_of_threads = 1;
815#endif
816
817 std::vector<SizeType> node_partition;
818 CreatePartition(number_of_threads, mObjectsSize, node_partition);
819
820 std::vector<PointType> Max(number_of_threads);
821 std::vector<PointType> Min(number_of_threads);
822
823 for(SizeType k=0; k<number_of_threads; k++ )
824 {
825 Max[k] = mMaxPoint;
826 Min[k] = mMinPoint;
827 }
828
829 IteratorType i_begin = mObjectsBegin;
830 IteratorType i_end = mObjectsEnd;
831
832 for (IteratorType i_object = i_begin ; i_object != i_end ; i_object++ )
833 {
834 TConfigure::CalculateBoundingBox(*i_object, Low, High);
835 for(SizeType i = 0 ; i < Dimension ; i++)
836 {
837 mMaxPoint[i] = (mMaxPoint[i] < High[i]) ? High[i] : mMaxPoint[i];
838 mMinPoint[i] = (mMinPoint[i] > Low[i]) ? Low[i] : mMinPoint[i];
839 }
840 }
841
842 auto Epsilon = PointType{mMaxPoint - mMinPoint};
843
844 for(SizeType i = 0 ; i < Dimension ; i++)
845 {
846 mMaxPoint[i] += Epsilon[i] * 0.01;
847 mMinPoint[i] -= Epsilon[i] * 0.01;
848 }
849 }

858 void CalculateCellSize(std::size_t ApproximatedSize)
859 {
860 std::size_t average_number_of_cells = static_cast<std::size_t>(std::pow(static_cast<double>(ApproximatedSize), 1.00 / Dimension));
861
862 std::array<double, 3> lengths;
863 double average_length = 0.00;
864
865 for (int i = 0; i < Dimension; i++) {
866 lengths[i] = mMaxPoint[i] - mMinPoint[i];
867 average_length += lengths[i];
868 }
869 average_length *= 1.00 / 3.00;
870
871 if (average_length < std::numeric_limits<double>::epsilon()) {
872 for(int i = 0; i < Dimension; i++) {
873 mN[i] = 1;
874 }
875 return;
876 }
877
878 for (int i = 0; i < Dimension; i++) {
879 mN[i] = static_cast<std::size_t>(lengths[i] / average_length * (double)average_number_of_cells) + 1;
880
881 if (mN[i] > 1) {
882 mCellSize[i] = lengths[i] / mN[i];
883 } else {
884 mCellSize[i] = average_length;
885 }
886
887 mInvCellSize[i] = 1.00 / mCellSize[i];
888 }
889 }
