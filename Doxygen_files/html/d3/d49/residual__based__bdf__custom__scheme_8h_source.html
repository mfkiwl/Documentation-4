178 void Initialize(ModelPart& rModelPart) override
179 {
180 KRATOS_TRY
181
182 BDFBaseType::Initialize(rModelPart);
183
184 // The current process info
185 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
186
187 // Getting dimension
188 KRATOS_WARNING_IF("ResidualBasedBDFCustomScheme", !r_current_process_info.Has(DOMAIN_SIZE)) << "DOMAIN_SIZE not defined. Please define DOMAIN_SIZE. 3D case will be assumed" << std::endl;
189 const std::size_t domain_size = r_current_process_info.Has(DOMAIN_SIZE) ? r_current_process_info.GetValue(DOMAIN_SIZE) : 3;
190 if (domain_size != mDomainSize) {
191 const std::size_t total_number_of_variables = mDoubleVariable.size();
192
193 // We remove the third component
194 if (domain_size == 2) {
195 const std::size_t number_variables_added = total_number_of_variables/3;
196 for (std::size_t i = 0; i < number_variables_added; ++i) {
197 mDoubleVariable.erase(mDoubleVariable.begin() + (2 + 2 * i));
198 mFirstDoubleDerivatives.erase(mDoubleVariable.begin() + (2 + 2 * i));
199 mSecondDoubleDerivatives.erase(mDoubleVariable.begin() + (2 + 2 * i));
200 }
201 } else if (domain_size == 3) { // We need to add the third component
202 const std::size_t number_variables_added = total_number_of_variables/2;
203 for (std::size_t i = 0; i < number_variables_added; ++i) {
204 const std::string variable_name = ((*(mDoubleVariable.begin() + 2 * i))->GetSourceVariable()).Name();
205 const auto& r_var_z = KratosComponents<Variable<double>>::Get(variable_name + "_Z");
206 mDoubleVariable.push_back(&r_var_z);
207 mFirstDoubleDerivatives.push_back(&(r_var_z.GetTimeDerivative()));
208 mSecondDoubleDerivatives.push_back(&((r_var_z.GetTimeDerivative()).GetTimeDerivative()));
209 }
210 } else {
211 KRATOS_ERROR << "DOMAIN_SIZE can onbly be 2 or 3. It is: " << domain_size << std::endl;
212 }
213
214 mDomainSize = domain_size;
215 }
216
217 KRATOS_CATCH("")
218 }

227 void InitializeSolutionStep(
228 ModelPart& rModelPart,
229 TSystemMatrixType& rA,
230 TSystemVectorType& rDx,
231 TSystemVectorType& rb
232 ) override
233 {
234 KRATOS_TRY;
235
236 BDFBaseType::InitializeSolutionStep(rModelPart, rA, rDx, rb);
237
238 // Updating time derivatives (nodally for efficiency)
239 const int num_nodes = static_cast<int>( rModelPart.Nodes().size() );
240 const auto it_node_begin = rModelPart.Nodes().begin();
241
242 // Auxiliar fixed value
243 bool fixed = false;
244
245 IndexPartition<std::size_t>(num_nodes).for_each([&](std::size_t Index){
246 auto it_node = it_node_begin + Index;
247
248 std::size_t counter = 0;
249 for (auto p_var : mDoubleVariable) {
250
251 fixed = false;
252
253 // Derivatives
254 const auto& dvar = *mFirstDoubleDerivatives[counter];
255 const auto& d2var = *mSecondDoubleDerivatives[counter];
256
257 if (it_node->HasDofFor(d2var)) {
258 if (it_node->IsFixed(d2var)) {
259 it_node->Fix(*p_var);
260 fixed = true;
261 }
262 }
263
264 if (it_node->HasDofFor(dvar)) {
265 if (it_node->IsFixed(dvar) && !fixed) {
266 it_node->Fix(*p_var);
267 }
268 }
269
270 counter++;
271 }
272 });
273
274 KRATOS_CATCH("ResidualBasedBDFCustomScheme.InitializeSolutionStep");
275 }

286 void Predict(
287 ModelPart& rModelPart,
288 DofsArrayType& rDofSet,
289 TSystemMatrixType& A,
290 TSystemVectorType& Dx,
291 TSystemVectorType& b
292 ) override
293 {
294 KRATOS_TRY;
295
296 // Getting process info
297 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
298
299 // Getting delta time
300 const double delta_time = r_current_process_info[DELTA_TIME];
301
302 // Updating time derivatives (nodally for efficiency)
303 const int num_nodes = static_cast<int>( rModelPart.Nodes().size() );
304
305 // Getting first node iterator
306 const auto it_node_begin = rModelPart.Nodes().begin();
307
308 IndexPartition<std::size_t>(num_nodes).for_each([&](std::size_t Index){
309 auto it_node = it_node_begin + Index;
310
311 std::size_t counter = 0;
312 for (auto p_var : mDoubleVariable) {
313 // Derivatives
314 const auto& dvar = *mFirstDoubleDerivatives[counter];
315 const auto& d2var = *mSecondDoubleDerivatives[counter];
316
317 ComputePredictComponent(it_node, *p_var, dvar, d2var, delta_time);
318
319 counter++;
320 }
321
322 // Updating time derivatives
323 UpdateFirstDerivative(it_node);
324 UpdateSecondDerivative(it_node);
325 });
326
327 KRATOS_CATCH( "" );
328 }

339 int Check(const ModelPart& rModelPart) const override
340 {

342
343 const int err = BDFBaseType::Check(rModelPart);
344 if(err!=0) return err;
345

354
355 for ( auto p_var : mDoubleVariable)
356 KRATOS_CHECK_DOF_IN_NODE((*p_var), r_node)
357 }

359 KRATOS_CATCH( "" );

452 inline void UpdateFirstDerivative(NodesArrayType::iterator itNode) override
453 {
454 // DOUBLES
455 std::size_t counter = 0;
456 for (auto p_var : mDoubleVariable) {
457 double& dotun0 = itNode->FastGetSolutionStepValue(*mFirstDoubleDerivatives[counter]);
458 dotun0 = BDFBaseType::mBDF[0] * itNode->FastGetSolutionStepValue(*p_var);
459 for (std::size_t i_order = 1; i_order < BDFBaseType::mOrder + 1; ++i_order)
460 dotun0 += BDFBaseType::mBDF[i_order] * itNode->FastGetSolutionStepValue(*p_var, i_order);
461 counter++;
462 }

522 template<class TClassVar>
523 void ComputePredictComponent(
524 NodesArrayType::iterator itNode,
525 const TClassVar& rVariable,
526 const TClassVar& rDerivedVariable,
527 const TClassVar& rDerived2Variable,
528 const double DeltaTime
529 )
530 {
531 // Values
532 const double dot2un1 = itNode->FastGetSolutionStepValue(rDerived2Variable, 1);
533 const double dotun1 = itNode->FastGetSolutionStepValue(rDerivedVariable, 1);
534 const double un1 = itNode->FastGetSolutionStepValue(rVariable, 1);
535 const double dot2un0 = itNode->FastGetSolutionStepValue(rDerived2Variable);
536 double& dotun0 = itNode->FastGetSolutionStepValue(rDerivedVariable);
537 double& un0 = itNode->FastGetSolutionStepValue(rVariable);
538
539 if (itNode->HasDofFor(rDerived2Variable) && itNode->IsFixed(rDerived2Variable)) {
540 dotun0 = dot2un0;
541 for (std::size_t i_order = 1; i_order < BDFBaseType::mOrder + 1; ++i_order)
542 dotun0 -= BDFBaseType::mBDF[i_order] * itNode->FastGetSolutionStepValue(rDerivedVariable, i_order);
543 dotun0 /= BDFBaseType::mBDF[0];
544
545 un0 = dotun0;
546 for (std::size_t i_order = 1; i_order < BDFBaseType::mOrder + 1; ++i_order)
547 un0 -= BDFBaseType::mBDF[i_order] * itNode->FastGetSolutionStepValue(rVariable, i_order);
548 un0 /= BDFBaseType::mBDF[0];
549 } else if (itNode->HasDofFor(rDerivedVariable) && itNode->IsFixed(rDerivedVariable)) {
550 un0 = dotun0;
551 for (std::size_t i_order = 1; i_order < BDFBaseType::mOrder + 1; ++i_order)
552 un0 -= BDFBaseType::mBDF[i_order] * itNode->FastGetSolutionStepValue(rVariable, i_order);
553 un0 /= BDFBaseType::mBDF[0];
554 } else if (!itNode->IsFixed(rVariable)) {
555 un0 = un1 + DeltaTime * dotun1 + 0.5 * std::pow(DeltaTime, 2) * dot2un1;
556 }
557 }

563 void CreateVariablesList(Parameters ThisParameters)
564 {
565 const std::size_t n_variables = ThisParameters["solution_variables"].size();
566
567 // The current dimension
568 mDomainSize = ThisParameters["domain_size"].GetInt();
569
570 const auto variable_names = ThisParameters["solution_variables"].GetStringArray();
571
572 for (std::size_t p_var = 0; p_var < n_variables; ++p_var){
573 const std::string& variable_name = variable_names[p_var];
574
575 if(KratosComponents<Variable<double>>::Has(variable_name)){
576 const auto& r_var = KratosComponents<Variable<double>>::Get(variable_name);
577 mDoubleVariable.push_back(&r_var);
578 mFirstDoubleDerivatives.push_back(&(r_var.GetTimeDerivative()));
579 mSecondDoubleDerivatives.push_back(&((r_var.GetTimeDerivative()).GetTimeDerivative()));
580 } else if (KratosComponents< Variable< array_1d< double, 3> > >::Has(variable_name)) {
581 // Components
582 const auto& r_var_x = KratosComponents<Variable<double>>::Get(variable_name+"_X");
583 const auto& r_var_y = KratosComponents<Variable<double>>::Get(variable_name+"_Y");
584 mDoubleVariable.push_back(&r_var_x);
585 mDoubleVariable.push_back(&r_var_y);
586 mFirstDoubleDerivatives.push_back(&(r_var_x.GetTimeDerivative()));
587 mFirstDoubleDerivatives.push_back(&(r_var_y.GetTimeDerivative()));
588 mSecondDoubleDerivatives.push_back(&((r_var_x.GetTimeDerivative()).GetTimeDerivative()));
589 mSecondDoubleDerivatives.push_back(&((r_var_y.GetTimeDerivative()).GetTimeDerivative()));
590 if (mDomainSize == 3) {
591 const auto& r_var_z = KratosComponents<Variable<double>>::Get(variable_name+"_Z");
592 mDoubleVariable.push_back(&r_var_z);
593 mFirstDoubleDerivatives.push_back(&(r_var_z.GetTimeDerivative()));
594 mSecondDoubleDerivatives.push_back(&((r_var_z.GetTimeDerivative()).GetTimeDerivative()));
595 }
596 } else {
597 KRATOS_ERROR << "Only double and vector variables are allowed in the variables list." ;
598 }
599 }
600 }
