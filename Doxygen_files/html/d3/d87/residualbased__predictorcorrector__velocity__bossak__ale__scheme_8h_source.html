218 //predicts the solution at the current step as
219 // v = vold
220 void Predict(ModelPart& rModelPart,
221 DofsArrayType& rDofSet,
222 TSystemMatrixType& A,
223 TSystemVectorType& Dv,
224 TSystemVectorType& b) override
225 {
226 // if (rModelPart.GetCommunicator().MyPID() == 0)
227 // std::cout << "prediction" << std::endl;
228
229 int NumThreads = OpenMPUtils::GetNumThreads();
230 OpenMPUtils::PartitionVector NodePartition;
231 OpenMPUtils::DivideInPartitions(rModelPart.Nodes().size(), NumThreads, NodePartition);
232
233 #pragma omp parallel
234 {
235 //array_1d<double, 3 > DeltaDisp;
236
237 int k = OpenMPUtils::ThisThread();
238
239 ModelPart::NodeIterator NodesBegin = rModelPart.NodesBegin() + NodePartition[k];
240 ModelPart::NodeIterator NodesEnd = rModelPart.NodesBegin() + NodePartition[k + 1];
241
242 for (ModelPart::NodeIterator itNode = NodesBegin; itNode != NodesEnd; itNode++) {
243 array_1d<double, 3 > & OldVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY, 1);
244 double& OldPressure = (itNode)->FastGetSolutionStepValue(PRESSURE, 1);
245
246 //predicting velocity
247 //ATTENTION::: the prediction is performed only on free nodes
248 array_1d<double, 3 > & CurrentVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY);
249 double& CurrentPressure = (itNode)->FastGetSolutionStepValue(PRESSURE);
250
251 if ((itNode->pGetDof(VELOCITY_X))->IsFree())
252 (CurrentVelocity[0]) = OldVelocity[0];
253 if (itNode->pGetDof(VELOCITY_Y)->IsFree())
254 (CurrentVelocity[1]) = OldVelocity[1];
255 if (itNode->HasDofFor(VELOCITY_Z))
256 if (itNode->pGetDof(VELOCITY_Z)->IsFree())
257 (CurrentVelocity[2]) = OldVelocity[2];
258
259 if (itNode->pGetDof(PRESSURE)->IsFree())
260 CurrentPressure = OldPressure;
261
262 // updating time derivatives ::: please note that displacements and
263 // their time derivatives can not be consistently fixed separately
264 array_1d<double, 3 > DeltaVel;
265 noalias(DeltaVel) = CurrentVelocity - OldVelocity;
266 array_1d<double, 3 > & OldAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 1);
267 array_1d<double, 3 > & CurrentAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION);
268
269 SchemeType::UpdateAcceleration(CurrentAcceleration, DeltaVel, OldAcceleration);
270
271 // Pfem2 ALE update to eliminate convective term
272 noalias(itNode->FastGetSolutionStepValue(MESH_VELOCITY)) = itNode->FastGetSolutionStepValue(VELOCITY);
273 }
274 }
