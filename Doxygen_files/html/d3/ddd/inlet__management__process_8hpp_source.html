100

103

106

109 double timeInterval = rCurrentProcessInfo[DELTA_TIME];
110
111 unsigned int numberOfEulerianInletNodes = mrRemesh.Info->NumberOfEulerianInletNodes;
112 unsigned int numberOfLagrangianInletNodes = mrRemesh.Info->NumberOfLagrangianInletNodes;
113
114 if (currentTime < 2 * timeInterval)
115 {
116 mrRemesh.Info->RemovedNodes = 0;
117 if (mEchoLevel > 1)
118 std::cout << " First meshes: I repare the mesh without adding new nodes" << std::endl;
119 mrRemesh.Info->InitialNumberOfNodes = mrRemesh.Info->NumberOfNodes;
120 }
121
122 if (currentTime > 1.5 * timeInterval && numberOfLagrangianInletNodes > 0)
123 {
124 CheckAndCreateNewInletLayer();
125 }
126 else
127 {
128 CountInletNodes();
129 }
130
131 if (mEchoLevel > 1)
132 std::cout << " INLET MANAGEMENT PROCESS ]; " << std::endl;
133
134 KRATOS_CATCH(" ")
135 }

195 void CheckAndCreateNewInletLayer()
196
197 {
198 KRATOS_TRY
199
200 if (mEchoLevel > 1)
201 std::cout << " CheckAndCreateNewInletLayer " << std::endl;
202 const unsigned int dimension = mrModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
203 double maxSeparation = mrRemesh.Refine->CriticalRadius;
204
205 std::vector<Node<3>::Pointer> clonedNodes;
206 clonedNodes.clear();
207 clonedNodes.resize(1);
208 unsigned int numberClonedNodes = 0;
209 unsigned int sizeClonedNodes = 0;
210
211 for (ModelPart::NodesContainerType::iterator i_node = mrModelPart.NodesBegin(); i_node != mrModelPart.NodesEnd(); i_node++)
212 {
213 if (i_node->GetValue(LAGRANGIAN_INLET) == true)
214 {
215
216 ElementWeakPtrVectorType &neighb_elems = i_node->GetValue(NEIGHBOUR_ELEMENTS);
217 NodeWeakPtrVectorType &rN = i_node->GetValue(NEIGHBOUR_NODES);
218
219 if ((neighb_elems.size() == 0 && rN.size() == 0) || i_node->Is(RIGID))
220 {
221
222 const array_1d<double, 3> &inletDisplacement = i_node->FastGetSolutionStepValue(DISPLACEMENT);
223 double distanceFromOrigin = sqrt(inletDisplacement[0] * inletDisplacement[0] +
224 inletDisplacement[1] * inletDisplacement[1]);
225 if (dimension == 3)
226 {
227 distanceFromOrigin = sqrt(inletDisplacement[0] * inletDisplacement[0] +
228 inletDisplacement[1] * inletDisplacement[1] +
229 inletDisplacement[2] * inletDisplacement[2]);
230 }
231
232 if (distanceFromOrigin > maxSeparation)
233 {
234
235 if (i_node->Is(FLUID))
236 {
237 Node<3>::Pointer pnode = i_node->Clone();
238 sizeClonedNodes = numberClonedNodes + 1;
239 clonedNodes.resize(sizeClonedNodes);
240 clonedNodes[numberClonedNodes] = pnode;
241 numberClonedNodes++;
242 }
243 // inlet nodes will be replaced at their initial position
244 i_node->X() = i_node->X0();
245 i_node->Y() = i_node->Y0();
246 i_node->FastGetSolutionStepValue(DISPLACEMENT_X, 0) = 0;
247 i_node->FastGetSolutionStepValue(DISPLACEMENT_X, 1) = 0;
248 i_node->FastGetSolutionStepValue(DISPLACEMENT_Y, 0) = 0;
249 i_node->FastGetSolutionStepValue(DISPLACEMENT_Y, 1) = 0;
250 if (dimension == 3)
251 {
252 i_node->Z() = i_node->Z0();
253 i_node->FastGetSolutionStepValue(DISPLACEMENT_Z, 0) = 0;
254 i_node->FastGetSolutionStepValue(DISPLACEMENT_Z, 1) = 0;
255 }
256
257 }
258 }
259 }
260 }
261
262 for (unsigned int i = 0; i < sizeClonedNodes; i++)
263 {
264
265 Node<3>::Pointer pnode = clonedNodes[i];
266 double NodeIdParent = MesherUtilities::GetMaxNodeId(mrModelPart.GetParentModelPart());
267 double NodeId = MesherUtilities::GetMaxNodeId(mrModelPart);
268 unsigned int id = NodeIdParent + 1; // total model part node size
269
270 if (NodeId > NodeIdParent)
271 {
272 id = NodeId + 1;
273 }
274 pnode->SetId(id);
275 pnode->Free(VELOCITY_X);
276 pnode->Free(VELOCITY_Y);
277 if (dimension == 3)
278 {
279 pnode->Free(VELOCITY_Z);
280 }
281 pnode->GetValue(LAGRANGIAN_INLET) = false;
282 pnode->Reset(INLET);
283 pnode->Reset(RIGID);
284 pnode->Reset(BOUNDARY);
285 mrRemesh.NodalPreIds.push_back(pnode->Id());
286 mrModelPart.AddNode(pnode);
287 }
288
289 KRATOS_CATCH("")
290 }

292 void CountInletNodes()
293
294 {
295 KRATOS_TRY
296
297 unsigned int eulerianInletNodes = 0;
298 unsigned int lagrangianInletNodes = 0;
299 for (ModelPart::NodesContainerType::iterator i_node = mrModelPart.NodesBegin(); i_node != mrModelPart.NodesEnd(); i_node++)
300 {
301 if (i_node->GetValue(EULERIAN_INLET) == true)
302 {
303 eulerianInletNodes += 1;
304 }
305 if (i_node->GetValue(LAGRANGIAN_INLET) == true)
306 {
307 lagrangianInletNodes += 1;
308 }
309 }
310 mrRemesh.Info->NumberOfEulerianInletNodes = eulerianInletNodes;
311 mrRemesh.Info->NumberOfLagrangianInletNodes = lagrangianInletNodes;
312
313 KRATOS_CATCH("")
314 }
