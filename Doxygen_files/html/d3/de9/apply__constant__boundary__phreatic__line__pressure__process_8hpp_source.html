36 ApplyConstantBoundaryPhreaticLinePressureProcess(ModelPart& model_part,
37 Parameters rParameters
38 ) : Process(Flags()) , mrModelPart(model_part)
39 {
40 KRATOS_TRY
41
42 //only include validation with c++11 since raw_literals do not exist in c++03
43 Parameters default_parameters( R"(
44 {
45 "model_part_name":"PLEASE_CHOOSE_MODEL_PART_NAME",
46 "variable_name": "PLEASE_PRESCRIBE_VARIABLE_NAME",
47 "is_fixed": false,
48 "gravity_direction": 1,
49 "out_of_plane_direction": 2,
50 "first_reference_coordinate": [0.0,1.0,0.0],
51 "second_reference_coordinate": [1.0,0.5,0.0],
52 "specific_weight" : 10000.0,
53 "table" : 1
54 } )" );
55
56 // Some values need to be mandatorily prescribed since no meaningful default value exist. For this reason try accessing to them
57 // So that an error is thrown if they don't exist
58 rParameters["first_reference_coordinate"];
59 rParameters["second_reference_coordinate"];
60 rParameters["variable_name"];
61 rParameters["model_part_name"];
62
63 mIsFixedProvided = rParameters.Has("is_fixed");
64
65 // Now validate agains defaults -- this also ensures no type mismatch
66 rParameters.ValidateAndAssignDefaults(default_parameters);
67
68 mVariableName = rParameters["variable_name"].GetString();
69 mIsFixed = rParameters["is_fixed"].GetBool();
70 mGravityDirection = rParameters["gravity_direction"].GetInt();
71 mOutOfPlaneDirection = rParameters["out_of_plane_direction"].GetInt();
72 KRATOS_ERROR_IF(mGravityDirection == mOutOfPlaneDirection)
73 << "Gravity direction cannot be the same as Out-of-Plane directions "
74 << rParameters
75 << std::endl;
76
77 for (unsigned int i=0; i<N_DIM_3D; ++i)
78 if (i!=mGravityDirection && i!=mOutOfPlaneDirection) mHorizontalDirection = i;
79
80 mFirstReferenceCoordinate = rParameters["first_reference_coordinate"].GetVector();
81 mSecondReferenceCoordinate= rParameters["second_reference_coordinate"].GetVector();
82
83 mMinHorizontalCoordinate = std::min(mFirstReferenceCoordinate[mHorizontalDirection], mSecondReferenceCoordinate[mHorizontalDirection]);
84 mMaxHorizontalCoordinate = std::max(mFirstReferenceCoordinate[mHorizontalDirection], mSecondReferenceCoordinate[mHorizontalDirection]);
85
86 KRATOS_ERROR_IF_NOT(mMaxHorizontalCoordinate > mMinHorizontalCoordinate)
87 << "First and second point on the phreatic line have the same horizontal coordinate"
88 << rParameters
89 << std::endl;
90
91 mSlope = (mSecondReferenceCoordinate[mGravityDirection] - mFirstReferenceCoordinate[mGravityDirection])
92 /(mSecondReferenceCoordinate[mHorizontalDirection] - mFirstReferenceCoordinate[mHorizontalDirection]);
93
94 mSpecificWeight = rParameters["specific_weight"].GetDouble();
95
96 KRATOS_CATCH("")
97 }

113 void ExecuteInitialize() override
114 {
115 KRATOS_TRY
116
117 if (mrModelPart.NumberOfNodes() > 0) {
118 const Variable<double> &var = KratosComponents< Variable<double> >::Get(mVariableName);
119
120 block_for_each(mrModelPart.Nodes(), [&var, this](Node<3>& rNode){
121 if (mIsFixed) rNode.Fix(var);
122 else if (mIsFixedProvided) rNode.Free(var);
123
124 double height = 0.0;
125 if (rNode.Coordinates()[mHorizontalDirection] >= mMinHorizontalCoordinate && rNode.Coordinates()[mHorizontalDirection] <= mMaxHorizontalCoordinate) {
126 height = mSlope * (rNode.Coordinates()[mHorizontalDirection] - mFirstReferenceCoordinate[mHorizontalDirection]) + mFirstReferenceCoordinate[mGravityDirection];
127 } else if (rNode.Coordinates()[mHorizontalDirection] < mMinHorizontalCoordinate) {
128 height = mSlope * (mMinHorizontalCoordinate - mFirstReferenceCoordinate[mHorizontalDirection]) + mFirstReferenceCoordinate[mGravityDirection];
129 } else if (rNode.Coordinates()[mHorizontalDirection] > mMaxHorizontalCoordinate) {
130 height = mSlope * (mMaxHorizontalCoordinate - mFirstReferenceCoordinate[mHorizontalDirection]) + mFirstReferenceCoordinate[mGravityDirection];
131 }
132 const double distance = height - rNode.Coordinates()[mGravityDirection];
133 const double pressure = mSpecificWeight * distance ;
134
135 if (pressure > 0.0) {
136 rNode.FastGetSolutionStepValue(var) = pressure;
137 } else {
138 rNode.FastGetSolutionStepValue(var) = 0.0;
139 }
140 });
141
142 }
143
144 KRATOS_CATCH("")
145 }
