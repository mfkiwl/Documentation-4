721 void ComputeDihedralAngles(Vector& rDihedralAngles) const override
722 {
723 if(rDihedralAngles.size() != 24) {
724 rDihedralAngles.resize(24, false);
725 }
726 const auto faces = this->GenerateFaces();
727 // The three faces that contain the node i can be obtained by doing:
728 // faces[faces_0[i]], faces[faces_1[i]], faces[faces_2[i]]
729 const std::array<unsigned int, 8> faces_0 = {0,0,0,0,5,5,5,5};
730 const std::array<unsigned int, 8> faces_1 = {1,1,3,3,1,1,3,3};
731 const std::array<unsigned int, 8> faces_2 = {4,2,2,4,4,2,2,4};
732
733 array_1d<double, 3> normal_0, normal_1, normal_2;
734 double dihedral_angle_0, dihedral_angle_1, dihedral_angle_2;
735 for (unsigned int i = 0; i < 8; ++i) {
736 const TPointType& r_point_i = this->GetPoint(i);
737 noalias(normal_0) = faces[faces_0[i]].UnitNormal(r_point_i);
738 noalias(normal_1) = faces[faces_1[i]].UnitNormal(r_point_i);
739 noalias(normal_2) = faces[faces_2[i]].UnitNormal(r_point_i);
740 dihedral_angle_0 = std::acos(inner_prod(normal_0, -normal_1));
741 dihedral_angle_1 = std::acos(inner_prod(normal_0, -normal_2));
742 dihedral_angle_2 = std::acos(inner_prod(normal_2, -normal_1));
743 rDihedralAngles[i*3] = dihedral_angle_0;
744 rDihedralAngles[i*3 + 1] = dihedral_angle_1;
745 rDihedralAngles[i*3 + 2] = dihedral_angle_2;
746 }
747 }
