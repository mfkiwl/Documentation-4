168 // Do the local Dx vector import
169 Epetra_Vector local_dx(mpDofImport->TargetMap());
170 int i_err = local_dx.Import(rDx, *mpDofImport, Insert);
171 KRATOS_ERROR_IF_NOT(i_err == 0) << "Local Dx import failed!" << std::endl;
172
173 // Local thread variables
174 int dof_id;
175 TDataType dof_dx;
176
177 // Loop over Dofs
178 for (int i = 0; i < n_dofs; i++) {
179 auto it_dof = rDofSet.begin() + i;
180 if (it_dof->IsFree() && it_dof->GetSolutionStepValue(PARTITION_INDEX) == rank) {
181 dof_id = it_dof->EquationId();
182 const TDataType& r_dof_value = it_dof->GetSolutionStepValue(0);
183 dof_dx = local_dx[mpDofImport->TargetMap().LID(dof_id)];
184
185 const auto &r_current_variable = it_dof->GetVariable();
186 KeyType var_key = r_current_variable.IsComponent() ? r_current_variable.GetSourceVariable().Key() : r_current_variable.Key();
187 int var_local_key = r_local_key_map[var_key];
188
189 rSolutionNormsVector[var_local_key] += r_dof_value * r_dof_value;
190 rIncreaseNormsVector[var_local_key] += dof_dx * dof_dx;
191 rDofsCount[var_local_key]++;
192 }
193 }

209 void InitializeEpetraImport(
210 const DofsArrayType &rDofSet,
211 const TSystemVectorType &rDx)
212 {
213 int number_of_dofs = rDofSet.size();
214 int system_size = TSparseSpace::Size(rDx);
215 std::vector<int> index_array(number_of_dofs);
216
217 // Filling the array with the global ids
218 unsigned int counter = 0;
219 for (typename DofsArrayType::const_iterator i_dof = rDofSet.begin(); i_dof != rDofSet.end(); ++i_dof) {
220 const int id = i_dof->EquationId();
221 if (id < system_size) {
222 index_array[counter++] = id;
223 }
224 }
225
226 std::sort(index_array.begin(), index_array.end());
227 std::vector<int>::iterator new_end = std::unique(index_array.begin(), index_array.end());
228 index_array.resize(new_end - index_array.begin());
229
230 int check_size = -1;
231 int tot_update_dofs = index_array.size();
232 rDx.Comm().SumAll(&tot_update_dofs, &check_size, 1);
233 KRATOS_ERROR_IF(check_size < system_size)
234 << "DOF count is not correct. There are less dofs then expected.\n"
235 << "Expected number of active dofs: " << system_size << ", DOFs found: " << check_size << std::endl;
236
237 // Defining a map as needed
238 Epetra_Map dof_update_map(-1, index_array.size(), &(*(index_array.begin())), 0, rDx.Comm());
239
240 // Defining the import instance
241 std::unique_ptr<Epetra_Import> p_dof_import(new Epetra_Import(dof_update_map, rDx.Map()));
242 mpDofImport.swap(p_dof_import);
243
244 mEpetraImportIsInitialized = true;
245 }
