150 ShellT3_LocalCoordinateSystem CreateLocalCoordinateSystem() const override
151 {
152 const GeometryType & geom = GetGeometry();
153
154#ifdef COROT_POLAR_DECOMPOSITION
155
156 // reference coordinate system
157 ShellT3_LocalCoordinateSystem a( CreateReferenceCoordinateSystem() );
158
159 // current coordinate system using the 1-2 side alignement
160 ShellT3_LocalCoordinateSystem b( geom[0], geom[1], geom[2] );
161
162 double aX1 = a.X1(); double aY1 = a.Y1();
163 double bX1 = b.X1(); double bY1 = b.Y1();
164 double aX2 = a.X2(); double aY2 = a.Y2();
165 double bX2 = b.X2(); double bY2 = b.Y2();
166 double aX3 = a.X3(); double aY3 = a.Y3();
167 double bX3 = b.X3(); double bY3 = b.Y3();
168
169 // now we are in the local coordinate systems (reference and current), i.e. we are looking in the local Z direction
170 // which is the same for both coordinate systems.
171 // now we can compute the 2D deformation gradient between the 2 configurations, at the element center.
172
173 double C1 = 1/(aX1*aY2 - aX2*aY1 - aX1*aY3 + aX3*aY1 + aX2*aY3 - aX3*aY2);
174 double f11 = C1*(aY1 - aY3)*(bX1 - bX2) - C1*(aY1 - aY2)*(bX1 - bX3);
175 double f12 = C1*(aX1 - aX2)*(bX1 - bX3) - C1*(aX1 - aX3)*(bX1 - bX2);
176 double f21 = C1*(aY1 - aY3)*(bY1 - bY2) - C1*(aY1 - aY2)*(bY1 - bY3);
177 double f22 = C1*(aX1 - aX2)*(bY1 - bY3) - C1*(aX1 - aX3)*(bY1 - bY2);
178
179 // now we can extrapolate the rotation angle that makes this deformation gradient symmetric.
180 // F = R*U -> find R such that R'*F = U
181 double alpha = std::atan2( f21 - f12, f11 + f22 );
182
183 // this final coordinate system is the one in which
184 // the deformation gradient is equal to the stretch tensor
185 return ShellT3_LocalCoordinateSystem( geom[0], geom[1], geom[2], alpha );
186
187#else
188
189 return ShellT3_LocalCoordinateSystem(geom[0], geom[1], geom[2]);
190
191#endif // COROT_POLAR_DECOMPOSITION

195 VectorType CalculateLocalDisplacements(const ShellT3_LocalCoordinateSystem & LCS,
196 const VectorType & globalDisplacements) override
197 {
198 const GeometryType & geom = GetGeometry();
199
200 Vector3Type deformationalDisplacements;
201
202 QuaternionType Q = QuaternionType::FromRotationMatrix( LCS.Orientation() );
203
204 const Vector3Type & C = LCS.Center();
205
206 VectorType localDisplacements(18);
207
208 TransformationMatrixType T, T0;
209 Q.ToRotationMatrix( T );
210 mQ0.ToRotationMatrix( T0 );
211
212 for(unsigned int i = 0; i < 3; i++)
213 {
214 unsigned int index = i * 6;
215
216 // get deformational displacements
217
218 noalias( deformationalDisplacements ) = prod( T , geom[i] - C );
219 noalias( deformationalDisplacements ) -= prod( T0, geom[i].GetInitialPosition() - mC0 );
220
221 localDisplacements[index] = deformationalDisplacements[0];
222 localDisplacements[index + 1] = deformationalDisplacements[1];
223 localDisplacements[index + 2] = deformationalDisplacements[2];
224
225 // get deformational rotations
226
227 QuaternionType Qd = Q * mQN[i] * mQ0.conjugate();
228
229 Qd.ToRotationVector( localDisplacements[index + 3],
230 localDisplacements[index + 4],
231 localDisplacements[index + 5] );
232 }
233
234 return localDisplacements;
235 }

237 void FinalizeCalculations(const ShellT3_LocalCoordinateSystem & LCS,
238 const VectorType & globalDisplacements,
239 const VectorType & localDisplacements,
240 MatrixType & rLeftHandSideMatrix,
241 VectorType & rRightHandSideVector,
242 const bool RHSrequired,
243 const bool LHSrequired) override
244 {
245 // Get the total rotation matrix (local - to - global)
246 // Note: do NOT include the warpage correction matrix!
247 // Explanation:
248 // The Warpage correction matrix computed by the LocalCoordinateSystem is a Linear Projector.
249 // It should be used in a LinearCoordinateTransformation.
250 // Here instead we already calculate a nonlinear Projector (P = Pu - S * G)!
251
252 MatrixType T(18, 18);
253 LCS.ComputeTotalRotationMatrix( T );
254
255 // Form all matrices:
256 // S: Spin-Fitter matrix
257 // G: Spin-Lever matrix
258 // P: Projector (Translational & Rotational)
259
260 MatrixType P( EICR::Compute_Pt(3) );
261 MatrixType S( EICR::Compute_S(LCS.Nodes()) );
262 MatrixType G( RotationGradient(LCS) );
263 noalias( P ) -= prod( S, G );
264
265 // Compute the projected local forces ( pe = P' * RHS ).
266 // Note: here the RHS is already given as a residual vector -> - internalForces -> (pe = - Ke * U)
267 // so projectedLocalForces = - P' * Ke * U
268
269 VectorType projectedLocalForces( prod( trans( P ), rRightHandSideVector ) );
270
271 // Compute the Right-Hand-Side vector in global coordinate system (- T' * P' * Km * U).
272 // At this point the computation of the Right-Hand-Side is complete.
273
274 noalias( rRightHandSideVector ) = prod( trans( T ), projectedLocalForces );
275
276 // Begin the computation of the Left-Hand-Side Matrix :
277
278 if(!LHSrequired) return; // avoid useless calculations!
279
280 // This is a temporary matrix to store intermediate values
281 // to avoid extra dynamic memory allocations!
282
283 MatrixType temp(18, 18);
284
285 // H: Axial Vector Jacobian
286 MatrixType H( EICR::Compute_H(localDisplacements) );
287
288 // Step 1: ( K.M : Material Stiffness Matrix )
289 // Apply the projector to the Material Stiffness Matrix (Ke = P' * Km * H * P)
290 // At this point 'LHS' contains the 'projected' Material Stiffness matrix
291 // in local corotational coordinate system
292
293 noalias( temp ) = prod( rLeftHandSideMatrix, H );
294 noalias( rLeftHandSideMatrix ) = prod( temp, P );
295 noalias( temp ) = prod( trans( P ), rLeftHandSideMatrix );
296 rLeftHandSideMatrix.swap( temp );
297
298 // Step 2: ( K.GP: Equilibrium Projection Geometric Stiffness Matrix )
299 // First assemble the 'Fnm' matrix with the Spins of the nodal forces.
300 // Actually at this point the 'Fnm' Matrix is the 'Fn' Matrix,
301 // because it only contains the spins of the 'translational' forces.
302 // At this point 'LHS' contains also this term of the Geometric stiffness
303 // (Ke = (P' * Km * H * P) - (G' * Fn' * P))
304
305 MatrixType Fnm(18, 3, 0.0);
306
307 EICR::Spin_AtRow( projectedLocalForces, Fnm, 0 );
308 EICR::Spin_AtRow( projectedLocalForces, Fnm, 6 );
309 EICR::Spin_AtRow( projectedLocalForces, Fnm, 12 );
310
311 noalias( temp ) = prod( trans( G ), trans( Fnm ) );
312 noalias( rLeftHandSideMatrix ) += prod( temp, P ); // note: '+' not '-' because the RHS already has the negative sign
313
314 // Step 3: ( K.GR: Rotational Geometric Stiffness Matrix )
315 // Add the Spins of the nodal moments to 'Fnm'.
316 // At this point 'LHS' contains also this term of the Geometric stiffness
317 // (Ke = (P' * Km * H * P) - (G' * Fn' * P) - (Fnm * G))
318
319 EICR::Spin_AtRow( projectedLocalForces, Fnm, 3 );
320 EICR::Spin_AtRow( projectedLocalForces, Fnm, 9 );
321 EICR::Spin_AtRow( projectedLocalForces, Fnm, 15 );
322
323 noalias( rLeftHandSideMatrix ) += prod( Fnm, G ); // note: '+' not '-' because the RHS already has the negative sign
324
325 // Step 4: (Global Stiffness Matrix)
326 // Transform the LHS to the Global coordinate system.
327 // T' * [(P' * Km * H * P) - (G' * Fn' * P) - (Fnm * G)] * T
328 noalias( temp ) = prod( rLeftHandSideMatrix, T );
329 noalias( rLeftHandSideMatrix ) = prod( trans( T ), temp );
330 }

332 MatrixType GetNodalDeformationalRotationTensor(const ShellT3_LocalCoordinateSystem & LCS,
333 const Vector& globalDisplacements,
334 size_t nodeid) override
335 {
336 if(nodeid>2) return IdentityMatrix(3,3);
337
338 QuaternionType Q = QuaternionType::FromRotationMatrix( LCS.Orientation() );
339
340 QuaternionType Qd = Q * mQN[nodeid] * mQ0.conjugate();
341
342 MatrixType R(3,3);
343 Qd.ToRotationMatrix(R);
344 return R;

347 MatrixType GetNodalDeformationalRotationTensor(const ShellT3_LocalCoordinateSystem & LCS,
348 const Vector& globalDisplacements,
349 const Vector& N) override
350 {
351 QuaternionType Q = QuaternionType::FromRotationMatrix( LCS.Orientation() );
352
353 RealType qx(0.0);
354 RealType qy(0.0);
355 RealType qz(0.0);
356 RealType qw(0.0);
357
358 for(int i = 0; i < 3; i++)
359 {
360 QuaternionType iQd = Q * mQN[i] * mQ0.conjugate();
361 iQd.normalize();
362 RealType iN = N[i];
363 qx += iN * iQd.X();
364 qy += iN * iQd.Y();
365 qz += iN * iQd.Z();
366 qw += iN * iQd.W();
367 }
368
369 MatrixType R(3, 3);
370
371 QuaternionType Qd(qw, qx, qy, qz);
372 Qd.normalize();
373 Qd.ToRotationMatrix(R);
374
375 return R;
376 }

390 inline MatrixType RotationGradient(const ShellT3_LocalCoordinateSystem& LCS)
391 {
392 Matrix G(3, 18, 0.0);
393 //const GeometryType & geom = GetGeometry();
394
395 ShellT3_LocalCoordinateSystem a( CreateReferenceCoordinateSystem() ); // REF coordinate system
396
397 ShellT3_LocalCoordinateSystem::Vector3ContainerType nodes = a.Nodes();
398
399 double aX1 = a.X1(); double aY1 = a.Y1();
400 double aX2 = a.X2(); double aY2 = a.Y2();
401 double aX3 = a.X3(); double aY3 = a.Y3();
402
403 double pert = std::sqrt(2.0*a.Area()) * 1.0E-3;
404
405 Vector3Type e3;
406 Vector3Type e1;
407
408 for(int i = 0; i < 3; i++) // for each node
409 {
410 Vector3Type& iNode = nodes[i];
411
412 int index = i * 6;
413
414 for(int j = 0; j < 3; j++) // for each component in [x, y, z]
415 {
416 double saved_coord = iNode[j]; // save the current coordinate
417
418 iNode[j] += pert; // apply perturbation
419
420 ShellT3_LocalCoordinateSystem b( nodes[0], nodes[1], nodes[2] ); // perturbed coordinate system (1-2 side alignement)
421
422 double bX1 = b.X1(); double bY1 = b.Y1();
423 double bX2 = b.X2(); double bY2 = b.Y2();
424 double bX3 = b.X3(); double bY3 = b.Y3();
425
426 // now we are in the local coordinate systems (reference and current), i.e. we are looking in the local Z direction
427 // which is the same for both coordinate systems.
428 // now we can compute the 2D deformation gradient between the 2 configurations, at the element center.
429
430 double C1 = 1/(aX1*aY2 - aX2*aY1 - aX1*aY3 + aX3*aY1 + aX2*aY3 - aX3*aY2);
431 double f11 = C1*(aY1 - aY3)*(bX1 - bX2) - C1*(aY1 - aY2)*(bX1 - bX3);
432 double f12 = C1*(aX1 - aX2)*(bX1 - bX3) - C1*(aX1 - aX3)*(bX1 - bX2);
433 double f21 = C1*(aY1 - aY3)*(bY1 - bY2) - C1*(aY1 - aY2)*(bY1 - bY3);
434 double f22 = C1*(aX1 - aX2)*(bY1 - bY3) - C1*(aX1 - aX3)*(bY1 - bY2);
435
436 double alpha = std::atan2( f21 - f12, f11 + f22 );
437
438 ShellT3_LocalCoordinateSystem c( nodes[0], nodes[1], nodes[2], alpha );// perturbed coordinate system (polar decomposition)
439
440 // save the (numerical) rotation gradient
441
442 noalias( e3 ) = c.Vz();
443 noalias( e1 ) = c.Vx();
444
445 G(0, index + j) = -e3(1) / pert; // - d(e3.y)/dxj , where xj is x,y,z for j=0,1,2
446 G(1, index + j) = e3(0) / pert; // + d(e3.x)/dxj , where xj is x,y,z for j=0,1,2
447 G(2, index + j) = e1(1) / pert; // + d(e1.y)/dxj , where xj is x,y,z for j=0,1,2
448
449 iNode[j] = saved_coord; // restore the current coordinate
450 }
451 }
452
453 return G;
454 }
