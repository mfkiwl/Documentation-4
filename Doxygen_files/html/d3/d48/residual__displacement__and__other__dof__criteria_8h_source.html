140 bool PostCriteria(
141 ModelPart& rModelPart,
142 DofsArrayType& rDofSet,
143 const TSystemMatrixType& rA,
144 const TSystemVectorType& rDx,
145 const TSystemVectorType& rb
146 ) override
147 {
148 if (TSparseSpace::Size(rb) != 0) { //if we are solving for something
149 TDataType ratio_displacement = 0.0;
150 TDataType ratio_other_dof = 0.0;
151
152 SizeType disp_size;
153 CalculateResidualNorm(rModelPart, mCurrentResidualDispNorm, mCurrentResidualOtherDoFNorm, disp_size, rDofSet, rb);
154
155 if (mInitialResidualDispNorm == 0.0) {
156 ratio_displacement = 0.0;
157 } else {
158 ratio_displacement = mCurrentResidualDispNorm/mInitialResidualDispNorm;
159 }
160
161 if (mInitialResidualOtherDoFNorm == 0.0) {
162 ratio_other_dof = 0.0;
163 } else {
164 ratio_other_dof = mCurrentResidualOtherDoFNorm/mInitialResidualOtherDoFNorm;
165 }
166
167 const std::size_t system_size = TSparseSpace::Size(rb);
168 const TDataType absolute_norm_disp = (mCurrentResidualDispNorm/static_cast<TDataType>(disp_size));
169 const TDataType absolute_norm_other_dof = (mCurrentResidualOtherDoFNorm/static_cast<TDataType>(system_size - disp_size));
170
171 KRATOS_INFO_IF("ResidualDisplacementAndOtherDoFCriteria", this->GetEchoLevel() > 0) << "RESIDUAL DISPLACEMENT CRITERION :: Ratio = "<< ratio_displacement << "; Norm = " << absolute_norm_disp << std::endl;
172 KRATOS_INFO_IF("ResidualDisplacementAndOtherDoFCriteria", this->GetEchoLevel() > 0) << "RESIDUAL " << mOtherDoFName << " CRITERION :: Ratio = "<< ratio_other_dof << "; Norm = " << absolute_norm_other_dof << std::endl;
173
174 rModelPart.GetProcessInfo()[CONVERGENCE_RATIO] = ratio_displacement;
175 rModelPart.GetProcessInfo()[RESIDUAL_NORM] = absolute_norm_disp;
176
177 if ((ratio_displacement <= mRatioTolerance || absolute_norm_disp < mAbsoluteTolerance) && (ratio_other_dof <= mRatioTolerance || absolute_norm_other_dof < mAbsoluteTolerance)) {
178 KRATOS_INFO_IF("ResidualDisplacementAndOtherDoFCriteria", this->GetEchoLevel() > 0) << "Convergence is achieved" << std::endl;
179 return true;
180 } else {
181 return false;
182 }
183 } else {
184 return true;
185 }

209 void InitializeSolutionStep(
210 ModelPart& rModelPart,
211 DofsArrayType& rDofSet,
212 const TSystemMatrixType& rA,
213 const TSystemVectorType& rDx,
214 const TSystemVectorType& rb
215 ) override
216 {
217 BaseType::InitializeSolutionStep(rModelPart, rDofSet, rA, rDx, rb);
218
219 // Filling mActiveDofs when MPC exist
220 if (rModelPart.NumberOfMasterSlaveConstraints() > 0) {
221 mActiveDofs.resize(rDofSet.size());
222
223 #pragma omp parallel for
224 for(int i=0; i<static_cast<int>(mActiveDofs.size()); ++i) {
225 mActiveDofs[i] = true;
226 }
227
228 #pragma omp parallel for
229 for (int i=0; i<static_cast<int>(rDofSet.size()); ++i) {
230 const auto it_dof = rDofSet.begin() + i;
231 if (it_dof->IsFixed()) {
232 mActiveDofs[it_dof->EquationId()] = false;
233 }
234 }
235
236 for (const auto& r_mpc : rModelPart.MasterSlaveConstraints()) {
237 for (const auto& r_dof : r_mpc.GetMasterDofsVector()) {
238 mActiveDofs[r_dof->EquationId()] = false;
239 }
240 for (const auto& r_dof : r_mpc.GetSlaveDofsVector()) {
241 mActiveDofs[r_dof->EquationId()] = false;
242 }
243 }
244 }
245
246 SizeType size_residual;
247 CalculateResidualNorm(rModelPart, mInitialResidualDispNorm, mInitialResidualOtherDoFNorm, size_residual, rDofSet, rb);
248 }

335 virtual void CalculateResidualNorm(
336 ModelPart& rModelPart,
337 TDataType& rResidualSolutionNormDisp,
338 TDataType& rResidualSolutionNormOtherDof,
339 SizeType& rDofNumDisp,
340 DofsArrayType& rDofSet,
341 const TSystemVectorType& rb
342 )
343 {
344 // Initialize
345 TDataType residual_solution_norm_disp = TDataType();
346 TDataType residual_solution_norm_other_dof = TDataType();
347 SizeType disp_dof_num = 0;
348
349 // Auxiliary values
350 TDataType residual_dof_value = 0.0;
351 const auto it_dof_begin = rDofSet.begin();
352 const int number_of_dof = static_cast<int>(rDofSet.size());
353
354 // Loop over Dofs
355 if (rModelPart.NumberOfMasterSlaveConstraints() > 0) {
356 #pragma omp parallel for firstprivate(residual_dof_value) reduction(+:residual_solution_norm_disp, residual_solution_norm_other_dof, disp_dof_num)
357 for (int i = 0; i < number_of_dof; i++) {
358 auto it_dof = it_dof_begin + i;
359
360 const IndexType dof_id = it_dof->EquationId();
361 residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
362
363 if (mActiveDofs[dof_id]) {
364 if (it_dof->GetVariable() == DISPLACEMENT_X || it_dof->GetVariable() == DISPLACEMENT_Y || it_dof->GetVariable() == DISPLACEMENT_Z) {
365 residual_solution_norm_disp += std::pow(residual_dof_value, 2);
366 disp_dof_num++;
367 } else {
368 residual_solution_norm_other_dof += std::pow(residual_dof_value, 2);
369 }
370 }
371 }
372 } else {
373 #pragma omp parallel for firstprivate(residual_dof_value) reduction(+:residual_solution_norm_disp, residual_solution_norm_other_dof, disp_dof_num)
374 for (int i = 0; i < number_of_dof; i++) {
375 auto it_dof = it_dof_begin + i;
376
377 if (!it_dof->IsFixed()) {
378 const IndexType dof_id = it_dof->EquationId();
379 residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
380
381 if (it_dof->GetVariable() == DISPLACEMENT_X || it_dof->GetVariable() == DISPLACEMENT_Y || it_dof->GetVariable() == DISPLACEMENT_Z) {
382 residual_solution_norm_disp += std::pow(residual_dof_value, 2);
383 disp_dof_num++;
384 } else {
385 residual_solution_norm_other_dof += std::pow(residual_dof_value, 2);
386 }
387 }
388 }
389 }
390
391 rDofNumDisp = disp_dof_num;
392 rResidualSolutionNormDisp = std::sqrt(residual_solution_norm_disp);
393 rResidualSolutionNormOtherDof = std::sqrt(residual_solution_norm_other_dof);
394 }
