353  // Loop over Dofs

355  #pragma omp parallel for firstprivate(residual_dof_value) reduction(+:residual_solution_norm_disp, residual_solution_norm_other_dof, disp_dof_num)
356  for (int i = 0; i < number_of_dof; i++) {
357  auto it_dof = it_dof_begin + i;
358 
359  const IndexType dof_id = it_dof->EquationId();
360  residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
361 
362  if (mActiveDofs[dof_id]) {
363  if (it_dof->GetVariable() == DISPLACEMENT_X || it_dof->GetVariable() == DISPLACEMENT_Y || it_dof->GetVariable() == DISPLACEMENT_Z) {
364  residual_solution_norm_disp += std::pow(residual_dof_value, 2);
365  disp_dof_num++;

370  }
371  } else {

374  auto it_dof = it_dof_begin + i;
375 
376  if (!it_dof->IsFixed()) {
377  const IndexType dof_id = it_dof->EquationId();
378  residual_dof_value = TSparseSpace::GetValue(rb,dof_id);
379 
380  if (it_dof->GetVariable() == DISPLACEMENT_X || it_dof->GetVariable() == DISPLACEMENT_Y || it_dof->GetVariable() == DISPLACEMENT_Z) {
381  residual_solution_norm_disp += std::pow(residual_dof_value, 2);
