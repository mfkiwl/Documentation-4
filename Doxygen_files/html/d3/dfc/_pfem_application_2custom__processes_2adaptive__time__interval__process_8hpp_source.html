117 updatedTime -= initialTimeInterval;
118 unsigned int previousMilestoneStep=updatedTime/initialTimeInterval;

121 updatedTimeInterval =currentTimeInterval;

123 bool badVelocityConvergence=rCurrentProcessInfo[BAD_VELOCITY_CONVERGENCE];

144 milestoneTimeReached=true;
145 }else{
146 milestoneTimeReached=false;
147 rCurrentProcessInfo.SetValue(DELTA_TIME,updatedTimeInterval);
148 }
149
150 if(timeIntervalReduced==false){
151
152 if(updatedTimeInterval>(2.0*minimumTimeInterval)){
153
154 const unsigned int dimension = mrModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
155 if(dimension==2){
156 CheckNodalConditionForTimeStepReduction(updatedTimeInterval,increaseTimeInterval,timeIntervalReduced);
157 if(timeIntervalReduced==false){
158 CheckElementalConditionForTimeStepReduction(increaseTimeInterval);

165 IncreaseTimeInterval(updatedTimeInterval,deltaTimeToNewMilestone,tolerance,increaseTimeInterval);
166 }
167 else{
168 increaseTimeInterval=false;
169 }
170
171 }
172
173 double newTimeInterval = rCurrentProcessInfo[DELTA_TIME];
174 double milestoneGap=fabs(newTimeInterval-deltaTimeToNewMilestone);
175
176 if(milestoneGap<0.49*newTimeInterval && milestoneTimeReached==false){
177 /* std::cout<<"the milestone is very close, I add "<<milestoneGap<<" to "<<newTimeInterval<<std::endl;*/
178 newTimeInterval+=milestoneGap;
179 rCurrentProcessInfo.SetValue(DELTA_TIME,newTimeInterval);

184 rCurrentProcessInfo.SetValue(TIME,updatedTime);
185 rCurrentProcessInfo.SetValue(CURRENT_DELTA_TIME,newTimeInterval);

192 rCurrentProcessInfo.SetValue(CURRENT_DELTA_TIME,currentTimeInterval);
193 }

195

206 void CheckNodalConditionForTimeStepReduction(double updatedTimeInterval,
207 bool &increaseTimeInterval,
208 bool &timeIntervalReduced)
209 {
210
211 ProcessInfo& rCurrentProcessInfo = mrModelPart.GetProcessInfo();
212
213#pragma omp parallel
214 {
215 ModelPart::NodeIterator NodeBegin;
216 ModelPart::NodeIterator NodeEnd;
217 OpenMPUtils::PartitionedIterators(mrModelPart.Nodes(),NodeBegin,NodeEnd);
218 for(ModelPart::NodeIterator itNode = NodeBegin; itNode != NodeEnd; ++itNode)
219 {
220 if(itNode->IsNot(TO_ERASE) && itNode->IsNot(ISOLATED) && itNode->IsNot(SOLID)){
221 const array_1d<double,3> &Vel = itNode->FastGetSolutionStepValue(VELOCITY);
222 double NormVelNode=0;
223 for (unsigned int d = 0; d < 3; ++d){
224 NormVelNode+=Vel[d] * Vel[d];
225 }
226 double motionInStep=sqrt(NormVelNode)*updatedTimeInterval;
227 double unsafetyFactor=0;
228 NodeWeakPtrVectorType& nNodes = itNode->GetValue(NEIGHBOUR_NODES);
229 for(auto i_nnode : nNodes)
230 {
231 array_1d<double,3> CoorNeighDifference=itNode->Coordinates()-i_nnode.Coordinates();
232 double squaredDistance=0;
233 for (unsigned int d = 0; d < 3; ++d){
234 squaredDistance+=CoorNeighDifference[d]*CoorNeighDifference[d];
235 }
236 double nodeDistance=sqrt(squaredDistance);
237 double tempUnsafetyFactor=motionInStep/nodeDistance;
238 if(tempUnsafetyFactor>unsafetyFactor){
239 unsafetyFactor=tempUnsafetyFactor;
240 }
241 }
242
243 if(unsafetyFactor>0.35){
244 increaseTimeInterval=false;
245 if(unsafetyFactor>1.0){
246 double temporaryTimeInterval = rCurrentProcessInfo[DELTA_TIME];
247 double reducedTimeInterval=0.5*updatedTimeInterval;
248 if(reducedTimeInterval<temporaryTimeInterval){
249 rCurrentProcessInfo.SetValue(DELTA_TIME,reducedTimeInterval);
250 /* std::cout<<"reducing time step (nodal criterion)"<<reducedTimeInterval<<std::endl; */
251 rCurrentProcessInfo.SetValue(TIME_INTERVAL_CHANGED,true);
252 timeIntervalReduced=true;
253 break;
254 }
255 }
256 }
257 }
258 }
259 }

427 {
