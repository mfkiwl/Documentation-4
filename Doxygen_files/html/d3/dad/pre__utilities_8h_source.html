470 void ApplyConcentricForceOnParticles(ModelPart& r_model_part, const array_1d<double, 3>& center, const double density_for_artificial_gravity) {
471 auto& pElements = r_model_part.GetCommunicator().LocalMesh().Elements();
472
473 #pragma omp parallel for
474 for (int k = 0; k < (int)pElements.size(); k++) {
475 auto it = pElements.begin() + k;
476 auto& node = it->GetGeometry()[0];
477 const array_1d<double, 3>& coords = node.Coordinates();
478 array_1d<double, 3> vector_particle_to_center;
479 noalias(vector_particle_to_center) = center - coords;
480 const double distance_to_center = MathUtils<double>::Norm3(vector_particle_to_center);
481 const double inv_dist = 1.0 / distance_to_center;
482 array_1d<double, 3> force;
483 SphericParticle* spheric_p_particle = dynamic_cast<SphericParticle*> (&*it);
