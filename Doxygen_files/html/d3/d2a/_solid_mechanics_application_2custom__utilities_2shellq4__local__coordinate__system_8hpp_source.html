48 template< class TVector3 >
49 inline static double NormalizeVector3(TVector3 & V)
50 {
51 double vn = V(0) * V(0) + V(1) * V(1) + V(2) * V(2);
52 if(vn != 0.0 && vn != 1.0) {
53 vn = std::sqrt(vn);
54 V /= vn;
55 }
56 return vn;
57 }

61 ShellQ4_LocalCoordinateSystem(const Vector3Type & P1global,
62 const Vector3Type & P2global,
63 const Vector3Type & P3global,
64 const Vector3Type & P4global)
65 : mP(4)
66 , mOrientation(3, 3)
67 {
68 // Form the basis vectors alligning the local X direction
69 // with the 1-2 side
70
71 // compute the central point
72 noalias( mCenter ) = P1global;
73 noalias( mCenter ) += P2global;
74 noalias( mCenter ) += P3global;
75 noalias( mCenter ) += P4global;
76 mCenter *= 0.25;
77
78 // compute the diagonal vectors
79 Vector3Type d13( P3global - P1global );
80 Vector3Type d24( P4global - P2global );
81
82 // compute the Normal vector at the element center
83 // as the cross product of the 2 diagonals.
84 // While normalizing the normal vector save its norm to compute the area.
85 // Note: the norm should be divided by 2 because its computed from the
86 // cross product of the diagonals, which gives twice the area!
87 Vector3Type e3;
88 MathUtils<RealType>::CrossProduct(e3, d13, d24);
89 mArea = NormalizeVector3( e3 );
90 mArea /= 2.0;
91
92 // compute the local X direction parallel to the projection of the side 1-2 onto
93 // the local xy plane.
94 Vector3Type e1( P2global - P1global );
95 RealType e1_dot_e3 = MathUtils<RealType>::Dot(e1, e3);
96 noalias( e1 ) -= e1_dot_e3 * e3;
97 NormalizeVector3( e1 );
98
99 // finally compute the local Y direction to be orthogonal to both X and Z local directions
100 Vector3Type e2;
101 MathUtils<RealType>::CrossProduct(e2, e3, e1);
102 NormalizeVector3( e2 );
103
104 // form the 3x3 transformation matrix
105 for(int i = 0; i < 3; i++)
106 {
107 mOrientation(0, i) = e1(i);
108 mOrientation(1, i) = e2(i);
109 mOrientation(2, i) = e3(i);
110 }
111
112 // transform global coordinates to the local coordinate system
113 for(int i = 0; i < 3; i++)
114 {
115 mP[0](i) = mOrientation(i, 0) * (P1global(0) - mCenter(0)) + mOrientation(i, 1) * (P1global(1) - mCenter(1)) + mOrientation(i, 2) * (P1global(2) - mCenter(2));
116 mP[1](i) = mOrientation(i, 0) * (P2global(0) - mCenter(0)) + mOrientation(i, 1) * (P2global(1) - mCenter(1)) + mOrientation(i, 2) * (P2global(2) - mCenter(2));
117 mP[2](i) = mOrientation(i, 0) * (P3global(0) - mCenter(0)) + mOrientation(i, 1) * (P3global(1) - mCenter(1)) + mOrientation(i, 2) * (P3global(2) - mCenter(2));
118 mP[3](i) = mOrientation(i, 0) * (P4global(0) - mCenter(0)) + mOrientation(i, 1) * (P4global(1) - mCenter(1)) + mOrientation(i, 2) * (P4global(2) - mCenter(2));
119 }
120 }

221 inline void ComputeTotalRotationMatrix( MatrixType & R )const
222 {
223 size_t mat_size = 24;
224 if(R.size1() != mat_size || R.size2() != mat_size)
225 R.resize(mat_size, mat_size, false);
226
227 noalias( R ) = ZeroMatrix(mat_size, mat_size);
228
229 for(size_t k = 0; k < 8; k++)
230 {
231 size_t i = k * 3;
232 R(i , i ) = mOrientation(0, 0); R(i , i+1) = mOrientation(0, 1); R(i , i+2) = mOrientation(0, 2);
233 R(i+1, i ) = mOrientation(1, 0); R(i+1, i+1) = mOrientation(1, 1); R(i+1, i+2) = mOrientation(1, 2);
234 R(i+2, i ) = mOrientation(2, 0); R(i+2, i+1) = mOrientation(2, 1); R(i+2, i+2) = mOrientation(2, 2);
235 }

264 inline void ComputeLocalToGlobalTransformationMatrix( MatrixType & R )
265 {
266 size_t mat_size = 24;
267 if(R.size1() != mat_size || R.size2() != mat_size)
268 R.resize(mat_size, mat_size, false);
269
270 noalias( R ) = ZeroMatrix(mat_size, mat_size);
271
272 // form the global transformation matrix
273 for(size_t k = 0; k < 8; k++)
274 {
275 size_t i = k * 3;
276 R(i , i ) = mOrientation(0, 0); R(i , i+1) = mOrientation(0, 1); R(i , i+2) = mOrientation(0, 2);
277 R(i+1, i ) = mOrientation(1, 0); R(i+1, i+1) = mOrientation(1, 1); R(i+1, i+2) = mOrientation(1, 2);
278 R(i+2, i ) = mOrientation(2, 0); R(i+2, i+1) = mOrientation(2, 1); R(i+2, i+2) = mOrientation(2, 2);
279 }
280
281 // if needed, include the warpage correction.
282 if(this->IsWarped())
283 {
284 MatrixType W( IdentityMatrix(24, 24) );
285 RealType z1 = this->Z1();
286 RealType z2 = this->Z2();
287 RealType z3 = this->Z3();
288 RealType z4 = this->Z4();
289 W( 0, 4) = -z1;
290 W( 1, 3) = z1;
291 W( 6, 10) = -z2;
292 W( 7, 9) = z2;
293 W(12, 16) = -z3;
294 W(13, 15) = z3;
295 W(18, 22) = -z4;
296 W(19, 21) = z4;
297 R = prod( W, R );
298 }

307 template<int TComponent>
308 inline const double Coord_ij(int Tid1, int Tid2)const {
309 KRATOS_ERROR_IF_NOT(TComponent >= 0 && TComponent < 3) <<
310 "The component index should be 0-based, from 0 to 2" <<std::endl;
311 double ci, cj;
312 if(Tid1 < 5) {
313 int i = Tid1 - 1;
314 ci = mP[i][TComponent];
315 }
316 else {
317 int i = Tid1 - 1 - 4;
318 ci = mP[i][TComponent] + mP[i == 3 ? 0 : i + 1][TComponent];
319 }
320 if(Tid2 < 5) {
321 int i = Tid2 - 1;
322 cj = mP[i][TComponent];
323 }
324 else {
325 int i = Tid2 - 1 - 4;
326 cj = mP[i][TComponent] + mP[i == 3 ? 0 : i + 1][TComponent];
327 }
328 return cj - ci;
329 }
