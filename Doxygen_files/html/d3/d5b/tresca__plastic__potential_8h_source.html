123 double J3, lode_angle;
124 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(rDeviator, J3);
125 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
126
127 const double checker = std::abs(lode_angle * 180.0 / Globals::Pi);
128
129 double c2, c3;
130 if (checker < 29.0) {
131 c2 = 2.0 * (std::cos(lode_angle) + std::sin(lode_angle) * std::tan(3.0 * lode_angle));
132 c3 = std::sqrt(3.0) * std::sin(lode_angle) / (J2 * std::cos(3.0 * lode_angle));
133 } else {
134 c2 = std::sqrt(3.0);
135 c3 = 0.0;
136 }
137
138 noalias(rGFlux) = c2 * second_vector + c3 * third_vector;
