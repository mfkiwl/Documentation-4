678  template<class TClassType, bool TImplicit = false>
679  void PerformMortarOperations(
680  MatrixType& rA,
681  std::vector<VectorType>& rb,
682  IntMap& rInverseConectivityDatabase,
683  typename TClassType::Pointer pIndexesPairs,
684  GeometricalObject& rGeometricalObject,
685  ExactMortarIntegrationUtilityType& rIntegrationUtility,
686  MortarKinematicVariablesType& rThisKineticVariables,
687  MortarOperatorType& rThisMortarOperators,
688  const IndexType Iteration
689  )
690  {
691  // The root model part
692  ModelPart& r_root_model_part = mOriginModelPart.GetRootModelPart();
693 
694  // Getting the auxiliar variable
695  const TVarType& r_aux_variable = KratosComponents<TVarType>::Get(MortarUtilities::GetAuxiliarVariable<TVarType>());
696 
697  // Indexes of the pair to be removed
698  std::vector<IndexType> indexes_to_remove, geometrical_objects_to_erase;
699 
700  // Getting discontinous factor
701  const double discontinous_interface_factor = mOptions.Is(DISCONTINOUS_INTERFACE) ? mThisParameters["discontinous_interface_factor"].GetDouble() : 1.0;
702 
703  // Declare auxiliar coordinates
704  GeometryType::CoordinatesArrayType aux_coords;
705 
706  // Geometrical values
707  auto& r_slave_geometry = rGeometricalObject.GetGeometry();
708  r_slave_geometry.PointLocalCoordinates(aux_coords, r_slave_geometry.Center());
709  const array_1d<double, 3> slave_normal = r_slave_geometry.UnitNormal(aux_coords);
710 
711  // The model part as const to avoid race conditions
712  const auto& r_const_origin_model_part = mOriginModelPart;
713  const auto& r_const_destination_model_part = mDestinationModelPart;
714 
715  for (auto it_pair = pIndexesPairs->begin(); it_pair != pIndexesPairs->end(); ++it_pair ) {
716  const IndexType master_id = pIndexesPairs->GetId(it_pair); // MASTER
717 
718  const auto& r_master_geometry = mOptions.Is(ORIGIN_SKIN_IS_CONDITION_BASED) ? r_const_origin_model_part.GetCondition(master_id).GetGeometry() : r_const_origin_model_part.GetElement(master_id).GetGeometry();
719  r_master_geometry.PointLocalCoordinates(aux_coords, r_master_geometry.Center());
720  const array_1d<double, 3> master_normal = r_master_geometry.UnitNormal(aux_coords);
721 
722  const IntegrationMethod& r_integration_method = GetIntegrationMethod();
723 
724  // Reading integration points
725  std::vector<array_1d<PointType,TDim>> geometrical_objects_points_slave; // These are the segmentation points, with this points it is possible to create the lines or triangles used on the mapping
726  const bool is_inside = rIntegrationUtility.GetExactIntegration(r_slave_geometry, slave_normal, r_master_geometry, master_normal, geometrical_objects_points_slave);
727 
728  if (is_inside) {
729  // Initialize general variables for the current master element
730  rThisKineticVariables.Initialize();
731 
732  // Initialize the mortar operators
733  rThisMortarOperators.Initialize();
734 
735  const BoundedMatrixType Ae = CalculateAe(r_slave_geometry, rThisKineticVariables, geometrical_objects_points_slave, r_integration_method);
736 
737  AssemblyMortarOperators( geometrical_objects_points_slave, r_slave_geometry, r_master_geometry,master_normal, rThisKineticVariables, rThisMortarOperators, r_integration_method, Ae);
738 
739  /* We compute the residual */
740  const IndexType size_to_compute = MortarUtilities::SizeToCompute<TDim, TVarType>();
741  Matrix residual_matrix(TNumNodes, size_to_compute);
742  ComputeResidualMatrix(residual_matrix, r_slave_geometry, r_master_geometry, rThisMortarOperators);
743 
744  if (!TImplicit) {
745  MortarUtilities::AddValue<TVarType, MortarUtilitiesSettings::SaveAsNonHistoricalVariable>(r_slave_geometry, r_aux_variable, residual_matrix);
746  }
747 
748  // We check if DOperator is diagonal
749  if (mEchoLevel > 1) {
750  BoundedMatrixType aux_copy_D = rThisMortarOperators.DOperator;
751  LumpMatrix(aux_copy_D);
752  const BoundedMatrixType aux_diff = aux_copy_D - rThisMortarOperators.DOperator;
753  const double norm_diff = norm_frobenius(aux_diff);
754  if (norm_diff > 1.0e-4)
755  KRATOS_WARNING("D OPERATOR") << " THE MORTAR OPERATOR D IS NOT DIAGONAL" << std::endl;
756  if (mEchoLevel == 3) {
757  KRATOS_WATCH(norm_diff);
758  KRATOS_WATCH(rThisMortarOperators.DOperator);
759  }
760  }
761 
762  if (Iteration == 0) { // Just assembled the first iteration
763  if constexpr (TImplicit) {
764  /* We compute the residual and assemble */
765  const SizeType variable_size = MortarUtilities::SizeToCompute<TDim, TVarType>();
766  AssembleRHSAndLHS(rA, rb, variable_size, residual_matrix, r_slave_geometry, rInverseConectivityDatabase, rThisMortarOperators);
767  } else {
768  for (IndexType i_node = 0; i_node < TNumNodes; ++i_node) {
769  double& r_nodal_area = r_slave_geometry[i_node].GetValue(NODAL_AREA);
770  AtomicAdd(r_nodal_area, rThisMortarOperators.DOperator(i_node, i_node));
771  }
772  // In case of discontinous interface we add contribution to near nodes
773  if (mOptions.Is(DISCONTINOUS_INTERFACE)) {
774  const double element_length = r_slave_geometry.Length();
775 
776  // Iterating over nodes
777  for (IndexType i_node = 0; i_node < TNumNodes; ++i_node) {
778  const double nodal_area_contribution = rThisMortarOperators.DOperator(i_node, i_node);
779 
780  // The original node coordinates
781  const auto& r_slave_node = r_slave_geometry[i_node];
782 
783  // Iterating over other paired geometrical objects
784  const auto& r_index_masp_master = mOptions.Is(ORIGIN_SKIN_IS_CONDITION_BASED) ? r_const_origin_model_part.GetCondition(master_id).GetValue(INDEX_SET) : r_const_origin_model_part.GetElement(master_id).GetValue(INDEX_SET);
785  for (auto it_master_pair = r_index_masp_master->begin(); it_master_pair != r_index_masp_master->end(); ++it_master_pair ) {
786 
787  const IndexType auxiliar_slave_id = r_index_masp_master->GetId(it_master_pair);
788  if (rGeometricalObject.Id() != auxiliar_slave_id) {
789  GeometryType& r_auxiliar_slave_geometry = const_cast<GeometryType&>(mOptions.Is(DESTINATION_SKIN_IS_CONDITION_BASED) ? r_const_destination_model_part.GetCondition(auxiliar_slave_id).GetGeometry() : r_const_destination_model_part.GetElement(auxiliar_slave_id).GetGeometry());
790 
791  for (IndexType j_node = 0; j_node < TNumNodes; ++j_node) {
792  // The auxiliary node distance
793  auto& r_auxiliary_slave_node = r_auxiliar_slave_geometry[j_node];
794  const double distance = r_auxiliary_slave_node.Distance(r_slave_node);
795  const double contribution_coeff = 1.0/std::pow((1.0 + distance/(discontinous_interface_factor * element_length)), 2);
796 
797  double& r_nodal_area = r_auxiliary_slave_node.GetValue(NODAL_AREA);
798  AtomicAdd(r_nodal_area, contribution_coeff * nodal_area_contribution);
799  }
800  }
801  }
802  }
803  }
804  }
805  } else if constexpr (TImplicit) {
806  const SizeType variable_size = MortarUtilities::SizeToCompute<TDim, TVarType>();
807  AssembleRHS(rb, variable_size, residual_matrix, r_slave_geometry, rInverseConectivityDatabase);
808  }
809  } else { // NOTE: The geometrical object considered maybe is to tight
810  indexes_to_remove.push_back(master_id);
811  const IndexType other_id = pIndexesPairs->GetOtherId(it_pair);
812  if (std::is_same<TClassType, IndexMap>::value && other_id != 0) {
813  geometrical_objects_to_erase.push_back(other_id);
814  }
815  }
816  }
817 
818  // Clear indexes
819  for (IndexType i_to_remove = 0; i_to_remove < indexes_to_remove.size(); ++i_to_remove) {
820  if (mOptions.Is(ORIGIN_SKIN_IS_CONDITION_BASED)) {
821  for (auto& id : geometrical_objects_to_erase ) {
822  auto& r_cond = r_root_model_part.GetCondition(id);
823  r_cond.Set(TO_ERASE, true);
824  }
825  } else {
826  for (auto& id : geometrical_objects_to_erase ) {
827  auto& r_elem = r_root_model_part.GetElement(id);
828  r_elem.Set(TO_ERASE, true);
829  }
830  }
831  pIndexesPairs->RemoveId(indexes_to_remove[i_to_remove]);
832  }
833  }
