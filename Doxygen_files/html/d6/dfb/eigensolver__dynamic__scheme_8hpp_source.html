99 ) override

101 KRATOS_TRY
102
103 if (CurrentProcessInfo[BUILD_LEVEL] == 1)
104 { // mass matrix
105 rCurrentElement.CalculateMassMatrix(LHS_Contribution,CurrentProcessInfo);
106 std::size_t LocalSize = LHS_Contribution.size1();

111 else if (CurrentProcessInfo[BUILD_LEVEL] == 2) // stiffness matrix
112 {
113 rCurrentElement.CalculateLocalSystem(LHS_Contribution,RHS_Contribution,CurrentProcessInfo);
114 }
115 else
116 {
117 KRATOS_ERROR <<"Invalid BUILD_LEVEL" << std::endl;
118 }
119
120 rCurrentElement.EquationIdVector(EquationId,CurrentProcessInfo);
121
122 KRATOS_CATCH("")
