234 void CalculateFluidFraction(
235 ModelPart& r_model_part,
236 ProcessInfo& r_current_process_info)
237 {
238
239 double delta_time = r_current_process_info[DELTA_TIME];
240 double delta_time_inv = 1.0 / delta_time;
241
242 //for(int k = 0; k<static_cast<int>(r_model_part.Nodes().size()); k++)
243 block_for_each(r_model_part.Nodes(), [&](Node<3>& rNode)
244 {
245 const double fluid_fraction = rNode.FastGetSolutionStepValue(FLUID_FRACTION);
246 const double fluid_fraction_old = rNode.FastGetSolutionStepValue(FLUID_FRACTION_OLD);
247 const double fluid_fraction_rate = delta_time_inv * (fluid_fraction - fluid_fraction_old);
248 rNode.FastGetSolutionStepValue(FLUID_FRACTION_RATE) = fluid_fraction_rate;
249 });
250 }
