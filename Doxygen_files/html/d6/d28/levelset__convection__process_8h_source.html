162 void Execute() override
163 {
164 KRATOS_TRY;
165
166 // Fill the auxiliary convection model part if not done yet
167 if(mDistancePartIsInitialized == false){
168 ReGenerateConvectionModelPart(mrBaseModelPart);
169 }
170
171 // Evaluate steps needed to achieve target max_cfl
172 const auto n_substep = EvaluateNumberOfSubsteps();
173
174 auto& rCurrentProcessInfo = mpDistanceModelPart->GetProcessInfo();
175
176 // Save the variables to be employed so that they can be restored after the solution
177 const auto& r_previous_var = rCurrentProcessInfo.GetValue(CONVECTION_DIFFUSION_SETTINGS)->GetUnknownVariable();
178 const double previous_delta_time = rCurrentProcessInfo.GetValue(DELTA_TIME);
179
180 // Save current level set value and current and previous step velocity values
181 IndexPartition<int>(mpDistanceModelPart->NumberOfNodes()).for_each(
182 [&](int i_node){
183 const auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
184 mVelocity[i_node] = it_node->FastGetSolutionStepValue(*mpConvectVar);
185 mVelocityOld[i_node] = it_node->FastGetSolutionStepValue(*mpConvectVar,1);
186 mOldDistance[i_node] = it_node->FastGetSolutionStepValue(*mpLevelSetVar,1);
187 }
188 );
189
190 const double dt = previous_delta_time / static_cast<double>(n_substep);
191 rCurrentProcessInfo.SetValue(DELTA_TIME, dt);
192 rCurrentProcessInfo.GetValue(CONVECTION_DIFFUSION_SETTINGS)->SetUnknownVariable(*mpLevelSetVar);
193
194 // We set these values at every time step as other processes/solvers also use them
195 auto fill_process_info_function = GetFillProcessInfoFunction();
196 fill_process_info_function(mrBaseModelPart);
197
198 for(unsigned int step = 1; step <= n_substep; ++step){
199 KRATOS_INFO_IF("LevelSetConvectionProcess", mLevelSetConvectionSettings["echo_level"].GetInt() > 0) <<
200 "Doing step "<< step << " of " << n_substep << std::endl;
201
202 if (mIsBfecc || mElementRequiresLevelSetGradient){
203 mpGradientCalculator->Execute();
204 }
205
206 // Compute shape functions of old and new step
207 const double Nold = 1.0 - static_cast<double>(step) / static_cast<double>(n_substep);
208 const double Nnew = 1.0 - Nold;
209
210 const double Nold_before = 1.0 - static_cast<double>(step-1) / static_cast<double>(n_substep);
211 const double Nnew_before = 1.0 - Nold_before;
212
213 // Emulate clone time step by copying the new distance onto the old one
214 IndexPartition<int>(mpDistanceModelPart->NumberOfNodes()).for_each(
215 [&](int i_node){
216 auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
217
218 const array_1d<double,3>& r_v = mVelocity[i_node];
219 const array_1d<double,3>& r_v_old = mVelocityOld[i_node];
220
221 noalias(it_node->FastGetSolutionStepValue(*mpConvectVar)) = Nold * r_v_old + Nnew * r_v;
222 noalias(it_node->FastGetSolutionStepValue(*mpConvectVar, 1)) = Nold_before * r_v_old + Nnew_before * r_v;
223 it_node->FastGetSolutionStepValue(*mpLevelSetVar, 1) = it_node->FastGetSolutionStepValue(*mpLevelSetVar);
224 }
225 );
226
227 // Storing the levelset variable for error calculation and Evaluating the limiter
228 if (mEvaluateLimiter) {
229 EvaluateLimiter();
230 }
231
232 mpSolvingStrategy->InitializeSolutionStep();
233 mpSolvingStrategy->Predict();
234 mpSolvingStrategy->SolveSolutionStep(); // forward convection to reach phi_n+1
235 mpSolvingStrategy->FinalizeSolutionStep();
236
237 // Error Compensation and Correction
238 if (mIsBfecc) {
239 ErrorCalculationAndCorrection();
240 }
241 }
242
243 // Reset the processinfo to the original settings
244 rCurrentProcessInfo.SetValue(DELTA_TIME, previous_delta_time);
245 rCurrentProcessInfo.GetValue(CONVECTION_DIFFUSION_SETTINGS)->SetUnknownVariable(r_previous_var);
246
247 // Reset the velocities and levelset values to the one saved before the solution process
248 IndexPartition<int>(mpDistanceModelPart->NumberOfNodes()).for_each(
249 [&](int i_node){
250 auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
251 it_node->FastGetSolutionStepValue(*mpConvectVar) = mVelocity[i_node];
252 it_node->FastGetSolutionStepValue(*mpConvectVar,1) = mVelocityOld[i_node];
253 it_node->FastGetSolutionStepValue(*mpLevelSetVar,1) = mOldDistance[i_node];
254 }
255 );
256
257 KRATOS_CATCH("")
258 }

459 virtual void ReGenerateConvectionModelPart(ModelPart& rBaseModelPart){
460
461 KRATOS_TRY
462
463 if (mrModel.HasModelPart(mAuxModelPartName)) {
464 mrModel.DeleteModelPart(mAuxModelPartName);
465 }
466
467 mpDistanceModelPart= &(mrModel.CreateModelPart(mAuxModelPartName));
468
469
470 // Check buffer size
471 const auto base_buffer_size = rBaseModelPart.GetBufferSize();
472 KRATOS_ERROR_IF(base_buffer_size < 2) <<
473 "Base model part buffer size is " << base_buffer_size << ". Set it to a minimum value of 2." << std::endl;
474
475 // Generate
476 mpDistanceModelPart->Nodes().clear();
477 mpDistanceModelPart->Conditions().clear();
478 mpDistanceModelPart->Elements().clear();
479
480 mpDistanceModelPart->SetProcessInfo(rBaseModelPart.pGetProcessInfo());
481 mpDistanceModelPart->SetBufferSize(base_buffer_size);
482 for(auto it_properties = rBaseModelPart.PropertiesBegin() ; it_properties != rBaseModelPart.PropertiesEnd() ; ++it_properties){
483 mpDistanceModelPart->AddProperties(*(it_properties).base());
484 }
485 mpDistanceModelPart->Tables() = rBaseModelPart.Tables();
486
487 // Assigning the nodes to the new model part
488 mpDistanceModelPart->Nodes() = rBaseModelPart.Nodes();
489
490 // Ensure that the nodes have distance as a DOF
491 VariableUtils().AddDof<Variable<double>>(*mpLevelSetVar, rBaseModelPart);
492
493 // Generating the elements
494 mpDistanceModelPart->Elements().reserve(rBaseModelPart.NumberOfElements());
495 KRATOS_ERROR_IF(mpConvectionFactoryElement == nullptr) << "Convection factory element has not been set yet." << std::endl;
496 for (auto it_elem = rBaseModelPart.ElementsBegin(); it_elem != rBaseModelPart.ElementsEnd(); ++it_elem){
497 // Create the new element from the factory registered one
498 auto p_element = mpConvectionFactoryElement->Create(
499 it_elem->Id(),
500 it_elem->pGetGeometry(),
501 it_elem->pGetProperties());
502
503 mpDistanceModelPart->Elements().push_back(p_element);
504 }
505
506 // Initialize the nodal and elemental databases
507 InitializeDistanceModelPartDatabases();
508
509 // Resize the arrays
510 const auto n_nodes = mpDistanceModelPart->NumberOfNodes();
511 mVelocity.resize(n_nodes);
512 mVelocityOld.resize(n_nodes);
513 mOldDistance.resize(n_nodes);
514
515 if (mEvaluateLimiter){
516 mSigmaPlus.resize(n_nodes);
517 mSigmaMinus.resize(n_nodes);
518 mLimiter.resize(n_nodes);
519 }
520
521 if (mIsBfecc){
522 mError.resize(n_nodes);
523 }
524
525 mDistancePartIsInitialized = true;
526
527 KRATOS_CATCH("")
528 }

547 unsigned int EvaluateNumberOfSubsteps(){
548 // First of all compute the maximum local CFL number
549 const double dt = mpDistanceModelPart->GetProcessInfo()[DELTA_TIME];
550 double max_cfl_found = block_for_each<MaxReduction<double>>(mpDistanceModelPart->Elements(), [&](Element& rElement){
551 double vol;
552 array_1d<double, TDim+1 > N;
553 BoundedMatrix<double, TDim+1, TDim > DN_DX;
554 auto& r_geom = rElement.GetGeometry();
555 GeometryUtils::CalculateGeometryData(r_geom, DN_DX, N, vol);
556
557 // Compute h
558 double h=0.0;
559 for(unsigned int i=0; i<TDim+1; i++){
560 double h_inv = 0.0;
561 for(unsigned int k=0; k<TDim; k++){
562 h_inv += DN_DX(i,k)*DN_DX(i,k);
563 }
564 h += 1.0/h_inv;
565 }
566 h = sqrt(h)/static_cast<double>(TDim+1);
567
568 // Get average velocity at the nodes
569 array_1d<double, 3 > vgauss = ZeroVector(3);
570 for(unsigned int i=0; i<TDim+1; i++){
571 vgauss += N[i]* r_geom[i].FastGetSolutionStepValue(*mpConvectVar);
572 }
573
574 double cfl_local = norm_2(vgauss) / h;
575 return cfl_local;
576 });
577 max_cfl_found *= dt;
578
579 // Synchronize maximum CFL between processes
580 max_cfl_found = mpDistanceModelPart->GetCommunicator().GetDataCommunicator().MaxAll(max_cfl_found);
581
582 unsigned int n_steps = static_cast<unsigned int>(max_cfl_found / mMaxAllowedCFL);
583 if(n_steps < 1){
584 n_steps = 1;
585 }
586
587 // Now we compare with the maximum set
588 if (mMaxSubsteps > 0 && mMaxSubsteps < n_steps){
589 n_steps = mMaxSubsteps;
590 }
591
592 return n_steps;
593 }

600 void EvaluateLimiter()
601 {
602 const double epsilon = 1.0e-12;
603
604 auto& r_default_comm = mpDistanceModelPart->GetCommunicator().GetDataCommunicator();
605 GlobalPointersVector< Node<3 > > gp_list;
606
607 for (int i_node = 0; i_node < static_cast<int>(mpDistanceModelPart->NumberOfNodes()); ++i_node){
608 auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
609 GlobalPointersVector< Node<3 > >& global_pointer_list = it_node->GetValue(NEIGHBOUR_NODES);
610
611 for (unsigned int j = 0; j< global_pointer_list.size(); ++j)
612 {
613 auto& global_pointer = global_pointer_list(j);
614 gp_list.push_back(global_pointer);
615 }
616 }
617
618 GlobalPointerCommunicator< Node<3 > > pointer_comm(r_default_comm, gp_list);
619
620 auto coordinate_proxy = pointer_comm.Apply(
621 [](GlobalPointer<Node<3> >& global_pointer) -> Point::CoordinatesArrayType
622 {
623 return global_pointer->Coordinates();
624 }
625 );
626
627 IndexPartition<int>(mpDistanceModelPart->NumberOfNodes()).for_each(
628 [&](int i_node){
629 auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
630
631 it_node->SetValue(*mpLevelSetVar, it_node->FastGetSolutionStepValue(*mpLevelSetVar)); //Store mpLevelSetVar
632
633 const auto& X_i = it_node->Coordinates();
634 const auto& grad_i = it_node->GetValue(*mpLevelSetGradientVar);
635
636 double S_plus = 0.0;
637 double S_minus = 0.0;
638
639 GlobalPointersVector< Node<3 > >& global_pointer_list = it_node->GetValue(NEIGHBOUR_NODES);
640
641 for (unsigned int j = 0; j< global_pointer_list.size(); ++j)
642 {
643 // if (it_node->Id() == j_node->Id())
644 // continue;
645
646 auto& global_pointer = global_pointer_list(j);
647 auto X_j = coordinate_proxy.Get(global_pointer);
648
649 S_plus += std::max(0.0, inner_prod(grad_i, X_i-X_j));
650 S_minus += std::min(0.0, inner_prod(grad_i, X_i-X_j));
651 }
652
653 mSigmaPlus[i_node] = std::min(1.0, (std::abs(S_minus)+epsilon)/(S_plus+epsilon));
654 mSigmaMinus[i_node] = std::min(1.0, (S_plus+epsilon)/(std::abs(S_minus)+epsilon));
655 }
656 );
657
658 auto combined_proxy = pointer_comm.Apply(
659 [&](GlobalPointer<Node<3>> &global_pointer) -> std::pair<double, array_1d<double,3>> {
660 return std::make_pair(
661 global_pointer->FastGetSolutionStepValue(*mpLevelSetVar),
662 global_pointer->Coordinates());
663 });
664
665 //Calculating beta_ij in a way that the linearity is preserved on non-symmetrical meshes
666 IndexPartition<int>(mpDistanceModelPart->NumberOfNodes()).for_each(
667 [&](int i_node){
668 auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
669 const double distance_i = it_node->FastGetSolutionStepValue(*mpLevelSetVar);
670 const auto& X_i = it_node->Coordinates();
671 const auto& grad_i = it_node->GetValue(*mpLevelSetGradientVar);
672
673 double numerator = 0.0;
674 double denominator = 0.0;
675
676 GlobalPointersVector< Node<3 > >& global_pointer_list = it_node->GetValue(NEIGHBOUR_NODES);
677
678 for (unsigned int j = 0; j< global_pointer_list.size(); ++j)
679 {
680 auto& global_pointer = global_pointer_list(j);
681 auto result = combined_proxy.Get(global_pointer);
682 const auto X_j = std::get<1>(result);
683 const double distance_j = std::get<0>(result);
684
685 double beta_ij = 1.0;
686
687 if (inner_prod(grad_i, X_i-X_j) > 0)
688 beta_ij = mSigmaPlus[i_node];
689 else if (inner_prod(grad_i, X_i-X_j) < 0)
690 beta_ij = mSigmaMinus[i_node];
691
692 numerator += beta_ij*(distance_i - distance_j);
693 denominator += beta_ij*std::abs(distance_i - distance_j);
694 }
695
696 const double fraction = std::abs(numerator) / (denominator + epsilon);
697
698 if (mIsBfecc){
699 mLimiter[i_node] = 1.0 - std::pow(fraction, mPowerBfeccLimiter);
700 }
701
702 if (mElementRequiresLimiter){
703 it_node->GetValue(LIMITER_COEFFICIENT) = (1.0 - std::pow(fraction, mPowerElementalLimiter));
704 }
705 }
706 );
707
708 if (mElementRequiresLimiter){
709 block_for_each(mpDistanceModelPart->Elements(), [&](Element& rElement){
710 const auto& r_geometry = rElement.GetGeometry();
711 double elemental_limiter = 1.0;
712 for(unsigned int i_node=0; i_node< TDim+1; ++i_node) {
713 elemental_limiter = std::min(r_geometry[i_node].GetValue(LIMITER_COEFFICIENT), elemental_limiter);
714 }
715 rElement.GetValue(LIMITER_COEFFICIENT) = elemental_limiter;
716 }
717 );
718 }
719 }

727 void ErrorCalculationAndCorrection()
728 {
729 block_for_each(mpDistanceModelPart->Nodes(), [this](Node<3>& rNode){
730 noalias(rNode.FastGetSolutionStepValue(*mpConvectVar)) = -1.0 * rNode.FastGetSolutionStepValue(*mpConvectVar);
731 noalias(rNode.FastGetSolutionStepValue(*mpConvectVar, 1)) = -1.0 * rNode.FastGetSolutionStepValue(*mpConvectVar, 1);
732 rNode.FastGetSolutionStepValue(*mpLevelSetVar, 1) = rNode.FastGetSolutionStepValue(*mpLevelSetVar);
733 });
734
735 mpSolvingStrategy->InitializeSolutionStep();
736 mpSolvingStrategy->Predict();
737 mpSolvingStrategy->SolveSolutionStep(); // backward convetion to obtain phi_n*
738 mpSolvingStrategy->FinalizeSolutionStep();
739
740 // Calculating the raw error without a limiter, etc.
741 IndexPartition<int>(mpDistanceModelPart->NumberOfNodes()).for_each(
742 [&](int i_node){
743 auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
744 mError[i_node] =
745 0.5*(it_node->GetValue(*mpLevelSetVar) - it_node->FastGetSolutionStepValue(*mpLevelSetVar));
746 });
747
748 IndexPartition<int>(mpDistanceModelPart->NumberOfNodes()).for_each(
749 [&](int i_node){
750 auto it_node = mpDistanceModelPart->NodesBegin() + i_node;
751 noalias(it_node->FastGetSolutionStepValue(*mpConvectVar)) = -1.0 * it_node->FastGetSolutionStepValue(*mpConvectVar);
752 noalias(it_node->FastGetSolutionStepValue(*mpConvectVar, 1)) = -1.0 * it_node->FastGetSolutionStepValue(*mpConvectVar, 1);
753 const double phi_n_star = it_node->GetValue(*mpLevelSetVar) + mLimiter[i_node]*mError[i_node];

755 it_node->FastGetSolutionStepValue(*mpLevelSetVar, 1) = phi_n_star;
756 });

758 mpSolvingStrategy->InitializeSolutionStep();
759 mpSolvingStrategy->Predict();
760 mpSolvingStrategy->SolveSolutionStep(); // forward convection to obtain the corrected phi_n+1
761 mpSolvingStrategy->FinalizeSolutionStep();

945 void InitializeConvectionStrategy(BuilderAndSolverPointerType pBuilderAndSolver)
946 {
947 // Check that there is at least one element and node in the model
948 KRATOS_ERROR_IF(mrBaseModelPart.NumberOfNodes() == 0) << "The model has no nodes." << std::endl;
949 KRATOS_ERROR_IF(mrBaseModelPart.NumberOfElements() == 0) << "The model has no elements." << std::endl;
950
951 // Check that the level set and convection variables are in the nodal database
952 VariableUtils().CheckVariableExists<Variable<double>>(*mpLevelSetVar, mrBaseModelPart.Nodes());
953 VariableUtils().CheckVariableExists<Variable<array_1d<double,3>>>(*mpConvectVar, mrBaseModelPart.Nodes());
954
955 // Check the base model part element family (only simplex elements are supported)
956 if constexpr (TDim == 2){
957 KRATOS_ERROR_IF(mrBaseModelPart.ElementsBegin()->GetGeometry().GetGeometryFamily() != GeometryData::KratosGeometryFamily::Kratos_Triangle) <<
958 "In 2D the element type is expected to be a triangle" << std::endl;
959 } else if constexpr (TDim == 3) {
960 KRATOS_ERROR_IF(mrBaseModelPart.ElementsBegin()->GetGeometry().GetGeometryFamily() != GeometryData::KratosGeometryFamily::Kratos_Tetrahedra) <<
961 "In 3D the element type is expected to be a tetrahedra" << std::endl;
962 }
963
964 // Generate an auxilary model part and populate it by elements of type DistanceCalculationElementSimplex
965 ReGenerateConvectionModelPart(mrBaseModelPart);
966
967 // Generate a linear strategy
968 bool CalculateReactions = false;
969 bool ReformDofAtEachIteration = false;
970 bool CalculateNormDxFlag = false;
971 auto p_scheme = Kratos::make_shared<ResidualBasedIncrementalUpdateStaticScheme<TSparseSpace,TDenseSpace>>();
972 mpSolvingStrategy = Kratos::make_unique< ResidualBasedLinearStrategy<TSparseSpace,TDenseSpace,TLinearSolver > >(
973 *mpDistanceModelPart,
974 p_scheme,
975 pBuilderAndSolver,
976 CalculateReactions,
977 ReformDofAtEachIteration,
978 CalculateNormDxFlag);
979
980 mpSolvingStrategy->SetEchoLevel(mLevelSetConvectionSettings["echo_level"].GetInt());
981 mpSolvingStrategy->Check();
982 mpSolvingStrategy->Initialize();
983 }
