472 template< class TMethodPointerType >
473 TMethodPointerType GetAssignmentMethod()
474 {
475 TMethodPointerType AssignmentMethod = nullptr;
476 switch( mAssignment )
477 {
478 case AssignmentType::DIRECT:
479 AssignmentMethod = &AssignScalarVariableToEntitiesProcess::DirectAssignValue;
480 break;
481 case AssignmentType::ADDITION:
482 AssignmentMethod = &AssignScalarVariableToEntitiesProcess::AddAssignValue;
483 break;
484 case AssignmentType::SUBTRACTION:
485 AssignmentMethod = &AssignScalarVariableToEntitiesProcess::SubtractAssignValue;
486 break;
487 case AssignmentType::MULTIPLICATION:
488 AssignmentMethod = &AssignScalarVariableToEntitiesProcess::MultiplyAssignValue;
489 break;
490 case AssignmentType::DIVISION:
491 AssignmentMethod = &AssignScalarVariableToEntitiesProcess::DivideAssignValue;
492 break;
493 default:
494 KRATOS_ERROR << "Unexpected value for Assignment method " << std::endl;
495 }
496 return AssignmentMethod;
