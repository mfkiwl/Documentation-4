84
85 void Initialize(ModelPart& r_model_part) override
86 {
87 KRATOS_TRY
88
89 mDeltaTime = r_model_part.GetProcessInfo()[DELTA_TIME];
90
91 // Pure Newmark scheme
92 // Note: this is a quasistatic scheme. The velocity and acceleration are stored without affecting the displacement
93 mbeta = 0.25;
94 mgamma = 0.5;
95
96 mNewmark0 = ( 1.0 / (mbeta * mDeltaTime * mDeltaTime) );
97 mNewmark1 = ( mgamma / (mbeta * mDeltaTime) );
98 mNewmark2 = ( 1.0 / (mbeta * mDeltaTime) );
99 mNewmark3 = ( 0.5 / (mbeta) - 1.0 );
100 mNewmark4 = ( (mgamma / mbeta) - 1.0 );
101 mNewmark5 = ( mDeltaTime * 0.5 * ( ( mgamma / mbeta ) - 2.0 ) );
102
103 r_model_part.GetProcessInfo()[RAYLEIGH_ALPHA] = mRayleighAlpha;
104 r_model_part.GetProcessInfo()[RAYLEIGH_BETA] = mRayleighBeta;
105
106 mSchemeIsInitialized = true;
107
108 KRATOS_CATCH("")
109 }

114 void Predict(
115 ModelPart& rModelPart,
116 DofsArrayType& rDofSet,
117 TSystemMatrixType& A,
118 TSystemVectorType& Dx,
119 TSystemVectorType& b
120 ) override
121 {
122 KRATOS_TRY;
123
124 //const double DeltaTime = rModelPart.GetProcessInfo()[DELTA_TIME];
125
126 // Updating time derivatives (nodally for efficiency)
127 const unsigned int NumThreads = ParallelUtilities::GetNumThreads();
128 OpenMPUtils::PartitionVector NodePartition;
129 OpenMPUtils::DivideInPartitions(rModelPart.Nodes().size(), NumThreads, NodePartition);
130
131 const int nnodes = static_cast<int>( rModelPart.Nodes().size() );
132 NodesArrayType::iterator NodeBegin = rModelPart.Nodes().begin();
133
134 #pragma omp parallel for firstprivate(NodeBegin)
135 for(int i = 0; i< nnodes; i++)
136 {
137 array_1d<double, 3 > DeltaDisplacement;
138
139 NodesArrayType::iterator itNode = NodeBegin + i;
140
141 const array_1d<double, 3 > & PreviousAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 1);
142 const array_1d<double, 3 > & PreviousVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY, 1);
143 const array_1d<double, 3 > & PreviousDisplacement = (itNode)->FastGetSolutionStepValue(DISPLACEMENT, 1);
144 array_1d<double, 3 > & CurrentAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 0);
145 array_1d<double, 3 > & CurrentVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY, 0);
146 array_1d<double, 3 > & CurrentDisplacement = (itNode)->FastGetSolutionStepValue(DISPLACEMENT, 0);
147
148 // Updating time derivatives ::: Please note that displacements and its time derivatives can not be consistently fixed separately
149 noalias(DeltaDisplacement) = CurrentDisplacement - PreviousDisplacement;
150
151 this->UpdateVelocity (CurrentVelocity, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
152
153 this->UpdateAcceleration (CurrentAcceleration, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
154 }
155
156 KRATOS_CATCH( "" );
157 }

161 void Update(
162 ModelPart& rModelPart,
163 DofsArrayType& rDofSet,
164 TSystemMatrixType& A,
165 TSystemVectorType& Dx,
166 TSystemVectorType& b ) override
167 {
168 KRATOS_TRY;
169
170 int NumThreads = ParallelUtilities::GetNumThreads();
171 OpenMPUtils::PartitionVector DofSetPartition;
172 OpenMPUtils::DivideInPartitions(rDofSet.size(), NumThreads, DofSetPartition);
173
174 #pragma omp parallel
175 {
176 int k = OpenMPUtils::ThisThread();
177
178 typename DofsArrayType::iterator DofsBegin = rDofSet.begin() + DofSetPartition[k];
179 typename DofsArrayType::iterator DofsEnd = rDofSet.begin() + DofSetPartition[k+1];
180
181 //Update Displacement and Pressure (DOFs)
182 for (typename DofsArrayType::iterator itDof = DofsBegin; itDof != DofsEnd; ++itDof)
183 {
184 if (itDof->IsFree())
185 itDof->GetSolutionStepValue() += TSparseSpace::GetValue(Dx, itDof->EquationId());
186 }
187 }
188
189 // Updating time derivatives (nodally for efficiency)
190 OpenMPUtils::PartitionVector NodePartition;
191 OpenMPUtils::DivideInPartitions(rModelPart.Nodes().size(), NumThreads, NodePartition);
192
193 const int nnodes = static_cast<int>(rModelPart.Nodes().size());
194 NodesArrayType::iterator NodeBegin = rModelPart.Nodes().begin();
195
196 #pragma omp parallel for firstprivate(NodeBegin)
197 for(int i = 0; i < nnodes; i++)
198 {
199 array_1d<double, 3 > DeltaDisplacement;
200
201 NodesArrayType::iterator itNode = NodeBegin + i;
202
203 noalias(DeltaDisplacement) = (itNode)->FastGetSolutionStepValue(DISPLACEMENT) - (itNode)->FastGetSolutionStepValue(DISPLACEMENT, 1);
204
205 array_1d<double, 3 > & CurrentVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY, 0);
206 const array_1d<double, 3 > & PreviousVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY, 1);
207
208 array_1d<double, 3 > & CurrentAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 0);
209 const array_1d<double, 3 > & PreviousAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 1);
210
211 this->UpdateVelocity (CurrentVelocity, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
212
213 this->UpdateAcceleration (CurrentAcceleration, DeltaDisplacement, PreviousVelocity, PreviousAcceleration);
214 }
215
216 KRATOS_CATCH( "" );
217 }

318 void AddDampingToLHS(LocalSystemMatrixType& LHS_Contribution, LocalSystemMatrixType& C, const ProcessInfo& CurrentProcessInfo)
319 {
320 // adding damping contribution
321 if (C.size1() != 0)
322 {
323 noalias(LHS_Contribution) += mgamma/(mbeta*mDeltaTime)*C;
324 }
325 }
326
327//----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
328
329 void AddDampingToRHS(Element& rCurrentElement,
330 LocalSystemVectorType& RHS_Contribution,
331 LocalSystemMatrixType& C,
332 const ProcessInfo& CurrentProcessInfo)
333 {
334 int thread = OpenMPUtils::ThisThread();
335
336 //adding damping contribution
337 if (C.size1() != 0)
338 {
339 rCurrentElement.GetFirstDerivativesVector(mVelocityVector[thread], 0);
340
341 noalias(RHS_Contribution) -= prod(C, mVelocityVector[thread]);
342
343 }
344 }

348 inline void UpdateVelocity(
349 array_1d<double, 3 > & CurrentVelocity,
350 const array_1d<double, 3 > & DeltaDisplacement,
351 const array_1d<double, 3 > & PreviousVelocity,
352 const array_1d<double, 3 > & PreviousAcceleration
353 )
354 {
355 noalias(CurrentVelocity) = (mNewmark1 * DeltaDisplacement - mNewmark4 * PreviousVelocity
356 - mNewmark5 * PreviousAcceleration);
357 }
358
359 //----------------------------------------------------------------------------------------------------------------
360
361 inline void UpdateAcceleration(
362 array_1d<double, 3 > & CurrentAcceleration,
363 const array_1d<double, 3 > & DeltaDisplacement,
364 const array_1d<double, 3 > & PreviousVelocity,
365 const array_1d<double, 3 > & PreviousAcceleration
366 )
367 {
368 noalias(CurrentAcceleration) = (mNewmark0 * DeltaDisplacement - mNewmark2 * PreviousVelocity
369 - mNewmark3 * PreviousAcceleration);
370 }
