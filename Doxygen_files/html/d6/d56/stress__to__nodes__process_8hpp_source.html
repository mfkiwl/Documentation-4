66 void StressExtrapolationAndSmoothing(NodeStresses *pNodeStressesVector)
67 {
68 Vector gauss_point_stress;
69 double gauss_point_damage;
70 // Loop over elements to extrapolate the stress to the nodes
71 for (ElementsArrayType::ptr_iterator it = mrModelPart.Elements().ptr_begin(); it != mrModelPart.Elements().ptr_end(); ++it) {
72 auto& r_geometry = (*it)->GetGeometry();
73 gauss_point_stress = (*it)->GetValue(STRESS_VECTOR);
74 gauss_point_damage = (*it)->GetValue(DAMAGE_ELEMENT);
75
76 if (r_geometry.PointsNumber() == 3) { // 2D version
77 for (unsigned int i = 0; i < 3; i++) {
78 const int node_id = r_geometry.GetPoint(i).Id();
79 pNodeStressesVector[node_id - 1].EffectiveStressVector[0] += gauss_point_stress[0];
80 pNodeStressesVector[node_id - 1].EffectiveStressVector[1] += gauss_point_stress[1];
81 pNodeStressesVector[node_id - 1].EffectiveStressVector[3] += gauss_point_stress[2];
82 pNodeStressesVector[node_id - 1].Damage += gauss_point_damage;
83 pNodeStressesVector[node_id - 1].NElems += 1;
84 }
85 } else { // 3D version
86 for (unsigned int i = 0; i < 4; i++) {
87 const int node_id = r_geometry.GetPoint(i).Id();
88 pNodeStressesVector[node_id - 1].EffectiveStressVector[0] += gauss_point_stress[0];
89 pNodeStressesVector[node_id - 1].EffectiveStressVector[1] += gauss_point_stress[1];
90 pNodeStressesVector[node_id - 1].EffectiveStressVector[2] += gauss_point_stress[2];
91 pNodeStressesVector[node_id - 1].EffectiveStressVector[3] += gauss_point_stress[3];
92 pNodeStressesVector[node_id - 1].EffectiveStressVector[4] += gauss_point_stress[4];
93 pNodeStressesVector[node_id - 1].EffectiveStressVector[5] += gauss_point_stress[5];
94 pNodeStressesVector[node_id - 1].Damage += gauss_point_damage;
95 pNodeStressesVector[node_id - 1].NElems += 1;
96 }
97 }
98 }
99
100 // Ponderate over the elements coincident on that node
101 for (unsigned int i = 0; i < mNNodes; i++) {
102 pNodeStressesVector[i].EffectiveStressVector /= pNodeStressesVector[i].NElems;
103 pNodeStressesVector[i].Damage /= pNodeStressesVector[i].NElems;
104 }
105
106 // Loop to compute the equivalent streses at each node
107 for (ModelPart::NodeIterator it = mrModelPart.NodesBegin(); it != mrModelPart.NodesEnd(); ++it) {
108 int Id = (*it).Id();
109 const Vector& r_nodal_stress = pNodeStressesVector[Id - 1].EffectiveStressVector;
110
111 // Compute the norm of the vector -> Put VonMises stress
112 double &norm = it->GetSolutionStepValue(EQUIVALENT_NODAL_STRESS);
113 norm = this->CalculateVonMisesStress(r_nodal_stress);
114 }
115
116 // Loop to compute the max eq. stress to normalize the inidicator
117 double max_equivalent_stress = 0.0;
118 for (ModelPart::NodeIterator it = mrModelPart.NodesBegin(); it != mrModelPart.NodesEnd(); ++it) {
119 const double &norm = it->GetSolutionStepValue(EQUIVALENT_NODAL_STRESS);
120 if (norm > max_equivalent_stress)
121 max_equivalent_stress = norm;
122 }
123
124 for (ModelPart::NodeIterator it = mrModelPart.NodesBegin(); it != mrModelPart.NodesEnd(); ++it) {
125 double &norm = it->GetSolutionStepValue(EQUIVALENT_NODAL_STRESS);
126 int Id = (*it).Id();
127 double nodal_damage = pNodeStressesVector[Id - 1].Damage;
128 // if (nodal_damage >= 0.9 && norm < max_equivalent_stress / 1.0e2) norm = max_equivalent_stress / 1.0e2;
129 norm /= (max_equivalent_stress * 1.0e4);
130 }
131 }

133 // --------------------------------------------------------------------
134 double CalculateVonMisesStress(const Vector& rStressVector) // Returns Von Mises stress
135 {
136 const double I1 = rStressVector[0] + rStressVector[1] + rStressVector[2];
137 Vector Deviator = rStressVector;
138 const double Pmean = I1 / 3.0;
139
140 Deviator[0] -= Pmean;
141 Deviator[1] -= Pmean;
142 Deviator[2] -= Pmean;
143
144 const double J2 = 0.5 * (Deviator[0] * Deviator[0] + Deviator[1] * Deviator[1] + Deviator[2] * Deviator[2]) +
145 (Deviator[3] * Deviator[3] + Deviator[4] * Deviator[4] + Deviator[5] * Deviator[5]);
146
147 return std::sqrt(3.0 * J2);
148 }
