287  void FixIsolatedNodes(ModelPart& rModelPart)
288  {
289  KRATOS_ERROR_IF_NOT(rModelPart.HasSubModelPart("Contact")) << "CONTACT MODEL PART NOT CREATED" << std::endl;
290  KRATOS_ERROR_IF_NOT(rModelPart.HasSubModelPart("ComputingContact")) << "CONTACT COMPUTING MODEL PART NOT CREATED" << std::endl;
291  ModelPart& r_contact_model_part = rModelPart.GetSubModelPart("Contact");
292  ModelPart& r_computing_contact_model_part = rModelPart.GetSubModelPart("ComputingContact");
293 
294  // We reset the flag
295  auto& r_nodes_array = r_contact_model_part.Nodes();
296  block_for_each(r_nodes_array, [&](NodeType& rNode) {
297  rNode.Set(VISITED, false);
298  rNode.Set(ISOLATED, false);
299  });
300 
301  // Now we set the flag in the nodes
302  auto& r_conditions_array = r_computing_contact_model_part.Conditions();
303  block_for_each(r_conditions_array, [&](Condition& rCond) {
304  auto& r_parent_geometry = rCond.GetGeometry().GetGeometryPart(0);
305  for (std::size_t i_node = 0; i_node < r_parent_geometry.size(); ++i_node) {
306  r_parent_geometry[i_node].SetLock();
307  if (r_parent_geometry[i_node].Is(VISITED) == false) {
308  r_parent_geometry[i_node].Set(ISOLATED, rCond.Is(ISOLATED));
309  r_parent_geometry[i_node].Set(VISITED, true);
310  } else {
311  r_parent_geometry[i_node].Set(ISOLATED, r_parent_geometry[i_node].Is(ISOLATED) && rCond.Is(ISOLATED));
312  }
313  r_parent_geometry[i_node].UnSetLock();
314  }
315  });
316 
317  // We fix the LM
318  block_for_each(r_nodes_array, [&](NodeType& rNode) {
319  if (rNode.Is(ISOLATED)) {
320  if (rNode.SolutionStepsDataHas(LAGRANGE_MULTIPLIER_CONTACT_PRESSURE)) {
321  rNode.Fix(LAGRANGE_MULTIPLIER_CONTACT_PRESSURE);
322  } else if (rNode.SolutionStepsDataHas(VECTOR_LAGRANGE_MULTIPLIER_X)) {
323  rNode.Fix(VECTOR_LAGRANGE_MULTIPLIER_X);
324  rNode.Fix(VECTOR_LAGRANGE_MULTIPLIER_Y);
325  rNode.Fix(VECTOR_LAGRANGE_MULTIPLIER_Z);
326  }
327  }
328  });
329  }
330 
335  void FreeIsolatedNodes(ModelPart& rModelPart)
336  {
337  KRATOS_ERROR_IF_NOT(rModelPart.HasSubModelPart("Contact")) << "CONTACT MODEL PART NOT CREATED" << std::endl;
338  ModelPart& r_contact_model_part = rModelPart.GetSubModelPart("Contact");
339 
340  // We release the LM
341  auto& r_nodes_array = r_contact_model_part.Nodes();
342  block_for_each(r_nodes_array, [&](NodeType& rNode) {
343  if (rNode.Is(ISOLATED)) {
344  if (rNode.SolutionStepsDataHas(LAGRANGE_MULTIPLIER_CONTACT_PRESSURE)) {
345  rNode.Free(LAGRANGE_MULTIPLIER_CONTACT_PRESSURE);
346  } else if (rNode.SolutionStepsDataHas(VECTOR_LAGRANGE_MULTIPLIER_X)) {
347  rNode.Free(VECTOR_LAGRANGE_MULTIPLIER_X);
348  rNode.Free(VECTOR_LAGRANGE_MULTIPLIER_Y);
349  rNode.Free(VECTOR_LAGRANGE_MULTIPLIER_Z);
350  }
351  }
352  });
353  }
