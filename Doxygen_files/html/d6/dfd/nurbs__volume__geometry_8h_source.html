648  Matrix& Jacobian( Matrix& rResult, const CoordinatesArrayType& rCoordinates ) const override
649  {
650  const SizeType working_space_dimension = this->WorkingSpaceDimension();
651  const SizeType local_space_dimension = this->LocalSpaceDimension();
652  const SizeType points_number = this->PointsNumber();
653 
654  rResult = ZeroMatrix(working_space_dimension,local_space_dimension);
655 
656  Matrix shape_functions_gradients(points_number, local_space_dimension);
657  ShapeFunctionsLocalGradients( shape_functions_gradients, rCoordinates );
658 
659  // Get control point indices
660  NurbsVolumeShapeFunction shape_function_container(mPolynomialDegreeU, mPolynomialDegreeV, mPolynomialDegreeW, 0);
661  shape_function_container.ComputeBSplineShapeFunctionValues(mKnotsU,mKnotsV,mKnotsW,
662  rCoordinates[0], rCoordinates[1], rCoordinates[2]);
663 
664  SizeType number_cp_u = NurbsUtilities::GetNumberOfControlPoints(mPolynomialDegreeU, mKnotsU.size());
665  SizeType number_cp_v = NurbsUtilities::GetNumberOfControlPoints(mPolynomialDegreeV, mKnotsV.size());
666  SizeType number_cp_w = NurbsUtilities::GetNumberOfControlPoints(mPolynomialDegreeW, mKnotsW.size());
667 
668  std::vector<int> cp_indices = shape_function_container.ControlPointIndices(number_cp_u, number_cp_v, number_cp_w);
669  SizeType number_of_cp = shape_function_container.NumberOfNonzeroControlPoints();
670 
671  for (IndexType i = 0; i < number_of_cp; ++i ) {
672  const array_1d<double, 3>& r_coordinates = (*this)[cp_indices[i]].Coordinates();
673  for(IndexType k = 0; k< working_space_dimension; ++k) {
674  const double value = r_coordinates[k];
675  for(IndexType m = 0; m < local_space_dimension; ++m) {
676  rResult(k,m) += value * shape_functions_gradients(i,m);
677  }
678  }
679  }
680 
681  return rResult;
682  }
