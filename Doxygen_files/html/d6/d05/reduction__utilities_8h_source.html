281template <class... Reducer>
282struct CombinedReduction {
283 typedef std::tuple<typename Reducer::value_type...> value_type;
284 typedef std::tuple<typename Reducer::return_type...> return_type;
285
286 std::tuple<Reducer...> mChild;
287
288 CombinedReduction() {}
289
291 return_type GetValue(){
292 return_type return_value;
293 fill_value<0>(return_value);
294 return return_value;
295 }
296
297 template <int I, class T>
298 typename std::enable_if<(I < sizeof...(Reducer)), void>::type
299 fill_value(T& v) {
300 std::get<I>(v) = std::get<I>(mChild).GetValue();
301 fill_value<I+1>(v);
302 };
303
304 template <int I, class T>
305 typename std::enable_if<(I == sizeof...(Reducer)), void>::type
306 fill_value(T& v) {}
307
309 template <class... T>
310 void LocalReduce(const std::tuple<T...> &&v) {
311 // Static recursive loop over tuple elements
312 reduce_local<0>(v);
313 }
314
316 void ThreadSafeReduce(const CombinedReduction &other) {
317 reduce_global<0>(other);
318 }
319
320 private:
321
322 template <int I, class T>
323 typename std::enable_if<(I < sizeof...(Reducer)), void>::type
324 reduce_local(T &&v) {
325 std::get<I>(mChild).LocalReduce(std::get<I>(v));
326 reduce_local<I+1>(std::forward<T>(v));
327 };
328
329 template <int I, class T>
330 typename std::enable_if<(I == sizeof...(Reducer)), void>::type
331 reduce_local(T &&v) {
332 // Exit static recursion
333 }
334
335 template <int I>
336 typename std::enable_if<(I < sizeof...(Reducer)), void>::type
337 reduce_global(const CombinedReduction &other) {
338 std::get<I>(mChild).ThreadSafeReduce(std::get<I>(other.mChild));
339 reduce_global<I+1>(other);
340 }
341
342 template <int I>
343 typename std::enable_if<(I == sizeof...(Reducer)), void>::type
344 reduce_global(const CombinedReduction &other) {
345 // Exit static recursion
346 }
347};
