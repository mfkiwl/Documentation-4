338 void CalculateNodeNeighbourCount()
339 {
340 for(int model_i =0;model_i<mpVMModelParts.size();model_i++)
341 {
342 ModelPart* mpVMModePart = mpVMModelParts[model_i];
343 auto& r_nodes = mpVMModePart->Nodes();
344 int mNumNodes = r_nodes.size();
345
346 VariableUtils variable_utils;
347 variable_utils.SetFlag(STRUCTURE,true,r_nodes);
348
349 // Note: this should not be parallel, the operation is not threadsafe if the variable is uninitialized
350 for (auto& r_node : r_nodes)
351 {
352 r_node.SetValue(NUMBER_OF_NEIGHBOUR_ELEMENTS,0);
353 }
354
355 mNumNodes = mpVMModePart->GetCommunicator().GetDataCommunicator().SumAll(mNumNodes);
356
357 auto& r_elements = mpVMModePart->Elements();
358 const int num_elements = r_elements.size();
359
360 #pragma omp parallel for
361 for (int i = 0; i < num_elements; i++)
362 {
363 auto i_elem = r_elements.begin() + i;
364 auto& r_geom = i_elem->GetGeometry();
365 for (unsigned int i = 0; i < r_geom.PointsNumber(); i++)
366 {
367 auto& r_node = r_geom[i];
368 if (r_node.Is(STRUCTURE))
369 {
370 r_node.SetLock();
371 r_node.GetValue(NUMBER_OF_NEIGHBOUR_ELEMENTS) += 1;
372 r_node.UnSetLock();
373 }
374 }
375 }
376
377 mpVMModePart->GetCommunicator().AssembleNonHistoricalData(NUMBER_OF_NEIGHBOUR_ELEMENTS);
378
379 }

386 ModelPart& r_controlling_object = mrModel.GetModelPart(control_obj.GetString());

391
392 if (root_model_part.HasSubModelPart(vm_model_part_name)){
393 p_vm_model_part = &(root_model_part.GetSubModelPart(vm_model_part_name));
394 for(int i =0; i<mpVMModelParts.size(); i++)
395 if(mpVMModelParts[i]->Name()==p_vm_model_part->Name())

452 void ComputeFilteredThickness(){
453
454 for(int model_i=0;model_i<mpVMModelParts.size();model_i++){
455
456 //first update control thickness
457 AddVariable1ToVarible2(mpVMModelParts[model_i],D_CT,CT);
458
459 //now filter nodal control thickness
460 //first we need to multiply with mass matrix
461 SetVariable(mpVMModelParts[model_i],HELMHOLTZ_VAR_THICKNESS,0.0);
462 SetVariable(mpVMModelParts[model_i],HELMHOLTZ_SOURCE_THICKNESS,0.0);
463 //now we need to multiply with the mass matrix
464 #pragma omp parallel for
465 for(auto& elem_i : mpVMModelParts[model_i]->Elements())
466 {
467 VectorType origin_values;
468 GetElementVariableValuesVector(elem_i,CT,origin_values);
469 MatrixType mass_matrix;
470 elem_i.Calculate(HELMHOLTZ_MASS_MATRIX,mass_matrix,mpVMModelParts[model_i]->GetProcessInfo());
471 VectorType int_vals = prod(mass_matrix,origin_values);
472 AddElementVariableValuesVector(elem_i,HELMHOLTZ_SOURCE_THICKNESS,int_vals);
473 }
474
475 //now we need to apply BCs
476 #pragma omp parallel for
477 for(auto& node_i : mpVMModelParts[model_i]->Nodes())
478 {
479 auto& r_nodal_variable_hl_var = node_i.FastGetSolutionStepValue(HELMHOLTZ_VAR_THICKNESS);
480 if(node_i.IsFixed(HELMHOLTZ_VAR_THICKNESS))
481 r_nodal_variable_hl_var = node_i.FastGetSolutionStepValue(FT);
482 }
483 mpStrategies[model_i]->Solve();
484 SetVariable1ToVarible2(mpVMModelParts[model_i],HELMHOLTZ_VAR_THICKNESS,FT);
485 }

488 void ComputePhyiscalThickness(){
489 for(int model_i=0;model_i<mpVMModelParts.size();model_i++){
490 //now do the projection and then set the PT
491 #pragma omp parallel for
492 for(auto& node_i : mpVMModelParts[model_i]->Nodes()){
493 const auto& filtered_thick = node_i.FastGetSolutionStepValue(FT);
494 auto& thickness = node_i.FastGetSolutionStepValue(PT);
495 auto& p_thickness = node_i.FastGetSolutionStepValue(PPT);
496 auto& thickness_der = node_i.FastGetSolutionStepValue(D_PT_D_FT);
497 auto& p_thickness_der = node_i.FastGetSolutionStepValue(D_PPT_D_FT);
498 thickness = ProjectForward(filtered_thick,filtered_thicknesses,physical_thicknesses,beta,1);
499 p_thickness = ProjectForward(filtered_thick,filtered_thicknesses,physical_thicknesses,beta,SIMP_pow_fac);
500 thickness_der = ProjectionDerivative(filtered_thick,filtered_thicknesses,physical_thicknesses,beta,1);
501 p_thickness_der = ProjectionDerivative(filtered_thick,filtered_thicknesses,physical_thicknesses,beta,SIMP_pow_fac);
502 }
503 }
504
505 // update elements' t
506 for(auto& control_obj : mControlSettings["controlling_objects"]){
507 ModelPart& r_controlling_object = mrModel.GetModelPart(control_obj.GetString());
508 #pragma omp parallel for
509 for (int i = 0; i < (int)r_controlling_object.Elements().size(); i++) {
510 ModelPart::ElementsContainerType::iterator it = r_controlling_object.ElementsBegin() + i;
511 double elem_i_thicknes = 0.0;
512 double elem_i_p_thicknes = 0.0;
513 for(unsigned int node_element = 0; node_element<it->GetGeometry().size(); node_element++){
514 elem_i_thicknes += it->GetGeometry()[node_element].FastGetSolutionStepValue(PT);
515 elem_i_p_thicknes += it->GetGeometry()[node_element].FastGetSolutionStepValue(PPT);
516 }
517
518 elem_i_thicknes /= it->GetGeometry().size();
519 elem_i_p_thicknes /= it->GetGeometry().size();
520
521 it->GetProperties().SetValue(THICKNESS,elem_i_p_thicknes);
522 it->GetProperties().SetValue(PT,elem_i_thicknes);
523 }
524 }

528 double ProjectForward(double x,Vector x_limits,Vector y_limits,double beta, int penal_fac = 1){
529
530 double x1,x2,y1,y2;
531 int index_x1 = 0;
532 if(x>=x_limits[x_limits.size()-1]){
533 x1=x_limits[x_limits.size()-2];
534 index_x1 = x_limits.size()-2;
535 x2=x_limits[x_limits.size()-1];
536 y1=y_limits[y_limits.size()-2];
537 y2=y_limits[y_limits.size()-1];
538 }
539 else if(x<=x_limits[0]){
540 x1=x_limits[0];
541 index_x1 = 0;
542 x2=x_limits[1];
543 y1=y_limits[0];
544 y2=y_limits[1];
545 }
546 else{
547 for(int i=0;i<x_limits.size()-1;i++)
548 if((x>=x_limits[i]) && (x<=x_limits[i+1]))
549 {
550 y1 = y_limits[i];
551 y2 = y_limits[i+1];
552 x1 = x_limits[i];
553 index_x1 = i;
554 x2 = x_limits[i+1];
555 break;
556 }
557 }
558
559 double pow_val = -2.0*beta*(x-(x1+x2)/2);
560
561 if(pow_val>700)
562 pow_val = 700;
563 if(pow_val<-700)
564 pow_val = -700;
565
566 return (y2-y1)/(std::pow(1+std::exp(pow_val),penal_fac)) + y1;
567 }

570 double ProjectBackward(double y,Vector x_limits,Vector y_limits,double beta){
571
572 double x = 0;
573 if(y>=y_limits[y_limits.size()-1])
574 x = x_limits[y_limits.size()-1];
575 else if(y<=y_limits[0])
576 x = x_limits[0];
577 else{
578 for(int i=0;i<y_limits.size()-1;i++)
579 if((y>y_limits[i]) && (y<y_limits[i+1]))
580 {
581 double y1 = y_limits[i];
582 double y2 = y_limits[i+1];
583 double x1 = x_limits[i];
584 double x2 = x_limits[i+1];
585 x = ((x2+x1)/2.0) + (1.0/(-2.0*beta)) * std::log(((y2-y1)/(y-y1))-1);
586 break;
587 }
588 else if(y==y_limits[i]){
589 x = x_limits[i];
590 break;
591 }
592 else if(y==y_limits[i+1]){
593 x = x_limits[i+1];
594 break;
595 }
596 }
597 return x;
598 }

600 double ProjectionDerivative(double x,Vector x_limits,Vector y_limits,double beta,int penal_fac = 1){
601
602 double dfdx = 0;
603 double x1,x2,y1,y2;
604 if(x>=x_limits[x_limits.size()-1]){
605 x1=x_limits[x_limits.size()-2];
606 x2=x_limits[x_limits.size()-1];
607 y1=y_limits[y_limits.size()-2];
608 y2=y_limits[y_limits.size()-1];
609 }
610 else if(x<=x_limits[0]){
611 x1=x_limits[0];
612 x2=x_limits[1];
613 y1=y_limits[0];
614 y2=y_limits[1];
615 }
616 else{
617 for(int i=0;i<x_limits.size()-1;i++)
618 if((x>=x_limits[i]) && (x<=x_limits[i+1]))
619 {
620 y1 = y_limits[i];
621 y2 = y_limits[i+1];
622 x1 = x_limits[i];
623 x2 = x_limits[i+1];
624 break;
625 }
626 }
627
628 double pow_val = -2.0*beta*(x-(x1+x2)/2);
629
630 if(pow_val>700)
631 pow_val = 700;
632 if(pow_val<-700)
633 pow_val = -700;
634
635 double dydx = (y2-y1) * (1.0/std::pow(1+std::exp(pow_val),penal_fac+1)) * penal_fac * 2.0 * beta * std::exp(pow_val);
636
637 // if (y2<y1)
638 // dydx *=-1;
639
640 return dydx;
