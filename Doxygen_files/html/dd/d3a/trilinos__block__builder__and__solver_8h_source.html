166 void Build(typename TSchemeType::Pointer pScheme,
167 ModelPart& rModelPart,
168 TSystemMatrixType& rA,
169 TSystemVectorType& rb) override
170 {
171 KRATOS_TRY
172
173 KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
174
175 // Resetting to zero the vector of reactions
176 TSparseSpace::SetToZero(*BaseType::mpReactionsVector);
177
178 // Contributions to the system
179 LocalSystemMatrixType LHS_Contribution = LocalSystemMatrixType(0, 0);
180 LocalSystemVectorType RHS_Contribution = LocalSystemVectorType(0);
181
182 // Vector containing the localization in the system of the different terms
183 Element::EquationIdVectorType equation_ids_vector;
184 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
185
186 BuiltinTimer build_timer;
187
188 // Assemble all elements
189 for (auto it = rModelPart.Elements().ptr_begin(); it < rModelPart.Elements().ptr_end(); it++) {
190 // Detect if the element is active or not. If the user did not make
191 // any choice the element is active by default
192 const bool element_is_active = !((*it)->IsDefined(ACTIVE)) || (*it)->Is(ACTIVE);
193
194 if (element_is_active) {
195 // Calculate elemental contribution
196 pScheme->CalculateSystemContributions(**it, LHS_Contribution, RHS_Contribution, equation_ids_vector, r_current_process_info);
197
198 // Assemble the elemental contribution
199 TSparseSpace::AssembleLHS(rA, LHS_Contribution, equation_ids_vector);
200 TSparseSpace::AssembleRHS(rb, RHS_Contribution, equation_ids_vector);
201 }
202 }
203
204 LHS_Contribution.resize(0, 0, false);
205 RHS_Contribution.resize(0, false);
206
207 // Assemble all conditions
208 for (auto it = rModelPart.Conditions().ptr_begin(); it < rModelPart.Conditions().ptr_end(); it++) {
209 // detect if the element is active or not. If the user did not make
210 // any choice the element is active by default
211 const bool condition_is_active = !((*it)->IsDefined(ACTIVE)) || (*it)->Is(ACTIVE);
212 if (condition_is_active) {
213 // Calculate elemental contribution
214 pScheme->CalculateSystemContributions(**it, LHS_Contribution, RHS_Contribution, equation_ids_vector, r_current_process_info);
215
216 // Assemble the condition contribution
217 TSparseSpace::AssembleLHS(rA, LHS_Contribution, equation_ids_vector);
218 TSparseSpace::AssembleRHS(rb, RHS_Contribution, equation_ids_vector);
219 }
220 }
221
222 // Finalizing the assembly
223 rA.GlobalAssemble();
224 rb.GlobalAssemble();
225
226 KRATOS_INFO_IF("TrilinosBlockBuilderAndSolver", BaseType::GetEchoLevel() >= 1) << "Build time: " << build_timer.ElapsedSeconds() << std::endl;
227
228 KRATOS_CATCH("")
229 }

483 void SetUpDofSet(
484 typename TSchemeType::Pointer pScheme,
485 ModelPart& rModelPart
486 ) override
487 {
488 KRATOS_TRY
489
490 using DofsVectorType = Element::DofsVectorType;
491
492 // Gets the array of elements from the modeler
493 DofsVectorType dof_list;
494 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
495
496 DofsArrayType temp_dofs_array;
497 IndexType guess_num_dofs = rModelPart.GetCommunicator().LocalMesh().NumberOfNodes() * 3;
498 temp_dofs_array.reserve(guess_num_dofs);
499 BaseType::mDofSet = DofsArrayType();
500
501 // Taking dofs of elements
502 ElementsArrayType& r_elements_array = rModelPart.GetCommunicator().LocalMesh().Elements();
503 for (auto it_elem = r_elements_array.ptr_begin(); it_elem != r_elements_array.ptr_end(); ++it_elem) {
504 pScheme->GetDofList(**it_elem, dof_list, r_current_process_info);
505 for (auto i_dof = dof_list.begin(); i_dof != dof_list.end(); ++i_dof)
506 temp_dofs_array.push_back(*i_dof);
507 }
508
509 // Taking dofs of conditions
510 auto& r_conditions_array = rModelPart.GetCommunicator().LocalMesh().Conditions();
511 for (auto it_cond = r_conditions_array.ptr_begin(); it_cond != r_conditions_array.ptr_end(); ++it_cond) {
512 pScheme->GetDofList(**it_cond, dof_list, r_current_process_info);
513 for (auto i_dof = dof_list.begin(); i_dof != dof_list.end(); ++i_dof)
514 temp_dofs_array.push_back(*i_dof);
515 }
516
517 temp_dofs_array.Unique();
518 BaseType::mDofSet = temp_dofs_array;
519
520 // Throws an exception if there are no Degrees of freedom involved in
521 // the analysis
522 KRATOS_ERROR_IF(rModelPart.GetCommunicator().GetDataCommunicator().SumAll(BaseType::mDofSet.size()) == 0) << "No degrees of freedom!";
523
524#ifdef KRATOS_DEBUG
525 // If reactions are to be calculated, we check if all the dofs have
526 // reactions defined This is to be done only in debug mode
527 if (BaseType::GetCalculateReactionsFlag()) {
528 for (auto dof_iterator = BaseType::mDofSet.begin();
529 dof_iterator != BaseType::mDofSet.end(); ++dof_iterator) {
530 KRATOS_ERROR_IF_NOT(dof_iterator->HasReaction())
531 << "Reaction variable not set for the following : " << std::endl
532 << "Node : " << dof_iterator->Id() << std::endl
533 << "Dof : " << (*dof_iterator) << std::endl
534 << "Not possible to calculate reactions." << std::endl;
535 }
536 }
537#endif
538 BaseType::mDofSetIsInitialized = true;
539
540 KRATOS_CATCH("")
541 }

548 void SetUpSystem(ModelPart& rModelPart) override
549 {
550 int free_size = 0;
551 auto& r_comm = rModelPart.GetCommunicator();
552 const auto& r_data_comm = r_comm.GetDataCommunicator();
553 int current_rank = r_comm.MyPID();
554
555 // Calculating number of fixed and free dofs
556 for (const auto& r_dof : BaseType::mDofSet)
557 if (r_dof.GetSolutionStepValue(PARTITION_INDEX) == current_rank)
558 free_size++;
559
560 // Calculating the total size and required offset
561 int free_offset;
562 int global_size;
563
564 // The correspounding offset by the sum of the sizes in thread with
565 // inferior current_rank
566 free_offset = r_data_comm.ScanSum(free_size);
567
568 // The total size by the sum of all size in all threads
569 global_size = r_data_comm.SumAll(free_size);
570
571 // finding the offset for the begining of the partition
572 free_offset -= free_size;
573
574 // Now setting the equation id with .
575 for (auto& r_dof : BaseType::mDofSet)
576 if (r_dof.GetSolutionStepValue(PARTITION_INDEX) == current_rank)
577 r_dof.SetEquationId(free_offset++);
578
579 BaseType::mEquationSystemSize = global_size;
580 mLocalSystemSize = free_size;
581 KRATOS_INFO_IF_ALL_RANKS("TrilinosBlockBuilderAndSolver", BaseType::GetEchoLevel() > 1)
582 << "\n BaseType::mEquationSystemSize = " << BaseType::mEquationSystemSize
583 << "\n mLocalSystemSize = " << mLocalSystemSize
584 << "\n free_offset = " << free_offset << std::endl;
585
586 // by Riccardo ... it may be wrong!
587 mFirstMyId = free_offset - mLocalSystemSize;
588 mLastMyId = mFirstMyId + mLocalSystemSize;
589
590 r_comm.SynchronizeDofs();
591 }

603 void ResizeAndInitializeVectors(typename TSchemeType::Pointer pScheme,
604 TSystemMatrixPointerType& rpA,
605 TSystemVectorPointerType& rpDx,
606 TSystemVectorPointerType& rpb,
607 ModelPart& rModelPart) override
608 {
609 KRATOS_TRY
610
611 // Resizing the system vectors and matrix
612 if (rpA == nullptr || TSparseSpace::Size1(*rpA) == 0 ||
613 BaseType::GetReshapeMatrixFlag() == true) { // if the matrix is not initialized
614 IndexType number_of_local_dofs = mLastMyId - mFirstMyId;
615 int temp_size = number_of_local_dofs;
616 if (temp_size < 1000) {
617 temp_size = 1000;
618 }
619 std::vector<int> temp(temp_size, 0);
620
621 // TODO: Check if these should be local elements and conditions
622 auto& r_elements_array = rModelPart.Elements();
623 auto& r_conditions_array = rModelPart.Conditions();
624
625 // Generate map - use the "temp" array here
626 for (IndexType i = 0; i != number_of_local_dofs; i++) {
627 temp[i] = mFirstMyId + i;
628 }
629 Epetra_Map my_map(-1, number_of_local_dofs, temp.data(), 0, mrComm);
630
631 // Create and fill the graph of the matrix --> the temp array is
632 // reused here with a different meaning
633 Epetra_FECrsGraph Agraph(Copy, my_map, mGuessRowSize);
634 Element::EquationIdVectorType equation_ids_vector;
635 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
636
637 // Assemble all elements
638 for (auto it_elem = r_elements_array.ptr_begin(); it_elem != r_elements_array.ptr_end(); ++it_elem) {
639 pScheme->EquationId(**it_elem, equation_ids_vector, r_current_process_info);
640
641 // Filling the list of active global indices (non fixed)
642 IndexType num_active_indices = 0;
643 for (IndexType i = 0; i < equation_ids_vector.size(); i++) {
644 temp[num_active_indices] = equation_ids_vector[i];
645 num_active_indices += 1;
646 }
647
648 if (num_active_indices != 0) {
649 const int ierr = Agraph.InsertGlobalIndices(num_active_indices, temp.data(), num_active_indices, temp.data());
650 KRATOS_ERROR_IF(ierr < 0) << ": Epetra failure in Graph.InsertGlobalIndices. Error code: " << ierr << std::endl;
651 }
652 std::fill(temp.begin(), temp.end(), 0);
653 }
654
655 // assemble all conditions
656 for (auto it_cond = r_conditions_array.ptr_begin(); it_cond != r_conditions_array.ptr_end(); ++it_cond) {
657 pScheme->EquationId(**it_cond, equation_ids_vector, r_current_process_info);
658
659 // filling the list of active global indices (non fixed)
660 IndexType num_active_indices = 0;
661 for (IndexType i = 0; i < equation_ids_vector.size(); i++) {
662 temp[num_active_indices] = equation_ids_vector[i];
663 num_active_indices += 1;
664 }
665
666 if (num_active_indices != 0) {
667 const int ierr = Agraph.InsertGlobalIndices(num_active_indices, temp.data(), num_active_indices, temp.data());
668 KRATOS_ERROR_IF(ierr < 0) << ": Epetra failure in Graph.InsertGlobalIndices. Error code: " << ierr << std::endl;
669 }
670 std::fill(temp.begin(), temp.end(), 0);
671 }
672
673 // Finalizing graph construction
674 const int ierr = Agraph.GlobalAssemble();
675 KRATOS_ERROR_IF(ierr < 0) << ": Epetra failure in Graph.InsertGlobalIndices. Error code: " << ierr << std::endl;
676
677 // Generate a new matrix pointer according to this graph
678 TSystemMatrixPointerType p_new_A = TSystemMatrixPointerType(new TSystemMatrixType(Copy, Agraph));
679 rpA.swap(p_new_A);
680
681 // Generate new vector pointers according to the given map
682 if (rpb == nullptr || TSparseSpace::Size(*rpb) != BaseType::mEquationSystemSize) {
683 TSystemVectorPointerType p_new_b = TSystemVectorPointerType(new TSystemVectorType(my_map));
684 rpb.swap(p_new_b);
685 }
686 if (rpDx == nullptr || TSparseSpace::Size(*rpDx) != BaseType::mEquationSystemSize) {
687 TSystemVectorPointerType p_new_Dx = TSystemVectorPointerType(new TSystemVectorType(my_map));
688 rpDx.swap(p_new_Dx);
689 }
690 if (BaseType::mpReactionsVector == nullptr) { // if the pointer is not initialized initialize it to an
691 // empty matrix
692 TSystemVectorPointerType pNewReactionsVector = TSystemVectorPointerType(new TSystemVectorType(my_map));
693 BaseType::mpReactionsVector.swap(pNewReactionsVector);
694 }
695 } else if (BaseType::mpReactionsVector == nullptr && this->mCalculateReactionsFlag) {
696 TSystemVectorPointerType pNewReactionsVector =
697 TSystemVectorPointerType(new TSystemVectorType(rpDx->Map()));
698 BaseType::mpReactionsVector.swap(pNewReactionsVector);
699 } else {
700 if (TSparseSpace::Size1(*rpA) == 0 ||
701 TSparseSpace::Size1(*rpA) != BaseType::mEquationSystemSize ||
702 TSparseSpace::Size2(*rpA) != BaseType::mEquationSystemSize) {
703 KRATOS_ERROR << "It should not come here resizing is not allowed this way!!!!!!!! ... " << std::endl;
704 }
705 }
706
707 KRATOS_CATCH("")
708 }

712 void CalculateReactions(typename TSchemeType::Pointer pScheme,
713 ModelPart& rModelPart,
714 TSystemMatrixType& rA,
715 TSystemVectorType& rDx,
716 TSystemVectorType& rb) override
717 {
718 TSparseSpace::SetToZero(rb);
719
720 // Refresh RHS to have the correct reactions
721 BuildRHS(pScheme, rModelPart, rb);
722
723 // Initialize the Epetra importer
724 // TODO: this part of the code has been pasted until a better solution
725 // is found
726 int system_size = TSparseSpace::Size(rb);
727 int number_of_dofs = BaseType::mDofSet.size();
728 std::vector<int> index_array(number_of_dofs);
729
730 // Filling the array with the global ids
731 int counter = 0;
732 int id = 0;
733 for (const auto& dof : BaseType::mDofSet) {
734 id = dof.EquationId();
735 if (id < system_size) {
736 index_array[counter++] = id;
737 }
738 }
739
740 std::sort(index_array.begin(), index_array.end());
741 std::vector<int>::iterator NewEnd = std::unique(index_array.begin(), index_array.end());
742 index_array.resize(NewEnd - index_array.begin());
743
744 int check_size = -1;
745 int tot_update_dofs = index_array.size();
746 rb.Comm().SumAll(&tot_update_dofs, &check_size, 1);
747 KRATOS_ERROR_IF(check_size < system_size)
748 << "Dof count is not correct. There are less dofs than expected.\n"
749 << "Expected number of active dofs = " << system_size
750 << " dofs found = " << check_size;
751
752 // Defining a map as needed
753 Epetra_Map dof_update_map(-1, index_array.size(),
754 &(*(index_array.begin())), 0, rb.Comm());
755
756 // Defining the importer class
757 Kratos::shared_ptr<Epetra_Import> pDofImporter = Kratos::make_shared<Epetra_Import>(dof_update_map, rb.Map());
758
759 // Defining a temporary vector to gather all of the values needed
760 Epetra_Vector temp_RHS(pDofImporter->TargetMap());
761
762 // Importing in the new temp_RHS vector the values
763 int ierr = temp_RHS.Import(rb, *pDofImporter, Insert);
764 KRATOS_ERROR_IF(ierr != 0) << "Epetra failure found - error code: " << ierr << std::endl;
765
766 double* temp_RHS_values; // DO NOT make delete of this one!!
767 temp_RHS.ExtractView(&temp_RHS_values);
768
769 rb.Comm().Barrier();
770
771 const int ndofs = static_cast<int>(BaseType::mDofSet.size());
772
773 // Store the RHS values in the reaction variable
774 // NOTE: dofs are assumed to be numbered consecutively in the
775 // BlockBuilderAndSolver
776 for (int k = 0; k < ndofs; k++) {
777 auto dof_iterator = BaseType::mDofSet.begin() + k;
778
779 const int i = (dof_iterator)->EquationId();
780 // (dof_iterator)->GetSolutionStepReactionValue() = -(*b[i]);
781 const double react_val = temp_RHS[pDofImporter->TargetMap().LID(i)];
782 (dof_iterator->GetSolutionStepReactionValue()) = -react_val;
783 }
784 }

798 void ApplyDirichletConditions(typename TSchemeType::Pointer pScheme,
799 ModelPart& rModelPart,
800 TSystemMatrixType& rA,
801 TSystemVectorType& rDx,
802 TSystemVectorType& rb) override
803 {
804 KRATOS_TRY
805
806 // loop over all dofs to find the fixed ones
807 std::vector<int> global_ids(BaseType::mDofSet.size());
808 std::vector<int> is_dirichlet(BaseType::mDofSet.size());
809
810 IndexType i = 0;
811 for (const auto& dof : BaseType::mDofSet) {
812 const int global_id = dof.EquationId();
813 global_ids[i] = global_id;
814 is_dirichlet[i] = dof.IsFixed();
815 ++i;
816 }
817
818 // here we construct and fill a vector "fixed local" which cont
819 Epetra_Map localmap(-1, global_ids.size(), global_ids.data(), 0, rA.Comm());
820 Epetra_IntVector fixed_local(Copy, localmap, is_dirichlet.data());
821
822 Epetra_Import dirichlet_importer(rA.ColMap(), fixed_local.Map());
823
824 // defining a temporary vector to gather all of the values needed
825 Epetra_IntVector fixed(rA.ColMap());
826
827 // importing in the new temp vector the values
828 int ierr = fixed.Import(fixed_local, dirichlet_importer, Insert);
829 if (ierr != 0)
830 KRATOS_ERROR << "Epetra failure found";
831
832 for (int i = 0; i < rA.NumMyRows(); i++) {
833 int numEntries; // number of non-zero entries
834 double* vals; // row non-zero values
835 int* cols; // column indices of row non-zero values
836 rA.ExtractMyRowView(i, numEntries, vals, cols);
837
838 int row_gid = rA.RowMap().GID(i);
839 int row_lid = localmap.LID(row_gid);
840
841 if (fixed_local[row_lid] == 0) // not a dirichlet row
842 {
843 for (int j = 0; j < numEntries; j++) {
844 if (fixed[cols[j]] == true)
845 vals[j] = 0.0;
846 }
847 }
848 else // this IS a dirichlet row
849 {
850 // set to zero the rhs
851 rb[0][i] = 0.0; // note that the index of i is expected to be
852 // coherent with the rows of A
853
854 // set to zero the whole row
855 for (int j = 0; j < numEntries; j++) {
856 int col_gid = rA.ColMap().GID(cols[j]);
857 if (col_gid != row_gid)
858 vals[j] = 0.0;
859 }
860 }
861 }
862
863 KRATOS_CATCH("");
864 }
