615 template<class TType>
616 void CalculateOnConstitutiveLaw(
617 const Variable<TType>& rVariable,
618 std::vector<TType>& rOutput,
619 const ProcessInfo& rCurrentProcessInfo)
620 {
621 const auto& r_geometry = GetGeometry();
622 const SizeType n_nodes = r_geometry.size();
623 const SizeType dim = r_geometry.WorkingSpaceDimension();
624 const SizeType strain_size = mConstitutiveLawVector[0]->GetStrainSize();
625 const SizeType n_gauss = r_geometry.IntegrationPointsNumber(GetIntegrationMethod());
626 const auto& r_integration_points = r_geometry.IntegrationPoints(GetIntegrationMethod());
627
628 // Create the kinematics container and fill the nodal data
629 KinematicVariables kinematic_variables(strain_size, dim, n_nodes);
630 for (IndexType i_node = 0; i_node < n_nodes; ++i_node) {
631 const auto& r_disp = r_geometry[i_node].FastGetSolutionStepValue(DISPLACEMENT);
632 for (IndexType d = 0; d < dim; ++d) {
633 kinematic_variables.Displacements(i_node * dim + d) = r_disp[d];
634 }
635 kinematic_variables.VolumetricNodalStrains[i_node] = r_geometry[i_node].FastGetSolutionStepValue(VOLUMETRIC_STRAIN);
636 }
637
638 // Create the constitutive variables and values containers
639 ConstitutiveVariables constitutive_variables(strain_size);
640 ConstitutiveLaw::Parameters cons_law_values(r_geometry, GetProperties(), rCurrentProcessInfo);
641 auto& r_cons_law_options = cons_law_values.GetOptions();
642 r_cons_law_options.Set(ConstitutiveLaw::COMPUTE_STRESS, true);
643 r_cons_law_options.Set(ConstitutiveLaw::USE_ELEMENT_PROVIDED_STRAIN, true);
644 r_cons_law_options.Set(ConstitutiveLaw::COMPUTE_CONSTITUTIVE_TENSOR, false);
645
646 // Call the initialize material response
647 for (IndexType i_gauss = 0; i_gauss < n_gauss; ++i_gauss) {
648 // Recompute the kinematics
649 CalculateKinematicVariables(kinematic_variables, i_gauss, GetIntegrationMethod());
650
651 // Set the constitutive variables
652 SetConstitutiveVariables(kinematic_variables, constitutive_variables, cons_law_values, i_gauss, r_integration_points);
653
654 // Calculate the output value
655 rOutput[i_gauss] = mConstitutiveLawVector[i_gauss]->CalculateValue(cons_law_values, rVariable, rOutput[i_gauss] );
656 }
657 }
