208 void SetTimeCoefficients(ProcessInfo &rCurrentProcessInfo)
209 {
210 KRATOS_TRY;
211
212 if (mTimeOrder == 2)
213 {
214 // calculate the BDF coefficients
215 double Dt = rCurrentProcessInfo[DELTA_TIME];
216 double OldDt = rCurrentProcessInfo.GetPreviousTimeStepInfo(1)[DELTA_TIME];
217
218 double Rho = OldDt / Dt;
219 double TimeCoeff = 1.0 / (Dt * Rho * Rho + Dt * Rho);
220
221 Vector &BDFcoeffs = rCurrentProcessInfo[BDF_COEFFICIENTS];
222 BDFcoeffs.resize(3, false);
223
224 BDFcoeffs[0] = TimeCoeff * (Rho * Rho + 2.0 * Rho); // coefficient for step n+1 (3/2Dt if Dt is constant)
225 BDFcoeffs[1] = -TimeCoeff * (Rho * Rho + 2.0 * Rho + 1.0); // coefficient for step n (-4/2Dt if Dt is constant)
226 BDFcoeffs[2] = TimeCoeff; // coefficient for step n-1 (1/2Dt if Dt is constant)
227 }
228 else if (mTimeOrder == 1)
229 {
230 double Dt = rCurrentProcessInfo[DELTA_TIME];
231 double TimeCoeff = 1.0 / Dt;
232
233 Vector &BDFcoeffs = rCurrentProcessInfo[BDF_COEFFICIENTS];
234 BDFcoeffs.resize(2, false);
235
236 BDFcoeffs[0] = TimeCoeff; // coefficient for step n+1 (1/Dt)
237 BDFcoeffs[1] = -TimeCoeff; // coefficient for step n (-1/Dt)
238 }
239
240 KRATOS_CATCH("");

748
749 ModelPart &rModelPart = BaseType::GetModelPart();
750 const unsigned int dimension = rModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
751 double currFirstLame = itNode->FastGetSolutionStepValue(VOLUMETRIC_COEFFICIENT);
752 double deviatoricCoeff = 0;
753
754 Matrix Fgrad = itNode->FastGetSolutionStepValue(NODAL_DEFORMATION_GRAD);
755 Matrix FgradVel = itNode->FastGetSolutionStepValue(NODAL_DEFORMATION_GRAD_VEL);
756 double detFgrad = 1.0;
757 Matrix InvFgrad = ZeroMatrix(dimension, dimension);
758 Matrix SpatialVelocityGrad = ZeroMatrix(dimension, dimension);
759
760 if (dimension == 2)
761 {
762 MathUtils<double>::InvertMatrix2(Fgrad, InvFgrad, detFgrad);
763 }
764 else if (dimension == 3)
765 {
766 MathUtils<double>::InvertMatrix3(Fgrad, InvFgrad, detFgrad);
767 }
768
769 // it computes the spatial velocity gradient tensor --> [L_ij]=dF_ik*invF_kj
770 SpatialVelocityGrad = prod(FgradVel, InvFgrad);
771
772 if (dimension == 2)
773 {
774 auto &r_stain_tensor2D = itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE);
775 r_stain_tensor2D[0] = SpatialVelocityGrad(0, 0);
776 r_stain_tensor2D[1] = SpatialVelocityGrad(1, 1);
777 r_stain_tensor2D[2] = 0.5 * (SpatialVelocityGrad(1, 0) + SpatialVelocityGrad(0, 1));
778
779 this->ComputeDeviatoricCoefficientForFluid(itNode, dimension, deviatoricCoeff);
780
781 double DefVol = itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] + itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1];
782
783 itNode->GetSolutionStepValue(NODAL_VOLUMETRIC_DEF_RATE) = DefVol;
784
785 double nodalSigmaTot_xx = currFirstLame * DefVol + 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0];
786 double nodalSigmaTot_yy = currFirstLame * DefVol + 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1];
787 double nodalSigmaTot_xy = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2];
788
789 double nodalSigmaDev_xx = 2.0 * deviatoricCoeff * (itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] - DefVol / 3.0);
790 double nodalSigmaDev_yy = 2.0 * deviatoricCoeff * (itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] - DefVol / 3.0);
791 double nodalSigmaDev_xy = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2];
792
793 auto &r_stress_tensor2D = itNode->GetSolutionStepValue(NODAL_CAUCHY_STRESS, 0);
794 r_stress_tensor2D[0] = nodalSigmaTot_xx;
795 r_stress_tensor2D[1] = nodalSigmaTot_yy;
796 r_stress_tensor2D[2] = nodalSigmaTot_xy;
797
798 auto &r_dev_stress_tensor2D = itNode->GetSolutionStepValue(NODAL_DEVIATORIC_CAUCHY_STRESS, 0);
799 r_dev_stress_tensor2D[0] = nodalSigmaDev_xx;
800 r_dev_stress_tensor2D[1] = nodalSigmaDev_yy;
801 r_dev_stress_tensor2D[2] = nodalSigmaDev_xy;
802 }
803 else if (dimension == 3)
804 {
805 auto &r_stain_tensor3D = itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE);
806 r_stain_tensor3D[0] = SpatialVelocityGrad(0, 0);
807 r_stain_tensor3D[1] = SpatialVelocityGrad(1, 1);
808 r_stain_tensor3D[2] = SpatialVelocityGrad(2, 2);
809 r_stain_tensor3D[3] = 0.5 * (SpatialVelocityGrad(1, 0) + SpatialVelocityGrad(0, 1));
810 r_stain_tensor3D[4] = 0.5 * (SpatialVelocityGrad(2, 0) + SpatialVelocityGrad(0, 2));
811 r_stain_tensor3D[5] = 0.5 * (SpatialVelocityGrad(2, 1) + SpatialVelocityGrad(1, 2));
812
813 this->ComputeDeviatoricCoefficientForFluid(itNode, dimension, deviatoricCoeff);
814
815 double DefVol = itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] + itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] + itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2];
816 itNode->GetSolutionStepValue(NODAL_VOLUMETRIC_DEF_RATE) = DefVol;
817
818 double nodalSigmaTot_xx = currFirstLame * DefVol + 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0];
819 double nodalSigmaTot_yy = currFirstLame * DefVol + 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1];
820 double nodalSigmaTot_zz = currFirstLame * DefVol + 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2];
821 double nodalSigmaTot_xy = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3];
822 double nodalSigmaTot_xz = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4];
823 double nodalSigmaTot_yz = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5];
824
825 double nodalSigmaDev_xx = 2.0 * deviatoricCoeff * (itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] - DefVol / 3.0);
826 double nodalSigmaDev_yy = 2.0 * deviatoricCoeff * (itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] - DefVol / 3.0);
827 double nodalSigmaDev_zz = 2.0 * deviatoricCoeff * (itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] - DefVol / 3.0);
828 double nodalSigmaDev_xy = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3];
829 double nodalSigmaDev_xz = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4];
830 double nodalSigmaDev_yz = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5];
831
832 auto &r_stress_tensor3D = itNode->GetSolutionStepValue(NODAL_CAUCHY_STRESS, 0);
833 r_stress_tensor3D[0] = nodalSigmaTot_xx;
834 r_stress_tensor3D[1] = nodalSigmaTot_yy;
835 r_stress_tensor3D[2] = nodalSigmaTot_zz;
836 r_stress_tensor3D[3] = nodalSigmaTot_xy;
837 r_stress_tensor3D[4] = nodalSigmaTot_xz;
838 r_stress_tensor3D[5] = nodalSigmaTot_yz;
839
840 auto &r_dev_stress_tensor3D = itNode->GetSolutionStepValue(NODAL_DEVIATORIC_CAUCHY_STRESS, 0);
841 r_dev_stress_tensor3D[0] = nodalSigmaDev_xx;
842 r_dev_stress_tensor3D[1] = nodalSigmaDev_yy;
843 r_dev_stress_tensor3D[2] = nodalSigmaDev_zz;
844 r_dev_stress_tensor3D[3] = nodalSigmaDev_xy;
845 r_dev_stress_tensor3D[4] = nodalSigmaDev_xz;
846 r_dev_stress_tensor3D[5] = nodalSigmaDev_yz;
847 }

850 void ComputeDeviatoricCoefficientForFluid(ModelPart::NodeIterator itNode, const unsigned int dimension, double &deviatoricCoefficient)
851 {
852 ModelPart &rModelPart = BaseType::GetModelPart();
853 const double tolerance = 1e-12;
854
855 if (rModelPart.GetNodalSolutionStepVariablesList().Has(STATIC_FRICTION)) // mu(I)-rheology
856 {
857 const double static_friction = itNode->FastGetSolutionStepValue(STATIC_FRICTION);
858 const double dynamic_friction = itNode->FastGetSolutionStepValue(DYNAMIC_FRICTION);
859 const double delta_friction = dynamic_friction - static_friction;
860 const double inertial_number_zero = itNode->FastGetSolutionStepValue(INERTIAL_NUMBER_ZERO);
861 const double grain_diameter = itNode->FastGetSolutionStepValue(GRAIN_DIAMETER);
862 const double grain_density = itNode->FastGetSolutionStepValue(GRAIN_DENSITY);
863 const double regularization_coeff = itNode->FastGetSolutionStepValue(REGULARIZATION_COEFFICIENT);
864
865 const double theta = 0.5;
866 double mean_pressure = itNode->FastGetSolutionStepValue(PRESSURE, 0) * theta + itNode->FastGetSolutionStepValue(PRESSURE, 1) * (1 - theta);
867
868 double pressure_tolerance = -1.0e-07;
869 if (mean_pressure > pressure_tolerance)
870 {
871 mean_pressure = pressure_tolerance;
872 }
873
874 if (dimension == 2)
875 {
876 itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE) = sqrt((2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] +
877 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] +
878 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2]));
879 }
880 else if (dimension == 3)
881 {
882 itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE) = sqrt(2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] +
883 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] +
884 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] +
885 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3] +
886 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4] +
887 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5]);
888 }
889 const double equivalent_strain_rate = itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE);
890 const double exponent = -equivalent_strain_rate / regularization_coeff;
891 const double second_viscous_term = delta_friction * grain_diameter / (inertial_number_zero * std::sqrt(std::fabs(mean_pressure) / grain_density) + equivalent_strain_rate * grain_diameter);
892
893 if (std::fabs(equivalent_strain_rate) > tolerance)
894 {
895 const double first_viscous_term = static_friction * (1 - std::exp(exponent)) / equivalent_strain_rate;
896 deviatoricCoefficient = (first_viscous_term + second_viscous_term) * std::fabs(mean_pressure);
897 }
898 else
899 {
900 deviatoricCoefficient = 1.0; // this is for the first iteration and first time step
901 }
902 }
903 else if (rModelPart.GetNodalSolutionStepVariablesList().Has(INTERNAL_FRICTION_ANGLE)) // frictiional viscoplastic model
904 {
905 const double dynamic_viscosity = itNode->FastGetSolutionStepValue(DYNAMIC_VISCOSITY);
906 const double friction_angle = itNode->FastGetSolutionStepValue(INTERNAL_FRICTION_ANGLE);
907 const double cohesion = itNode->FastGetSolutionStepValue(COHESION);
908 const double adaptive_exponent = itNode->FastGetSolutionStepValue(ADAPTIVE_EXPONENT);
909
910 const double theta = 0.5;
911 double mean_pressure = itNode->FastGetSolutionStepValue(PRESSURE, 0) * theta + itNode->FastGetSolutionStepValue(PRESSURE, 1) * (1 - theta);
912
913 double pressure_tolerance = -1.0e-07;
914 if (mean_pressure > pressure_tolerance)
915 {
916 mean_pressure = pressure_tolerance;
917 }
918
919 if (dimension == 2)
920 {
921 itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE) = sqrt((2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] +
922 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] +
923 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2]));
924 }
925 else if (dimension == 3)
926 {
927 itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE) = sqrt(2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] +
928 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] +
929 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] +
930 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3] +
931 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4] +
932 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5]);
933 }
934
935 const double equivalent_strain_rate = itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE);
936
937 // Ensuring that the case of equivalent_strain_rate = 0 is not problematic
938 if (std::fabs(equivalent_strain_rate) > tolerance)
939 {
940 const double friction_angle_rad = friction_angle * Globals::Pi / 180.0;
941 const double tanFi = std::tan(friction_angle_rad);
942 double regularization = 1.0 - std::exp(-adaptive_exponent * equivalent_strain_rate);
943 deviatoricCoefficient = dynamic_viscosity + regularization * ((cohesion + tanFi * fabs(mean_pressure)) / equivalent_strain_rate);
944 }
945 else
946 {
947 deviatoricCoefficient = dynamic_viscosity;
948 }
949 }
950 else if (rModelPart.GetNodalSolutionStepVariablesList().Has(YIELD_SHEAR)) // bingham model
951 {
952 if (dimension == 2)
953 {
954 itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE) = sqrt((2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] +
955 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] +
956 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2]));
957 }
958 else if (dimension == 3)
959 {
960 itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE) = sqrt(2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] +
961 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] +
962 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] +
963 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3] +
964 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4] +
965 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[5]);
966 }
967
968 const double yieldShear = itNode->FastGetSolutionStepValue(YIELD_SHEAR);
969 const double equivalentStrainRate = itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE);
970 const double adaptiveExponent = itNode->FastGetSolutionStepValue(ADAPTIVE_EXPONENT);
971 const double exponent = -adaptiveExponent * equivalentStrainRate;
972 deviatoricCoefficient = itNode->FastGetSolutionStepValue(DYNAMIC_VISCOSITY);
973 if (equivalentStrainRate > tolerance)
974 {
975 deviatoricCoefficient += (yieldShear / equivalentStrainRate) * (1 - exp(exponent));
976 }
977 }
978 else if (rModelPart.GetNodalSolutionStepVariablesList().Has(DYNAMIC_VISCOSITY))
979 {
980 deviatoricCoefficient = itNode->FastGetSolutionStepValue(DYNAMIC_VISCOSITY);
981 }
982 itNode->FastGetSolutionStepValue(DEVIATORIC_COEFFICIENT) = deviatoricCoefficient;
983 }

986 {
987
988 ModelPart &rModelPart = BaseType::GetModelPart();
989 const unsigned int dimension = rModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
990
991 double detFgrad = 1.0;
992 Matrix nodalFgrad = ZeroMatrix(dimension, dimension);
993 Matrix FgradVel = ZeroMatrix(dimension, dimension);
994 Matrix InvFgrad = ZeroMatrix(dimension, dimension);
995 Matrix SpatialVelocityGrad = ZeroMatrix(dimension, dimension);
996
997 nodalFgrad = itNode->FastGetSolutionStepValue(NODAL_DEFORMATION_GRAD);
998 FgradVel = itNode->FastGetSolutionStepValue(NODAL_DEFORMATION_GRAD_VEL);
999
1000 // Inverse
1001
1002 if (dimension == 2)
1003 {
1004 MathUtils<double>::InvertMatrix2(nodalFgrad, InvFgrad, detFgrad);
1005 }
1006 else if (dimension == 3)
1007 {
1008 MathUtils<double>::InvertMatrix3(nodalFgrad, InvFgrad, detFgrad);
1009 }
1010
1011 // it computes the spatial velocity gradient tensor --> [L_ij]=dF_ik*invF_kj
1012 SpatialVelocityGrad = prod(FgradVel, InvFgrad);
1013
1014 if (dimension == 2)
1015 {
1016
1017 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] = SpatialVelocityGrad(0, 0);
1018 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] = SpatialVelocityGrad(1, 1);
1019 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] = 0.5 * (SpatialVelocityGrad(1, 0) + SpatialVelocityGrad(0, 1));
1020
1021 itNode->FastGetSolutionStepValue(NODAL_EQUIVALENT_STRAIN_RATE) = sqrt((2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] +
1022 2.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] +
1023 4.0 * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] * itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2]));
1024
1025 double DefX = itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0];
1026 double DefY = itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1];
1027
1028 double DefVol = DefX + DefY;
1029
1030 itNode->GetSolutionStepValue(NODAL_VOLUMETRIC_DEF_RATE) = DefVol;
1031 }
1032 else if (dimension == 3)
1033 {
1034
1035 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] = SpatialVelocityGrad(0, 0);
1036 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] = SpatialVelocityGrad(1, 1);
1037 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] = SpatialVelocityGrad(2, 2);
1038 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[3] = 0.5 * (SpatialVelocityGrad(1, 0) + SpatialVelocityGrad(0, 1));
1039 itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[4] = 0.5 * (SpatialVelocityGrad(2, 0) + SpatialVelocityGrad(0, 2));
