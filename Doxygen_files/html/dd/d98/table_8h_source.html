603 TResultType& Interpolate(TArgumentType const& X, TArgumentType const& X1, TResultType const& Y1, TArgumentType const& X2, TResultType const& Y2, TResultType& Result) const
604 {
605 const double epsilon = 1e-12;
606
607 double dx = X2 - X1;
608 TResultType dy = Y2 - Y1;
609
610 double scale = 0.00;
611
612 if (dx > epsilon)
613 scale = (X - X1) / dx;
614
615 Result = Y1 + dy * scale;
616
