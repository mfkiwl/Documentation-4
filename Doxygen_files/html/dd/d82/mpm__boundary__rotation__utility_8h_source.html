137 void ApplySlipCondition(TLocalMatrixType& rLocalMatrix,
138 TLocalVectorType& rLocalVector,
139 GeometryType& rGeometry) const override
140 {
141 const unsigned int LocalSize = rLocalVector.size();
142
143 if (LocalSize > 0)
144 {
145 for(unsigned int itNode = 0; itNode < rGeometry.PointsNumber(); ++itNode)
146 {
147 if(this->IsSlip(rGeometry[itNode]) )
148 {
149 // We fix the first displacement dof (normal component) for each rotated block
150 unsigned int j = itNode * this->GetBlockSize();
151
152 // Get the displacement of the boundary mesh, this does not assume that the mesh is moving.
153 // If the mesh is moving, need to consider the displacement of the moving mesh into account.
154 const array_1d<double,3> & displacement = rGeometry[itNode].FastGetSolutionStepValue(DISPLACEMENT);
155
156 // Get Normal Vector of the boundary
157 array_1d<double,3> rN = rGeometry[itNode].FastGetSolutionStepValue(NORMAL);
158 this->Normalize(rN);
159
160 for( unsigned int i = 0; i < j; ++i)// Skip term (i,i)
161 {
162 rLocalMatrix(i,j) = 0.0;
163 rLocalMatrix(j,i) = 0.0;
164 }
165 for( unsigned int i = j+1; i < LocalSize; ++i)
166 {
167 rLocalMatrix(i,j) = 0.0;
168 rLocalMatrix(j,i) = 0.0;
169 }
170
171 rLocalVector[j] = inner_prod(rN,displacement);
172 rLocalMatrix(j,j) = 1.0;
173 }
174 }
175 }
176 }

229 // An extra function to distinguish the application of slip in condition considering penalty imposition
230 void ConditionApplySlipCondition(TLocalMatrixType& rLocalMatrix,
231 TLocalVectorType& rLocalVector,
232 GeometryType& rGeometry) const
233 {
234 // If it is not a penalty condition, do as standard
235 if (!this->IsPenalty(rGeometry))
236 {
237 this->ApplySlipCondition(rLocalMatrix, rLocalVector, rGeometry);
238 }
239 // Otherwise, do the following modification
240 else
241 {
242 const unsigned int LocalSize = rLocalVector.size();
243
244 if (LocalSize > 0)
245 {
246 const unsigned int block_size = this->GetBlockSize();
247 TLocalMatrixType temp_matrix = ZeroMatrix(rLocalMatrix.size1(),rLocalMatrix.size2());
248 for(unsigned int itNode = 0; itNode < rGeometry.PointsNumber(); ++itNode)
249 {
250 if(this->IsSlip(rGeometry[itNode]) )
251 {
252 // We fix the first displacement dof (normal component) for each rotated block
253 unsigned int j = itNode * block_size;
254
255 // Copy all normal value in LHS to the temp_matrix
256 for (unsigned int i = j; i < rLocalMatrix.size1(); i+= block_size)
257 {
258 temp_matrix(i,j) = rLocalMatrix(i,j);
259 temp_matrix(j,i) = rLocalMatrix(j,i);
260 }
261
262 // Remove all other value in RHS than the normal component
263 for(unsigned int i = j; i < (j + block_size); ++i)
264 {
265 if (i!=j) rLocalVector[i] = 0.0;
266 }
267 }
268 }
269 rLocalMatrix = temp_matrix;
270 }
271 }

308 // Checking whether it is normal element or penalty element
309 bool IsPenalty(GeometryType& rGeometry) const
310 {
311 bool is_penalty = false;
312 for(unsigned int itNode = 0; itNode < rGeometry.PointsNumber(); ++itNode)
313 {
314 if(this->IsSlip(rGeometry[itNode]) )
315 {
316 const double identifier = rGeometry[itNode].FastGetSolutionStepValue(mrFlagVariable);
317 const double tolerance = 1.e-6;
318 if (identifier > 1.00 + tolerance)
319 {
320 is_penalty = true;
321 break;
322 }
323 }
324 }
325
326 return is_penalty;

337 void RotateVelocities(ModelPart& rModelPart) const override
338 {
339 TLocalVectorType displacement(this->GetDomainSize());
340 TLocalVectorType Tmp(this->GetDomainSize());
341
342 ModelPart::NodeIterator it_begin = rModelPart.NodesBegin();
343 #pragma omp parallel for firstprivate(displacement,Tmp)
344 for(int iii=0; iii<static_cast<int>(rModelPart.Nodes().size()); iii++)
345 {
346 ModelPart::NodeIterator itNode = it_begin+iii;
347 if( this->IsSlip(*itNode) )
348 {
349 //this->RotationOperator<TLocalMatrixType>(Rotation,);
350 if(this->GetDomainSize() == 3)
351 {
352 BoundedMatrix<double,3,3> rRot;
353 this->LocalRotationOperatorPure(rRot,*itNode);
354
355 array_1d<double,3>& rDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT);
356 for(unsigned int i = 0; i < 3; i++) displacement[i] = rDisplacement[i];
357 noalias(Tmp) = prod(rRot,displacement);
358 for(unsigned int i = 0; i < 3; i++) rDisplacement[i] = Tmp[i];
359 }
360 else
361 {
362 BoundedMatrix<double,2,2> rRot;
363 this->LocalRotationOperatorPure(rRot,*itNode);
364
365 array_1d<double,3>& rDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT);
366 for(unsigned int i = 0; i < 2; i++) displacement[i] = rDisplacement[i];
367 noalias(Tmp) = prod(rRot,displacement);
368 for(unsigned int i = 0; i < 2; i++) rDisplacement[i] = Tmp[i];
369 }
