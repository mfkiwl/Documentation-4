112 if (IsFull())

115 lock();
116 BlockType * p_result = mFirstAvailableBlock;
117 mFirstAvailableBlock = (p_result >= mpUninitializedMemory) ? mFirstAvailableBlock + mBlockSizeAfterAlignment : (BlockType*)*p_result;
118 // if(p_result >= mpUninitializedMemory) {

128 KRATOS_DEBUG_CHECK(Has(p_result));

130 mNumberOfAvailableBlocks--;
131 unlock();
132
133 mpUninitializedMemory += mBlockSizeAfterAlignment * ((p_result >= mpUninitializedMemory) && (mpUninitializedMemory < mpEnd));

138 void Deallocate(void* pPointrerToRelease) {
139 if (pPointrerToRelease == nullptr)
140 return;
141
142 KRATOS_DEBUG_CHECK_NOT_EQUAL(mpData, nullptr);
143 // Range check at least in lower bound.
144 KRATOS_DEBUG_CHECK_GREATER_EQUAL(pPointrerToRelease, mpData);
145 BlockType* p_to_release = static_cast<BlockType*>(pPointrerToRelease);
146
147 // Alignment check
148 KRATOS_DEBUG_CHECK_EQUAL((p_to_release - mpData) % mBlockSizeAfterAlignment, 0);
149
150 lock();
151 *p_to_release = (BlockType)mFirstAvailableBlock;
152 mFirstAvailableBlock = p_to_release;
153
154 mNumberOfAvailableBlocks++;
155 unlock();
156 pPointrerToRelease = nullptr;
157
158 }
