139 static inline void GetBoxCenter(double box_center[3], const double min_point[3], const double max_point[3])
140 {
141 for (unsigned int i = 0; i < 3; ++i){
142 box_center[i] = 0.5 * (min_point[i] + max_point[i]);
143
144 if (min_point[i] > max_point[i]){ // the box is broken by the boundary in this dimension: ] x * [ (outside), ] x [ * (inside)
145 const double& min = mDomainMin[i];
146 const double& max = mDomainMax[i];
147 box_center[i] += 0.5 * (max - min); // The center of the box and of its complementary are always half a domain period apart: |..]..x..[...*.| (our bet: we assume we should ADD to go from x to *)
148 if (box_center[i] > max){ // we made a mistake, we should have gone the opposite way
149 box_center[i] -= max - min;
150 }
151 }
152 }

194 static inline bool IntersectionBox(const PointerType& rObject, const PointType& rLowPoint, const PointType& rHighPoint, const double Radius)
195 {
196 const array_1d<double, 3>& center_of_particle = *rObject;
197
198 bool intersect;
199
200 if (mDomainIsPeriodic){
201 double expanded_box_min[3];
202 double expanded_box_max[3];
203
204 for (unsigned int i = 0; i < 3; ++i){
205 expanded_box_min[i] = rLowPoint[i] - Radius;
206 expanded_box_max[i] = rHighPoint[i] + Radius;
207 }
208
209 double box_center[3];
210 GetBoxCenter(box_center, expanded_box_min, expanded_box_max);
211 //double box_center[3] = {0.5 * (rLowPoint[0] + rHighPoint[0]), 0.5 * (rLowPoint[1] + rHighPoint[1]), 0.5 * (rLowPoint[2] + rHighPoint[2])};
212 double representative_center_of_particle[3] = {center_of_particle[0],
213 center_of_particle[1],
214 center_of_particle[2]};
215 TransformToClosestPeriodicCoordinates(box_center, representative_center_of_particle);
216
217 for (unsigned int i = 0; i < 3; ++i){
218 const bool is_broken = rLowPoint[i] > rHighPoint[i];
219
220 if (is_broken){ // i.e., we have | ] [ | in this direction
221 intersect = floatge(expanded_box_min[i], representative_center_of_particle[i])
222 && floatle(expanded_box_max[i], representative_center_of_particle[i]);
223 }
224
225 else { // i.e., we have | [ ] | in this direction
226 intersect = floatle(expanded_box_min[i], representative_center_of_particle[i])
227 && floatge(expanded_box_max[i], representative_center_of_particle[i]);
228 }
229 }
230 }
231
232 else {
233 for (unsigned int i = 0; i < 3; ++i){
234 intersect = floatle(rLowPoint[i] - Radius, center_of_particle[i])
235 && floatge(rHighPoint[i] + Radius, center_of_particle[i]);
236 }
237 }
238 return intersect;
239 }

248 static inline void SquaredDistance(const PointerType& rObj_1, const PointerType& rObj_2, double& squared_distance)
249 {

251 PointType& point_1 = *rObj_1;

286 static inline void PeriodicSubstract(const double a[3], const double b[3], double c[3])
287 {
288 for (unsigned int i = 0; i < 3; ++i){
289 c[i] = a[i] - b[i];
290 }
291
292 if (mDomainIsPeriodic){ // Periods have been set (the domain is periodic)
293 for (unsigned int i = 0; i < 3; ++i){
294 if (std::fabs(c[i]) > 0.5 * mDomainPeriods[i]){ // the objects are closer through the boundary
295 c[i] -= GetSign(c[i]) * mDomainPeriods[i];
296 }
297 }
298 }
