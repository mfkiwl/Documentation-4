168  void calculateEquationParameters()
169  {
170  Vector3 v1;
171  Vector3 v2;
172  v1 = mSecondReferenceCoordinate - mFirstReferenceCoordinate;
173  v2 = mThirdReferenceCoordinate - mFirstReferenceCoordinate;
174 
175  mNormalVector[0] = v1[1] * v2[2] - v1[2] * v2[1];
176  mNormalVector[1] = v1[2] * v2[0] - v1[0] * v2[2];
177  mNormalVector[2] = v1[0] * v2[1] - v1[1] * v2[0];
178 
179  double sum=0;
180  for (unsigned int i=0; i < mNormalVector.size(); ++i)
181  {
182  sum += mNormalVector[i]*mNormalVector[i];
183  }
