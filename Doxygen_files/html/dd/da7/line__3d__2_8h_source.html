845  CoordinatesArrayType& PointLocalCoordinates(
846  CoordinatesArrayType& rResult,
847  const CoordinatesArrayType& rPoint
848  ) const override
849  {
850  rResult.clear();
851 
852  const TPointType& r_first_point = BaseType::GetPoint(0);
853  const TPointType& r_second_point = BaseType::GetPoint(1);
854 
855  // Project point
856  const double tolerance = 1e-14; // Tolerance
857 
858  const double length = Length();
859 
860  const double length_1 = std::sqrt( std::pow(rPoint[0] - r_first_point[0], 2)
861  + std::pow(rPoint[1] - r_first_point[1], 2) + std::pow(rPoint[2] - r_first_point[2], 2));
862 
863  const double length_2 = std::sqrt( std::pow(rPoint[0] - r_second_point[0], 2)
864  + std::pow(rPoint[1] - r_second_point[1], 2) + std::pow(rPoint[2] - r_second_point[2], 2));
865 
866  if (length_1 <= (length + tolerance) && length_2 <= (length + tolerance)) {
867  rResult[0] = 2.0 * length_1/(length + tolerance) - 1.0;
868  } else if (length_1 > (length + tolerance)) {
869  rResult[0] = 2.0 * length_1/(length + tolerance) - 1.0; // NOTE: The same value as before, but it will be > than 1
870  } else if (length_2 > (length + tolerance)) {
871  rResult[0] = 1.0 - 2.0 * length_2/(length + tolerance);
872  } else {
873  rResult[0] = 2.0; // Out of the line!!!
874  }
875 
876  return rResult ;
877  }
