403 void ApplyDirichletConditions(
404 typename TSchemeType::Pointer pScheme,
405 ModelPart& rModelPart,
406 TSystemMatrixType& rA,
407 TSystemVectorType& rDx,
408 TSystemVectorType& rb
409 ) override
410 {
411 const std::size_t system_size = rA.size1();
412 Vector scaling_factors (system_size);
413
414 const auto it_dof_iterator_begin = BaseType::mDofSet.begin();
415 const std::size_t ndofs = BaseType::mDofSet.size();
416
417 // NOTE: dofs are assumed to be numbered consecutively in the BlockBuilderAndSolver
418 IndexPartition<std::size_t>(ndofs).for_each([&](std::size_t Index){
419 auto it_dof_iterator = it_dof_iterator_begin + Index;
420 if (it_dof_iterator->IsFixed()) {
421 scaling_factors[Index] = 0.0;
422 } else {
423 scaling_factors[Index] = 1.0;
424 }
425 });
426
427 // Filling with ones the LM dofs
428 const std::size_t loop_size = system_size - ndofs;
429
430 IndexPartition<std::size_t>(loop_size).for_each([&](std::size_t Index){
431 scaling_factors[ndofs + Index] = 1.0;
432 });
433
434
435 double* Avalues = rA.value_data().begin();
436 std::size_t* Arow_indices = rA.index1_data().begin();
437 std::size_t* Acol_indices = rA.index2_data().begin();
438
439 // Define zero value tolerance
440 const double zero_tolerance = std::numeric_limits<double>::epsilon();
441
442 // The diagonal considered
443 BaseType::mScaleFactor = TSparseSpace::GetScaleNorm(rModelPart.GetProcessInfo(), rA, BaseType::mScalingDiagonal);
444
445 // Detect if there is a line of all zeros and set the diagonal to a 1 if this happens
446 IndexPartition<std::size_t>(system_size).for_each([&](std::size_t Index){
447 std::size_t col_begin = 0, col_end = 0;
448 bool empty = true;
449
450 col_begin = Arow_indices[Index];
451 col_end = Arow_indices[Index + 1];
452 empty = true;
453 for (std::size_t j = col_begin; j < col_end; ++j) {
454 if(std::abs(Avalues[j]) > zero_tolerance) {
455 empty = false;
456 break;
457 }
458 }
459
460 if(empty) {
461 rA(Index, Index) = this->mScaleFactor;
462 rb[Index] = 0.0;
463 }
464 });
465
466 IndexPartition<std::size_t>(system_size).for_each([&](std::size_t Index){
467 const std::size_t col_begin = Arow_indices[Index];
468 const std::size_t col_end = Arow_indices[Index+1];
469 const double k_factor = scaling_factors[Index];
470 if (k_factor == 0.0) {
471 // Zero out the whole row, except the diagonal
472 for (std::size_t j = col_begin; j < col_end; ++j)
473 if (Acol_indices[j] != Index )
474 Avalues[j] = 0.0;
475
476 // Zero out the RHS
477 rb[Index] = 0.0;
478 } else {
479 // Zero out the column which is associated with the zero'ed row
480 for (std::size_t j = col_begin; j < col_end; ++j)
481 if(scaling_factors[ Acol_indices[j] ] == 0 )
482 Avalues[j] = 0.0;
483 }
484 });
485 }

573
574 if (rModelPart.MasterSlaveConstraints().size() != 0) {
575 // Getting process info
576 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
577
578 // First build the relation matrix
579 BuildMasterSlaveConstraints(rModelPart);
580
581 // Copy the LHS to avoid memory errors
582 TSystemMatrixType copy_of_A;
583 copy_of_A.swap(rA);
584
585 TSystemMatrixType copy_of_T(BaseType::mT);
586 TSystemMatrixType transpose_of_T(TSparseSpace::Size2(BaseType::mT), TSparseSpace::Size1(BaseType::mT));

588
589 // Some common values
590 const SizeType number_of_slave_dofs = TSparseSpace::Size1(BaseType::mT);
591
592 // Definition of the total size of the system
593 const SizeType total_size_of_system = BaseType::mEquationSystemSize + (BaseType::mOptions.Is(DOUBLE_LAGRANGE_MULTIPLIER) ? 2 * number_of_slave_dofs : number_of_slave_dofs);

603
604 // Fill with zeros
605 IndexPartition<std::size_t>(loop_size).for_each([&](std::size_t Index){
606 b_modified[start_index + Index] = 0.0;

612

670 contribution_coefficients(1, 2) = mAuxiliarConstraintFactor;
671 contribution_coefficients(2, 1) = mAuxiliarConstraintFactor;

836 // Vector containing the localization in the system of the different terms
837 DofsVectorType slave_dof_list, master_dof_list;

839 // Constraint initial iterator

842 const std::size_t size_indices = BaseType::mDofSet.size();

844
845 std::vector<LockObject> lock_array(size_indices);
846
847 #pragma omp parallel firstprivate(slave_dof_list, master_dof_list)
848 {
849 Element::EquationIdVectorType slave_ids(3);
850 Element::EquationIdVectorType master_ids(3);
851 std::unordered_map<IndexType, std::unordered_set<IndexType>> temp_indices;
852
853 #pragma omp for schedule(guided, 512) nowait
854 for (int i_const = 0; i_const < static_cast<int>(rModelPart.MasterSlaveConstraints().size()); ++i_const) {
855 auto it_const = it_const_begin + i_const;
856
857 // Detect if the constraint is active or not. If the user did not make any choice the constraint
858 // It is active by default
859 bool constraint_is_active = true;

861 constraint_is_active = it_const->Is(ACTIVE);
862 }

865 it_const->EquationIdVector(slave_ids, master_ids, r_current_process_info);
866
867 // Slave DoFs
868 for (auto &id_i : slave_ids) {
869 temp_indices[id_i].insert(id_i);
870 temp_indices[id_i].insert(master_ids.begin(), master_ids.end());
871 }
872 }
873 }
874
875 // Merging all the temporal indexes
876 for (int i = 0; i < static_cast<int>(temp_indices.size()); ++i) {
877 lock_array[i].lock();
878 indices[i].insert(temp_indices[i].begin(), temp_indices[i].end());
879 lock_array[i].unlock();
880 }
881 }
882
883 IndexType counter = 0;
884 mCorrespondanceDofsSlave.clear();
885 BaseType::mSlaveIds.clear();
886 BaseType::mMasterIds.clear();
887 for (int i = 0; i < static_cast<int>(size_indices); ++i) {
888 if (indices[i].size() == 0) { // Master dof!
889 BaseType::mMasterIds.push_back(i);
890 } else { // Slave dof
891 BaseType::mSlaveIds.push_back(i);
892 mCorrespondanceDofsSlave.insert(std::pair<IndexType, IndexType>(i, counter));
893 ++counter;
894 }
895 }
896
897 // The slave size
898 const std::size_t slave_size = BaseType::mSlaveIds.size();
899
900 // Count the row sizes
901 std::size_t nnz = 0;
902 nnz = IndexPartition<std::size_t>(slave_size).for_each<SumReduction<std::size_t>>([&, this](std::size_t Index){
903 return indices[this->mSlaveIds[Index]].size();
904 });
905
906 BaseType::mT = TSystemMatrixType(slave_size, size_indices, nnz);
907 BaseType::mConstantVector.resize(slave_size, false);
908 mLagrangeMultiplierVector.resize(slave_size, false);
909 TSparseSpace::SetToZero(mLagrangeMultiplierVector);
910
911 double *Tvalues = BaseType::mT.value_data().begin();
912 IndexType *Trow_indices = BaseType::mT.index1_data().begin();
913 IndexType *Tcol_indices = BaseType::mT.index2_data().begin();
914
915 // Filling the index1 vector - DO NOT MAKE PARALLEL THE FOLLOWING LOOP!
916 Trow_indices[0] = 0;
917 for (int i = 0; i < static_cast<int>(slave_size); ++i) {
918 Trow_indices[i + 1] = Trow_indices[i] + indices[BaseType::mSlaveIds[i]].size();
919 }
920
921 IndexPartition<std::size_t>(slave_size).for_each([&, this](std::size_t Index){
922 const IndexType row_begin = Trow_indices[Index];
923 const IndexType row_end = Trow_indices[Index + 1];
924 IndexType k = row_begin;
925 const IndexType i_slave = this->mSlaveIds[Index];
926 for (auto it = indices[i_slave].begin(); it != indices[i_slave].end(); ++it) {
927 Tcol_indices[k] = *it;
928 Tvalues[k] = 0.0;
929 ++k;
930 }
931
932 indices[i_slave].clear(); //deallocating the memory
933
934 std::sort(&Tcol_indices[row_begin], &Tcol_indices[row_end]);
935 });
936
937 BaseType::mT.set_filled(slave_size + 1, nnz);
938
939 // Reset flag
940 BaseType::mOptions.Set(TRANSFORMATION_MATRIX_COMPUTED, false);
941
942 Timer::Stop("ConstraintsRelationMatrixStructure");
943 }

961

963 DofsVectorType slave_dof_list, master_dof_list;

967 Vector constant_vector = LocalSystemVectorType(0);

970 Element::EquationIdVectorType slave_equation_ids, master_equation_ids;

975 {
976 #pragma omp for schedule(guided, 512)
977 for (int i_const = 0; i_const < number_of_constraints; ++i_const) {
978 auto it_const = rModelPart.MasterSlaveConstraints().begin() + i_const;
979
980 // Detect if the constraint is active or not. If the user did not make any choice the constraint
981 // It is active by default
982 bool constraint_is_active = true;
983 if (it_const->IsDefined(ACTIVE))
984 constraint_is_active = it_const->Is(ACTIVE);
985
986 if (constraint_is_active) {
987 it_const->CalculateLocalSystem(transformation_matrix, constant_vector, r_current_process_info);
988 it_const->EquationIdVector(slave_equation_ids, master_equation_ids, r_current_process_info);
989
990 for (IndexType i = 0; i < slave_equation_ids.size(); ++i) {
991 const IndexType i_global = mCorrespondanceDofsSlave[slave_equation_ids[i]];
992
993 // Assemble matrix row
994 BaseType::AssembleRowContribution(BaseType::mT, - transformation_matrix, i_global, i, master_equation_ids);

999 AtomicAdd(r_value, constant_value);
1000 }
1001 }
1002 }
1003 }
1004
1005 // Setting the slave dofs into the T system
1006 for (auto eq_id : BaseType::mSlaveIds) {
1007 BaseType::mT(mCorrespondanceDofsSlave[eq_id], eq_id) = 1.0;
1008 }

1011 BaseType::mOptions.Set(TRANSFORMATION_MATRIX_COMPUTED, true);

1129 TSystemVectorType& rbLM,

1134
1135 const auto it_dof_begin = BaseType::mDofSet.begin();
1136 const int ndofs = static_cast<int>(BaseType::mDofSet.size());
1137
1138 // Our auxiliar vector
1139 const SizeType number_of_slave_dofs = TSparseSpace::Size1(BaseType::mT);
1140 const SizeType total_size_of_system = BaseType::mEquationSystemSize + (BaseType::mOptions.Is(DOUBLE_LAGRANGE_MULTIPLIER) ? 2 * number_of_slave_dofs : number_of_slave_dofs);
1141 if (TSparseSpace::Size(rbLM) != total_size_of_system)
1142 rbLM.resize(total_size_of_system, false);

1145
1146 // NOTE: dofs are assumed to be numbered consecutively in the BlockBuilderAndSolver
1147 IndexPartition<std::size_t>(ndofs).for_each([&](std::size_t Index){
1148 auto it_dof = it_dof_begin + Index;
1149 aux_whole_dof_vector[Index] = it_dof->GetSolutionStepValue();
1150 });
1151
1152 // Compute auxiliar contribution
1153 TSystemVectorType aux_slave_dof_vector(number_of_slave_dofs);
1154 TSparseSpace::Mult(BaseType::mT, aux_whole_dof_vector, aux_slave_dof_vector);
1155
1156 // Finally compute the RHS LM contribution
1157 noalias(aux_lm_rhs_contribution) = ScaleFactor * (BaseType::mConstantVector - aux_slave_dof_vector);
1158
1159 if (BaseType::mOptions.Is(DOUBLE_LAGRANGE_MULTIPLIER)) {
1160 IndexPartition<std::size_t>(number_of_slave_dofs).for_each([&](std::size_t Index){
1161 rbLM[ndofs + Index] = aux_lm_rhs_contribution[Index];
1162 rbLM[ndofs + number_of_slave_dofs + Index] = aux_lm_rhs_contribution[Index];
1163 });
1164 } else {
1165 IndexPartition<std::size_t>(number_of_slave_dofs).for_each([&](std::size_t Index){
1166 rbLM[ndofs + Index] = aux_lm_rhs_contribution[Index];
1167 });
1168 }
1169
1170 // We compute the transposed matrix of the global relation matrix
1171 TSystemMatrixType T_transpose_matrix(ndofs, number_of_slave_dofs);
1172 SparseMatrixMultiplicationUtility::TransposeMatrix<TSystemMatrixType, TSystemMatrixType>(T_transpose_matrix, BaseType::mT, -ScaleFactor);
1173
1174 TSparseSpace::Mult(T_transpose_matrix, mLagrangeMultiplierVector, aux_whole_dof_vector);
1175 IndexPartition<std::size_t>(ndofs).for_each([&](std::size_t Index){
1176 rbLM[Index] = aux_whole_dof_vector[Index];
1177 });
