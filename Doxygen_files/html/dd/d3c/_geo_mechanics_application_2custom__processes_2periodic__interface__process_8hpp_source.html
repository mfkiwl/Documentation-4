103  void ExecuteFinalizeSolutionStep() override
104  {
105  KRATOS_TRY
106 
107  if (mrModelPart.NumberOfConditions() > 0) {
108  SizeType StressTensorSize = STRESS_TENSOR_SIZE_2D;
109  if (mDimension == N_DIM_3D) StressTensorSize = STRESS_TENSOR_SIZE_3D;
110 
111  block_for_each(mrModelPart.Conditions(), [&StressTensorSize](Condition& rCondition) {
112  Condition::GeometryType& rGeom = rCondition.GetGeometry();
113 
114  Matrix NodalStressMatrix(StressTensorSize, StressTensorSize);
115  noalias(NodalStressMatrix) = 0.5 * ( rGeom[0].FastGetSolutionStepValue(NODAL_CAUCHY_STRESS_TENSOR)
116  + rGeom[1].FastGetSolutionStepValue(NODAL_CAUCHY_STRESS_TENSOR) );
117  Vector PrincipalStresses(StressTensorSize);
118  noalias(PrincipalStresses) = SolidMechanicsMathUtilities<double>::EigenValuesDirectMethod(NodalStressMatrix);
119 
120  // Check whether the principal stress S1 at the node is higher than the prescribed limit to activate the joints
121  if (PrincipalStresses[0] >= mStressLimit) {
122  rCondition.Set(PERIODIC,false);
123  rGeom[0].FastGetSolutionStepValue(PERIODIC_PAIR_INDEX) = 0;
124  rGeom[1].FastGetSolutionStepValue(PERIODIC_PAIR_INDEX) = 0;
125 
126  GlobalPointersVector<Element>& rE = rGeom[0].GetValue(NEIGHBOUR_ELEMENTS);
127  for (unsigned int ie = 0; ie < rE.size(); ie++) {
128  #pragma omp critical
129  {
130  rE[ie].Set(ACTIVE,true);
131  }
132  }
133  }
134  });
135  }
