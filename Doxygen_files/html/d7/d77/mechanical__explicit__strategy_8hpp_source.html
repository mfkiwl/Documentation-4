211 {
212 KRATOS_TRY
213
214 if (!this->mInitializeWasPerformed){
215 // Pointers needed in the solution
216 typename TSchemeType::Pointer pScheme = GetScheme();
217 ModelPart& r_model_part = BaseType::GetModelPart();
218
219 TSystemMatrixType matrix_a_dummy = TSystemMatrixType();
220
221 // Initialize The Scheme - OPERATIONS TO BE DONE ONCE
222 if (!pScheme->SchemeIsInitialized())pScheme->Initialize(r_model_part);
223
224 // Initialize The Elements - OPERATIONS TO BE DONE ONCE
225 if (!pScheme->ElementsAreInitialized())pScheme->InitializeElements(r_model_part);
226
227 // Initialize The Conditions- OPERATIONS TO BE DONE ONCE
228 if (!pScheme->ConditionsAreInitialized())pScheme->InitializeConditions(r_model_part);
229
230 // Set Nodal Mass and Damping to zero
231 NodesArrayType& r_nodes = r_model_part.Nodes();
232 VariableUtils().SetNonHistoricalVariable(NODAL_MASS, 0.0, r_nodes);
233 VariableUtils().SetNonHistoricalVariable(NODAL_DISPLACEMENT_DAMPING, 0.0, r_nodes);
234
235 // Iterate over the elements
236 ElementsArrayType& r_elements = r_model_part.Elements();
237 ProcessInfo& r_current_process_info = r_model_part.GetProcessInfo();
238
239 Vector dummy_vector;
240 // If we consider the rotation DoF
241 const bool has_dof_for_rot_z = !r_nodes.empty() && r_nodes.begin()->HasDofFor(ROTATION_Z);
242 if (has_dof_for_rot_z) {
243 const array_1d<double, 3> zero_array = ZeroVector(3);
244 VariableUtils().SetNonHistoricalVariable(NODAL_INERTIA, zero_array, r_nodes);
245 VariableUtils().SetNonHistoricalVariable(NODAL_ROTATION_DAMPING, zero_array, r_nodes);
246
247 block_for_each(r_elements, dummy_vector, [&r_current_process_info](Element& r_element, Vector& r_dummy_vector){
248 // Getting nodal mass and inertia from element
249 // this function needs to be implemented in the respective
250 // element to provide inertias and nodal masses
251 r_element.AddExplicitContribution(r_dummy_vector, RESIDUAL_VECTOR, NODAL_INERTIA, r_current_process_info);
252 });
253 } else { // Only NODAL_MASS is needed
254 block_for_each(r_elements, dummy_vector, [&r_current_process_info](Element& r_element, Vector& r_dummy_vector){
255 // Getting nodal mass and inertia from element
256 // this function needs to be implemented in the respective
257 // element to provide inertias and nodal masses
258 r_element.AddExplicitContribution(r_dummy_vector, RESIDUAL_VECTOR, NODAL_MASS, r_current_process_info);
259 });
260 }
261
262 // Precompute for masses and inertias
263 if(r_model_part.MasterSlaveConstraints().size() > 0) {
264 ConstraintUtilities::PreComputeExplicitConstraintMassAndInertia(r_model_part);
265 }

334 bool SolveSolutionStep() override
335 {
336 typename TSchemeType::Pointer pScheme = GetScheme();
337 ModelPart& r_model_part = BaseType::GetModelPart();
338
339 // Some dummy sets and matrices
340 DofsArrayType dof_set_dummy;
341 TSystemMatrixType rA = TSystemMatrixType();
342 TSystemVectorType rDx = TSystemVectorType();
343 TSystemVectorType rb = TSystemVectorType();
344
345 // Initialize the non linear iteration
346 pScheme->InitializeNonLinIteration(BaseType::GetModelPart(), rA, rDx, rb);
347
348 pScheme->Predict(r_model_part, dof_set_dummy, rA, rDx, rb);
349
350 // Pre-compute MPC contributions
351 if(r_model_part.MasterSlaveConstraints().size() > 0) {
352 std::vector<std::string> dof_variable_names(2);
353 dof_variable_names[0] = "DISPLACEMENT";
354 dof_variable_names[1] = "ROTATION";
355 std::vector<std::string> residual_variable_names(2);
356 residual_variable_names[0] = "FORCE_RESIDUAL";
357 residual_variable_names[1] = "MOMENT_RESIDUAL";
358 ConstraintUtilities::PreComputeExplicitConstraintConstribution(r_model_part, dof_variable_names, residual_variable_names);
359 }
360
361 // Explicitly integrates the equation of motion.
362 pScheme->Update(r_model_part, dof_set_dummy, rA, rDx, rb);
363
364 // Finalize the non linear iteration
365 pScheme->FinalizeNonLinIteration(BaseType::GetModelPart(), rA, rDx, rb);
366
367 // We compute the MPC contributions
368 if(r_model_part.MasterSlaveConstraints().size() > 0) {
369 ComputeExplicitConstraintConstribution(pScheme, r_model_part);
370 }
371
372 // Calculate reactions if required
373 if (mCalculateReactionsFlag) {
374 CalculateReactions(pScheme, r_model_part, rA, rDx, rb);
375 }
376
377 return true;
378 }

496 void CalculateReactions(
497 typename TSchemeType::Pointer pScheme,
498 ModelPart& rModelPart,
499 TSystemMatrixType& rA,
500 TSystemVectorType& rDx,
501 TSystemVectorType& rb
502 )
503 {
504 // We iterate over the nodes
505 auto& r_nodes = rModelPart.Nodes();
506
507 if (!r_nodes.empty()) {
508 // If we consider rotation dofs
509 const bool has_dof_for_rot_z = (r_nodes.begin())->HasDofFor(ROTATION_Z);
510
511 // Auxiliary values
512 const array_1d<double,3> zero_array = ZeroVector(3);
513
514 // Getting
515 const auto it_node_begin = r_nodes.begin();
516 const IndexType disppos = it_node_begin->GetDofPosition(DISPLACEMENT_X);
517 const IndexType rotppos = it_node_begin->GetDofPosition(ROTATION_X);
518
519 // Construct loop lambda depending on whether nodes have rot_z
520 std::function<void(Node<3>&)> loop_base, loop;
521
522 loop_base = [&disppos](Node<3>& rNode){
523 const auto force_residual = rNode.FastGetSolutionStepValue(FORCE_RESIDUAL);
524
525 if (rNode.GetDof(DISPLACEMENT_X, disppos).IsFixed()) {
526 double& r_reaction = rNode.FastGetSolutionStepValue(REACTION_X);
527 r_reaction = force_residual[0];
528 }
529 if (rNode.GetDof(DISPLACEMENT_Y, disppos + 1).IsFixed()) {
530 double& r_reaction = rNode.FastGetSolutionStepValue(REACTION_Y);
531 r_reaction = force_residual[1];
532 }
533 if (rNode.GetDof(DISPLACEMENT_Z, disppos + 2).IsFixed()) {
534 double& r_reaction = rNode.FastGetSolutionStepValue(REACTION_Z);
535 r_reaction = force_residual[2];
536 }
537 };
538
539 if (has_dof_for_rot_z) {
540 loop = [&rotppos, &loop_base](Node<3>& rNode){
541 loop_base(rNode);
542 const auto moment_residual = rNode.FastGetSolutionStepValue(MOMENT_RESIDUAL);
543 if (rNode.GetDof(ROTATION_X, rotppos).IsFixed()) {
544 double& r_reaction = rNode.FastGetSolutionStepValue(REACTION_MOMENT_X);
545 r_reaction = moment_residual[0];
546 }
547 if (rNode.GetDof(ROTATION_Y, rotppos + 1).IsFixed()) {
548 double& r_reaction = rNode.FastGetSolutionStepValue(REACTION_MOMENT_Y);
549 r_reaction = moment_residual[1];
550 }
551 if (rNode.GetDof(ROTATION_Z, rotppos + 2).IsFixed()) {
552 double& r_reaction = rNode.FastGetSolutionStepValue(REACTION_MOMENT_Z);
553 r_reaction = moment_residual[2];
554 }
555 };
556 } else {
557 loop = loop_base;
558 }
559
560 // Compute on nodes
561 block_for_each(r_nodes, loop);
562 } // if not nodes.empty()
563 }
