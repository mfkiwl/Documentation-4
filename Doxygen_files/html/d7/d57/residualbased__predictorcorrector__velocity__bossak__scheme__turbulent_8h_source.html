305 //updating time derivatives (nodally for efficiency)
306 #pragma omp parallel
307 {

319 array_1d<double, 3 > & OldAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 1);

322
323 if (mMeshVelocity == 2)//Lagrangian

325 if((itNode)->FastGetSolutionStepValue(IS_LAGRANGIAN_INLET) < 1e-15)

330
331 noalias(itNode->FastGetSolutionStepValue(MESH_VELOCITY)) = itNode->FastGetSolutionStepValue(VELOCITY);
332 UpdateDisplacement(CurrentDisplacement, OldDisplacement, OldVelocity, OldAcceleration, CurrentAcceleration);

336 noalias(itNode->FastGetSolutionStepValue(MESH_VELOCITY)) = ZeroVector(3);

350 // v = vold
351 void Predict(ModelPart& rModelPart,
352 DofsArrayType& rDofSet,
353 TSystemMatrixType& A,
354 TSystemVectorType& Dv,
355 TSystemVectorType& b) override
356 {
357 // if (rModelPart.GetCommunicator().MyPID() == 0)
358 // std::cout << "prediction" << std::endl;
359
360 int NumThreads = ParallelUtilities::GetNumThreads();
361 OpenMPUtils::PartitionVector NodePartition;
362 OpenMPUtils::DivideInPartitions(rModelPart.Nodes().size(), NumThreads, NodePartition);
363
364 #pragma omp parallel
365 {
366 //array_1d<double, 3 > DeltaDisp;
367
368 int k = OpenMPUtils::ThisThread();
369
370 ModelPart::NodeIterator NodesBegin = rModelPart.NodesBegin() + NodePartition[k];
371 ModelPart::NodeIterator NodesEnd = rModelPart.NodesBegin() + NodePartition[k + 1];
372
373 for (ModelPart::NodeIterator itNode = NodesBegin; itNode != NodesEnd; itNode++) {
374 array_1d<double, 3 > & OldVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY, 1);
375 double& OldPressure = (itNode)->FastGetSolutionStepValue(PRESSURE, 1);
376
377 //predicting velocity
378 //ATTENTION::: the prediction is performed only on free nodes
379 array_1d<double, 3 > & CurrentVelocity = (itNode)->FastGetSolutionStepValue(VELOCITY);
380 double& CurrentPressure = (itNode)->FastGetSolutionStepValue(PRESSURE);
381
382 if ((itNode->pGetDof(VELOCITY_X))->IsFree())
383 (CurrentVelocity[0]) = OldVelocity[0];
384 if (itNode->pGetDof(VELOCITY_Y)->IsFree())
385 (CurrentVelocity[1]) = OldVelocity[1];
386 if (itNode->HasDofFor(VELOCITY_Z))
387 if (itNode->pGetDof(VELOCITY_Z)->IsFree())
388 (CurrentVelocity[2]) = OldVelocity[2];
389
390 if (itNode->pGetDof(PRESSURE)->IsFree())
391 CurrentPressure = OldPressure;
392
393 // updating time derivatives ::: please note that displacements and
394 // their time derivatives can not be consistently fixed separately
395 array_1d<double, 3 > DeltaVel;
396 noalias(DeltaVel) = CurrentVelocity - OldVelocity;
397 array_1d<double, 3 > & OldAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 1);
398 array_1d<double, 3 > & CurrentAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION);
399
400 UpdateAcceleration(CurrentAcceleration, DeltaVel, OldAcceleration);
401
402 if (mMeshVelocity == 2) //Lagrangian
403 {
404 array_1d<double, 3 > & OldDisplacement = (itNode)->FastGetSolutionStepValue(DISPLACEMENT, 1);
405 array_1d<double, 3 > & CurrentDisplacement = (itNode)->FastGetSolutionStepValue(DISPLACEMENT, 0);
406
407 if((itNode)->FastGetSolutionStepValue(IS_LAGRANGIAN_INLET) < 1e-15)
408 {
409 noalias(itNode->FastGetSolutionStepValue(MESH_VELOCITY)) = itNode->FastGetSolutionStepValue(VELOCITY);
410 UpdateDisplacement(CurrentDisplacement, OldDisplacement, OldVelocity, OldAcceleration, CurrentAcceleration);
411 }
412 else
413 {
414 itNode->FastGetSolutionStepValue(MESH_VELOCITY_X) = 0.0;
415 itNode->FastGetSolutionStepValue(MESH_VELOCITY_Y) = 0.0;
416 itNode->FastGetSolutionStepValue(DISPLACEMENT_X) = 0.0;
417 itNode->FastGetSolutionStepValue(DISPLACEMENT_Y) = 0.0;
418 }
419 }
420 }
421 }
422
423// if (rModelPart.GetCommunicator().MyPID() == 0)
424// std::cout << "end of prediction" << std::endl;
425
426 }

558 void InitializeSolutionStep(ModelPart& r_model_part,
559 TSystemMatrixType& A,
560 TSystemVectorType& Dx,
561 TSystemVectorType& b) override
562 {
563 ProcessInfo& CurrentProcessInfo = r_model_part.GetProcessInfo();
564
565 Scheme<TSparseSpace, TDenseSpace>::InitializeSolutionStep(r_model_part, A, Dx, b);
566
567 double DeltaTime = CurrentProcessInfo[DELTA_TIME];
568 KRATOS_ERROR_IF(DeltaTime < 1.0e-12) << "Detected delta_time = 0 in the Bossak scheme. Check if the time step is created correctly for the current model part" << std::endl;
569
570 //initializing constants
571 ma0 = 1.0 / (mGammaNewmark * DeltaTime);
572 ma1 = DeltaTime * mBetaNewmark / mGammaNewmark;
573 ma2 = (-1 + mGammaNewmark) / mGammaNewmark;
574 ma3 = DeltaTime;
575 ma4 = pow(DeltaTime, 2)*(-2.0 * mBetaNewmark + 1.0) / 2.0;
576 ma5 = pow(DeltaTime, 2) * mBetaNewmark;
577 mam = (1.0 - mAlphaBossak) / (mGammaNewmark * DeltaTime);
578 }

582 void FinalizeNonLinIteration(ModelPart &rModelPart, TSystemMatrixType &A, TSystemVectorType &Dx, TSystemVectorType &b) override
583 {
584 const auto& r_current_process_info = rModelPart.GetProcessInfo();
585
586 if (mpTurbulenceModel) // If not null
587 mpTurbulenceModel->Execute();
588
589 //if orthogonal subscales are computed
590 if (r_current_process_info[OSS_SWITCH] == 1.0) {
591
592 KRATOS_INFO("Bossak Scheme") << "Computing OSS projections" << std::endl;
593
594 const int nnodes = static_cast<int>(rModelPart.Nodes().size());
595 auto nbegin = rModelPart.NodesBegin();
596 #pragma omp parallel for firstprivate(nbegin,nnodes)
597 for(int i=0; i<nnodes; ++i)
598 {
599 auto ind = nbegin + i;
600 noalias(ind->FastGetSolutionStepValue(ADVPROJ)) = ZeroVector(3);
601
602 ind->FastGetSolutionStepValue(DIVPROJ) = 0.0;
603
604 ind->FastGetSolutionStepValue(NODAL_AREA) = 0.0;
605
606
607 }//end of loop over nodes
608
609 //loop on nodes to compute ADVPROJ CONVPROJ NODALAREA
610 array_1d<double, 3 > output = ZeroVector(3);
611
612 const int nel = static_cast<int>(rModelPart.Elements().size());
613 auto elbegin = rModelPart.ElementsBegin();
614 #pragma omp parallel for firstprivate(elbegin,nel,output)
615 for(int i=0; i<nel; ++i)
616 {
617 auto elem = elbegin + i;
618 elem->Calculate(ADVPROJ, output, r_current_process_info);
619 }
620
621 rModelPart.GetCommunicator().AssembleCurrentData(NODAL_AREA);
622 rModelPart.GetCommunicator().AssembleCurrentData(DIVPROJ);
623 rModelPart.GetCommunicator().AssembleCurrentData(ADVPROJ);
624
625 // Correction for periodic conditions
626 this->PeriodicConditionProjectionCorrection(rModelPart);
627
628 #pragma omp parallel for firstprivate(nbegin,nnodes)
629 for(int i=0; i<nnodes; ++i)
630 {
631 auto ind = nbegin + i;
632 if (ind->FastGetSolutionStepValue(NODAL_AREA) == 0.0)
633 {
634 ind->FastGetSolutionStepValue(NODAL_AREA) = 1.0;
635 //KRATOS_WATCH("*********ATTENTION: NODAL AREA IS ZERRROOOO************");
636 }
637 const double Area = ind->FastGetSolutionStepValue(NODAL_AREA);
638 ind->FastGetSolutionStepValue(ADVPROJ) /= Area;
639 ind->FastGetSolutionStepValue(DIVPROJ) /= Area;
640 }
641 }

644 void FinalizeSolutionStep(ModelPart &rModelPart, TSystemMatrixType &A, TSystemVectorType &Dx, TSystemVectorType &b) override
645 {
646 Element::EquationIdVectorType EquationId;
647 LocalSystemVectorType RHS_Contribution;
648 LocalSystemMatrixType LHS_Contribution;
649 const ProcessInfo& CurrentProcessInfo = rModelPart.GetProcessInfo();
650
651 //for (ModelPart::NodeIterator itNode = rModelPart.NodesBegin(); itNode != rModelPart.NodesEnd(); ++itNode)
652
653 #pragma omp parallel for
654 for(int k = 0; k<static_cast<int>(rModelPart.Nodes().size()); k++)
655 {
656 auto itNode = rModelPart.NodesBegin() + k;
657 (itNode->FastGetSolutionStepValue(REACTION)).clear();
658
659 // calculating relaxed acceleration
660 const array_1d<double, 3 > & CurrentAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 0);
661 const array_1d<double, 3 > & OldAcceleration = (itNode)->FastGetSolutionStepValue(ACCELERATION, 1);
662 const array_1d<double, 3> relaxed_acceleration = (1 - mAlphaBossak) * CurrentAcceleration
663 + mAlphaBossak * OldAcceleration;
664 (itNode)->SetValue(RELAXED_ACCELERATION, relaxed_acceleration);
665 }
666
667 //for (ModelPart::ElementsContainerType::ptr_iterator itElem = rModelPart.Elements().ptr_begin(); itElem != rModelPart.Elements().ptr_end(); ++itElem)
668
669 #pragma omp parallel for firstprivate(EquationId,RHS_Contribution,LHS_Contribution)
670 for(int k = 0; k<static_cast<int>(rModelPart.Elements().size()); k++)
671 {
672 auto itElem = rModelPart.Elements().ptr_begin()+k;
673 int thread_id = OpenMPUtils::ThisThread();
674
675 //KRATOS_WATCH(LHS_Contribution);
676 //basic operations for the element considered
677 (*itElem)->CalculateLocalSystem(LHS_Contribution, RHS_Contribution, CurrentProcessInfo);
678
679 //std::cout << rCurrentElement->Id() << " RHS = " << RHS_Contribution << std::endl;
680 (*itElem)->CalculateMassMatrix(mMass[thread_id], CurrentProcessInfo);
681 (*itElem)->CalculateLocalVelocityContribution(mDamp[thread_id], RHS_Contribution, CurrentProcessInfo);
682
683 (*itElem)->EquationIdVector(EquationId, CurrentProcessInfo);
684
685 //adding the dynamic contributions (statics is already included)

797 void PeriodicConditionProjectionCorrection(ModelPart& rModelPart)
798 {
799 const int num_nodes = rModelPart.NumberOfNodes();
800 const int num_conditions = rModelPart.NumberOfConditions();
801
802 #pragma omp parallel for
803 for (int i = 0; i < num_nodes; i++) {
804 auto it_node = rModelPart.NodesBegin() + i;
805
806 it_node->SetValue(NODAL_AREA,0.0);
807 it_node->SetValue(ADVPROJ,ZeroVector(3));
808 it_node->SetValue(DIVPROJ,0.0);
809 }
810
811 #pragma omp parallel for
812 for (int i = 0; i < num_conditions; i++) {
813 auto it_cond = rModelPart.ConditionsBegin() + i;
814
815 if(it_cond->Is(PERIODIC)) {
816 this->AssemblePeriodicContributionToProjections(it_cond->GetGeometry());
817 }
818 }
819
820 rModelPart.GetCommunicator().AssembleNonHistoricalData(NODAL_AREA);
821 rModelPart.GetCommunicator().AssembleNonHistoricalData(ADVPROJ);
822 rModelPart.GetCommunicator().AssembleNonHistoricalData(DIVPROJ);
823
824 #pragma omp parallel for
825 for (int i = 0; i < num_nodes; i++) {
826 auto it_node = rModelPart.NodesBegin() + i;
827 this->CorrectContributionsOnPeriodicNode(*it_node);
828 }
829 }
830
831 void AssemblePeriodicContributionToProjections(Geometry< Node<3> >& rGeometry)
832 {
833 unsigned int nodes_in_cond = rGeometry.PointsNumber();
834
835 double nodal_area = 0.0;
836 array_1d<double,3> momentum_projection = ZeroVector(3);
837 double mass_projection = 0.0;
838 for ( unsigned int i = 0; i < nodes_in_cond; i++ )
839 {
840 auto& r_node = rGeometry[i];
841 nodal_area += r_node.FastGetSolutionStepValue(NODAL_AREA);
842 noalias(momentum_projection) += r_node.FastGetSolutionStepValue(ADVPROJ);
843 mass_projection += r_node.FastGetSolutionStepValue(DIVPROJ);
844 }
845
846 for ( unsigned int i = 0; i < nodes_in_cond; i++ )
847 {
848 auto& r_node = rGeometry[i];
849 /* Note that this loop is expected to be threadsafe in normal conditions,
850 * since each node should belong to a single periodic link. However, I am
851 * setting the locks for openmp in case that we try more complicated things
852 * in the future (like having different periodic conditions for different
853 * coordinate directions).
854 */
855 r_node.SetLock();
856 r_node.GetValue(NODAL_AREA) = nodal_area;
857 noalias(r_node.GetValue(ADVPROJ)) = momentum_projection;
858 r_node.GetValue(DIVPROJ) = mass_projection;
859 r_node.UnSetLock();
860 }
861 }
