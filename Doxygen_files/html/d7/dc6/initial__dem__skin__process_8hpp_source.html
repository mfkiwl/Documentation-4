39 /* It will create a submodel part containing nodes to include DEM on them,
40 with the DEM radius assigned to each node to be created the DEM afterwards
41 The modelpart must include the skinModelpart and damage extrapolated to nodes*/
42 void Execute()
43 {
44 const std::string &name_dem_model_part = "InitialDemSkin";
45
46 if (mrModelPart.HasSubModelPart(name_dem_model_part))
47 {
48 mrModelPart.RemoveSubModelPart(name_dem_model_part);
49 }
50
51 FindNodalNeighboursProcess neighbour_finder(mrModelPart);
52 neighbour_finder.Execute();
53
54 mrModelPart.CreateSubModelPart(name_dem_model_part);
55 ModelPart* p_auxiliar_model_part = mrModelPart.pGetSubModelPart(name_dem_model_part);
56
57 ModelPart* p_skin_model_part = mrModelPart.pGetSubModelPart("SkinDEMModelPart");
58
59 for (ModelPart::NodeIterator it = (*p_skin_model_part).NodesBegin(); it != (*p_skin_model_part).NodesEnd(); ++it) {
60 p_auxiliar_model_part->AddNode(*(it.base()));
61 } // InitialDemSkin SubModelPart Filled with nodes
62
63 // Let's assign the DEM radius to those nodes...
64 for (ModelPart::NodeIterator it = (*p_auxiliar_model_part).NodesBegin(); it != (*p_auxiliar_model_part).NodesEnd(); ++it) {
65 GlobalPointersVector<Node<3>> &rneigh = (*it).GetValue(NEIGHBOUR_NODES);
66 KRATOS_ERROR_IF(rneigh.size() == 0) << "Nodal neighbours not computed..." << std::endl;
67 std::vector<double> radius_is_dems, radius_not_dem;
68 double distance, radius_dem, min_radius, min_radius_is_dem, min_radius_no_dem;
69
70 for (int i = 0; i < rneigh.size(); i++)
71 {
72
73 distance = this->CalculateDistanceBetweenNodes((*it), rneigh[i]);
74
75 if (rneigh[i].GetValue(DEM_RADIUS) != 0.0)
76 {
77 radius_dem = distance - rneigh[i].GetValue(DEM_RADIUS);
78 radius_is_dems.push_back(radius_dem);
79 }
80 else
81 {
82 radius_dem = 0.5 * distance;
83 radius_not_dem.push_back(radius_dem);
84 }
85 }
86 if (radius_is_dems.size() == 0) {
87 min_radius = this->GetMinimumValue(radius_not_dem);
88 (*it).SetValue(DEM_RADIUS, min_radius);
89 } else {
90 min_radius_is_dem = this->GetMinimumValue(radius_is_dems);
91
92 if (radius_not_dem.size() != 0)
93 min_radius_no_dem = this->GetMinimumValue(radius_not_dem);
94 else
95 min_radius_no_dem = 1000.0;
96
97 if (min_radius_is_dem > 1.5 * min_radius_no_dem)
98 min_radius = min_radius_no_dem;
99 else
100 min_radius = min_radius_is_dem;
101 (*it).SetValue(DEM_RADIUS, min_radius);
102 }
103 }
104 }

106 double CalculateDistanceBetweenNodes(
107 const Node<3> &Node1,
108 const Node<3> &Node2)
109 {
110 const double X1 = Node1.X();
111 const double X2 = Node2.X();
112 const double Y1 = Node1.Y();
113 const double Y2 = Node2.Y();
114 const double Z1 = Node1.Z();
115 const double Z2 = Node2.Z();
116 return std::sqrt(std::pow((X1 - X2), 2) + std::pow((Y1 - Y2), 2) + std::pow((Z1 - Z2), 2));
117 }
118
119 double GetMinimumValue(std::vector<double> array_values)
120 {
121 int size = array_values.size();
122
123 double aux = array_values[0];
124 for (int i = 1; i < size; i++)
125 {
126 if (array_values[i] < aux)
127 aux = array_values[i];
128 }
129 return aux;
130 }
