827 CoordinatesArrayType& PointLocalCoordinates(
828 CoordinatesArrayType& rResult,
829 const CoordinatesArrayType& rPoint
830 ) const override {
831
832 rResult = ZeroVector(3);
833
834 const TPointType& point_0 = this->GetPoint(0);
835
836 // Compute the Jacobian matrix and its determinant
837 BoundedMatrix<double, 2, 2> J;
838 J(0,0) = this->GetPoint(1).X() - this->GetPoint(0).X();
839 J(0,1) = this->GetPoint(2).X() - this->GetPoint(0).X();
840 J(1,0) = this->GetPoint(1).Y() - this->GetPoint(0).Y();
841 J(1,1) = this->GetPoint(2).Y() - this->GetPoint(0).Y();
842 const double det_J = J(0,0)*J(1,1) - J(0,1)*J(1,0);
843
844 // Compute eta and xi
845 const double eta = (J(1,0)*(point_0.X() - rPoint(0)) +
846 J(0,0)*(rPoint(1) - point_0.Y())) / det_J;
847 const double xi = (J(1,1)*(rPoint(0) - point_0.X()) +
848 J(0,1)*(point_0.Y() - rPoint(1))) / det_J;
849
850 rResult(0) = xi;
851 rResult(1) = eta;
852 rResult(2) = 0.0;
853
854 return rResult;
855 }

1595 /* Triangle/triangle intersection test routine,
1596 * by Tomas Moller, 1997.
1597 * See article "A Fast Triangle-Triangle Intersection Test",
1598 * Journal of Graphics Tools, 2(2), 1997
1599 *
1600 * Updated June 1999: removed the divisions -- a little faster now!
1601 * Updated October 1999: added {} to CROSS and SUB macros
1602 *
1603 * int NoDivTriTriIsect(float V0[3],float V1[3],float V2[3],
1604 * float U0[3],float U1[3],float U2[3])
1605 *
1606 * parameters: vertices of triangle 1: V0,V1,V2
1607 * vertices of triangle 2: U0,U1,U2
1608 * result : returns 1 if the triangles intersect, otherwise 0
1609 *
1610 */
1611
1612 bool NoDivTriTriIsect( const Point& V0,
1613 const Point& V1,
1614 const Point& V2,
1615 const Point& U0,
1616 const Point& U1,
1617 const Point& U2) const
1618 {
1619 short index;
1620 double d1,d2;
1621 double du0,du1,du2,dv0,dv1,dv2;
1622 double du0du1,du0du2,dv0dv1,dv0dv2;
1623 double vp0,vp1,vp2;
1624 double up0,up1,up2;
1625 double bb,cc,max;
1626 array_1d<double,2> isect1, isect2;
1627 array_1d<double,3> D;
1628 array_1d<double,3> E1,E2;
1629 array_1d<double,3> N1,N2;
1630
1631
1632 const double epsilon = 1E-6;
1633
1634// compute plane equation of triangle(V0,V1,V2) //
1635 noalias(E1) = V1-V0;
1636 noalias(E2) = V2-V0;
1637 MathUtils<double>::CrossProduct(N1, E1, E2);
1638 d1=-inner_prod(N1,V0);
1639// plane equation 1: N1.X+d1=0 //
1640
1641// put U0,U1,U2 into plane equation 1 to compute signed distances to the plane//
1642 du0=inner_prod(N1,U0)+d1;
1643 du1=inner_prod(N1,U1)+d1;
1644 du2=inner_prod(N1,U2)+d1;
1645
1646// coplanarity robustness check //
1647 if(fabs(du0)<epsilon) du0=0.0;
1648 if(fabs(du1)<epsilon) du1=0.0;
1649 if(fabs(du2)<epsilon) du2=0.0;
1650
1651 du0du1=du0*du1;
1652 du0du2=du0*du2;
1653
1654 if(du0du1>0.00 && du0du2>0.00)// same sign on all of them + not equal 0 ? //
1655 return false; // no intersection occurs //
1656
1657// compute plane of triangle (U0,U1,U2) //
1658 noalias(E1) = U1 - U0;
1659 noalias(E2) = U2 - U0;
1660 MathUtils<double>::CrossProduct(N2, E1, E2);
1661 d2=-inner_prod(N2,U0);
1662// plane equation 2: N2.X+d2=0 //
1663
1664// put V0,V1,V2 into plane equation 2 //
1665 dv0=inner_prod(N2,V0)+d2;
1666 dv1=inner_prod(N2,V1)+d2;
1667 dv2=inner_prod(N2,V2)+d2;
1668
1669 if(fabs(dv0)<epsilon) dv0=0.0;
1670 if(fabs(dv1)<epsilon) dv1=0.0;
1671 if(fabs(dv2)<epsilon) dv2=0.0;
1672
1673 dv0dv1=dv0*dv1;
1674 dv0dv2=dv0*dv2;
1675
1676 if(dv0dv1>0.00 && dv0dv2>0.00)// same sign on all of them + not equal 0 ? //
1677 return false; // no intersection occurs //
1678
1679// compute direction of intersection line //
1680 MathUtils<double>::CrossProduct(D, N1, N2);
1681
1682// compute and index to the largest component of D //
1683 max=(double)fabs(D[0]);
1684 index=0;
1685 bb=(double)fabs(D[1]);
1686 cc=(double)fabs(D[2]);
1687 if(bb>max)
1688 {
1689 max=bb,index=1;
1690 }
1691 if(cc>max)
1692 {
1693 max=cc,index=2;
1694 }
1695
1696// this is the simplified projection onto L//
1697 vp0=V0[index];
1698 vp1=V1[index];
1699 vp2=V2[index];
1700
1701 up0=U0[index];
1702 up1=U1[index];
1703 up2=U2[index];
1704
1705
1706// compute interval for triangle 1 //
1707 double a,b,c,x0,x1;
1708 if( New_Compute_Intervals(vp0,vp1,vp2,dv0, dv1,dv2,dv0dv1,dv0dv2, a,b,c,x0,x1)==true )
1709 {
1710 return coplanar_tri_tri(N1,V0,V1,V2,U0,U1,U2);
1711 }
1712
1713// compute interval for triangle 2 //
1714 double d,e,f,y0,y1;
1715 if( New_Compute_Intervals(up0, up1, up2, du0, du1, du2, du0du1, du0du2, d, e, f, y0, y1)==true )
1716 {
1717 return coplanar_tri_tri(N1,V0,V1,V2,U0,U1,U2);
1718 }
1719
1720
1721 double xx,yy,xxyy,tmp;
1722 xx=x0*x1;
1723 yy=y0*y1;
1724 xxyy=xx*yy;
1725
1726 tmp=a*xxyy;
1727 isect1[0]=tmp+b*x1*yy;
1728 isect1[1]=tmp+c*x0*yy;
1729
1730 tmp=d*xxyy;
1731 isect2[0]=tmp+e*xx*y1;
1732 isect2[1]=tmp+f*xx*y0;
1733
1734
1735 Sort(isect1[0],isect1[1]);
1736 Sort(isect2[0],isect2[1]);
1737
1738 if(isect1[1]<isect2[0] || isect2[1]<isect1[0]) return false;
1739 return true;
1740 }

1835 const Point& V0,
1836 const Point& V1,
1837 const Point& V2,
1838 const Point& U0,
1839 const Point& U1,
1840 const Point& U2) const
1841 {
1842 array_1d<double, 3 > A;
1843 short i0,i1;
1844
1845 // first project onto an axis-aligned plane, that maximizes the area //
1846 // of the triangles, compute indices: i0,i1. //
1847 A[0]=fabs(N[0]);
1848 A[1]=fabs(N[1]);
1849 A[2]=fabs(N[2]);
1850 if(A[0]>A[1])
1851 {
1852 if(A[0]>A[2])
1853 {
1854 i0=1; // A[0] is greatest //
1855 i1=2;
1856 }
1857 else
1858 {
1859 i0=0; // A[2] is greatest //
1860 i1=1;
1861 }
1862 }
1863 else // A[0]<=A[1] //
1864 {
1865 if(A[2]>A[1])
1866 {
1867 i0=0; // A[2] is greatest //
1868 i1=1;
1869 }
1870 else
1871 {
1872 i0=0; // A[1] is greatest //
1873 i1=2;
1874 }
1875 }
1876
1877 // test all edges of triangle 1 against the edges of triangle 2 //
1878 //std::cout<< "Proof One " << std::endl;
1879 if ( Edge_Against_Tri_Edges(i0, i1, V0,V1,U0,U1,U2)==true) return true;
1880

1882 if ( Edge_Against_Tri_Edges(i0, i1, V1,V2,U0,U1,U2)==true) return true;
1883

1885 if ( Edge_Against_Tri_Edges(i0, i1, V2,V0,U0,U1,U2)==true) return true;

1888 if (Point_In_Tri( i0, i1, V0, U0, U1, U2)==true) return true;
1889 if (Point_In_Tri( i0, i1, U0, V0, V1, V2)==true) return true;
1890
1891 return false;

2016 inline bool TriBoxOverlap(Point& rBoxCenter, Point& rBoxHalfSize) const
2017 {
2018 double abs_ex, abs_ey;
2019 array_1d<double,3 > vert0, vert1, vert2;
2020 array_1d<double,3 > edge0, edge1, edge2;
2021 std::pair<double, double> min_max;
2022
2023 // move everything so that the boxcenter is in (0,0,0)
2024 noalias(vert0) = this->GetPoint(0) - rBoxCenter;
2025 noalias(vert1) = this->GetPoint(1) - rBoxCenter;
2026 noalias(vert2) = this->GetPoint(2) - rBoxCenter;
2027
2028 // compute triangle edges
2029 noalias(edge0) = vert1 - vert0;
2030 noalias(edge1) = vert2 - vert1;
2031 noalias(edge2) = vert0 - vert2;
2032
2033 // Bullet 3:
2034 // test the 9 tests first (this was faster)
2035 // We are only interested on z-axis test, which defines the {x,y} plane
2036 // Note that projections onto crossproduct tri-{x,y} are always 0:
2037 // that means there is no separating axis on X,Y-axis tests
2038 abs_ex = std::abs(edge0[0]);
2039 abs_ey = std::abs(edge0[1]);
2040 if (AxisTestZ(edge0[0],edge0[1],abs_ex,abs_ey,vert0,vert2,rBoxHalfSize)) return false;
2041
2042 abs_ex = std::abs(edge1[0]);
2043 abs_ey = std::abs(edge1[1]);
2044 if (AxisTestZ(edge1[0],edge1[1],abs_ex,abs_ey,vert1,vert0,rBoxHalfSize)) return false;
2045
2046 abs_ex = std::abs(edge2[0]);
2047 abs_ey = std::abs(edge2[1]);
2048 if (AxisTestZ(edge2[0],edge2[1],abs_ex,abs_ey,vert2,vert1,rBoxHalfSize)) return false;
2049
2050 // Bullet 1:
2051 // first test overlap in the {x,y,(z)}-directions
2052 // find min, max of the triangle each direction, and test for overlap in
2053 // that direction -- this is equivalent to testing a minimal AABB around
2054 // the triangle against the AABB
2055
2056 // test in X-direction
2057 min_max = std::minmax({vert0[0],vert1[0],vert2[0]});
2058 if(min_max.first>rBoxHalfSize[0] || min_max.second<-rBoxHalfSize[0]) return false;
2059
2060 // test in Y-direction
2061 min_max = std::minmax({vert0[1],vert1[1],vert2[1]});
2062 if(min_max.first>rBoxHalfSize[1] || min_max.second<-rBoxHalfSize[1]) return false;
2063
2064 // test in Z-direction
2065 // we do not consider rBoxHalfSize[2] since we are working in 2D
2066
2067 // Bullet 2:
2068 // test if the box intersects the plane of the triangle
2069 // compute plane equation of triangle: normal*x+d=0
2070 // Triangle and box are on the {x-y} plane, so this test won't return false
2071
2072 return true; // box and triangle overlaps
2073 }
