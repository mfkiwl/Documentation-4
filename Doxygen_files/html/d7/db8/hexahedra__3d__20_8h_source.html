699 const CoordinatesArrayType& rPoint ) const override
700 {
701 switch ( ShapeFunctionIndex )

707 case 1 :

709 *( 1.0 + rPoint[1] )*( 2.0

711 case 2 :
712 return -(( 1.0 + rPoint[0] )
713 *( 1.0 + rPoint[1] )*( 1.0 - rPoint[2] )*( 2.0

721 *( 1.0 - rPoint[1] )*( 2.0
