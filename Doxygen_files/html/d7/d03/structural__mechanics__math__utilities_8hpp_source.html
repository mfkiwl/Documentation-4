288 static inline bool SolveSecondOrderEquation(
289 const RealType& a,
290 const RealType& b,
291 const RealType& c,
292 std::vector<RealType>& solution
293 )
294 {
295 const RealType disc = b*b - 4.00*a*c;
296 RealType q = 0.0;
297
298 solution.resize(2, false);
299
300 if (b > 0.00)
301 {
302 q = -0.5 * (b + std::sqrt(disc));
303 }
304 else
305 {
306 q = -0.5 * (b - std::sqrt(disc));
307 }
308
309 solution[0] = q / a;
310 solution[1] = c / q;
311
312 return true;
313 }

322 static inline double CalculateRadius(
323 const Vector& N,
324 const GeometryType& Geom,
325 const Configuration ThisConfiguration = Current
326 )
327 {
328 double Radius = 0.0;
329
330 for (unsigned int iNode = 0; iNode < Geom.size(); iNode++)
331 {
332 // Displacement from the reference to the current configuration
333 if (ThisConfiguration == Current)
334 {
335 const array_1d<double, 3 > CurrentPosition = Geom[iNode].Coordinates();
336 Radius += CurrentPosition[0] * N[iNode];
337 }
338 else
339 {
340 const array_1d<double, 3 > DeltaDisplacement = Geom[iNode].FastGetSolutionStepValue(DISPLACEMENT) - Geom[iNode].FastGetSolutionStepValue(DISPLACEMENT,1);
341 const array_1d<double, 3 > CurrentPosition = Geom[iNode].Coordinates();
342 const array_1d<double, 3 > ReferencePosition = CurrentPosition - DeltaDisplacement;
343 Radius += ReferencePosition[0] * N[iNode];
344 }
345 }
346
347 return Radius;
348 }

385 template<int TDim>
386 static inline void TensorTransformation(
387 BoundedMatrix<double,TDim,TDim>& rOriginLeft,
388 BoundedMatrix<double,TDim,TDim>& rOriginRight,
389 BoundedMatrix<double,TDim,TDim>& rTargetLeft,
390 BoundedMatrix<double,TDim,TDim>& rTargetRight,
391 BoundedMatrix<double,TDim,TDim>& rTensor)
392 {
393 // metric computation (of the target systems)
394 BoundedMatrix<double,TDim,TDim> metric_left = ZeroMatrix(TDim,TDim);
395 BoundedMatrix<double,TDim,TDim> metric_right = ZeroMatrix(TDim,TDim);
396 for(int i=0;i<TDim;i++){
397 for(int j=0;j<TDim;j++){
398 metric_left(i,j) += inner_prod(column(rTargetLeft,i),column(rTargetLeft,j));
399 metric_right(i,j) += inner_prod(column(rTargetRight,i),column(rTargetRight,j));
400 }
401 }
402
403 // invert metric
404 double det;
405 Matrix inv_metric_left = Matrix(TDim,TDim);
406 Matrix inv_metric_right = Matrix(TDim,TDim);
407 MathUtils<double>::InvertMatrix(Matrix(metric_left),inv_metric_left,det);
408 MathUtils<double>::InvertMatrix(metric_right,inv_metric_right,det);
409
410 // Compute dual target base vectors
411 BoundedMatrix<double,TDim,TDim> target_left_dual = ZeroMatrix(TDim,TDim); // Anna noalias?
412 BoundedMatrix<double,TDim,TDim> target_right_dual = ZeroMatrix(TDim,TDim); // Anna noalias?
413 for(int i=0;i<TDim;i++){
414 for(int j=0;j<TDim;j++){
415 column(target_left_dual,i) += inv_metric_left(i,j)*column(rTargetLeft,j);
416 column(target_right_dual,i) += inv_metric_right(i,j)*column(rTargetRight,j);
417 }
418 }
419
420 // Tensor transformation
421 BoundedMatrix<double, TDim, TDim> transformed_tensor = ZeroMatrix(TDim, TDim); // Anna noalias?
422 for(int k=0;k<TDim;k++){
423 for(int l=0;l<TDim;l++){
424 for(int i=0;i<TDim;i++){
425 for(int j=0;j<TDim;j++){
426 transformed_tensor(k,l) += rTensor(i,j)*inner_prod(column(target_left_dual,k),column(rOriginLeft,i))*inner_prod(column(target_right_dual,l),column(rOriginRight,j));
427 }
428 }
429 }
430 }
431 rTensor = transformed_tensor;
432 }
