85 SimpleMortarMapperProcessWrapper(
86 ModelPart& rOriginModelPart,
87 ModelPart& rDestinationModelPart,
88 Parameters ThisParameters = Parameters(R"({})" ),
89 LinearSolverType::Pointer pThisLinearSolver = nullptr
90 )
91 {
92 // The default parameters
93 const Parameters default_parameters = GetDefaultParameters();
94 const bool origin_are_conditions_is_defined = ThisParameters.Has("origin_are_conditions");
95 const bool destination_are_conditions_is_defined = ThisParameters.Has("destination_are_conditions");
96 ThisParameters.ValidateAndAssignDefaults(default_parameters);
97
98 // Automatic detect the entities
99 if (!origin_are_conditions_is_defined) {
100 if (rOriginModelPart.NumberOfElements() > 0) {
101 ThisParameters["origin_are_conditions"].SetBool(false);
102 KRATOS_WARNING("SimpleMortarMapperProcessWrapper") << "\'origin_are_conditions\' changed to \'False\'. Mapping from elements." << std::endl;
103 } else if (rOriginModelPart.NumberOfConditions() == 0) {
104 KRATOS_ERROR << "No conditions defined on origin model part" << std::endl;
105 }
106 }
107 if (!destination_are_conditions_is_defined) {
108 if (rDestinationModelPart.NumberOfElements() > 0) {
109 ThisParameters["destination_are_conditions"].SetBool(false);
110 KRATOS_WARNING("SimpleMortarMapperProcessWrapper") << "\'destination_are_conditions\' changed to \'False\'. Mapping from elements." << std::endl;
111 } else if (rDestinationModelPart.NumberOfConditions() == 0) {
112 KRATOS_ERROR << "No conditions defined on destination model part" << std::endl;
113 }
114 }
115
116 // The condition iterators
117 auto& r_geometry_origin = ThisParameters["origin_are_conditions"].GetBool() ? rOriginModelPart.Conditions().begin()->GetGeometry() : rOriginModelPart.Elements().begin()->GetGeometry();
118 auto& r_geometry_destination = ThisParameters["destination_are_conditions"].GetBool() ? rDestinationModelPart.Conditions().begin()->GetGeometry() : rDestinationModelPart.Elements().begin()->GetGeometry();
119
120 // The dimensions
121 const SizeType dimension = r_geometry_origin.WorkingSpaceDimension();
122 const SizeType size_1 = r_geometry_origin.size();
123 const SizeType size_2 = r_geometry_destination.size();
124
125 // The variable names
126 const std::string& r_origin_variable_name = ThisParameters["origin_variable"].GetString();
127 const std::string& r_destination_variable_name = ThisParameters["destination_variable"].GetString();
128
129 bool double_variable = true;
130 if(KratosComponents<Variable<double>>::Has(r_origin_variable_name)) {
131 if (r_destination_variable_name != "" && !(KratosComponents<Variable<double>>::Has(r_destination_variable_name))) {
132 KRATOS_ERROR << "The destination variable is not the same type (double) as the origin" << std::endl;
133 }
134 } else if (KratosComponents< Variable< array_1d< double, 3> > >::Has(r_origin_variable_name)) {
135 double_variable = false;
136 if (r_destination_variable_name != "" && !(KratosComponents<Variable<array_1d< double, 3>>>::Has(r_destination_variable_name)))
137 KRATOS_ERROR << "The destination variable is not the same type (array_1d< double, 3>) as the origin" << std::endl;
138 } else {
139 KRATOS_ERROR << "The types of the variables are not supported array_1d< double, 3> or double" << std::endl;
140 }
141
142 // Creating the mapper
143 if (dimension == 2) {
144 // 2D
145 if (double_variable) {
146 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<2, 2, Variable<double>>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
147 } else {
148 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<2, 2, Variable<array_1d< double, 3>>>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
149 }
150 } else {
151 // 3D
152 if (size_1 == 3 && size_2 == 3) {
153 if (double_variable) {
154 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 3, Variable<double>>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
155 } else {
156 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 3, Variable<array_1d< double, 3>>>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
157 }
158 } else if (size_1 == 4 && size_2 == 4) {
159 if (double_variable) {
160 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 4, Variable<double>>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
161 } else {
162 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 4, Variable<array_1d< double, 3>>>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
163 }
164 } else if (size_1 == 4 && size_2 == 3) {
165 if (double_variable) {
166 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 3, Variable<double>, 4>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
167 } else {
168 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 3, Variable<array_1d< double, 3>>, 4>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
169 }
170 } else if (size_1 == 3 && size_2 == 4) {
171 if (double_variable) {
172 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 4, Variable<double>, 3>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
173 } else {
174 mpMapperProcess = Kratos::make_shared<SimpleMortarMapperProcess<3, 4, Variable<array_1d< double, 3>>, 3>>(rOriginModelPart, rDestinationModelPart, ThisParameters, pThisLinearSolver);
175 }
176 } else {
177 KRATOS_ERROR << "Combination of dimensions and sizes not compatible. Dimension: " << dimension << " Size Origin: " << size_1 << " Size Destination: " << size_2 << std::endl;
178 }
179 }
