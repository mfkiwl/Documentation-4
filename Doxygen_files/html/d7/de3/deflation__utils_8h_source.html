109 void VisualizeAggregates(ModelPart::NodesContainerType& rNodes, Variable<double>& rVariable, const int max_reduced_size)
110 {
111 SparseMatrixType A(rNodes.size(),rNodes.size());
112 SparseMatrixType Adeflated;
113
114 //first of all build the connectivty matrix
115 std::vector< std::vector<int> > index_list(rNodes.size());
116
117 std::size_t total_size = 0;
118
119 //renumber nodes consecutively
120 int new_id = 1;
121 for(ModelPart::NodesContainerType::iterator in = rNodes.begin(); in!=rNodes.end(); in++)
122 in->SetId(new_id++);
123
124
125 //constructing the system matrix row by row
126 std::size_t index_i;
127 for(ModelPart::NodesContainerType::iterator in = rNodes.begin();
128 in!=rNodes.end(); in++)
129 {
130 index_i = (in)->Id()-1;
131 auto& neighb_nodes = in->GetValue(NEIGHBOUR_NODES);
132
133 std::vector<int>& indices = index_list[index_i];
134 indices.reserve(neighb_nodes.size()+1);
135
136 //filling the first neighbours list
137 indices.push_back(index_i);
138 for( auto i = neighb_nodes.begin();
139 i != neighb_nodes.end(); i++)
140 {
141
142 int index_j = i->Id()-1;
143 indices.push_back(index_j);
144 }
145
146 //sorting the indices and elminating the duplicates
147 std::sort(indices.begin(),indices.end());
148 std::vector<int>::iterator new_end = std::unique(indices.begin(),indices.end());
149
150 indices.erase(new_end,indices.end());
151
152 total_size += indices.size();
153
154 }
155
156 A.reserve(total_size,false);
157
158 //setting to zero the matrix (and the diagonal matrix)
159 for(unsigned int i=0; i<A.size1(); i++)
160 {
161 std::vector<int>& indices = index_list[i];
162 for(unsigned int j=0; j<indices.size(); j++)
163 {
164 A.push_back(i,indices[j] , 0.00);
165 }
166 }
167// std::cout << "matrix constructed" << std::endl;
168
169 //now call aggregation function to color the nodes
170 std::vector<int> w(rNodes.size());
171 ConstructW(max_reduced_size, A, w, Adeflated);
172// std::cout << "aggregates constructed" << std::endl;
173
174 //finally write the color to the nodes so that it can be visualized
175 int counter = 0;
176 for(ModelPart::NodesContainerType::iterator in=rNodes.begin(); in!=rNodes.end(); in++)
177 {
178 in->FastGetSolutionStepValue(rVariable) = w[counter++];
179 }
180// std::cout << "finished" << std::endl;

277 static void ConstructW(const int max_reduced_size, SparseMatrixType& rA, std::vector<int>& w, SparseMatrixType& deflatedA, const std::size_t block_size)
278 {
279 if(block_size == 1)
280 {
281 ConstructW(max_reduced_size,rA, w, deflatedA);
282 }
283 else
284 {
285 //simple checks to verify blocks are effectively respected
286 if(rA.size1()%block_size != 0 || rA.size2()%block_size != 0)
287 KRATOS_THROW_ERROR(std::logic_error,"the number of rows is not a multiple of block_size. Can not use the block deflation","")
288 if(rA.nnz()%block_size != 0)
289 KRATOS_THROW_ERROR(std::logic_error,"the number of non zeros is not a multiple of block_size. Can not use the block deflation","")
290
291 //construct Ascalar
292 SparseMatrixType Ascalar;
293 ConstructScalarMatrix(rA.size1(),block_size,rA.index1_data().begin(), rA.index2_data().begin(), Ascalar);
294
295 //deflate it using the standard methods
296 SparseMatrixType deflatedAscalar;
297 std::vector<int> wscalar;
298 ConstructW(max_reduced_size/block_size,Ascalar, wscalar, deflatedAscalar);
299
300 //compute w for the block structured problem
301
302 std::vector<int> w(wscalar.size()*block_size);
303 for(std::size_t i=0; i<wscalar.size(); i++)
304 {
305 for(std::size_t j=0; j<block_size; j++)
306 {
307 w[i*block_size + j] = wscalar[i]*block_size+j;
308 }
309 }
310
311 //compute deflatedA
312 SparseMatrixType deflatedA(deflatedAscalar.size1()*block_size,deflatedAscalar.size2()*block_size);
313 //do reserve!!
314 deflatedA.reserve(deflatedAscalar.nnz()*block_size*block_size);
315 ExpandScalarMatrix(rA.size1(),block_size,rA.index1_data().begin(), rA.index2_data().begin(), deflatedA);

445 /*
446 * Compute aggregates for a matrix A stored in CSR format
447 *
448 * Parameters:
449 * n_row - number of rows in A
450 * Ap[n_row + 1] - CSR row pointer
451 * Aj[nnz] - CSR column indices
452 * x[n_row] - aggregate numbers for each node
453 *
454 * Returns:
455 * The number of aggregates (== max(x[:]) + 1 )
456 *
457 * Notes:
458 * It is assumed that A is structurally symmetric.
459 * A may contain diagonal entries (self loops)
460 * Unaggregated nodes are marked with a -1
461 *
462 */
463 template <class I>
464 static I standard_aggregation(const I n_row,
465 const std::size_t Ap[],
466 const std::size_t Aj[],
467 I x[])
468 {
469 // Bj[n] == -1 means i-th node has not been aggregated
470 std::fill(x, x + n_row, 0);
471
472 I next_aggregate = 1; // number of aggregates + 1
473
474 //Pass #1
475 for(I i = 0; i < n_row; i++)
476 {
477 if(x[i])
478 {
479 continue; //already marked
480 }
481
482 const I row_start = Ap[i];
483 const I row_end = Ap[i+1];
484
485 //Determine whether all neighbors of this node are free (not already aggregates)
486 bool has_aggregated_neighbors = false;
487 bool has_neighbors = false;
488 for(I jj = row_start; jj < row_end; jj++)
489 {
490 const I j = Aj[jj];
491 if( i != j )
492 {
493 has_neighbors = true;
494 if( x[j] )
495 {
496 has_aggregated_neighbors = true;
497 break;
498 }
499 }
500 }
501
502 if(!has_neighbors)
503 {
504 //isolated node, do not aggregate
505 x[i] = -n_row;
506 }
507 else if (!has_aggregated_neighbors)
508 {
509 //Make an aggregate out of this node and its neighbors
510 x[i] = next_aggregate;
511 for(I jj = row_start; jj < row_end; jj++)
512 {
513 x[Aj[jj]] = next_aggregate;
514 }
515 next_aggregate++;
516 }
517 }
518
519 //Pass #2
520 // Add unaggregated nodes to any neighboring aggregate
521 for(I i = 0; i < n_row; i++)
522 {
523 if(x[i])
524 {
525 continue; //already marked
526 }
527
528 for(I jj = static_cast<I>(Ap[i]); jj < static_cast<I>(Ap[i+1]); jj++)
529 {
530 const I j = Aj[jj];
531
532 const I xj = x[j];
533 if(xj > 0)
534 {
535 x[i] = -xj;
536 break;
537 }
538 }
539 }
540
541 next_aggregate--;
542
543 //Pass #3
544 for(I i = 0; i < n_row; i++)
545 {
546 const I xi = x[i];
547
548 if(xi != 0)
549 {
550 // node i has been aggregated
551 if(xi > 0)
552 x[i] = xi - 1;
553 else if(xi == -n_row)
554 x[i] = -1;
555 else
556 x[i] = -xi - 1;
557 continue;
558 }
559
560 // node i has not been aggregated
561 const I row_start = Ap[i];
562 const I row_end = Ap[i+1];
563
564 x[i] = next_aggregate;
565
566 for(I jj = row_start; jj < row_end; jj++)
567 {
568 const I j = Aj[jj];
569
570 if(x[j] == 0) //unmarked neighbors
571 {
572 x[j] = next_aggregate;
573 }
574 }
575 next_aggregate++;
576 }
577
578
579 return next_aggregate; //number of aggregates
580 }

585 static void ConstructScalarMatrix(const std::size_t n_row, const std::size_t block_size,
586 const std::size_t Ap[],
587 const std::size_t Aj[],
588 SparseMatrixType& Ascalar
589 )
590 {
591 Ascalar.resize(n_row/block_size,n_row/block_size,0.0);
592 std::size_t scalar_size = (Ap[n_row]-Ap[0])/(block_size*block_size);
593 Ascalar.reserve(scalar_size);
594
595 for(std::size_t i = 0; i < n_row; i++)
596 {
597 if(i%block_size == 0)
598 {
599 std::size_t iscalar = i/block_size;
600 const std::size_t row_start = Ap[i];
601 const std::size_t row_end = Ap[i+1];
602
603 for(std::size_t jj = row_start; jj < row_end; jj++)
604 {
605 const std::size_t j = Aj[jj];
606 if(j%block_size == 0)
607 {
608 std::size_t jscalar = j/block_size;
609 Ascalar.push_back(iscalar,jscalar,0.0);
610 }
611 }
612 }
613 }

616 static void ExpandScalarMatrix(const std::size_t n_row, const std::size_t block_size,
617 const std::size_t Ap[],
618 const std::size_t Aj[],
619 SparseMatrixType& Aexpanded
620 )
621 {
622 Aexpanded.resize(n_row*block_size,n_row*block_size,0.0);
623 std::size_t expanded_size = (Ap[n_row]-Ap[0])*block_size*block_size;
624 Aexpanded.reserve(expanded_size);
625
626 for(std::size_t i = 0; i < n_row; i++)
627 {
628 const std::size_t row_start = Ap[i];
629 const std::size_t row_end = Ap[i+1];
630
631 for(std::size_t isub=0; isub<block_size; isub++)
632 {
633 std::size_t iexpanded = i*block_size + isub;
634 for(std::size_t jj = row_start; jj < row_end; jj++)
635 {
636 const std::size_t j = Aj[jj];
637
638 for(std::size_t jsub=0; jsub<block_size; jsub++)
639 {
640 std::size_t jexpanded = j*block_size+jsub;
641 Aexpanded.push_back(iexpanded,jexpanded,0.0);
642 }
643 }
644 }
