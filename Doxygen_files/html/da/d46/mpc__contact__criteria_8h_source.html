202  // Getting REACTION_CHECK_STIFFNESS_FACTOR
203  const double reaction_check_stiffness_factor = r_process_info.Has(REACTION_CHECK_STIFFNESS_FACTOR) ? r_process_info.GetValue(REACTION_CHECK_STIFFNESS_FACTOR) : 1.0e-12;

229  for (auto& r_elements : rModelPart.Elements()) {
230  if (r_elements.pGetProperties()->Has(YOUNG_MODULUS)) {
231  p_properties = r_elements.pGetProperties();
232  }
233  }

235  // Defining the convergence
236  IndexType is_active_set_converged = 0, is_slip_converged = 0;
237 
238  // Checking just after first iteration
239  // We get the YOUNG_MODULUS
240  const double young_modulus = p_properties->Has(YOUNG_MODULUS) ? p_properties->GetValue(YOUNG_MODULUS) : 0.0;
241  const double auxiliary_check = young_modulus > 0.0 ? -(reaction_check_stiffness_factor * young_modulus) : 0.0;
242 
243  // We check the active/inactive set during the first non-linear iteration or for the general semi-smooth case
244  auto& r_nodes_array = r_contact_model_part.Nodes();
245 
246  // If frictionaless or mesh tying
247  if (rModelPart.IsNot(SLIP)) {
248  is_active_set_converged = block_for_each<SumReduction<IndexType>>(r_nodes_array, [&](NodeType& rNode) {
249  if (rNode.Is(SLAVE)) {
250  // The contact force corresponds with the reaction in the normal direction
251  const array_1d<double, 3>& r_total_force = rNode.FastGetSolutionStepValue(REACTION);
252 
253  const double nodal_area = rNode.Has(NODAL_AREA) ? rNode.GetValue(NODAL_AREA) : 1.0;
254  const double gap = rNode.FastGetSolutionStepValue(WEIGHTED_GAP)/nodal_area;
255  const array_1d<double, 3>& r_normal = rNode.FastGetSolutionStepValue(NORMAL);
256  const double contact_force = inner_prod(r_total_force, r_normal);
257  const double contact_pressure = contact_force/rNode.GetValue(NODAL_MAUX);
258 
259  if (contact_pressure < auxiliary_check || gap < 0.0) { // NOTE: This could be conflictive (< or <=)
260  // We save the contact force
261  rNode.SetValue(CONTACT_FORCE, contact_force/rNode.GetValue(NODAL_PAUX) * r_normal);
262  rNode.SetValue(NORMAL_CONTACT_STRESS, contact_pressure);
263  if (rNode.IsNot(ACTIVE)) {
264  rNode.Set(ACTIVE, true);
265  return 1;
266  }
267  } else {
268  if (rNode.Is(ACTIVE)) {
269  rNode.Set(ACTIVE, false);
270  return 1;
271  }
272  }
273  }
274  return 0;
275  });
276  } else { // If frictional
277  using TwoReduction = CombinedReduction<SumReduction<IndexType>, SumReduction<IndexType>>;
278  std::tie(is_active_set_converged, is_slip_converged) = block_for_each<TwoReduction>(r_nodes_array, [&](NodeType& rNode) {
279  if (rNode.Is(SLAVE)) {
280  const double auxiliary_check = young_modulus > 0.0 ? -(reaction_check_stiffness_factor * young_modulus) : 0.0;
281  // The contact force corresponds with the reaction in the normal direction
282  const array_1d<double, 3>& r_total_force = rNode.FastGetSolutionStepValue(REACTION);
283 
284  const double nodal_area = rNode.Has(NODAL_AREA) ? rNode.GetValue(NODAL_AREA) : 1.0;
285  const double gap = rNode.FastGetSolutionStepValue(WEIGHTED_GAP)/nodal_area;
286  const array_1d<double, 3>& r_normal = rNode.FastGetSolutionStepValue(NORMAL);
287  const double contact_force = inner_prod(r_total_force, r_normal);
288  const double normal_contact_pressure = contact_force/rNode.GetValue(NODAL_MAUX);
289 
290  if (normal_contact_pressure < auxiliary_check || gap < 0.0) { // NOTE: This could be conflictive (< or <=)
291  // We save the contact force
292  rNode.SetValue(CONTACT_FORCE, r_total_force/rNode.GetValue(NODAL_PAUX));
293  rNode.SetValue(NORMAL_CONTACT_STRESS, normal_contact_pressure);
294  if (rNode.IsNot(ACTIVE)) {
295  rNode.Set(ACTIVE, true);
296  return std::make_tuple(1,0);
297  }
298 
299  // The friction coefficient
300  const double tangential_contact_pressure = norm_2(r_total_force - contact_force * r_normal)/rNode.GetValue(NODAL_MAUX);
301  const bool is_slip = rNode.Is(SLIP);
302  const double mu = rNode.GetValue(FRICTION_COEFFICIENT);
303 
304  if (tangential_contact_pressure <= - mu * contact_force) { // STICK CASE // TODO: Check the <=
305  rNode.SetValue(TANGENTIAL_CONTACT_STRESS, tangential_contact_pressure);
306  if (is_slip) {
307  rNode.Set(SLIP, false);
308  return std::make_tuple(0,1);
309  }
310  } else { // SLIP CASE
311  rNode.SetValue(TANGENTIAL_CONTACT_STRESS, - mu * contact_force);
312  if (!is_slip) {
313  rNode.Set(SLIP, true);
314  return std::make_tuple(0,1);
315  }
316  }
317  } else {
318  if (rNode.Is(ACTIVE)) {
319  rNode.Set(ACTIVE, false);
320  rNode.Reset(SLIP);
321  return std::make_tuple(1,0);
322  }
323  }
324  }
325  return std::make_tuple(0,0);
326  });
327  }
328 
329  // We set the constraints active and inactive in function of the active set
330  auto& r_conditions_array = rModelPart.GetSubModelPart("ComputingContact").Conditions();
331  block_for_each(r_conditions_array, [&](Condition& rCond) {

333  std::size_t counter = 0;
334  for (auto& r_node : r_slave_geometry) {
335  if (r_node.IsNot(ACTIVE)) {
336  ++counter;
337  }
338  }
339 
340  // In case of traction we deactivate
341  if (counter == r_slave_geometry.size()) {
342  rCond.Set(ACTIVE, false);
343  // We deactivate the constraints on inactive conditions
344  if (rCond.Has(CONSTRAINT_POINTER)) {
345  auto p_const = rCond.GetValue(CONSTRAINT_POINTER);
346 
347  // In case of traction we deactivate
348  p_const->Set(ACTIVE, false);
349  } else {

351  }
352  }
353  });
354 
355  // We save to the process info if the active set has converged
356  const bool active_set_converged = (is_active_set_converged == 0 ? true : false);
357  const bool slip_set_converged = (is_slip_converged == 0 ? true : false);
358 
359  if (rModelPart.GetCommunicator().MyPID() == 0 && this->GetEchoLevel() > 0) {
360  if (active_set_converged) {
361  KRATOS_INFO("MPCContactCriteria") << BOLDFONT("\tActive set") << " convergence is " << BOLDFONT(FGRN("achieved")) << std::endl;
362  } else {
363  KRATOS_INFO("MPCContactCriteria") << BOLDFONT("\tActive set") << " convergence is " << BOLDFONT(FRED("not achieved")) << std::endl;

367  } else {

369  }
370  }
371

373  }
374 
375  return true;
