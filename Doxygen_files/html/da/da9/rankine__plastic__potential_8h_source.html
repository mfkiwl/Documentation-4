109 static void CalculatePlasticPotentialDerivative(
110 const array_1d<double, VoigtSize>& rPredictiveStressVector,
111 const array_1d<double, VoigtSize>& rDeviator,
112 const double J2,
113 array_1d<double, VoigtSize>& rGFlux,
114 ConstitutiveLaw::Parameters& rValues
115 )
116 {
117 array_1d<double, VoigtSize> first_vector, second_vector, third_vector;
118 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateFirstVector(first_vector);
119 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateSecondVector(rDeviator, J2, second_vector);
120 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateThirdVector(rDeviator, J2, third_vector);
121
122 double J3, lode_angle;
123 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(rDeviator, J3);
124 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
125
126 double c1, c3, c2;
127 double checker = std::abs(lode_angle * 180.0 / Globals::Pi);
128 const double sqrt_3 = std::sqrt(3.0);
129
130 if (std::abs(checker) < 29.0) { // If it is not the edge
131 const double sqrt_J2 = std::sqrt(J2);
132 const double square_sin_3_lode = std::pow(std::sin(3.0 * lode_angle), 2);
133 const double angle = lode_angle + Globals::Pi / 6.0;
134 const double dLode_dJ2 = (3.0 * sqrt_3 * J3) / (4.0 * J2 * J2 * sqrt_J2 * std::sqrt(1.0 - square_sin_3_lode));
135 const double dLode_dJ3 = -sqrt_3 / (2.0 * J2 * sqrt_J2 * std::sqrt(1.0 - square_sin_3_lode));
136 c1 = 1.0 / 3.0;
137 c2 = 2.0 * sqrt_3 / 3.0 * (std::cos(angle) / (2.0 * sqrt_J2) - 2.0 * sqrt_3 * sqrt_J2 / 3.0 * std::sin(angle) * dLode_dJ2) * 2.0 * sqrt_J2;
138 c3 = -2.0 * std::sqrt(3.0 * J2) / 3.0 * std::sin(angle) * dLode_dJ3;
139 } else { // smoothing with drucker-praguer
140 const double friction_angle = rValues.GetMaterialProperties()[FRICTION_ANGLE] * Globals::Pi / 180.0;
141 const double sin_phi = std::sin(friction_angle);
142 const double CFL = -sqrt_3 * (3.0 - sin_phi) / (3.0 * sin_phi - 3.0);
143 c1 = CFL * 2.0 * sin_phi / (sqrt_3 * (3.0 - sin_phi));
144 c2 = CFL;
145 c3 = 0.0;
146 }
147 noalias(rGFlux) = c1 * first_vector + c2 * second_vector + c3 * third_vector;
148 }
