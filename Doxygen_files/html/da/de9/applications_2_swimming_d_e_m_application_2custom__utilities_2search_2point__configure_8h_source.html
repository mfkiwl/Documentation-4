176static inline void Distance(const PointerType& r_p_point_1, const PointerType& r_p_point_2, double& distance)
177{
178 array_1d<double, 3> center_1 = *r_p_point_1;
179 array_1d<double, 3> center_2 = *r_p_point_2;
180
181 distance = sqrt((center_1[0] - center_2[0]) * (center_1[0] - center_2[0]) +
182 (center_1[1] - center_2[1]) * (center_1[1] - center_2[1]) +
183 (center_1[2] - center_2[2]) * (center_1[2] - center_2[2]));
184}
