229 {

251 inline void ToEulerAngles(array_1d<double, 3>& EA)const
252 {
253 double test = W() * W() - X() * X() - Y() * Y() + Z() * Z();
254 if (test > (1.0 - 1.0e-6)) { // singularity at north pole
255 EA[0] = -atan2 (2 * Z() * W(), (W() * W() - Z() * Z()));
256 EA[1] = 0.0;
257 EA[2] = 0.0;
258 }
259 else if (test < (-1.0 + 1.0e-6)) { // singularity at south pole
260 EA[0] = atan2 (2 * Z() * W(), (W() * W() - Z() * Z()));
261 EA[1] = Globals::Pi;
262 EA[2] = 0.0;
263 }
264 else {
265 EA[0] = atan2((X() * Z() + Y() * W()), -(Y() * Z() - X() * W()));
266 EA[1] = -acos (test);
267 EA[2] = atan2((X() * Z() - Y() * W()), (Y() * Z() + X() * W()));
268 }

277 inline void ToRotationVector(T& rx, T& ry, T& rz)const
278 {
279 T xx, yy, zz, ww;
280
281 if(W() < 0.0) {
282 xx = -X();
283 yy = -Y();
284 zz = -Z();
285 ww = -W();
286 }
287 else {
288 xx = X();
289 yy = Y();
290 zz = Z();
291 ww = W();
292 }
293
294 T vNorm = xx*xx + yy*yy + zz*zz;
295 if(vNorm == 0.0) {
296 rx = 0.0;
297 ry = 0.0;
298 rz = 0.0;
299 return;
300 }
301
302 if(vNorm != 1.0)
303 vNorm = std::sqrt(vNorm);
304
305 T mult = (vNorm < ww) ? (2.0 / vNorm * std::asin(vNorm)) : (2.0 / vNorm * std::acos(ww));
306
307 rx = xx * mult;
308 ry = yy * mult;
309 rz = zz * mult;
310 }

339 template<class TVector3_A, class TVector3_B>
340 inline void RotateVector3(const TVector3_A& a, TVector3_B& b)const
341 {
342 // b = 2.0 * cross( this->VectorialPart, a )
343 b(0) = 2.0 * (Y() * a(2) - Z() * a(1));
344 b(1) = 2.0 * (Z() * a(0) - X() * a(2));
345 b(2) = 2.0 * (X() * a(1) - Y() * a(0));
346
347 // c = cross( this->VectorialPart, b )
348 T c0 = Y() * b(2) - Z() * b(1);
349 T c1 = Z() * b(0) - X() * b(2);
350 T c2 = X() * b(1) - Y() * b(0);
351
352 // set results
353 b(0) = a(0) + b(0)*W() + c0;
354 b(1) = a(1) + b(1)*W() + c1;
355 b(2) = a(2) + b(2)*W() + c2;
356 }

368 template<class TVector3>
369 inline void RotateVector3(TVector3& a)const
370 {
371 // b = 2.0 * cross( this->VectorialPart, a )
372 T b0 = 2.0 * (Y() * a(2) - Z() * a(1));
373 T b1 = 2.0 * (Z() * a(0) - X() * a(2));
374 T b2 = 2.0 * (X() * a(1) - Y() * a(0));
375
376 // c = cross( this->VectorialPart, b )
377 T c0 = Y() * b2 - Z() * b1;
378 T c1 = Z() * b0 - X() * b2;
379 T c2 = X() * b1 - Y() * b0;
380
381 // set results
382 a(0) += b0*W() + c0;
383 a(1) += b1*W() + c1;
384 a(2) += b2*W() + c2;

411 static inline Quaternion FromAxisAngle(T x, T y, T z, T radians)
412 {
413 T sqnorm = x*x + y*y + z*z;
414 if (sqnorm == 0.0)
415 return Quaternion::Identity();
416
417 if(sqnorm != 1.0) {
418 T norm = std::sqrt(sqnorm);
419 x /= norm;
420 y /= norm;
421 z /= norm;
422 }
423
424 T halfAngle = radians * 0.5;
425
426 T s = std::sin(halfAngle);
427 T q0 = std::cos(halfAngle);
428
429 Quaternion result(q0, s*x, s*y, s*z);
430 result.normalize();
431
432 return result;
433 }

442 static inline Quaternion FromRotationVector(T rx, T ry, T rz)
443 {
444 T rModulus = rx*rx + ry*ry + rz*rz;
445 if(rModulus == 0.0)
446 return Quaternion::Identity();
447
448 if(rModulus != 1.0) {
449 rModulus = std::sqrt(rModulus);
450 rx /= rModulus;
451 ry /= rModulus;
452 rz /= rModulus;

454

456
457 T q0 = std::cos(halfAngle);
458 T s = std::sin(halfAngle);
459
460 return Quaternion(q0, rx*s, ry*s, rz*s);
461 }

489 template<class TMatrix3x3>
490 static inline Quaternion FromRotationMatrix(const TMatrix3x3 & m)
491 {
492 T xx = m(0, 0);
493 T yy = m(1, 1);
494 T zz = m(2, 2);
495 T tr = xx + yy + zz;
496 Quaternion Q;
497 if ((tr > xx) && (tr > yy) && (tr > zz))
498 {
499 T S = std::sqrt(tr + 1.0) * 2.0;
500 Q = Quaternion(
501 0.25 * S,
502 (m(2, 1) - m(1, 2)) / S,
503 (m(0, 2) - m(2, 0)) / S,
504 (m(1, 0) - m(0, 1)) / S
505 );
506 }
507 else if ((xx > yy) && (xx > zz))
508 {
509 T S = std::sqrt(1.0 + xx - yy - zz) * 2.0;
510 Q = Quaternion(
511 (m(2, 1) - m(1, 2)) / S,
512 0.25 * S,
513 (m(0, 1) + m(1, 0)) / S,
514 (m(0, 2) + m(2, 0)) / S
515 );
516 }
517 else if (yy > zz)
518 {
519 T S = std::sqrt(1.0 + yy - xx - zz) * 2.0;
520 Q = Quaternion(
521 (m(0, 2) - m(2, 0)) / S,
522 (m(0, 1) + m(1, 0)) / S,
523 0.25 * S,
524 (m(1, 2) + m(2, 1)) / S
525 );
526 }
527 else
528 {
529 T S = std::sqrt(1.0 + zz - xx - yy) * 2.0;
530 Q = Quaternion(
531 (m(1, 0) - m(0, 1)) / S,
532 (m(0, 2) + m(2, 0)) / S,
533 (m(1, 2) + m(2, 1)) / S,
534 0.25 * S
535 );
536 }
537
538
539 Q.normalize();
540 return Q;

549 static inline Quaternion FromEulerAngles(const array_1d<double, 3> & EA)
550 {
551 Quaternion Q;
552 double c2 = cos(-EA[1]/2); double c1p3 = cos((EA[0]+EA[2])/2); double c1m3 = cos((EA[0]-EA[2])/2);
553 double s2 = sin(-EA[1]/2); double s1p3 = sin((EA[0]+EA[2])/2); double s1m3 = sin((EA[0]-EA[2])/2);
554
555 Q = Quaternion(
556 c2 * c1p3,
557 s2 * c1m3,
558 s2 * s1m3,
559 c2 * s1p3);
560 Q.normalize();
561 return Q;
562 }
