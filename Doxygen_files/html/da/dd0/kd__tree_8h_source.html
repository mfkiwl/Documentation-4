162 void SearchNearestPoint(PointType const& rThisPoint, PointerType& rResult, CoordinateType& rResultDistance, SearchStructureType& Auxiliar ) override
163 {
164 CoordinateType temp = Auxiliar.residual_distance[mCutingDimension];
165 CoordinateType distance_to_partition = rThisPoint[mCutingDimension] - mPosition;
166
167 if( distance_to_partition < 0.0 ) // The point is in the left partition
168 {
169 //searching in the left child
170 mpChilds[0]->SearchNearestPoint(rThisPoint, rResult, rResultDistance, Auxiliar );
171
172 // compare with distance to right partition
173 Auxiliar.residual_distance[mCutingDimension] = distance_to_partition * distance_to_partition;
174 Auxiliar.distance_to_partition2 = Auxiliar.residual_distance[0];
175 for(SizeType i = 1; i < Dimension; i++)
176 Auxiliar.distance_to_partition2 += Auxiliar.residual_distance[i];
177 if( rResultDistance > Auxiliar.distance_to_partition2 )
178 mpChilds[1]->SearchNearestPoint(rThisPoint, rResult, rResultDistance, Auxiliar );
179
180 }
181 else // The point is in the right partition
182 {
183 //Searching in the right child
184 mpChilds[1]->SearchNearestPoint(rThisPoint, rResult, rResultDistance, Auxiliar );
185
186 // compare with distance to left partition
187 Auxiliar.residual_distance[mCutingDimension] = distance_to_partition * distance_to_partition;
188 Auxiliar.distance_to_partition2 = Auxiliar.residual_distance[0];
189 for(SizeType i = 1; i < Dimension; i++)
190 Auxiliar.distance_to_partition2 += Auxiliar.residual_distance[i];
191 if( rResultDistance > Auxiliar.distance_to_partition2 )
192 mpChilds[0]->SearchNearestPoint( rThisPoint, rResult, rResultDistance, Auxiliar );
193 }
194 Auxiliar.residual_distance[mCutingDimension] = temp;

387 static IteratorType Partition( IteratorType PointsBegin, IteratorType PointsEnd, IndexType& rCuttingDimension, CoordinateType& rCuttingValue )
388 {
389 SizeType n = SearchUtils::PointerDistance(PointsBegin, PointsEnd);
390 // find dimension of maximum spread
391 rCuttingDimension = MaxSpread(PointsBegin, PointsEnd);
392 IteratorType partition = PointsBegin + n / 2;
393
394 MedianSplit(PointsBegin, partition, PointsEnd, rCuttingDimension, rCuttingValue);
395
396 return partition;

399 static SizeType MaxSpread( IteratorType PointsBegin, IteratorType PointsEnd )
400 {
401 SizeType max_dimension = 0; // dimension of max spread
402 CoordinateType max_spread = 0; // amount of max spread
403
404 /* if(PointsBegin == PointsEnd) // If there is no point return the first coordinate */
405 /* return max_dimension; */
406
407 CoordinateType min[Dimension];
408 CoordinateType max[Dimension];
409 for (SizeType d = 0; d < Dimension; d++)
410 {
411 min[d] = (**PointsBegin)[d];
412 max[d] = (**PointsBegin)[d];
413 }
414 for (IteratorType i_point = PointsBegin; i_point != PointsEnd; i_point++)
415 {
416 for (SizeType d = 0; d < Dimension; d++)
417 {
418 CoordinateType c = (**i_point)[d];
419 if (c < min[d])
420 min[d] = c;
421 else if (c > max[d])
422 max[d] = c;
423 }
424 }
425 max_dimension = 0;
426 max_spread = max[0] - min[0];
427 for (SizeType d = 1; d < Dimension; d++)
428 {
429 CoordinateType spread = max[d] - min[d];
430 if (spread > max_spread)
431 {
432 max_spread = spread;
433 max_dimension = d;
434 }
435 }
436
437 return max_dimension;
438 }

440 static void MedianSplit( IteratorType PointsBegin, IteratorType PartitionPosition, IteratorType PointsEnd,
441 IndexType CuttingDimension, CoordinateType& rCuttingValue )
442 {
443 IteratorType left = PointsBegin;
444 IteratorType right = PointsEnd - 1;
445 while (left < right) // Iterating to find the partition point
446 {
447
448 IteratorType i = left; // selecting left as pivot
449 if ((**i)[CuttingDimension] > (**right)[CuttingDimension])
450 std::swap(*i,*right);
451
452 CoordinateType value = (**i)[CuttingDimension];
453 IteratorType j = right;
454 for(;;)
455 {
456 while ((**(++i))[CuttingDimension] < value)
457 ;
458
459 while ((**(--j))[CuttingDimension] > value)
460 ;
461 if (i < j) std::swap(*i,*j);
462 else break;
463 }
464 std::swap(*left,*j);
465
466 if (j > PartitionPosition)
467 right = j - 1;
468 else if (j < PartitionPosition)
469 left = j + 1;
470 else break;
471 }
472
473 CoordinateType max = (**PointsBegin)[CuttingDimension];
474 IteratorType k = PointsBegin;
475 IteratorType last = PartitionPosition;
476
477 for(IteratorType i = PointsBegin ; i != PartitionPosition ; ++i)
478 if((**i)[CuttingDimension] > max)
479 {
480 max = (**i)[CuttingDimension];
481 k = i;
482 }
483
484 if(k != PointsBegin)
485 std::swap(--last, k);
486
487 rCuttingValue = ((**last)[CuttingDimension] + (**PartitionPosition)[CuttingDimension])/2.0;
488 }

613 static SizeType MaxSpread( IteratorType PointsBegin, IteratorType PointsEnd, CoordinateType& AverageValue )
614 {
615 SizeType max_dimension = 0; // dimension of max spread
616 CoordinateType max_spread = 0; // amount of max spread
617 AverageValue = 0.0;
618
619 CoordinateType size = static_cast<CoordinateType>(SearchUtils::PointerDistance(PointsBegin,PointsEnd));
620
621 CoordinateType min[Dimension];
622 CoordinateType max[Dimension];
623 CoordinateType Average[Dimension];
624 for (SizeType d = 0; d < Dimension; d++)
625 {
626 Average[d] = 0.0;
627 min[d] = (**PointsBegin)[d];
628 max[d] = (**PointsBegin)[d];
629 }
630 for (IteratorType i_point = PointsBegin; i_point != PointsEnd; i_point++)
631 {
632 for (SizeType d = 0; d < Dimension; d++)
633 {
634 CoordinateType c = (**i_point)[d];
635 Average[d] += c;
636 if (c < min[d])
637 min[d] = c;
638 else if (c > max[d])
639 max[d] = c;
640 }
641 }
642 max_dimension = 0;
643 max_spread = max[0] - min[0];
644 AverageValue = Average[0] / size;
645 for (SizeType d = 1; d < Dimension; d++)
646 {
647 CoordinateType spread = max[d] - min[d];
648 if (spread > max_spread)
649 {
650 max_spread = spread;
651 max_dimension = d;
652 AverageValue = Average[d] / size;
653 }
654 }
655
656 return max_dimension;
657 }

660 static void AverageSplit( IteratorType PointsBegin, IteratorType& PartitionPosition, IteratorType PointsEnd,
661 IndexType& CuttingDimension, CoordinateType& rCuttingValue )
662 {
663 // reorder by cutting_dimension
664 IteratorType left = PointsBegin;
665 IteratorType right = PointsEnd - 1;
666 for(;;)
667 {
668 while( left < PointsEnd && (**left)[CuttingDimension] < rCuttingValue ) left++;
669 while( right > PointsBegin && (**right)[CuttingDimension] >= rCuttingValue ) right--;
670 if (left <= right) std::swap(*left,*right);
671 else break;
672 }
673
674 PartitionPosition = left;
675 }

803
804 //CoordinateType size = static_cast<CoordinateType>(SearchUtils::PointerDistance(PointsBegin,PointsEnd));

808 for (SizeType d = 0; d < Dimension; d++)

838 static IteratorType Split( IteratorType PointsBegin, IteratorType PointsEnd, IndexType& CuttingDimension, CoordinateType& rCuttingValue )
839 {
840 // reorder by cutting_dimension
841 IteratorType left = PointsBegin;
842 IteratorType right = PointsEnd - 1;
843 for(;;)
844 {
845 while( (**left)[CuttingDimension] < rCuttingValue ) left++;
846 while( (**right)[CuttingDimension] >= rCuttingValue ) right--;
847 if (left < right) std::swap(*left,*right);
848 else break;
849 }
850
851 return left;
