49 template< class TVector3 >
50 inline static double NormalizeVector3(TVector3 & V)
51 {
52 double vn = V(0) * V(0) + V(1) * V(1) + V(2) * V(2);
53 if(vn != 0.0 && vn != 1.0) {
54 vn = std::sqrt(vn);
55 V /= vn;
56 }

62 ShellT3_LocalCoordinateSystem(const Vector3Type & P1global,
63 const Vector3Type & P2global,
64 const Vector3Type & P3global)
65 : mP(3)
66 , mOrientation(3, 3)
67 {
68 // Form the basis vectors alligning the local X direction
69 // with the 1-2 side
70
71 // compute the central point
72 noalias( mCenter ) = P1global;
73 noalias( mCenter ) += P2global;
74 noalias( mCenter ) += P3global;
75 mCenter /= 3.0;
76
77 // compute the local X direction parallel to the side 1-2
78 Vector3Type e1( P2global - P1global );
79
80 // compute the temporary local Y direction
81 Vector3Type e2( P3global - P1global );
82
83 // compute the Normal vector at the element center
84 // While normalizing the normal vector save its norm to compute the area.
85 Vector3Type e3;
86 MathUtils<RealType>::CrossProduct(e3, e1, e2);
87 mArea = NormalizeVector3( e3 );
88 mArea /= 2.0;
89
90 // finally compute the local Y direction to be orthogonal to both X and Z local directions
91 MathUtils<RealType>::CrossProduct(e2, e3, e1);
92
93 // normalize the X and Y directions
94 NormalizeVector3( e1 );
95 NormalizeVector3( e2 );
96
97 // form the 3x3 transformation matrix
98 for(int i = 0; i < 3; i++)
99 {
100 mOrientation(0, i) = e1(i);
101 mOrientation(1, i) = e2(i);
102 mOrientation(2, i) = e3(i);
103 }
104
105 // transform global coordinates to the local coordinate system
106 for(int i = 0; i < 3; i++)
107 {
108 mP[0](i) = mOrientation(i, 0) * (P1global(0) - mCenter(0)) + mOrientation(i, 1) * (P1global(1) - mCenter(1)) + mOrientation(i, 2) * (P1global(2) - mCenter(2));
109 mP[1](i) = mOrientation(i, 0) * (P2global(0) - mCenter(0)) + mOrientation(i, 1) * (P2global(1) - mCenter(1)) + mOrientation(i, 2) * (P2global(2) - mCenter(2));

111 }
112 }

199 inline void ComputeTotalRotationMatrix( MatrixType & R )const
200 {
201 size_t mat_size = 18;
202 if(R.size1() != mat_size || R.size2() != mat_size)
203 R.resize(mat_size, mat_size, false);
204
205 noalias( R ) = ZeroMatrix(mat_size, mat_size);
206
207 for(size_t k = 0; k < 6; k++)
208 {
209 size_t i = k * 3;
210 R(i , i ) = mOrientation(0, 0); R(i , i+1) = mOrientation(0, 1); R(i , i+2) = mOrientation(0, 2);
211 R(i+1, i ) = mOrientation(1, 0); R(i+1, i+1) = mOrientation(1, 1); R(i+1, i+2) = mOrientation(1, 2);
212 R(i+2, i ) = mOrientation(2, 0); R(i+2, i+1) = mOrientation(2, 1); R(i+2, i+2) = mOrientation(2, 2);
213 }
