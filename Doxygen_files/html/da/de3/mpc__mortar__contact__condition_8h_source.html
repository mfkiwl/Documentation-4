456 IntegrationMethod GetIntegrationMethod() const override
457 {
458 // Setting the auxiliary integration points
459 const IndexType integration_order = GetProperties().Has(INTEGRATION_ORDER_CONTACT) ? GetProperties().GetValue(INTEGRATION_ORDER_CONTACT) : 2;
460 switch (integration_order) {
461 case 1: return GeometryData::IntegrationMethod::GI_GAUSS_1;
462 case 2: return GeometryData::IntegrationMethod::GI_GAUSS_2;
463 case 3: return GeometryData::IntegrationMethod::GI_GAUSS_3;
464 case 4: return GeometryData::IntegrationMethod::GI_GAUSS_4;
465 case 5: return GeometryData::IntegrationMethod::GI_GAUSS_5;
466 default: return GeometryData::IntegrationMethod::GI_GAUSS_2;
