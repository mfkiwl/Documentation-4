442 double Volume() const override {
443 //Closed formula for the linear tetrahedra
444 const double onesixth = 1.0/6.0;
445
446 const CoordinatesArrayType& rP0 = this->Points()[0].Coordinates();
447 const CoordinatesArrayType& rP1 = this->Points()[1].Coordinates();
448 const CoordinatesArrayType& rP2 = this->Points()[2].Coordinates();
449 const CoordinatesArrayType& rP3 = this->Points()[3].Coordinates();
450
451 const double x10 = rP1[0] - rP0[0];
452 const double y10 = rP1[1] - rP0[1];
453 const double z10 = rP1[2] - rP0[2];
454
455 const double x20 = rP2[0] - rP0[0];
456 const double y20 = rP2[1] - rP0[1];
457 const double z20 = rP2[2] - rP0[2];
458

460 const double y30 = rP3[1] - rP0[1];
461 const double z30 = rP3[2] - rP0[2];
462
463 const double detJ = x10 * y20 * z30 - x10 * y30 * z20 + y10 * z20 * x30 - y10 * x20 * z30 + z10 * x20 * y30 - z10 * y20 * x30;
464 return detJ*onesixth;

544 double Circumradius() const override {
545 // | s10 y10 z10 | | s10 x10 z10 | | s10 x10 y10 | | x10 y10 z10 |
546 // MX | s20 y20 z20 | MY | s20 x20 z20 | MZ | s20 x20 y20 | AL | x20 y20 z20 |
547 // | s30 y30 z30 | | s30 x30 z30 | | s30 x30 y30 | | x30 y30 z30 |
548
549 const CoordinatesArrayType& rP0 = this->Points()[0].Coordinates();
550 const CoordinatesArrayType& rP1 = this->Points()[1].Coordinates();
551 const CoordinatesArrayType& rP2 = this->Points()[2].Coordinates();
552 const CoordinatesArrayType& rP3 = this->Points()[3].Coordinates();
553
554 const double aDot = rP0[0] * rP0[0] + rP0[1] * rP0[1] + rP0[2] * rP0[2];
555 const double bDot = rP1[0] * rP1[0] + rP1[1] * rP1[1] + rP1[2] * rP1[2];
556 const double cDot = rP2[0] * rP2[0] + rP2[1] * rP2[1] + rP2[2] * rP2[2];
557 const double dDot = rP3[0] * rP3[0] + rP3[1] * rP3[1] + rP3[2] * rP3[2];
558
559 // Build the simplified matrices
560 const double s10 = aDot - dDot;
561 const double x10 = rP0[0] - rP3[0];
562 const double y10 = rP0[1] - rP3[1];
563 const double z10 = rP0[2] - rP3[2];
564
565 const double s20 = bDot - dDot;
566 const double x20 = rP1[0] - rP3[0];
567 const double y20 = rP1[1] - rP3[1];
568 const double z20 = rP1[2] - rP3[2];
569
570 const double s30 = cDot - dDot;
571 const double x30 = rP2[0] - rP3[0];
572 const double y30 = rP2[1] - rP3[1];
573 const double z30 = rP2[2] - rP3[2];
574
575 const double detJMX = s10 * y20 * z30 + y10 * z20 * s30 + z10 * s20 * y30 - s30 * y20 * z10 - y30 * z20 * s10 - z30 * s20 * y10;
576 const double detJMY = s10 * x20 * z30 + x10 * z20 * s30 + z10 * s20 * x30 - s30 * x20 * z10 - x30 * z20 * s10 - z30 * s20 * x10;
577 const double detJMZ = s10 * x20 * y30 + x10 * y20 * s30 + y10 * s20 * x30 - s30 * x20 * y10 - x30 * y20 * s10 - y30 * s20 * x10;
578 const double detJAL = x10 * y20 * z30 + y10 * z20 * x30 + z10 * x20 * y30 - x30 * y20 * z10 - y30 * z20 * x10 - z30 * x20 * y10;
579
580 return std::sqrt( (detJMX*detJMX + detJMY*detJMY + detJMZ*detJMZ)) / (2*std::abs(detJAL));
581 }

588 double Inradius() const override {
589 // | x10 y10 z10 |
590 // AL | x20 y20 z20 |
591 // | x30 y30 z30 |
592
593 const CoordinatesArrayType& rP0 = this->Points()[0].Coordinates();
594 const CoordinatesArrayType& rP1 = this->Points()[1].Coordinates();
595 const CoordinatesArrayType& rP2 = this->Points()[2].Coordinates();
596 const CoordinatesArrayType& rP3 = this->Points()[3].Coordinates();
597
598 array_1d<double, 3> c012, c013, c023, c123;
599
600 MathUtils<double>::CrossProduct(c012, rP2-rP0, rP1-rP0);
601 MathUtils<double>::CrossProduct(c013, rP3-rP0, rP1-rP0);
602 MathUtils<double>::CrossProduct(c023, rP3-rP0, rP2-rP0);
603 MathUtils<double>::CrossProduct(c123, rP3-rP1, rP2-rP1);
604
605 const double n012 = std::sqrt(c012[0]*c012[0] + c012[1]*c012[1] + c012[2]*c012[2]);
606 const double n013 = std::sqrt(c013[0]*c013[0] + c013[1]*c013[1] + c013[2]*c013[2]);
607 const double n023 = std::sqrt(c023[0]*c023[0] + c023[1]*c023[1] + c023[2]*c023[2]);
608 const double n123 = std::sqrt(c123[0]*c123[0] + c123[1]*c123[1] + c123[2]*c123[2]);
609
610 // Build the simplified matrices
611 const double x10 = rP0[0] - rP3[0];
612 const double y10 = rP0[1] - rP3[1];
613 const double z10 = rP0[2] - rP3[2];
614
615 const double x20 = rP1[0] - rP3[0];
616 const double y20 = rP1[1] - rP3[1];
617 const double z20 = rP1[2] - rP3[2];
618
619 const double x30 = rP2[0] - rP3[0];
620 const double y30 = rP2[1] - rP3[1];
621 const double z30 = rP2[2] - rP3[2];
622
623 const double detJAL = x10 * y20 * z30 + y10 * z20 * x30 + z10 * x20 * y30 - x30 * y20 * z10 - y30 * z20 * x10 - z30 * x20 * y10;
624
625 return std::abs(detJAL) / (n012 + n013 + n023 + n123);
626 }

887 inline void ComputeDihedralAngles(Vector& rDihedralAngles) const override
888 {
889 if(rDihedralAngles.size() != 6)
890 rDihedralAngles.resize(6, false);
891
892 BoundedMatrix<double, 4, 3 > coords;
893 for (unsigned int i = 0; i < 4; i++) {
894 const array_1d<double, 3 > & xyz = this->GetPoint(i);
895 for (unsigned int j = 0; j < 3; j++)
896 coords(i, j) = xyz[j];
897 }
898 //we have to check 6 different angles (one per edge)
899 //to get an easy-to-read algorithm, we find the angles by creating the normal planes to each face and then find the angles between them
900 //to do so, we create a list of 3 nodes per angle. face1 is defined with nodes 0,1,2a , and face2 is defined with nodes 0,1,2b
901 const int node0[] = {0, 0, 0, 1, 1, 2};
902 const int node1[] = {1, 3, 2, 3, 2, 3};
903 const int node2a[] = {2, 1, 1, 0, 0, 0};
904 const int node2b[] = {3, 2, 3, 2, 3, 1};
905
906 array_1d<double,3> edge1,edge2a,edge2b,normal1,normal2;
907
908 //now we only loop through the six edges to see which one has the lowest angle
909 for (unsigned int i = 0; i < 6; i++) {
910 //first we find the edges
911 for (unsigned int j = 0; j < 3; ++j) {
912 edge1[j] = coords(node1[i],j) - coords(node0[i],j) ;
913 edge2a[j] = coords(node2a[i],j) - coords(node0[i],j) ;
914 edge2b[j] = coords(node2b[i],j) - coords(node0[i],j) ;
915 }
916 //now we find the normals to the planes
917 MathUtils<double>::CrossProduct(normal1,edge1,edge2a);
918 MathUtils<double>::CrossProduct(normal2,edge1,edge2b);
919 normal1 /= std::sqrt(normal1[0]*normal1[0] + normal1[1]*normal1[1] + normal1[2]*normal1[2]);
920 normal2 /= std::sqrt(normal2[0]*normal2[0] + normal2[1]*normal2[1] + normal2[2]*normal2[2]);
921
922 //and finally the cos of the angle:
923 const double angle_cos = ( normal1[0]*normal2[0] + normal1[1]*normal2[1] + normal1[2]*normal2[2] );
924 rDihedralAngles[i] = std::acos(angle_cos);
925 }

1471 void SplitAndDecompose(
1472 const BaseType& tetra, Plane& plane,
1473 std::vector<BaseType>& inside) const
1474 {
1475
1476 // Determine on which side of the plane the points of the tetrahedron lie.
1477 int i = 0;
1478 int positive = 0, negative = 0, zero = 0;
1479 array_1d<double,4> C;
1480 array_1d<int,4> pos, neg, zer;
1481
1482 noalias(C) = ZeroVector(4);
1483
1484 pos[0] = 0;
1485 neg[0] = 0;
1486 zer[0] = 0;
1487 pos[1] = 0;
1488 neg[1] = 0;
1489 zer[1] = 0;
1490 pos[2] = 0;
1491 neg[2] = 0;
1492 zer[2] = 0;
1493 pos[3] = 0;
1494 neg[3] = 0;
1495 zer[3] = 0;
1496
1497 for (i = 0; i < 4; ++i) {
1498 const array_1d<double,3>& p = tetra[i].Coordinates();
1499 C[i] = plane.DistanceTo(p);
1500 if (C[i] > 0.00)
1501 pos[positive++] = i;
1502 else if (C[i] < 0.00)
1503 neg[negative++] = i;
1504 else
1505 zer[zero++] = i;
1506 }
1507
1508 // For a split to occur, one of the c_i must be positive and one must
1509 // be negative.
1510 if (negative == 0) {
1511 // Tetrahedron is completely on the positive side of plane, full clip.
1512 return;
1513 }
1514
1515 if (positive == 0) {
1516 // Tetrahedron is completely on the negative side of plane.
1517 inside.push_back(tetra);
1518 return;
1519 }
1520
1521 double w0, w1, invCDiff;
1522 array_1d<array_1d<double, 3 >, 4> intp;
1523 array_1d<array_1d<double, 3 >, 4> V;
1524
1525 if (positive == 3) {
1526 // +++-
1527 for (i = 0; i < positive; ++i) {
1528 invCDiff = 1.00/(C[pos[i]] - C[neg[0]]);
1529 w0 = -C[neg[0]]*invCDiff;
1530 w1 = +C[pos[i]]*invCDiff;
1531 V[pos[i]] = w0*tetra[pos[i]].Coordinates() + w1*tetra[neg[0]].Coordinates();
1532 }
1533 inside.push_back(tetra);
1534 } else if (positive == 2) {
1535 if (negative == 2) {
1536 // ++--
1537 for (i = 0; i < positive; ++i) {
1538 invCDiff = (1.00)/(C[pos[i]] - C[neg[0]]);
1539 w0 = -C[neg[0]]*invCDiff;
1540 w1 = +C[pos[i]]*invCDiff;
1541 intp[i] = w0*tetra[pos[i]].Coordinates() + w1*tetra[neg[0]].Coordinates();
1542 }
1543 for (i = 0; i < negative; ++i) {
