485 bool IsInside(
486 const CoordinatesArrayType& rPoint,
487 CoordinatesArrayType& rResult,
488 const double Tolerance = std::numeric_limits<double>::epsilon()
489 ) const override
490 {
491 this->PointLocalCoordinates( rResult, rPoint );
492
493 if ( (rResult[0] >= (0.0 - Tolerance)) && (rResult[0] <= (1.0 + Tolerance)) )
494 if ( (rResult[1] >= (0.0 - Tolerance)) && (rResult[1] <= (1.0 + Tolerance)) )
495 if ( (rResult[2] >= (0.0 - Tolerance)) && (rResult[2] <= (1.0 + Tolerance)) )
496 if ((( 1.0 - ( rResult[0] + rResult[1] ) ) >= (0.0 - Tolerance) ) && (( 1.0 - ( rResult[0] + rResult[1] ) ) <= (1.0 + Tolerance) ) )
497 return true;
498
499 return false;
500 }

618 bool HasIntersection( const Point& rLowPoint, const Point& rHighPoint ) const override
619 {
620 // Check if faces have intersection
621 if(FaceType1(this->pGetPoint(0),this->pGetPoint(2), this->pGetPoint(1)).HasIntersection(rLowPoint, rHighPoint))
622 return true;
623 if(FaceType1(this->pGetPoint(3),this->pGetPoint(4), this->pGetPoint(5)).HasIntersection(rLowPoint, rHighPoint))
624 return true;
625 if(FaceType2(this->pGetPoint(1),this->pGetPoint(2), this->pGetPoint(5), this->pGetPoint(4)).HasIntersection(rLowPoint, rHighPoint))
626 return true;
627 if(FaceType2(this->pGetPoint(0),this->pGetPoint(3), this->pGetPoint(5), this->pGetPoint(2)).HasIntersection(rLowPoint, rHighPoint))
628 return true;
629 if(FaceType2(this->pGetPoint(0),this->pGetPoint(1), this->pGetPoint(4), this->pGetPoint(3)).HasIntersection(rLowPoint, rHighPoint))
630 return true;
631
632 CoordinatesArrayType local_coordinates;
633 // If there are no faces intersecting the box then or the box is inside the hexahedron or it does not have intersection
634 if(IsInside(rLowPoint,local_coordinates))
635 return true;
636
637 return false;
638 }

654 double ShapeFunctionValue( IndexType ShapeFunctionIndex,
655 const CoordinatesArrayType& rPoint ) const override
656 {
657 switch ( ShapeFunctionIndex )
658 {
659 case 0:
660 return( 1.0 -( rPoint[0] + rPoint[1] + rPoint[2] - ( rPoint[0]*rPoint[2] ) - ( rPoint[1]*rPoint[2] ) ) );
661 case 1:
662 return( rPoint[0] - ( rPoint[0]*rPoint[2] ) );
663 case 2:
664 return( rPoint[1] - ( rPoint[1]*rPoint[2] ) );
665 case 3:
666 return( rPoint[2] - ( rPoint[0]*rPoint[2] ) - ( rPoint[1]*rPoint[2] ) );
667 case 4:
668 return( rPoint[0]*rPoint[2] );
669 case 5:
670 return( rPoint[1]*rPoint[2] );
671 default:
672 KRATOS_ERROR << "Wrong index of shape function!" << *this << std::endl;
673 }
674
675 return 0;
676 }

856 static Matrix CalculateShapeFunctionsIntegrationPointsValues(typename BaseType::IntegrationMethod ThisMethod )
857 {
858 IntegrationPointsContainerType all_integration_points = AllIntegrationPoints();
859 IntegrationPointsArrayType integration_points = all_integration_points[static_cast<int>(ThisMethod)];
860 //number of integration points
861 const int integration_points_number = integration_points.size();
862 //number of nodes in current geometry
863 const int points_number = 6;
864 //setting up return matrix
865 Matrix shape_function_values( integration_points_number, points_number );
866 //loop over all integration points
867
868 for ( int pnt = 0; pnt < integration_points_number; pnt++ ) {
869 shape_function_values( pnt, 0 ) = ( 1.0
870 - integration_points[pnt].X()
871 - integration_points[pnt].Y()
872 - integration_points[pnt].Z()
873 + ( integration_points[pnt].X() * integration_points[pnt].Z() )
874 + ( integration_points[pnt].Y() * integration_points[pnt].Z() ) );
875 shape_function_values( pnt, 1 ) = integration_points[pnt].X()
876 - ( integration_points[pnt].X() * integration_points[pnt].Z() );
877 shape_function_values( pnt, 2 ) = integration_points[pnt].Y()
878 - ( integration_points[pnt].Y() * integration_points[pnt].Z() );
879 shape_function_values( pnt, 3 ) = integration_points[pnt].Z()
880 - ( integration_points[pnt].X() * integration_points[pnt].Z() )
881 - ( integration_points[pnt].Y() * integration_points[pnt].Z() );
882 shape_function_values( pnt, 4 ) = ( integration_points[pnt].X() * integration_points[pnt].Z() );
883 shape_function_values( pnt, 5 ) = ( integration_points[pnt].Y() * integration_points[pnt].Z() );
884 }
885
886 return shape_function_values;
887 }

901 static ShapeFunctionsGradientsType
902 CalculateShapeFunctionsIntegrationPointsLocalGradients(

904 {
905 IntegrationPointsContainerType all_integration_points =

907 IntegrationPointsArrayType integration_points = all_integration_points[static_cast<int>(ThisMethod)];
908 //number of integration points

911 //initialising container
912 //loop over all integration points
913
914 for ( int pnt = 0; pnt < integration_points_number; pnt++ )
915 {
916 Matrix result = ZeroMatrix( 6, 3 );
917 result( 0, 0 ) = -1.0 + integration_points[pnt].Z();
918 result( 0, 1 ) = -1.0 + integration_points[pnt].Z();
919 result( 0, 2 ) = -1.0 + integration_points[pnt].X() + integration_points[pnt].Y();
920 result( 1, 0 ) = 1.0 - integration_points[pnt].Z();
921 result( 1, 1 ) = 0.0;
922 result( 1, 2 ) = -integration_points[pnt].X();
923 result( 2, 0 ) = 0.0;
924 result( 2, 1 ) = 1.0 - integration_points[pnt].Z();
925 result( 2, 2 ) = -integration_points[pnt].Y();
926 result( 3, 0 ) = -integration_points[pnt].Z();
927 result( 3, 1 ) = -integration_points[pnt].Z();
928 result( 3, 2 ) = 1.0 - integration_points[pnt].X() - integration_points[pnt].Y();
929 result( 4, 0 ) = integration_points[pnt].Z();
930 result( 4, 1 ) = 0.0;
931 result( 4, 2 ) = integration_points[pnt].X();
932 result( 5, 0 ) = 0.0;
933 result( 5, 1 ) = integration_points[pnt].Z();
934 result( 5, 2 ) = integration_points[pnt].Y();
