668 inline static void AssembleLHS(
669 MatrixType& rA,
670 const Matrix& rLHSContribution,
671 const std::vector<std::size_t>& rEquationId
672 )
673 {
674 const unsigned int system_size = Size1(rA);
675
676 // Count active indices
677 unsigned int active_indices = 0;
678 for (unsigned int i = 0; i < rEquationId.size(); i++)
679 if (rEquationId[i] < system_size)
680 ++active_indices;
681
682 if (active_indices > 0) {
683 // Size Epetra vectors
684 Epetra_IntSerialDenseVector indices(active_indices);
685 Epetra_SerialDenseMatrix values(active_indices, active_indices);
686
687 // Fill epetra vectors
688 unsigned int loc_i = 0;
689 for (unsigned int i = 0; i < rEquationId.size(); i++) {
690 if (rEquationId[i] < system_size) {
691 indices[loc_i] = rEquationId[i];
692
693 unsigned int loc_j = 0;
694 for (unsigned int j = 0; j < rEquationId.size(); j++) {
695 if (rEquationId[j] < system_size) {
696 values(loc_i, loc_j) = rLHSContribution(i, j);
697 ++loc_j;
698 }
699 }
700 ++loc_i;
701 }
702 }
703
704 int ierr = rA.SumIntoGlobalValues(indices, values);
705 KRATOS_ERROR_IF(ierr != 0) << "Epetra failure found" << std::endl;
706 }
707 }

820 MatrixPointerType ReadMatrixMarket(
821 const std::string FileName,
822 Epetra_MpiComm& rComm
823 )
824 {
825 KRATOS_TRY
826
827 Epetra_CrsMatrix* pp = nullptr;
828
829 int error_code = EpetraExt::MatrixMarketFileToCrsMatrix(FileName.c_str(), rComm, pp);
830
831 KRATOS_ERROR_IF(error_code != 0) << "Eerror thrown while reading Matrix Market file "<<FileName<< " error code is : " << error_code;
832
833 rComm.Barrier();
834
835 const Epetra_CrsGraph& rGraph = pp->Graph();
836 MatrixPointerType paux = Kratos::make_shared<Epetra_FECrsMatrix>( ::Copy, rGraph, false );
837
838 IndexType NumMyRows = rGraph.RowMap().NumMyElements();
839
840 int* MyGlobalElements = new int[NumMyRows];
841 rGraph.RowMap().MyGlobalElements(MyGlobalElements);
842
843 for(IndexType i = 0; i < NumMyRows; ++i) {
844// std::cout << pA->Comm().MyPID() << " : I=" << i << std::endl;
845 IndexType GlobalRow = MyGlobalElements[i];
846
847 int NumEntries;
848 std::size_t Length = pp->NumGlobalEntries(GlobalRow); // length of Values and Indices
849
850 double* Values = new double[Length]; // extracted values for this row
851 int* Indices = new int[Length]; // extracted global column indices for the corresponding values
852
853 error_code = pp->ExtractGlobalRowCopy(GlobalRow, Length, NumEntries, Values, Indices);
854
855 KRATOS_ERROR_IF(error_code != 0) << "Error thrown in ExtractGlobalRowCopy : " << error_code;
856
857 error_code = paux->ReplaceGlobalValues(GlobalRow, Length, Values, Indices);
858
859 KRATOS_ERROR_IF(error_code != 0) << "Error thrown in ReplaceGlobalValues : " << error_code;
860
861 delete [] Values;
862 delete [] Indices;
863 }
864
865 paux->GlobalAssemble();
866
867 delete [] MyGlobalElements;
868 delete pp;
869
870 return paux;
871 KRATOS_CATCH("");
872 }

924 static double CheckAndCorrectZeroDiagonalValues(
925 const ProcessInfo& rProcessInfo,
926 MatrixType& rA,
927 VectorType& rb,
928 const SCALING_DIAGONAL ScalingDiagonal = SCALING_DIAGONAL::NO_SCALING
929 )
930 {
931 KRATOS_TRY
932
933 // Define zero value tolerance
934 const double zero_tolerance = std::numeric_limits<double>::epsilon();
935
936 // The diagonal considered
937 const double scale_factor = GetScaleNorm(rProcessInfo, rA, ScalingDiagonal);
938
939 for (int i = 0; i < rA.NumMyRows(); i++) {
940 int numEntries; // Number of non-zero entries
941 double* vals; // Row non-zero values
942 int* cols; // Column indices of row non-zero values
943 rA.ExtractMyRowView(i, numEntries, vals, cols);
944 const int row_gid = rA.RowMap().GID(i);
945 bool empty = true;
946 int j;
947 for (j = 0; j < numEntries; j++) {
948 const int col_gid = rA.ColMap().GID(cols[j]);
949 // Check diagonal value
950 if (col_gid == row_gid) {
951 if(std::abs(vals[j]) > zero_tolerance) {
952 empty = false;
953 }
954 break;
955 }
956 }
957
958 // If diagonal empty assign scale factor
959 if (empty) {
960 vals[j] = scale_factor;
961 rb[0][i] = 0.0;
962 }
963 }
964
965 // Global assembly
966 rb.GlobalAssemble();
967 rA.GlobalAssemble();
968
969 return scale_factor;
970
971 KRATOS_CATCH("");
972 }

981 static double GetScaleNorm(
982 const ProcessInfo& rProcessInfo,
983 const MatrixType& rA,
984 const SCALING_DIAGONAL ScalingDiagonal = SCALING_DIAGONAL::NO_SCALING
985 )
986 {
987 KRATOS_TRY
988
989 switch (ScalingDiagonal) {
990 case SCALING_DIAGONAL::NO_SCALING:
991 return 1.0;
992 case SCALING_DIAGONAL::CONSIDER_PRESCRIBED_DIAGONAL: {
993 KRATOS_ERROR_IF_NOT(rProcessInfo.Has(BUILD_SCALE_FACTOR)) << "Scale factor not defined at process info" << std::endl;
994 return rProcessInfo.GetValue(BUILD_SCALE_FACTOR);
995 }
996 case SCALING_DIAGONAL::CONSIDER_NORM_DIAGONAL:
997 return GetDiagonalNorm(rA)/static_cast<double>(Size1(rA));
998 case SCALING_DIAGONAL::CONSIDER_MAX_DIAGONAL:
999 return GetMaxDiagonal(rA);
1000 default:
1001 return GetMaxDiagonal(rA);
1002 }
1003
1004 KRATOS_CATCH("");
1005 }
