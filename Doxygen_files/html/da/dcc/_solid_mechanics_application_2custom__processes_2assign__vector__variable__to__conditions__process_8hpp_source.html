256
257 typedef void (AssignVectorVariableToConditionsProcess::*AssignmentMethodPointer) (ModelPart::ConditionType&, const Variable<array_1d<double,3> >&, const array_1d<double,3>&);
258
259 AssignmentMethodPointer AssignmentMethod = this->GetAssignmentMethod<AssignmentMethodPointer>();
260
261 const int nconditions = this->mrModelPart.GetMesh().Conditions().size();
262
263 if(nconditions != 0)
264 {
265 ModelPart::ConditionsContainerType::iterator it_begin = this->mrModelPart.GetMesh().ConditionsBegin();
266
267 #pragma omp parallel for
268 for(int i = 0; i<nconditions; i++)
269 {
270 ModelPart::ConditionsContainerType::iterator it = it_begin + i;
271
272 (this->*AssignmentMethod)(*it, rVariable, rvector_value);
273 }
274 }
