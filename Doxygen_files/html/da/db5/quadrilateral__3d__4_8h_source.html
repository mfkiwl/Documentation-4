777 Vector& rResult,
778 IntegrationMethod ThisMethod
779 ) const override
780 {
781 const unsigned int integration_points_number = msGeometryData.IntegrationPointsNumber( ThisMethod );
782 if(rResult.size() != integration_points_number)
783 {
784 rResult.resize(integration_points_number,false);
785 }
786
787 JacobiansType jacobian;
788 this->Jacobian( jacobian, ThisMethod);
789
790 for ( unsigned int pnt = 0; pnt < integration_points_number; pnt++ )
791 {
792 const double det_j = std::pow(jacobian[pnt](0,1),2)*(std::pow(jacobian[pnt](1,0),2) + std::pow(jacobian[pnt](2,0),2)) + std::pow(jacobian[pnt](1,1)*jacobian[pnt](2,0) - jacobian[pnt](1,0)*jacobian[pnt](2,1),2) - 2.0*jacobian[pnt](0,0)*jacobian[pnt](0,1)*(jacobian[pnt](1,0)*jacobian[pnt](1,1) + jacobian[pnt](2,0)*jacobian[pnt](2,1)) + std::pow(jacobian[pnt](0,0),2)*(std::pow(jacobian[pnt](1,1),2) + std::pow(jacobian[pnt](2,1),2));
793
794 if (det_j < 0.0) KRATOS_ERROR << "WARNING::NEGATIVE VALUE: NOT POSSIBLE TO EVALUATE THE JACOBIAN DETERMINANT" << std::endl;
795
796 rResult[pnt] = std::sqrt(det_j);

1515 CoordinatesArrayType& rResult,

1518 ) const

1522 for(IndexType i=0; i<this->size();i++) {
1523 X(0, i) = this->GetPoint( i ).X();

1526 }

1528 static constexpr double MaxNormPointLocalCoordinates = 300.0;
1529 static constexpr std::size_t MaxIteratioNumberPointLocalCoordinates = 500;
1530 static constexpr double MaxTolerancePointLocalCoordinates = 1.0e-8;
1531
1532 Matrix J = ZeroMatrix( 2, 2 );
1533 Matrix invJ = ZeroMatrix( 2, 2 );
1534
1535 // Starting with xi = 0
1536 rResult = ZeroVector( 3 );
1537 array_1d<double, 2> DeltaXi = ZeroVector( 2 );
1538 const array_1d<double, 3> zero_array = ZeroVector(3);
1539 array_1d<double, 3> CurrentGlobalCoords;
1540
1541 //Newton iteration:
1542 for ( IndexType k = 0; k < MaxIteratioNumberPointLocalCoordinates; k++ ) {
1543 noalias(CurrentGlobalCoords) = zero_array;
1544 this->GlobalCoordinates( CurrentGlobalCoords, rResult );
1545
1546 noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;
1547
1548 // Derivatives of shape functions
1549 Matrix shape_functions_gradients;
1550 shape_functions_gradients = ShapeFunctionsLocalGradients(shape_functions_gradients, rResult );
1551 noalias(DN) = prod(X,shape_functions_gradients);
1552
1553 noalias(J) = prod(trans(DN),DN);
1554 const array_1d<double, 2> res = prod(trans(DN), CurrentGlobalCoords);
1555
1556 // Deteminant of Jacobian
1557 const double det_j = J( 0, 0 ) * J( 1, 1 ) - J( 0, 1 ) * J( 1, 0 );
1558
1559 // Filling matrix
1560 invJ( 0, 0 ) = ( J( 1, 1 ) ) / ( det_j );
1561 invJ( 1, 0 ) = -( J( 1, 0 ) ) / ( det_j );
1562 invJ( 0, 1 ) = -( J( 0, 1 ) ) / ( det_j );
1563 invJ( 1, 1 ) = ( J( 0, 0 ) ) / ( det_j );
1564
1565 DeltaXi( 0 ) = invJ( 0, 0 ) * res[0] + invJ( 0, 1 ) * res[1];
1566 DeltaXi( 1 ) = invJ( 1, 0 ) * res[0] + invJ( 1, 1 ) * res[1];
1567
1568 rResult[0] += DeltaXi[0];
1569 rResult[1] += DeltaXi[1];
1570
1571 if ( norm_2( DeltaXi ) > MaxNormPointLocalCoordinates ) {
1572 KRATOS_WARNING_IF("Quadrilateral3D4", IsInside == false && k > 0) << "detJ =\t" << det_j << " DeltaX =\t" << DeltaXi << " stopping calculation. Iteration:\t" << k << std::endl;
1573 break;
1574 }
1575
1576 if ( norm_2( DeltaXi ) < MaxTolerancePointLocalCoordinates )
1577 break;
1578 }
1579
1580 return rResult;
