182 inline static MatrixType Compute_Pt(size_t num_nodes)
183 {
184 RealType a = RealType(num_nodes - 1) / RealType(num_nodes);
185 RealType b = -1.0 / RealType(num_nodes);
186
187 size_t num_dofs = num_nodes * 6;
188
189 MatrixType P( IdentityMatrix(num_dofs, num_dofs) );
190
191 for(size_t i = 0; i < num_nodes; i++)
192 {
193 size_t j = i * 6;
194
195 // diagonal block
196 P(j , j) = a;
197 P(j + 1, j + 1) = a;
198 P(j + 2, j + 2) = a;
199
200 // out-of-diagonal block
201 for(size_t k = i + 1; k < num_nodes; k++)
202 {
203 size_t w = k * 6;
204
205 P(j , w ) = b;
206 P(j + 1, w + 1) = b;
207 P(j + 2, w + 2) = b;
208
209 P(w , j ) = b;
210 P(w + 1, j + 1) = b;
211 P(w + 2, j + 2) = b;
212 }
213 }
214
215 return P;
216 }

257 inline static MatrixType Compute_H(const VectorType & displacements)
258 {
259 size_t num_dofs = displacements.size();
260 size_t num_nodes = num_dofs / 6;
261
262 MatrixType H( IdentityMatrix(num_dofs, num_dofs) );
263
264 MatrixType Omega(3, 3);
265 MatrixType Hi(3, 3);
266
267 for(size_t i = 0; i < num_nodes; i++)
268 {
269 size_t index = i * 6;
270 Vector3Type rv = MathUtilsType::project( displacements, MathUtilsType::range(index + 3, index + 6) );
271
272 double angle = norm_2(rv);
273
274 if(angle >= 2.0 * Globals::Pi)
275 angle = std::fmod(angle, 2.0 * Globals::Pi);
276
277 double eta;
278 if(angle < 0.05) {
279 double angle2 = angle * angle;
280 double angle4 = angle2 * angle2;
281 double angle6 = angle4 * angle2;
282 eta = 1.0 / 12.0 + 1.0 / 270.0 * angle2 + 1.0 / 30240.0 * angle4 + 1.0 / 1209600.0 * angle6;
283 }
284 else {
285 eta = ( 1.0 - 0.5 * angle * std::tan( 0.5 * Globals::Pi - 0.5 * angle ) ) / (angle * angle);
286 }
287
288 Spin( rv, Omega );
289
290 noalias( Hi ) = IdentityMatrix(3, 3);
291 noalias( Hi ) -= 0.5 * Omega;
292 noalias( Hi ) += eta * prod( Omega, Omega );
293
294 VectorType iRange = MathUtilsType::range(index + 3, index + 6);
295 MathUtilsType::project( H, iRange, iRange, Hi);
296 }
297
298 return H;
299 }

310 inline static MatrixType Compute_L(const VectorType & displacements, const VectorType & forces, const MatrixType & H)
311 {
312 size_t num_dofs = displacements.size();
313 size_t num_nodes = num_dofs / 6;
314
315 MatrixType L(num_dofs, num_dofs, 0.0);
316
317 Vector3Type rotationVector;
318 Vector3Type momentVector;
319 MatrixType Omega(3, 3);
320 MatrixType Omega2(3, 3);
321 MatrixType Li(3, 3);
322 MatrixType LiTemp1(3, 3);
323 MatrixType LiTemp2(3, 3);
324
325 for(size_t i = 0; i < num_nodes; i++)
326 {
327 size_t index = i * 6;
328 VectorType iRange = MathUtilsType::range(index + 3, index + 6);
329 noalias( rotationVector ) = MathUtilsType::project( displacements, iRange );
330 noalias( momentVector ) = MathUtilsType::project( forces, iRange );
331
332 double angle = norm_2(rotationVector);
333
334 if(angle >= 2.0 * Globals::Pi)
335 angle = std::fmod(angle, 2.0 * Globals::Pi);
336
337 double angle2 = angle * angle;
338 double angle4 = angle2 * angle2;
339 double angle6 = angle4 * angle2;
340
341 double eta;
342 double mu;
343 if(angle < 0.05) {
344 eta = 1.0 / 12.0 + angle2 / 270.0 + angle4 / 30240.0 + angle6 / 1209600.0;
345 mu = 1.0 / 360.0 + angle2 / 7560.0 + angle4 / 201600.0 + angle6 / 5987520.0;
346 }
347 else {
348 eta = ( 1.0 - 0.5 * angle * std::tan( 0.5 * Globals::Pi - 0.5 * angle ) ) / (angle * angle);
349 double sin_h_angle = std::sin(0.5 * angle);
350 mu = ( angle2 + 4.0 * std::cos(angle) + angle * std::sin(angle) - 4.0 ) / ( 4.0 * angle4 * sin_h_angle * sin_h_angle );
351 }
352
353 Spin( rotationVector, Omega );
354 noalias( Omega2 ) = prod( Omega, Omega );
355
356 noalias( LiTemp2 ) = outer_prod( momentVector, rotationVector );
357
358 noalias( Li ) = inner_prod( rotationVector, momentVector ) * IdentityMatrix(3, 3);
359 noalias( Li ) += outer_prod( rotationVector, momentVector );
360 noalias( Li ) -= LiTemp2;
361
362 noalias( LiTemp1 ) = mu * prod( Omega2, LiTemp2 );
363 Spin( momentVector, LiTemp2, 0.5 );
364 noalias( LiTemp1 ) -= LiTemp2;
365
366 noalias( LiTemp1 ) += eta * Li;
367
368 noalias( Li ) = prod( LiTemp1, MathUtilsType::project( H, iRange, iRange ) );
369
370 MathUtilsType::project( L, iRange, iRange, Li);
371 }
372
373 return L;
374 }
