61 void ExecuteInitializeSolutionStep() override
62 {
63 KRATOS_TRY
64
65 if (mrModelPart.NumberOfNodes() > 0) {
66 const Variable<double> &var = KratosComponents< Variable<double> >::Get(mVariableName);
67
68 if (mIsSeepage) {
69 block_for_each(mrModelPart.Nodes(), [&var, this](Node<3>& rNode) {
70 const double pressure = CalculatePressure(rNode);
71
72 if ((PORE_PRESSURE_SIGN_FACTOR * pressure) < 0.0) {
73 rNode.FastGetSolutionStepValue(var) = pressure;
74 if (mIsFixed) rNode.Fix(var);
75 } else {
76 if (mIsFixedProvided) rNode.Free(var);
77 }
78 });
79 } else {
80 block_for_each(mrModelPart.Nodes(), [&var, this](Node<3>& rNode) {
81 if (mIsFixed) rNode.Fix(var);
82 else if (mIsFixedProvided) rNode.Free(var);
83
84 const double pressure = CalculatePressure(rNode);
85
86 if ((PORE_PRESSURE_SIGN_FACTOR * pressure) < mPressureTensionCutOff) {
87 rNode.FastGetSolutionStepValue(var) = pressure;
88 } else {
89 rNode.FastGetSolutionStepValue(var) = mPressureTensionCutOff;
90 }
91 });
92 }
93 }
