148 static void CalculateDamageParameter(
149 ConstitutiveLaw::Parameters& rValues,
150 double& rAParameter,
151 const double CharacteristicLength
152 )
153 {
154 const Properties& r_material_properties = rValues.GetMaterialProperties();
155
156 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
157 const double young_modulus = r_material_properties[YOUNG_MODULUS];
158 const double yield_compression = r_material_properties.Has(YIELD_STRESS) ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
159
160 if (r_material_properties[SOFTENING_TYPE] == static_cast<int>(SofteningType::Exponential)) {
161 rAParameter = 1.00 / (fracture_energy * young_modulus / (CharacteristicLength * std::pow(yield_compression, 2)) - 0.5);
162 KRATOS_ERROR_IF(rAParameter < 0.0) << "Fracture energy is too low, increase FRACTURE_ENERGY..." << std::endl;
163 } else { // linear
164 rAParameter = -std::pow(yield_compression, 2) / (2.0 * young_modulus * fracture_energy / CharacteristicLength);
165 }

217 static int Check(const Properties& rMaterialProperties)
218 {
219 if (!rMaterialProperties.Has(YIELD_STRESS)) {
220 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
221 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
222
223 const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
224 const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
225
226 KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
227 KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
228 } else {
229 const double yield_stress = rMaterialProperties[YIELD_STRESS];
230
231 KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
232 }
233 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRACTURE_ENERGY)) << "FRACTURE_ENERGY is not a defined value" << std::endl;
234 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "YOUNG_MODULUS is not a defined value" << std::endl;
235
236 return TPlasticPotentialType::Check(rMaterialProperties);
237 }
