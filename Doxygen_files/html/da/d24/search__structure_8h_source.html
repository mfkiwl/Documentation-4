99template< std::size_t TDimension, class TPointType >
100bool PointInBox( TPointType const& BoxMinPoint, TPointType const& BoxMaxPoint, TPointType const& ThisPoint )
101{
102 for(std::size_t i = 0 ; i < TDimension ; i++)
103 if( ThisPoint[i] < BoxMinPoint[i] || ThisPoint[i] > BoxMaxPoint[i] )
104 return false;
105 return true;

108template< std::size_t TDimension, class TPointType >
109class SquaredDistanceFunction
110{
111public:
112 double operator()( TPointType const& p1, TPointType const& p2 )
113 {
114 double tmp = p1[0] - p2[0];
115 double dist = tmp*tmp;
116 for( std::size_t i = 1 ; i < TDimension ; i++)
117 {
118 tmp = p1[i] - p2[i];
119 dist += tmp*tmp;
120 }
121 return dist;
122 }
123};

255 ~SubBinAxis() {}

260 void Set(IndexType const& Min_, IndexType const& Max_, IndexType const& MaxSize_, IndexType const& Block_)

265 Max = std::min( Max_, MaxSize-1 );
266 }

281 return Max;
282 }
283 SizeType Size()
284 {
285 return static_cast<SizeType>(Max-Min);
286 }

288 {
289 if( Min > static_cast<IndexType>(0) ) Min--;
290 if( Max < MaxSize-1 ) Max++;
291 return *this;
292 }
293 SubBinAxis const& operator--()
294 {
295 Min++;
296 Max--;
297 return *this;
298 }
