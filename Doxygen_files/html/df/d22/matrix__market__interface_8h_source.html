80template <typename CompressedMatrixType> inline bool ReadMatrixMarketMatrix(const char *FileName, CompressedMatrixType &M)
81{
82 typedef typename CompressedMatrixType::value_type ValueType;
83
84 // Open MM file for reading
85 FILE *f = fopen(FileName, "r");
86
87 if (f == NULL)
88 {
89 printf("ReadMatrixMarketMatrix(): unable to open %s.\n", FileName);
90 return false;
91 }
92
93 // Process MM file header
94 MM_typecode mm_code;
95
96 if (mm_read_banner(f, &mm_code) != 0)
97 {
98 printf("ReadMatrixMarketMatrix(): unable to read MatrixMarket banner.\n");
99 fclose(f);
100 return false;
101 }
102
103 if (!mm_is_valid(mm_code))
104 {
105 printf("ReadMatrixMarketMatrix(): invalid MatrixMarket banner.\n");
106 fclose(f);
107 return false;
108 }
109
110 // Check for supported types of MM file
111 if (!(mm_is_coordinate(mm_code) && mm_is_sparse(mm_code)))
112 {
113 printf("ReadMatrixMarketMatrix(): unspported MatrixMarket type, \"%s\".\n", mm_typecode_to_str(mm_code));
114 fclose(f);
115 return false;
116 }
117
118 // Read MM dimensions and NNZ
119 int size1, size2, nnz;
120
121 if (mm_read_mtx_crd_size(f, &size1, &size2, &nnz) != 0)
122 {
123 printf("ReadMatrixMarketMatrix(): cannot read dimensions and NNZ.\n");
124 fclose(f);
125 return false;
126 }
127
128 // Allocate temporary arrays
129 int *I = new int[nnz];
130 int *J = new int[nnz];
131 ValueType *V = new ValueType[nnz];
132
133 // Check if matrix type matches MM file
134 if (!IsCorrectType<ValueType>(mm_code))
135 {
136 printf("ReadMatrixMarketMatrix(): MatrixMarket type, \"%s\" does not match provided matrix type.\n", mm_typecode_to_str(mm_code));
137 fclose(f);
138 return false;
139 }
140
141 // Read MM file
142
143 // Pattern file, only non-zero structure
144 if (mm_is_pattern(mm_code))
145 for (int i = 0; i < nnz; i++)
146 {
147 if (fscanf(f, "%d %d", &I[i], &J[i]) != 2)
148 {
149 printf("ReadMatrixMarketMatrix(): invalid data.\n");
150 fclose(f);
151
152 delete[] I;
153 delete[] J;
154 delete[] V;
155
156 return false;
157 }
158
159 // Adjust to 0-based
160 I[i]--;
161 J[i]--;
162
163 // Set all values to 1.00
164 V[i] = 1.00;
165 }
166 else
167 for (int i = 0; i < nnz; i++)
168 {
169 if (! ReadMatrixMarketMatrixEntry(f, I[i], J[i], V[i]))
170 {
171 printf("ReadMatrixMarketMatrix(): invalid data.\n");
172 fclose(f);
173
174 delete[] I;
175 delete[] J;
176 delete[] V;
177
178 return false;
179 }
180
181 // Adjust to 0-based
182 I[i]--;
183 J[i]--;
184 }
185
186 fclose(f);
187
188 // Second stage
189 int *nz = new int[size1];
190
191 for (int i = 0; i < size1; i++)
192 nz[i] = 0;
193
194 // Count non-zeros on each line
195 if (mm_is_symmetric(mm_code))
196 for (int i = 0; i < nnz; i++)
197 {
198 if (I[i] == J[i])
199 nz[I[i]]++;
200 else
201 {
202 nz[I[i]]++;
203 nz[J[i]]++;
204 }
205 }
206 else
207 for (int i = 0; i < nnz; i++)
208 nz[I[i]]++;
209
210 // Find out total number of non-zeros
211 int nnz2;
212
213 if (mm_is_symmetric(mm_code))
214 {
215 int diagonals = 0;
216
217 for (int i = 0; i < nnz; i++)
218 if (I[i] == J[i])
219 diagonals++;
220
221 nnz2 = diagonals + 2 * (nnz - diagonals);
222 }
223 else
224 nnz2 = nnz;
225
226 // Fill in an almost-CSR data structure
227 int *filled = new int[size1];
228 int *indices = new int[size1];
229 int *columns = new int[nnz2];
230 ValueType *values = new ValueType[nnz2];
231
232 indices[0] = 0;
233 for (int i = 1; i < size1; i++)
234 indices[i] = indices[i - 1] + nz[i - 1];
235
236 for (int i = 0; i < size1; i++)
237 filled[i] = 0;
238
239 if (mm_is_symmetric(mm_code))
240 for (int i = 0; i < nnz; i++)
241 if (I[i] == J[i])
242 {
243 int index;
244
245 index = indices[I[i]] + filled[I[i]];
246 columns[index] = J[i];
247 values[index] = V[i];
248 filled[I[i]]++;
249 }
250 else
251 {
252 int index;
253
254 index = indices[I[i]] + filled[I[i]];
255 columns[index] = J[i];
256 values[index] = V[i];
257 filled[I[i]]++;
258
259 index = indices[J[i]] + filled[J[i]];
260 columns[index] = I[i];
261 values[index] = V[i];
262 filled[J[i]]++;
263 }
264 else
265 for (int i = 0; i < nnz; i++)
266 {
267 int index;
268
269 index = indices[I[i]] + filled[I[i]];
270 columns[index] = J[i];
271 values[index] = V[i];
272 filled[I[i]]++;
273 }
274
275 // Create the matrix
276 CompressedMatrixType *m = new CompressedMatrixType(size1, size2, nnz2);
277
278 int k = 0;
279
280 for (int i = 0; i < size1; i++)
281 for (int j = 0; j < nz[i]; j++)
282 (*m)(i, columns[indices[i] + j]) = values[k++];
283
284 M = *m;
285
286 delete[] I;
287 delete[] J;
288 delete[] V;
289
290 delete[] filled;
291 delete[] indices;
292 delete[] columns;
293 delete[] values;
294 delete[] nz;
295
296 delete m;
297
298 return true;
299}

336

354
355 typename CompressedMatrixType::iterator1 a_iterator = M.begin1();
356
357 for (unsigned int i = 0; i < M.size1(); i++)

364 {
365 if (a_iterator.index1() >= row_iterator.index2())
366 nnz++;
367 }
368
369 a_iterator++;
370 }
371 }
372 else
373 nnz = M.nnz();
374
375 // Write MM file sizes
376 mm_write_mtx_crd_size(f, M.size1(), M.size2(), nnz);
377
378 if (Symmetric)
379 {
380 typename CompressedMatrixType::iterator1 a_iterator = M.begin1();

495 fclose(f);
496 return false;
497 }
498

500 if (size2 != 1)
501 {
502 printf("ReadMatrixMarketVector(): not a N x 1 array.\n");
503 fclose(f);
504 return false;
505 }
506
507 VectorType *v = new VectorType(size1);
508 ValueType T;
509
510 // Check if vector type matches MM file
511 if (!IsCorrectType<ValueType>(mm_code))
512 {

514 fclose(f);
515 return false;
516 }
517
518 // Read MM file
519
520 for (int i = 0; i < size1; i++)
521 {
522 if (! ReadMatrixMarketVectorEntry(f, T))
523 {
524 printf("ReadMatrixMarketVector(): invalid data.\n");
525 fclose(f);
526
527 return false;
528 }
529
530 (*v)(i) = T;
531 }
532
533 fclose(f);
534
535 V = *v;
536
537 delete v;
