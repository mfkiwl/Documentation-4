76  // Predict Displacements on free nodes and update Acceleration, Velocity and DtPressure
77  block_for_each(rModelPart.Nodes(), [&](Node& rNode){
78  if (rNode.IsFixed(ACCELERATION_X))
79  {
80  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_X, 1);
81  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_X, 1);
82  const double &PreviousAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_X, 1);
83  const double &CurrentAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_X);
84 
85  rNode.FastGetSolutionStepValue(DISPLACEMENT_X) = PreviousDisplacement
86  + mDeltaTime * PreviousVelocity
87  + mDeltaTime * mDeltaTime
88  * ( ( 0.5 - mBeta) * PreviousAcceleration + mBeta * CurrentAcceleration );
89  }
90  else if (rNode.IsFixed(VELOCITY_X))
91  {
92  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_X, 1);
93  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_X, 1);
94  const double &CurrentVelocity = rNode.FastGetSolutionStepValue(VELOCITY_X);
95  rNode.FastGetSolutionStepValue(DISPLACEMENT_X) = PreviousDisplacement
96  + mDeltaTime*(mBeta/mGamma*(CurrentVelocity - PreviousVelocity)
97  + PreviousVelocity);
98  }
99  else if (!rNode.IsFixed(DISPLACEMENT_X))
100  {
101  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_X, 1);
102  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_X, 1);
103  const double &PreviousAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_X, 1);
104 
105  rNode.FastGetSolutionStepValue(DISPLACEMENT_X) = PreviousDisplacement
106  + mDeltaTime * PreviousVelocity
107  + 0.5 * mDeltaTime * mDeltaTime
108  * PreviousAcceleration;
109  }
110 
111  if (rNode.IsFixed(ACCELERATION_Y))
112  {
113  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_Y, 1);
114  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_Y, 1);
115  const double &PreviousAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_Y, 1);
116  const double &CurrentAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_Y);
117

120  + mDeltaTime * mDeltaTime
121  * ( ( 0.5 - mBeta) * PreviousAcceleration + mBeta * CurrentAcceleration );
122  }
123  else if (rNode.IsFixed(VELOCITY_Y))
124  {
125  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_Y, 1);
126  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_Y, 1);

131  }
132  else if (!rNode.IsFixed(DISPLACEMENT_Y))

140  + 0.5 * mDeltaTime * mDeltaTime
141  * PreviousAcceleration;
142  }
143 
144  // For 3D cases
145  if (rNode.HasDofFor(DISPLACEMENT_Z))
146  {
147  if (rNode.IsFixed(ACCELERATION_Z))
148  {
149  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_Z, 1);
150  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_Z, 1);
151  const double &PreviousAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_Z, 1);
152  const double &CurrentAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_Z);
153 
154  rNode.FastGetSolutionStepValue(DISPLACEMENT_Z) = PreviousDisplacement
155  + mDeltaTime * PreviousVelocity
156  + mDeltaTime * mDeltaTime
157  * ( ( 0.5 - mBeta) * PreviousAcceleration + mBeta * CurrentAcceleration );
158  }
159  else if (rNode.IsFixed(VELOCITY_Z))
160  {
161  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_Z, 1);
162  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_Z, 1);
163  const double &CurrentVelocity = rNode.FastGetSolutionStepValue(VELOCITY_Z);
164  rNode.FastGetSolutionStepValue(DISPLACEMENT_Z) = PreviousDisplacement
165  + mDeltaTime*(mBeta/mGamma*(CurrentVelocity - PreviousVelocity)
166  + PreviousVelocity);
167  }
168  else if (!rNode.IsFixed(DISPLACEMENT_Z))
169  {
170  const double &PreviousDisplacement = rNode.FastGetSolutionStepValue(DISPLACEMENT_Z, 1);
171  const double &PreviousVelocity = rNode.FastGetSolutionStepValue(VELOCITY_Z, 1);
172  const double &PreviousAcceleration = rNode.FastGetSolutionStepValue(ACCELERATION_Z, 1);
173 
174  rNode.FastGetSolutionStepValue(DISPLACEMENT_Z) = PreviousDisplacement
175  + mDeltaTime * PreviousVelocity
176  + 0.5 * mDeltaTime * mDeltaTime
177  * PreviousAcceleration;
178  }
179  }
180 
181  noalias(rNode.FastGetSolutionStepValue(ACCELERATION)) = 1.0/(mBeta*mDeltaTime*mDeltaTime)
182  * ( (rNode.FastGetSolutionStepValue(DISPLACEMENT) - rNode.FastGetSolutionStepValue(DISPLACEMENT,1))
183  - mDeltaTime
184  * rNode.FastGetSolutionStepValue(VELOCITY, 1)
185  - (0.5-mBeta) * mDeltaTime * mDeltaTime
186  * rNode.FastGetSolutionStepValue(ACCELERATION,1));
187 
188  noalias(rNode.FastGetSolutionStepValue(VELOCITY)) = rNode.FastGetSolutionStepValue(VELOCITY, 1)
189  + (1.0-mGamma) * mDeltaTime
190  * rNode.FastGetSolutionStepValue(ACCELERATION,1)
191  + mGamma * mDeltaTime
192  * rNode.FastGetSolutionStepValue(ACCELERATION);
193 
194  const double DeltaPressure = rNode.FastGetSolutionStepValue(WATER_PRESSURE)
195  - rNode.FastGetSolutionStepValue(WATER_PRESSURE, 1);
196 
197  rNode.FastGetSolutionStepValue(DT_WATER_PRESSURE) = 1.0/(mTheta*mDeltaTime)
198  * ( DeltaPressure
199  - (1.0-mTheta) * mDeltaTime
200  * rNode.FastGetSolutionStepValue(DT_WATER_PRESSURE,1));
201  });
