126 static inline bool Intersection(const PointerType& rObj_1, const PointerType& rObj_2){
127 array_1d<double, 3> rObj_2_to_rObj_1 = rObj_1->GetGeometry().GetPoint(0) - rObj_2->GetGeometry().GetPoint(0);
128 double distance_2 = inner_prod(rObj_2_to_rObj_1, rObj_2_to_rObj_1);

131 const double& radius_2 = rObj_2->GetGeometry()[0].FastGetSolutionStepValue(RADIUS);

155 static inline bool IntersectionBox(const PointerType& rObject, const PointType& rLowPoint, const PointType& rHighPoint)
156 {
157
158// double separation_from_particle_radius_ratio = 0.1;
159
160 array_1d<double, 3> center_of_particle = rObject->GetGeometry().GetPoint(0);
161
162 const double& radius = rObject->GetGeometry()[0].FastGetSolutionStepValue(RADIUS);
163
164 bool intersect = (rLowPoint[0] - radius <= center_of_particle[0] && rLowPoint[1] - radius <= center_of_particle[1] && rLowPoint[2] - radius <= center_of_particle[2] &&
165 rHighPoint[0] + radius >= center_of_particle[0] && rHighPoint[1] + radius >= center_of_particle[1] && rHighPoint[2] + radius >= center_of_particle[2]);
166
167 return intersect;

183 static inline void Distance(const PointerType& rObj_1, const PointerType& rObj_2, double& distance)
184 {
185 array_1d<double, 3> center_of_particle1 = rObj_1->GetGeometry().GetPoint(0);
186 array_1d<double, 3> center_of_particle2 = rObj_2->GetGeometry().GetPoint(0);
187
188 distance = sqrt((center_of_particle1[0] - center_of_particle2[0]) * (center_of_particle1[0] - center_of_particle2[0]) +
189 (center_of_particle1[1] - center_of_particle2[1]) * (center_of_particle1[1] - center_of_particle2[1]) +
190 (center_of_particle1[2] - center_of_particle2[2]) * (center_of_particle1[2] - center_of_particle2[2]) );
191 }
