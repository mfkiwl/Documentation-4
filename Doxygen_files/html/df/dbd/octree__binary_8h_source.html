254 void Constrain2To1() {
255 std::vector<cell_type*> leaves;
256 std::vector<cell_type*> next_leaves;
257
258 //when the function will be at upper level (in mesher instead of octree) this vector (leaves) should be passed and copied instead of recomputed
259 GetAllLeavesVector(leaves);
260
261 for (char i_level = MIN_LEVEL; i_level < ROOT_LEVEL - 1; i_level++) {
262 for (std::size_t i_cell = 0; i_cell < leaves.size(); i_cell++) {
263 cell_type* current_leaf = leaves[i_cell];
264 if (current_leaf->GetLevel() == i_level) {
265 key_type neighbour_key[3];
266 //18 is the number of neighbours counting faces and edges of the cell
267 for (int i_direction = 0; i_direction < 18; i_direction++) {
268 if (current_leaf->GetNeighbourKey(i_direction, neighbour_key)) {
269 cell_type* neighbour_cell = pGetCell(neighbour_key);
270 if (neighbour_cell->GetLevel() > i_level + 1) {
271 cell_type* temp_neighbour_cell = neighbour_cell;
272 for (char j_level = neighbour_cell->GetLevel(); j_level > i_level + 1; j_level--) {
273 SubdivideCell(temp_neighbour_cell);
274 temp_neighbour_cell->TransferObjectsToSonsNormalized();
275 std::size_t child_index = temp_neighbour_cell->GetChildIndex(neighbour_key[0], neighbour_key[1], neighbour_key[2]);
276 for (std::size_t j = 0; j < CHILDREN_NUMBER; j++) {
277 if (j != child_index) {
278 next_leaves.push_back(temp_neighbour_cell->GetChildren() + j);
279 }
280 }
281 temp_neighbour_cell = temp_neighbour_cell->GetChildren() + child_index;
282 if (j_level == neighbour_cell->GetLevel() - 1) // the last loop we add all the child as leaf
283 next_leaves.push_back(temp_neighbour_cell);
284 }
285 }
286 }
287 }
288 } else if (current_leaf->IsLeaf()) { // becuase it may be divided
289 next_leaves.push_back(current_leaf);
290 }
291 }
292 leaves.swap(next_leaves);
293 next_leaves.clear();
294 }

342 void GetLeavesInBoundingBoxNormalized(const double* coord1, const double* coord2,
343 std::vector<cell_type*>& leaves) const
344 {
345 key_type min_x_key = CalcKeyNormalized(coord1[0]);
346 key_type min_y_key = CalcKeyNormalized(coord1[1]);
347 key_type min_z_key = CalcKeyNormalized(coord1[2]);
348
349 key_type max_x_key = CalcKeyNormalized(coord2[0]);
350 key_type max_y_key = CalcKeyNormalized(coord2[1]);
351 key_type max_z_key = CalcKeyNormalized(coord2[2]);
352
353 key_type delta_x = min_x_key^max_x_key;
354 key_type delta_y = min_y_key^max_y_key;
355 key_type delta_z = min_z_key^max_z_key;
356
357 // finding the level of the cell containing the entire region
358 std::size_t min_level_1 = ROOT_LEVEL;
359 std::size_t min_level_2 = ROOT_LEVEL;
360 std::size_t min_level = ROOT_LEVEL;
361
362 const std::size_t one = 1;
363 while (!(delta_x & (one << min_level_1)) && (min_level_1 > MIN_LEVEL)) min_level_1--;
364 while (!(delta_y & (one << min_level_2)) && (min_level_2 > min_level_1)) min_level_2--;
365 while (!(delta_z & (one << min_level)) && (min_level > min_level_2)) min_level--;
366 min_level++;
367
368 cell_type* range_cell = root_;
369
370 for (std::size_t i = ROOT_LEVEL; i > min_level; i--) {
371 if (range_cell->IsLeaf()) {
372 break;
373 }
374 range_cell = range_cell->pGetChild(min_x_key, min_y_key, min_z_key);
375
376 }
377 // Now we have the cell (or leaf) containing the entire range and from now on we have to gather the leaves
378 std::vector<cell_type*> cells_stack;
379 cells_stack.push_back(range_cell);
380 while (!cells_stack.empty()) {
381 cell_type* cell = cells_stack.back();
382 cells_stack.pop_back();
383 if (cell->HasChildren()) {
384 for (std::size_t i = 0; i < CHILDREN_NUMBER; i++){
385 //abel. to be optimized
386 cell_type* child=cell->pGetChild(i);
387
388 double low[3];
389 double high[3];
390 child->GetMinPoint(low);
391 child->GetMaxPoint(high);
392 if (Collides(coord1, coord2, low, high))
393 cells_stack.push_back(cell->pGetChild(i));
394 }
395 } else
396 leaves.push_back(cell);
397 }
398
399
400 return;
401 }

589 key_type z_key = CalcKeyNormalized(coord[2]);

594 // I'm assuming that the desired size is also normalized to the [0,1] span

596
597 if(scaled_size < (1 << MIN_LEVEL))

602 SubdivideCell(cell);
603 }
604 cell = cell->pGetChild(x_key, y_key, z_key);
605 }
606
607 return 0;
