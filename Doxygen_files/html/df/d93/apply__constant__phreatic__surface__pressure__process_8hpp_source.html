106 void ExecuteInitialize() override
107 {
108 KRATOS_TRY
109
110 if (mrModelPart.NumberOfNodes() > 0) {
111 const Variable<double> &var = KratosComponents< Variable<double> >::Get(mVariableName);
112 Vector3 direction=ZeroVector(3);
113 direction[mGravityDirection] = 1.0;
114
115 if (mIsSeepage) {
116 block_for_each(mrModelPart.Nodes(), [&var, &direction, this](Node<3>& rNode) {
117 double distance = 0.0;
118 double d = 0.0;
119 for (unsigned int j=0; j < rNode.Coordinates().size(); ++j) {
120 distance += mNormalVector[j] * rNode.Coordinates()[j];
121 d += mNormalVector[j]*direction[j];
122 }
123 distance = -(distance - mEqRHS) / d;
124
125 const double pressure = - PORE_PRESSURE_SIGN_FACTOR * mSpecificWeight * distance;
126
127 if ((PORE_PRESSURE_SIGN_FACTOR * pressure) < 0) {
128 rNode.FastGetSolutionStepValue(var) = pressure;
129 if (mIsFixed) rNode.Fix(var);
130 } else {
131 if (mIsFixedProvided) rNode.Free(var);
132 }
133 });
134 } else {
135 block_for_each(mrModelPart.Nodes(), [&var, &direction, this](Node<3>& rNode) {
136 if (mIsFixed) rNode.Fix(var);
137 else if (mIsFixedProvided) rNode.Free(var);
138
139 double distance = 0.0;
140 double d = 0.0;
141 for (unsigned int j=0; j < rNode.Coordinates().size(); ++j) {
142 distance += mNormalVector[j] * rNode.Coordinates()[j];
143 d += mNormalVector[j]*direction[j];
144 }
145 distance = -(distance - mEqRHS) / d;
146
147 const double pressure = - PORE_PRESSURE_SIGN_FACTOR * mSpecificWeight * distance;
148
149 if ((PORE_PRESSURE_SIGN_FACTOR * pressure) < mPressureTensionCutOff) {
150 rNode.FastGetSolutionStepValue(var) = pressure;
151 } else {
152 rNode.FastGetSolutionStepValue(var) = mPressureTensionCutOff;
153 }
154 });
155 }
156
157 }

203 void calculateEquationParameters()
204 {
205 Vector3 v1;
206 Vector3 v2;
207 v1 = mSecondReferenceCoordinate - mFirstReferenceCoordinate;
208 v2 = mThirdReferenceCoordinate - mFirstReferenceCoordinate;
209
210 mNormalVector[0] = v1[1] * v2[2] - v1[2] * v2[1];
211 mNormalVector[1] = v1[2] * v2[0] - v1[0] * v2[2];
212 mNormalVector[2] = v1[0] * v2[1] - v1[1] * v2[0];
213
214 double sum=0;
215 for (unsigned int i=0; i < mNormalVector.size(); ++i)
216 {
217 sum += mNormalVector[i]*mNormalVector[i];
218 }
219 double norm = std::sqrt(sum);
220 if (!(norm > 0.0))
221 KRATOS_ERROR << "Normal vector to phreatic surface has zero size!"
222 << std::endl;
223
224 mEqRHS = 0.0;
225 for (unsigned int i=0; i < mFirstReferenceCoordinate.size(); ++i)
226 {
227 mEqRHS += mNormalVector[i]*mFirstReferenceCoordinate[i];
228 }
229 }
