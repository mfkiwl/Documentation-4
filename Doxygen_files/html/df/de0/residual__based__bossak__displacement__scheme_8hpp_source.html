209  void Predict(
210  ModelPart& rModelPart,
211  DofsArrayType& rDofSet,
212  TSystemMatrixType& rA,
213  TSystemVectorType& rDx,
214  TSystemVectorType& rb
215  ) override
216  {

218 
219  // The current process info
220  const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
221  const double delta_time = r_current_process_info[DELTA_TIME];
222 
223  // Updating time derivatives (nodally for efficiency)
224  const auto it_node_begin = rModelPart.Nodes().begin();
225

227  KRATOS_ERROR_IF_NOT(it_node_begin->HasDofFor(DISPLACEMENT_X)) << "ResidualBasedBossakDisplacementScheme:: DISPLACEMENT is not added" << std::endl;

232  // Getting dimension
233  KRATOS_WARNING_IF("ResidualBasedBossakDisplacementScheme", !r_current_process_info.Has(DOMAIN_SIZE)) << "DOMAIN_SIZE not defined. Please define DOMAIN_SIZE. 3D case will be assumed" << std::endl;

235

237  array_1d<double, 3 > delta_displacement;
238  std::array<bool, 3> predicted = {false, false, false};
239  const std::array<const Variable<ComponentType>*, 3> disp_components = {&DISPLACEMENT_X, &DISPLACEMENT_Y, &DISPLACEMENT_Z};
240  const std::array<const Variable<ComponentType>*, 3> vel_components = {&VELOCITY_X, &VELOCITY_Y, &VELOCITY_Z};
241  const std::array<const Variable<ComponentType>*, 3> accel_components = {&ACCELERATION_X, &ACCELERATION_Y, &ACCELERATION_Z};

244  TLSContainerType aux_TLS = std::make_tuple(delta_displacement, predicted);
245 
246  block_for_each(rModelPart.Nodes(), aux_TLS, [&](Node<3>& rNode, TLSContainerType& rAuxTLS){
247  auto& r_delta_displacement = std::get<0>(rAuxTLS);
248  auto& r_predicted = std::get<1>(rAuxTLS);
249 
250  for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
251  r_predicted[i_dim] = false;
252  }
253 
254  // Predicting: NewDisplacement = r_previous_displacement + r_previous_velocity * delta_time;
255  const array_1d<double, 3>& r_previous_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION, 1);
256  const array_1d<double, 3>& r_previous_velocity = rNode.FastGetSolutionStepValue(VELOCITY, 1);
257  const array_1d<double, 3>& r_previous_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT, 1);
258  array_1d<double, 3>& r_current_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION);
259  array_1d<double, 3>& r_current_velocity = rNode.FastGetSolutionStepValue(VELOCITY);
260  array_1d<double, 3>& r_current_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT);
261 
262  if (accelpos > -1) {
263  for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
264  if (rNode.GetDof(*accel_components[i_dim], accelpos + i_dim).IsFixed()) {
265  r_delta_displacement[i_dim] = (r_current_acceleration[i_dim] + mBossak.c3 * r_previous_acceleration[i_dim] + mBossak.c2 * r_previous_velocity[i_dim])/mBossak.c0;
266  r_current_displacement[i_dim] = r_previous_displacement[i_dim] + r_delta_displacement[i_dim];
267  r_predicted[i_dim] = true;
268  }
269  }
270  }
271  if (velpos > -1) {
272  for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
273  if (rNode.GetDof(*vel_components[i_dim], velpos + i_dim).IsFixed() && !r_predicted[i_dim]) {
274  r_delta_displacement[i_dim] = (r_current_velocity[i_dim] + mBossak.c4 * r_previous_velocity[i_dim] + mBossak.c5 * r_previous_acceleration[i_dim])/mBossak.c1;
275  r_current_displacement[i_dim] = r_previous_displacement[i_dim] + r_delta_displacement[i_dim];
276  r_predicted[i_dim] = true;
277  }
278  }
279  }
280  for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
281  if (!rNode.GetDof(*disp_components[i_dim], disppos + i_dim).IsFixed() && !r_predicted[i_dim]) {
282  r_current_displacement[i_dim] = r_previous_displacement[i_dim] + delta_time * r_previous_velocity[i_dim] + 0.5 * std::pow(delta_time, 2) * r_previous_acceleration[i_dim];
283  }
284  }
285 
286  // Updating time derivatives ::: Please note that displacements and its time derivatives can not be consistently fixed separately
287  noalias(r_delta_displacement) = r_current_displacement - r_previous_displacement;
288  UpdateVelocity(r_current_velocity, r_delta_displacement, r_previous_velocity, r_previous_acceleration);
289  UpdateAcceleration(r_current_acceleration, r_delta_displacement, r_previous_velocity, r_previous_acceleration);
290  });
291 
292  KRATOS_CATCH( "" );
293  }

317  const double delta_time = r_current_process_info[DELTA_TIME];
318 
319  // Initializing Bossak constants
320  mBossak.c0 = ( 1.0 / (mBossak.beta * delta_time * delta_time) );
321  mBossak.c1 = ( mBossak.gamma / (mBossak.beta * delta_time) );
322  mBossak.c2 = ( 1.0 / (mBossak.beta * delta_time) );
323  mBossak.c3 = ( 0.5 / (mBossak.beta) - 1.0 );
324  mBossak.c4 = ( (mBossak.gamma / mBossak.beta) - 1.0 );
325  mBossak.c5 = ( delta_time * 0.5 * ( ( mBossak.gamma / mBossak.beta ) - 2.0 ) );
326

328  const auto it_node_begin = rModelPart.Nodes().begin();
329 
330  // Getting dimension
331  KRATOS_WARNING_IF("ResidualBasedBossakDisplacementScheme", !r_current_process_info.Has(DOMAIN_SIZE)) << "DOMAIN_SIZE not defined. Please define DOMAIN_SIZE. 3D case will be assumed" << std::endl;
332  const std::size_t dimension = r_current_process_info.Has(DOMAIN_SIZE) ? r_current_process_info.GetValue(DOMAIN_SIZE) : 3;
333 
334  // Getting position
335  const int velpos = it_node_begin->HasDofFor(VELOCITY_X) ? static_cast<int>(it_node_begin->GetDofPosition(VELOCITY_X)) : -1;
336  const int accelpos = it_node_begin->HasDofFor(ACCELERATION_X) ? static_cast<int>(it_node_begin->GetDofPosition(ACCELERATION_X)) : -1;
337 
338  std::array<bool, 3> fixed = {false, false, false};
339  const std::array<const Variable<ComponentType>*, 3> disp_components = {&DISPLACEMENT_X, &DISPLACEMENT_Y, &DISPLACEMENT_Z};
340  const std::array<const Variable<ComponentType>*, 3> vel_components = {&VELOCITY_X, &VELOCITY_Y, &VELOCITY_Z};
341  const std::array<const Variable<ComponentType>*, 3> accel_components = {&ACCELERATION_X, &ACCELERATION_Y, &ACCELERATION_Z};
342 
343  block_for_each(rModelPart.Nodes(), fixed, [&](Node<3>& rNode, std::array<bool,3>& rFixedTLS){
344  for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
345  rFixedTLS[i_dim] = false;
346  }
347 
348  if (accelpos > -1) {
349  for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
350  if (rNode.GetDof(*accel_components[i_dim], accelpos + i_dim).IsFixed()) {
351  rNode.Fix(*disp_components[i_dim]);
352  rFixedTLS[i_dim] = true;
353  }
354  }
355  }
356  if (velpos > -1) {
357  for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
358  if (rNode.GetDof(*vel_components[i_dim], velpos + i_dim).IsFixed() && !rFixedTLS[i_dim]) {
359  rNode.Fix(*disp_components[i_dim]);
360  }
361  }
362  }
363  });
