191 void CalculateBossakCoefficients()
192 {
193 mBossak.beta = (1.0 - mBossak.alpha) * (1.0 - mBossak.alpha) * mNewmark.beta;
194 mBossak.gamma = mNewmark.gamma - mBossak.alpha;

244 void Predict(
245 ModelPart& rModelPart,
246 DofsArrayType& rDofSet,
247 TSystemMatrixType& rA,
248 TSystemVectorType& rDx,
249 TSystemVectorType& rb
250 ) override
251 {
252 KRATOS_TRY;
253
254 // The current process info
255 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
256 const double delta_time = r_current_process_info[DELTA_TIME];
257
258 // Updating time derivatives (nodally for efficiency)
259 const auto it_node_begin = rModelPart.Nodes().begin();
260
261 // Getting position
262 KRATOS_ERROR_IF_NOT(it_node_begin->HasDofFor(DISPLACEMENT_X)) << "ResidualBasedBossakDisplacementScheme:: DISPLACEMENT is not added" << std::endl;
263 const int disppos = it_node_begin->GetDofPosition(DISPLACEMENT_X);
264 const int velpos = it_node_begin->HasDofFor(VELOCITY_X) ? static_cast<int>(it_node_begin->GetDofPosition(VELOCITY_X)) : -1;
265 const int accelpos = it_node_begin->HasDofFor(ACCELERATION_X) ? static_cast<int>(it_node_begin->GetDofPosition(ACCELERATION_X)) : -1;
266
267 // Getting dimension
268 KRATOS_WARNING_IF("ResidualBasedBossakDisplacementScheme", !r_current_process_info.Has(DOMAIN_SIZE)) << "DOMAIN_SIZE not defined. Please define DOMAIN_SIZE. 3D case will be assumed" << std::endl;
269 const std::size_t dimension = r_current_process_info.Has(DOMAIN_SIZE) ? r_current_process_info.GetValue(DOMAIN_SIZE) : 3;
270
271 // Auxiliar variables
272 array_1d<double, 3 > delta_displacement;
273 std::array<bool, 3> predicted = {false, false, false};
274 const std::array<const Variable<ComponentType>*, 3> disp_components = {&DISPLACEMENT_X, &DISPLACEMENT_Y, &DISPLACEMENT_Z};
275 const std::array<const Variable<ComponentType>*, 3> vel_components = {&VELOCITY_X, &VELOCITY_Y, &VELOCITY_Z};
276 const std::array<const Variable<ComponentType>*, 3> accel_components = {&ACCELERATION_X, &ACCELERATION_Y, &ACCELERATION_Z};
277
278 typedef std::tuple<array_1d<double,3>, std::array<bool,3>> TLSContainerType;
279 TLSContainerType aux_TLS = std::make_tuple(delta_displacement, predicted);
280
281 block_for_each(rModelPart.Nodes(), aux_TLS, [&](Node<3>& rNode, TLSContainerType& rAuxTLS){
282 auto& r_delta_displacement = std::get<0>(rAuxTLS);
283 auto& r_predicted = std::get<1>(rAuxTLS);
284
285 for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
286 r_predicted[i_dim] = false;
287 }
288
289 // Predicting: NewDisplacement = r_previous_displacement + r_previous_velocity * delta_time;
290 const array_1d<double, 3>& r_previous_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION, 1);
291 const array_1d<double, 3>& r_previous_velocity = rNode.FastGetSolutionStepValue(VELOCITY, 1);
292 const array_1d<double, 3>& r_previous_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT, 1);
293 array_1d<double, 3>& r_current_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION);
294 array_1d<double, 3>& r_current_velocity = rNode.FastGetSolutionStepValue(VELOCITY);
295 array_1d<double, 3>& r_current_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT);
296
297 if (accelpos > -1) {
298 for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
299 if (rNode.GetDof(*accel_components[i_dim], accelpos + i_dim).IsFixed()) {
300 r_delta_displacement[i_dim] = (r_current_acceleration[i_dim] + mBossak.c3 * r_previous_acceleration[i_dim] + mBossak.c2 * r_previous_velocity[i_dim])/mBossak.c0;
301 r_current_displacement[i_dim] = r_previous_displacement[i_dim] + r_delta_displacement[i_dim];
302 r_predicted[i_dim] = true;
303 }
304 }
305 }
306 if (velpos > -1) {
307 for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
308 if (rNode.GetDof(*vel_components[i_dim], velpos + i_dim).IsFixed() && !r_predicted[i_dim]) {
309 r_delta_displacement[i_dim] = (r_current_velocity[i_dim] + mBossak.c4 * r_previous_velocity[i_dim] + mBossak.c5 * r_previous_acceleration[i_dim])/mBossak.c1;
310 r_current_displacement[i_dim] = r_previous_displacement[i_dim] + r_delta_displacement[i_dim];
311 r_predicted[i_dim] = true;
312 }
313 }
314 }
315 for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
316 if (!rNode.GetDof(*disp_components[i_dim], disppos + i_dim).IsFixed() && !r_predicted[i_dim]) {
317 r_current_displacement[i_dim] = r_previous_displacement[i_dim] + delta_time * r_previous_velocity[i_dim] + 0.5 * std::pow(delta_time, 2) * r_previous_acceleration[i_dim];
318 }
319 }
320
321 // Updating time derivatives ::: Please note that displacements and its time derivatives can not be consistently fixed separately
322 noalias(r_delta_displacement) = r_current_displacement - r_previous_displacement;
323 UpdateVelocity(r_current_velocity, r_delta_displacement, r_previous_velocity, r_previous_acceleration);
324 UpdateAcceleration(r_current_acceleration, r_delta_displacement, r_previous_velocity, r_previous_acceleration);
325 });
326
327 KRATOS_CATCH( "" );
328 }

337 void InitializeSolutionStep(
338 ModelPart& rModelPart,
339 TSystemMatrixType& A,
340 TSystemVectorType& Dx,
341 TSystemVectorType& b
342 ) override
343 {
344 KRATOS_TRY;
345
346 // The current process info
347 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
348
349 ImplicitBaseType::InitializeSolutionStep(rModelPart, A, Dx, b);
350
351 const double delta_time = r_current_process_info[DELTA_TIME];
352
353 // Initializing Bossak constants
354 mBossak.c0 = ( 1.0 / (mBossak.beta * delta_time * delta_time) );
355 mBossak.c1 = ( mBossak.gamma / (mBossak.beta * delta_time) );
356 mBossak.c2 = ( 1.0 / (mBossak.beta * delta_time) );
357 mBossak.c3 = ( 0.5 / (mBossak.beta) - 1.0 );
358 mBossak.c4 = ( (mBossak.gamma / mBossak.beta) - 1.0 );
359 mBossak.c5 = ( delta_time * 0.5 * ( ( mBossak.gamma / mBossak.beta ) - 2.0 ) );
360
361 // Updating time derivatives (nodally for efficiency)
362 const auto it_node_begin = rModelPart.Nodes().begin();
363
364 // Getting dimension
365 KRATOS_WARNING_IF("ResidualBasedBossakDisplacementScheme", !r_current_process_info.Has(DOMAIN_SIZE)) << "DOMAIN_SIZE not defined. Please define DOMAIN_SIZE. 3D case will be assumed" << std::endl;
366 const std::size_t dimension = r_current_process_info.Has(DOMAIN_SIZE) ? r_current_process_info.GetValue(DOMAIN_SIZE) : 3;
367
368 // Getting position
369 const int velpos = it_node_begin->HasDofFor(VELOCITY_X) ? static_cast<int>(it_node_begin->GetDofPosition(VELOCITY_X)) : -1;
370 const int accelpos = it_node_begin->HasDofFor(ACCELERATION_X) ? static_cast<int>(it_node_begin->GetDofPosition(ACCELERATION_X)) : -1;
371
372 std::array<bool, 3> fixed = {false, false, false};
373 const std::array<const Variable<ComponentType>*, 3> disp_components = {&DISPLACEMENT_X, &DISPLACEMENT_Y, &DISPLACEMENT_Z};
374 const std::array<const Variable<ComponentType>*, 3> vel_components = {&VELOCITY_X, &VELOCITY_Y, &VELOCITY_Z};
375 const std::array<const Variable<ComponentType>*, 3> accel_components = {&ACCELERATION_X, &ACCELERATION_Y, &ACCELERATION_Z};
376
377 block_for_each(rModelPart.Nodes(), fixed, [&](Node<3>& rNode, std::array<bool,3>& rFixedTLS){
378 for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
379 rFixedTLS[i_dim] = false;
380 }
381
382 if (accelpos > -1) {
383 for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
384 if (rNode.GetDof(*accel_components[i_dim], accelpos + i_dim).IsFixed()) {
385 rNode.Fix(*disp_components[i_dim]);
386 rFixedTLS[i_dim] = true;
387 }
388 }
389 }
390 if (velpos > -1) {
391 for (std::size_t i_dim = 0; i_dim < dimension; ++i_dim) {
392 if (rNode.GetDof(*vel_components[i_dim], velpos + i_dim).IsFixed() && !rFixedTLS[i_dim]) {
393 rNode.Fix(*disp_components[i_dim]);
394 }
395 }
396 }
397 });
398
399 KRATOS_CATCH( "" );

619 LocalSystemMatrixType& LHS_Contribution,
620 LocalSystemMatrixType& D,

622 const ProcessInfo& rCurrentProcessInfo
623 ) override
624 {

642 void AddDynamicsToRHS(
643 Element& rElement,
644 LocalSystemVectorType& RHS_Contribution,
645 LocalSystemMatrixType& D,
646 LocalSystemMatrixType& M,
647 const ProcessInfo& rCurrentProcessInfo
648 ) override
649 {
650 const std::size_t this_thread = OpenMPUtils::ThisThread();
651
652 const auto& r_const_elem_ref = rElement;
653 // Adding inertia contribution
654 if (M.size1() != 0) {
655
656 r_const_elem_ref.GetSecondDerivativesVector(mVector.a[this_thread], 0);
657 mVector.a[this_thread] *= (1.00 - mBossak.alpha);
658
659 r_const_elem_ref.GetSecondDerivativesVector(mVector.ap[this_thread], 1);
660 noalias(mVector.a[this_thread]) += mBossak.alpha * mVector.ap[this_thread];
661
662 noalias(RHS_Contribution) -= prod(M, mVector.a[this_thread]);
663 }
664
665 // Adding damping contribution
666 if (D.size1() != 0) {
667 r_const_elem_ref.GetFirstDerivativesVector(mVector.v[this_thread], 0);
668 noalias(RHS_Contribution) -= prod(D, mVector.v[this_thread]);
669 }
