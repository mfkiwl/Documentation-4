512 // Slave element info

515 if constexpr (TTensor == ScalarValue) {

609 IntegrationMethod GetIntegrationMethod() const override
610 {
611 // Setting the auxiliary integration points
612 const IndexType integration_order = GetProperties().Has(INTEGRATION_ORDER_CONTACT) ? GetProperties().GetValue(INTEGRATION_ORDER_CONTACT) : 2;
613 switch (integration_order) {
614 case 1: return GeometryData::IntegrationMethod::GI_GAUSS_1;
615 case 2: return GeometryData::IntegrationMethod::GI_GAUSS_2;
616 case 3: return GeometryData::IntegrationMethod::GI_GAUSS_3;
617 case 4: return GeometryData::IntegrationMethod::GI_GAUSS_4;
618 case 5: return GeometryData::IntegrationMethod::GI_GAUSS_5;
