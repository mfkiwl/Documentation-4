116 static void CalculateEquivalentStress(
117 const array_1d<double, VoigtSize>& rPredictiveStressVector,
118 const Vector& rStrainVector,
119 double& rEquivalentStress,
120 ConstitutiveLaw::Parameters& rValues
121 )
122 {
123 const Properties& r_material_properties = rValues.GetMaterialProperties();
124
125 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
126 const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
127 const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
128 double friction_angle = r_material_properties[FRICTION_ANGLE] * Globals::Pi / 180.0; // In radians!
129
130 // Check input variables
131 if (friction_angle < tolerance) {
132 friction_angle = 32.0 * Globals::Pi / 180.0;
133 KRATOS_WARNING("ModifiedMohrCoulombYieldSurface") << "Friction Angle not defined, assumed equal to 32 deg " << std::endl;
134 }
135
136 double theta;
137 const double R = std::abs(yield_compression / yield_tension);
138 const double Rmorh = std::pow(std::tan((Globals::Pi / 4.0) + friction_angle / 2.0), 2);
139 const double alpha_r = R / Rmorh;
140 const double sin_phi = std::sin(friction_angle);
141
142 double I1, J2, J3;
143 array_1d<double, VoigtSize> deviator = ZeroVector(VoigtSize);
144 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateI1Invariant(rPredictiveStressVector, I1);
145 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ2Invariant(rPredictiveStressVector, I1, deviator, J2);
146 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(deviator, J3);
147
148 const double K1 = 0.5 * (1.0 + alpha_r) - 0.5 * (1.0 - alpha_r) * sin_phi;
149 const double K2 = 0.5 * (1.0 + alpha_r) - 0.5 * (1.0 - alpha_r) / sin_phi;
150 const double K3 = 0.5 * (1.0 + alpha_r) * sin_phi - 0.5 * (1.0 - alpha_r);
151
152 // Check Modified Mohr-Coulomb criterion
153 if (std::abs(I1) < tolerance) {
154 rEquivalentStress = 0.0;
155 } else {
156 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, theta);
157 rEquivalentStress = (2.0 * std::tan(Globals::Pi * 0.25 + friction_angle * 0.5) / std::cos(friction_angle)) * ((I1 * K3 / 3.0) +
158 std::sqrt(J2) * (K1 * std::cos(theta) - K2 * std::sin(theta) * sin_phi / std::sqrt(3.0)));
159 }
160 }

200 static void CalculateDamageParameter(
201 ConstitutiveLaw::Parameters& rValues,
202 double& rAParameter,
203 const double CharacteristicLength
204 )
205 {
206 const Properties& r_material_properties = rValues.GetMaterialProperties();
207
208 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
209 const double young_modulus = r_material_properties[YOUNG_MODULUS];
210 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
211 const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
212 const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
213 const double n = yield_compression / yield_tension;
214
215 if (r_material_properties[SOFTENING_TYPE] == static_cast<int>(SofteningType::Exponential)) {
216 rAParameter = 1.00 / (fracture_energy * n * n * young_modulus / (CharacteristicLength * std::pow(yield_compression, 2)) - 0.5);
217 KRATOS_DEBUG_ERROR_IF(rAParameter < 0.0) << "Fracture energy is too low, increase FRACTURE_ENERGY..." << std::endl;
218 } else { // linear
219 rAParameter = -std::pow(yield_compression, 2) / (2.0 * young_modulus * fracture_energy * n * n / CharacteristicLength);
220 }
221 }

234 static void CalculateYieldSurfaceDerivative(
235 const array_1d<double, VoigtSize>& rPredictiveStressVector,
236 const array_1d<double, VoigtSize>& rDeviator,
237 const double J2,
238 array_1d<double, VoigtSize>& rFFlux,
239 ConstitutiveLaw::Parameters& rValues)
240 {
241 const Properties& r_material_properties = rValues.GetMaterialProperties();
242
243 array_1d<double, VoigtSize> first_vector, second_vector, third_vector;
244
245 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateFirstVector(first_vector);
246 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateSecondVector(rDeviator, J2, second_vector);
247 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateThirdVector(rDeviator, J2, third_vector);
248
249 double J3, lode_angle;
250 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(rDeviator, J3);
251 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
252
253 const double checker = std::abs(lode_angle * 180.0 / Globals::Pi);
254
255 double c1, c2, c3;
256 double friction_angle = r_material_properties[FRICTION_ANGLE] * Globals::Pi / 180.0;
257
258 // Check input variables
259 if (friction_angle < tolerance) {
260 friction_angle = 32.0 * Globals::Pi / 180.0;
261 KRATOS_WARNING("ModifiedMohrCoulombYieldSurface") << "Friction Angle not defined, assumed equal to 32 deg " << std::endl;
262 }
263
264 const double sin_phi = std::sin(friction_angle);
265 const double cons_phi = std::cos(friction_angle);
266 const double sin_theta = std::sin(lode_angle);
267 const double cos_theta = std::cos(lode_angle);
268 const double cos_3theta = std::cos(3.0 * lode_angle);
269 const double tan_theta = std::tan(lode_angle);
270 const double tan_3theta = std::tan(3.0 * lode_angle);
271 const double Root3 = std::sqrt(3.0);
272
273 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
274 const double compr_yield = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
275 const double tens_yield = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
276 const double n = compr_yield / tens_yield;
277
278 const double angle_phi = (Globals::Pi * 0.25) + friction_angle * 0.5;
279 const double alpha = n / (std::tan(angle_phi) * std::tan(angle_phi));
280
281 const double CFL = 2.0 * std::tan(angle_phi) / cons_phi;
282
283 const double K1 = 0.5 * (1.0 + alpha) - 0.5 * (1.0 - alpha) * sin_phi;
284 const double K2 = 0.5 * (1.0 + alpha) - 0.5 * (1.0 - alpha) / sin_phi;
285 const double K3 = 0.5 * (1.0 + alpha) * sin_phi - 0.5 * (1.0 - alpha);
286
287 if (std::abs(sin_phi) > tolerance)
288 c1 = CFL * K3 / 3.0;
289 else
290 c1 = 0.0; // check
291
292 if (std::abs(checker) < 29.0) { // Lode angle not greater than pi/6
293 c2 = cos_theta * CFL * (K1 * (1.0 + tan_theta * tan_3theta) + K2 * sin_phi * (tan_3theta - tan_theta) / Root3);
294 c3 = CFL * (K1 * Root3 * sin_theta + K2 * sin_phi * cos_theta) / (2.0 * J2 * cos_3theta);
295 } else {
296 c3 = 0.0;
297 double aux = 1.0;
298 if (lode_angle > tolerance)
299 aux = -1.0;
300 c2 = 0.5 * CFL * (K1 * Root3 + aux * K2 * sin_phi / Root3);
301 }
302 noalias(rFFlux) = c1 * first_vector + c2 * second_vector + c3 * third_vector;
303 }

309 static int Check(const Properties& rMaterialProperties)
310 {
311 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRICTION_ANGLE)) << "FRICTION_ANGLE is not a defined value" << std::endl;
312 if (!rMaterialProperties.Has(YIELD_STRESS)) {
313 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
314 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
315
316 const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
317 const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
318
319 KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
320 KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
321 } else {
322 const double yield_stress = rMaterialProperties[YIELD_STRESS];
323
324 KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
325 }
326 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRACTURE_ENERGY)) << "FRACTURE_ENERGY is not a defined value" << std::endl;
327 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "YOUNG_MODULUS is not a defined value" << std::endl;
328
329 return TPlasticPotentialType::Check(rMaterialProperties);
330 }
