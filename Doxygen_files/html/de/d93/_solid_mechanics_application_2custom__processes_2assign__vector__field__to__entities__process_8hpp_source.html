334 template<class TDataType>
335 void CallFunction(const Condition::Pointer& pCondition, const double& time, TDataType& rValue)
336 {
337
338 Condition::GeometryType& rConditionGeometry = pCondition->GetGeometry();
339 unsigned int size = rConditionGeometry.size();
340 double value = 0;
341 unsigned int counter = 0;
342
343 rValue.resize(size*3,false);
344
345 if( mIsSpatialField ){
346
347 double x = 0.0, y = 0.0, z = 0.0;
348
349 for(unsigned int i=0; i<size; i++)
350 {
351 this->LocalAxesTransform(rConditionGeometry[i].X(), rConditionGeometry[i].Y(), rConditionGeometry[i].Z(), x, y, z);
352
353 value = mPyObject.attr(this->mPyMethodName.c_str())(x,y,z,time).cast<double>();
354
355 for(unsigned int j=0; j<3; j++)
356 {
357 rValue[counter] = value * mvector_value[j];
358 counter++;
359 }
360 }
361
362 }
363 else{
364
365 value = mPyObject.attr(this->mPyMethodName.c_str())(0.0,0.0,0.0,time).cast<double>();
366 for(unsigned int i=0; i<size; i++)
367 {
368 for(unsigned int j=0; j<3; j++)
369 {
370 rValue[counter] = value * mvector_value[j];
371 counter++;
372 }
373 }
374
375
376 }

381 template< class TVarType, class TDataType >
382 void AssignValueToConditions(TVarType& rVariable, TDataType& Value, const double& rTime )
383 {
384
385 typedef void (BaseType::*AssignmentMethodPointer) (ModelPart::ConditionType&, const TVarType&, const TDataType&);
386
387 AssignmentMethodPointer AssignmentMethod = this->GetAssignmentMethod<AssignmentMethodPointer>();
388
389 const int nconditions = mrModelPart.GetMesh().Conditions().size();
390
391 if(nconditions != 0)
392 {
393 ModelPart::ConditionsContainerType::iterator it_begin = mrModelPart.GetMesh().ConditionsBegin();
394
395 //#pragma omp parallel for //it does not work in parallel
396 for(int i = 0; i<nconditions; i++)
397 {
398 ModelPart::ConditionsContainerType::iterator it = it_begin + i;
399
400 this->CallFunction<TDataType>(*(it.base()), rTime, Value);
401
402 (this->*AssignmentMethod)(*it, rVariable, Value);
403 }
404 }
405
406 }
