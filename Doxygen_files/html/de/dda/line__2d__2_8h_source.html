938 bool IsInside(
939 const CoordinatesArrayType& rPoint,
940 CoordinatesArrayType& rResult,
941 const double Tolerance = std::numeric_limits<double>::epsilon()
942 ) const override
943 {
944 // We compute the distance, if it is not in the plane we project
945 const Point point_to_project(rPoint);
946 Point point_projected;
947 const double distance = GeometricalProjectionUtilities::FastProjectOnLine2D(*this, point_to_project, point_projected);
948
949 // We check if we are on the plane
950 if (std::abs(distance) > std::numeric_limits<double>::epsilon()) {
951 if (std::abs(distance) > 1.0e-6 * Length()) {
952 KRATOS_WARNING_FIRST_N("Line2D2", 10) << "The point of coordinates X: " << rPoint[0] << "\tY: " << rPoint[1] << " it is in a distance: " << std::abs(distance) << std::endl;
953 return false;
954 }
955 }
956
957 PointLocalCoordinates( rResult, point_projected );
958
959 if ( std::abs( rResult[0] ) <= (1.0 + Tolerance) ) {
960 return true;
961 }
962
963 return false;
964 }

972 bool HasIntersection(const BaseType& rOtherGeometry) const override
973 {
974 const double tolerance = std::numeric_limits<double>::epsilon();
975 // We get the local points
976 const TPointType& first_point = BaseType::GetPoint(0); //p1
977 const TPointType& second_point = BaseType::GetPoint(1); //p2
978
979 // We get the other line's points
980 const TPointType& first_point_other = *rOtherGeometry(0); //p3
981 const TPointType& second_point_other = *rOtherGeometry(1); //p4
982
983 // parametric coordinate of intersection on current line
984 const double numerator = ( (first_point[0]-first_point_other[0])*(first_point_other[1] - second_point_other[1]) - (first_point[1]-first_point_other[1])*(first_point_other[0]-second_point_other[0]) );
985 const double denominator = ( (first_point[0]-second_point[0])*(first_point_other[1] - second_point_other[1]) - (first_point[1]-second_point[1])*(first_point_other[0]-second_point_other[0]) );
986 if (std::abs(denominator) < tolerance) // this means parallel lines.
987 return false;
988 const double t = numerator / denominator;
989
990 return (0.0-tolerance<=t) && (t<=1.0+tolerance);
991 }

1001 bool HasIntersection(const Point& rLowPoint, const Point& rHighPoint) const override
1002 {
1003 const double tolerance = std::numeric_limits<double>::epsilon();
1004 // We get the local points
1005 const TPointType& first_point = BaseType::GetPoint(0);
1006 const TPointType& second_point = BaseType::GetPoint(1);
1007
1008 if ( // If one of the point is inside the box then there is an intersection. If none is inside then we check further.
1009 ( (first_point[0] >= rLowPoint[0] && first_point[0] <= rHighPoint[0])
1010 && (first_point[1] >= rLowPoint[1] && first_point[1] <= rHighPoint[1]) ) // IF the first point is inside the box
1011 ||
1012 ( (second_point[0] >= rLowPoint[0] && second_point[0] <= rHighPoint[0])
1013 && (second_point[1] >= rLowPoint[1] && second_point[1] <= rHighPoint[1]) ) // IF the second point is inside the box
1014 )
1015 return true;
1016
1017 const double high_x = rHighPoint[0];
1018 const double high_y = rHighPoint[1];
1019 const double low_x = rLowPoint[0];
1020 const double low_y = rLowPoint[1];
1021
1022 const double denominator = ( second_point[0] - first_point[0] );
1023 const double numerator = (second_point[1] - first_point[1]);
1024 const double slope = std::abs(denominator) > tolerance ? std::abs(numerator) > tolerance ? numerator / denominator : 1.0e-12 : 1.0e12;
1025
1026 // Intersection with left vertical line of the box that is x = low_x
1027 const double y_1 = slope*( low_x - first_point[0] ) + first_point[1];
1028 if(y_1 >= low_y - tolerance && y_1 <= high_y+tolerance) // If y intersection is between two y bounds there is an intersection
1029 return true;
1030 // Intersection with right vertical line of the box that is x = high_x
1031 const double y_2 = slope*( high_x - first_point[0] ) + first_point[1];
1032 if(y_2 >= low_y - tolerance && y_2 <= high_y+tolerance) // If y intersection is between two y bounds there is an intersection
1033 return true;
1034 // Intersection with bottom horizontal line of the box that is y = low_y
1035 const double x_1 = first_point[0] + ( (low_y - first_point[1]) / slope );
1036 if(x_1 >= low_x-tolerance && x_1 <= high_x+tolerance) // If x intersection is between two x bounds there is an intersection
1037 return true;
1038 // Intersection with top horizontal line of the box that is y = high_y
1039 const double x_2 = first_point[0] + ( (high_y - first_point[1]) / slope );
1040 if(x_2 >= low_x-tolerance && x_2 <= high_x+tolerance) // If x intersection is between two x bounds there is an intersection
1041 return true;
1042
1043
1044 return false;
1045 }

1054 CoordinatesArrayType& PointLocalCoordinates(
1055 CoordinatesArrayType& rResult,
1056 const CoordinatesArrayType& rPoint
1057 ) const override
1058 {
1059 rResult.clear();
1060
1061 const TPointType& r_first_point = BaseType::GetPoint(0);
1062 const TPointType& r_second_point = BaseType::GetPoint(1);
1063
1064 // Project point
1065 const double tolerance = 1e-14; // Tolerance
1066
1067 const double length = Length();
1068
1069 const double length_1 = std::sqrt( std::pow(rPoint[0] - r_first_point[0], 2)
1070 + std::pow(rPoint[1] - r_first_point[1], 2));
1071
1072 const double length_2 = std::sqrt( std::pow(rPoint[0] - r_second_point[0], 2)
1073 + std::pow(rPoint[1] - r_second_point[1], 2));
1074
1075 if (length_1 <= (length + tolerance) && length_2 <= (length + tolerance)) {
1076 rResult[0] = 2.0 * length_1/(length + tolerance) - 1.0;
1077 } else {
1078 if (length_1 > length_2) {
1079 rResult[0] = 2.0 * length_1/(length + tolerance) - 1.0;
1080 } else {
1081 rResult[0] = -2.0 * length_1/(length + tolerance) - 1.0;
1082 }
1083 }
1084
1085 return rResult ;
1086 }
