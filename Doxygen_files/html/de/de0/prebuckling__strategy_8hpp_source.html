414 bool SolveSolutionStep() override
415 {
416 KRATOS_TRY
417
418 ModelPart& rModelPart = BaseType::GetModelPart();
419 SchemePointerType& pScheme = this->pGetScheme();
420 BuilderAndSolverPointerType& pBuilderAndSolver = this->pGetBuilderAndSolver();
421 SparseMatrixType& rStiffnessMatrix = *mpStiffnessMatrix;
422 SparseMatrixType& rStiffnessMatrixPrevious = *mpStiffnessMatrixPrevious;
423 SparseVectorType& rRHS = *mpRHS;
424 SparseVectorType& rDx = *mpDx;
425
426 typename ConvergenceCriteriaType::Pointer pConvergenceCriteria = mpConvergenceCriteria;
427
428 // Initializing the parameters of the Newton-Raphson cycle
429 unsigned int iteration_number = 1;
430 rModelPart.GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
431 bool is_converged = false;
432
433 // Update loadfactor increment
434 double delta_load_multiplier = 0.0;
435 if( mLoadStepIteration == 1) // inital load increment
436 {
437 delta_load_multiplier = mInitialLoadIncrement*(mLambda + mLambdaPrev);
438 }
439 else if( mLoadStepIteration % 2 == 1 ) //small load increment
440 {
441 delta_load_multiplier = mSmallLoadIncrement*(mLambdaPrev );
442 }
443
444 BuiltinTimer system_solve_time;
445 // Initialize nonlinear iteration
446 this->pGetScheme()->InitializeNonLinIteration( rModelPart,rStiffnessMatrix, rDx, rRHS );
447 pConvergenceCriteria->InitializeNonLinearIteration(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
448 is_converged = mpConvergenceCriteria->PreCriteria(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
449
450 TSparseSpace::SetToZero(rStiffnessMatrix);
451 TSparseSpace::SetToZero(rRHS);
452 TSparseSpace::SetToZero(rDx);
453 // Build system and solve system
454 pBuilderAndSolver->BuildAndSolve(pScheme, rModelPart, rStiffnessMatrix, rDx, rRHS);
455 // Update internal variables
456 pScheme->Update(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS );
457 BaseType::MoveMesh();
458 // Finalize nonlinear solution step
459 pScheme->FinalizeNonLinIteration( rModelPart,rStiffnessMatrix, rDx, rRHS );
460 pConvergenceCriteria->FinalizeNonLinearIteration(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
461
462 if (is_converged){
463 is_converged = mpConvergenceCriteria->PostCriteria(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
464 }
465
466 // Start iteration cycle
467 while ( !is_converged &&
468 iteration_number++ < mMaxIteration)
469 {
470 // Setting the number of iteration
471 rModelPart.GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
472
473 // Initialize nonlinear iteration step
474 pScheme->InitializeNonLinIteration( rModelPart,rStiffnessMatrix, rDx, rRHS );
475 pConvergenceCriteria->InitializeNonLinearIteration(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
476 is_converged = mpConvergenceCriteria->PreCriteria(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
477
478 TSparseSpace::SetToZero(rStiffnessMatrix);
479 TSparseSpace::SetToZero(rDx);
480 TSparseSpace::SetToZero(rRHS);
481 // Build and solve system
482 pBuilderAndSolver->BuildAndSolve(pScheme, rModelPart, rStiffnessMatrix,rDx, rRHS);
483
484 // Update internal variables
485 this->pGetScheme()->Update(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
486 BaseType::MoveMesh();
487
488 // Finalize nonlinear iteration step
489 this->pGetScheme()->FinalizeNonLinIteration( rModelPart,rStiffnessMatrix, rDx, rRHS );
490 pConvergenceCriteria->FinalizeNonLinearIteration(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
491
492 if (is_converged){
493 is_converged = mpConvergenceCriteria->PostCriteria(rModelPart, pBuilderAndSolver->GetDofSet(), rStiffnessMatrix, rDx, rRHS);
494 }
495 }
496 KRATOS_INFO_IF("Nonlinear Loadstep Time: ", BaseType::GetEchoLevel() > 0)
497 << system_solve_time.ElapsedSeconds() << std::endl;
498
499 if ( !is_converged ) {
500 KRATOS_INFO_IF("Nonlinear Loadstep: ", this->GetEchoLevel() > 0)
501 << "Convergence not achieved after ( " << mMaxIteration
502 << " ) Iterations !" << std::endl;
503 } else {
504 KRATOS_INFO_IF("Nonlinear Loadstep: ", this->GetEchoLevel() > 0)
505 << "Convergence achieved after " << iteration_number << " / "
506 << mMaxIteration << " iterations" << std::endl;
507 }
508
509 // Vector and matrix are initialized by eigensolver
510 DenseVectorType Eigenvalues;
511 DenseMatrixType Eigenvectors;
512
513 if( mLoadStepIteration % 2 == 0 ){
514 //Copy matrices after path following step and initial step
515 rStiffnessMatrixPrevious = rStiffnessMatrix;
516
517 if( mLoadStepIteration > 0){
518 // Update mLambdaPrev
519 mLambdaPrev = mLambdaPrev + mPathFollowingStep*mLambda;
520 }
521 }
522 else if( mLoadStepIteration % 2 == 1 ){
523 // Evaluate eigenvalue problem after small load increment
524 //#####################################################################################
525 //Method 1
526 //Poy, Denise Lori-eng; On the Buckling Finite Element Analysis of Beam Structures
527 //#####################################################################################
528 // Solve Eigenvalue Problem
529 // BuiltinTimer eigen_solve_time;
530 // this->pGetEigenSolver()->GetLinearSystemSolver()->Solve(
531 // rStiffnessMatrix,
532 // rStiffnessMatrixPrevious,
533 // Eigenvalues,
534 // Eigenvectors);
535
536 // mLambda = 1.0 / ( 1.0 - Eigenvalues(0) ) * delta_load_multiplier;
537 // for(int i = 0; i < Eigenvalues.size(); i++ )
538 // {
539 // Eigenvalues[i] = mLambdaPrev + 1.0 / ( 1.0 - Eigenvalues[i] )*delta_load_multiplier;
540 // }
541 // this->AssignVariables(Eigenvalues, Eigenvectors);
542 //End Method 1#########################################################################
543
544 //#####################################################################################
545 //Method 2:
546 // Jia, X.Mang, H. A.; Assessment of solutions from the consistently linearized eigenproblem
547 // by means of finite difference approximations
548 //#####################################################################################
549 // Solve Eigenvalue Problem
550
551 // This implementation of the eigenvalue problem relies on the difference of the stiffness matrix
552 // between the current and previous step
553 rStiffnessMatrix = rStiffnessMatrixPrevious - rStiffnessMatrix;
554
555 // Symmetrice matrices if enabled
556 // The ublas transpose function is called manually, because it is missing in the corresponding space
557 if( mMakeMatricesSymmetricFlag ){
558 rStiffnessMatrix = 0.5 * ( rStiffnessMatrix + boost::numeric::ublas::trans(rStiffnessMatrix) );
559 rStiffnessMatrixPrevious = 0.5 * ( rStiffnessMatrixPrevious + boost::numeric::ublas::trans(rStiffnessMatrixPrevious) );
560 }
561
562 this->pGetEigenSolver()->GetLinearSystemSolver()->Solve(
563 rStiffnessMatrixPrevious,
564 rStiffnessMatrix,
565 Eigenvalues,
566 Eigenvectors);
567
568 // Update eigenvalues to loadfactors (Instead of dividing matrix by delta_load_multiplier, here eigenvalues are multiplied)
569 mLambda = Eigenvalues(0)*delta_load_multiplier;
570 for(unsigned int i = 0; i < Eigenvalues.size(); i++ )
571 {
572 Eigenvalues[i] = mLambdaPrev + Eigenvalues[i]*delta_load_multiplier;
573 }
574
575 this->AssignVariables(Eigenvalues, Eigenvectors);
576
577 // Reset elastic stiffness matrix for next loadstep
578 mpStiffnessMatrix = nullptr;
579 pBuilderAndSolver->ResizeAndInitializeVectors(
580 pScheme, mpStiffnessMatrix, mpDx, mpRHS, rModelPart);
581 // End Method 2#########################################################################
582
583 // Convergence criteria for buckling analysis
584 if( std::abs(mLambda/mLambdaPrev) < mConvergenceRatio ){
585 mSolutionFound = true;
586 KRATOS_INFO_IF("Prebuckling Analysis: ", BaseType::GetEchoLevel() > 0)
587 << "Convergence achieved in " << mLoadStepIteration + 1 << " Load Iterations!" << std::endl;
588 }
589 }
590
591 mLoadStepIteration++;
592 // Prepare load conditions for next solution step
593 UpdateLoadConditions();
594
595 return true;
596 KRATOS_CATCH("")
597 }

770 {

792 {
793 ModelPart &rModelPart = BaseType::GetModelPart();
794 const std::size_t NumEigenvalues = rEigenvalues.size();
795
796 // Store eigenvalues in process info
797 rModelPart.GetProcessInfo()[EIGENVALUE_VECTOR] = rEigenvalues;

799 for (ModelPart::NodeIterator itNode = rModelPart.NodesBegin(); itNode != rModelPart.NodesEnd(); itNode++)

802 const std::size_t NumNodeDofs = NodeDofs.size();
803
804 Matrix &rNodeEigenvectors = itNode->GetValue(EIGENVECTOR_MATRIX);
805 if (rNodeEigenvectors.size1() != NumEigenvalues || rNodeEigenvectors.size2() != NumNodeDofs){

807 }

819 for (std::size_t j = 0; j < NumNodeDofs; j++){
