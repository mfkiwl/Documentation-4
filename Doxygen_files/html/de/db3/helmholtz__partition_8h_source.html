443  double ProjectForward(double x,double beta){
444 
445  double x_c = 1.0;
446  double x_f = 0.0;
447 
448  double pow_val = -2.0*beta*(x-(x_c+x_f)/2);
449  double value = (x_c-x_f)/(1+std::exp(pow_val)) + x_f;
450  if(value>1.0)
451  value = 1.0;
452  if(value<0.0)
453  value = 0.0;
454  return value;
455  }

457  double ProjectBackward(double y,double beta){
458 
459  double y_c = 1.0;
460  double y_f = 0.0;
461 
462  if((y>y_f) && (y<y_c))
463  return ((y_f+y_c)/2.0) + (1.0/(-2.0*beta)) * std::log(((y_c-y_f)/(y-y_f))-1);
464  else if(y==y_c)
465  return y_c;
466  else if(y==y_f)
467  return y_f;
468  else if(y>y_c)
469  return y_c;
470  else if(y<y_f)
471  return y_f;
472  else
473  return 0.0;
474  }

476  double FirstFilterDerivative(double x, double beta){
477 
478  double x_c = 1.0;
479  double x_f = 0.0;
480 
481  double pow_val = -2.0*beta*(x-(x_c+x_f)/2);
482  return (1.0/(1+std::exp(pow_val))) * (1.0/(1+std::exp(pow_val))) * 2.0 * beta * std::exp(pow_val);
483  }
