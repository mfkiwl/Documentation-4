287 void CalculateNodeNeighbourCount()
288 {
289 for(int model_i =0;model_i<mpVMModelParts.size();model_i++)
290 {
291 ModelPart* mpVMModePart = mpVMModelParts[model_i];
292 auto& r_nodes = mpVMModePart->Nodes();
293 int mNumNodes = r_nodes.size();
294
295 VariableUtils variable_utils;
296 variable_utils.SetFlag(STRUCTURE,true,r_nodes);
297
298 // Note: this should not be parallel, the operation is not threadsafe if the variable is uninitialized
299 for (auto& r_node : r_nodes)
300 {
301 r_node.SetValue(NUMBER_OF_NEIGHBOUR_ELEMENTS,0);
302 }
303
304 mNumNodes = mpVMModePart->GetCommunicator().GetDataCommunicator().SumAll(mNumNodes);
305
306 auto& r_elements = mpVMModePart->Elements();
307 const int num_elements = r_elements.size();
308
309 #pragma omp parallel for
310 for (int i = 0; i < num_elements; i++)
311 {
312 auto i_elem = r_elements.begin() + i;
313 auto& r_geom = i_elem->GetGeometry();
314 for (unsigned int i = 0; i < r_geom.PointsNumber(); i++)
315 {
316 auto& r_node = r_geom[i];
317 if (r_node.Is(STRUCTURE))
318 {
319 r_node.SetLock();
320 r_node.GetValue(NUMBER_OF_NEIGHBOUR_ELEMENTS) += 1;
321 r_node.UnSetLock();
322 }
323 }
324 }
325
326 mpVMModePart->GetCommunicator().AssembleNonHistoricalData(NUMBER_OF_NEIGHBOUR_ELEMENTS);
327
328 }

389 void ComputeFilteredDensity(){
390
391 for(int model_i=0;model_i<mpVMModelParts.size();model_i++){
392
393 //first update control density
394 AddVariable1ToVarible2(mpVMModelParts[model_i],D_CD,CD);
395
396 //now filter nodal control desnity
397 //first we need to multiply with mass matrix
398 SetVariable(mpVMModelParts[model_i],HELMHOLTZ_VAR_DENSITY,0.0);
399 SetVariable(mpVMModelParts[model_i],HELMHOLTZ_SOURCE_DENSITY,0.0);
400 //now we need to multiply with the mass matrix
401 for(auto& elem_i : mpVMModelParts[model_i]->Elements())
402 {
403 VectorType origin_values;
404 GetElementVariableValuesVector(elem_i,CD,origin_values);
405 MatrixType mass_matrix;
406 elem_i.Calculate(HELMHOLTZ_MASS_MATRIX,mass_matrix,mpVMModelParts[model_i]->GetProcessInfo());
407 VectorType int_vals = prod(mass_matrix,origin_values);
408 AddElementVariableValuesVector(elem_i,HELMHOLTZ_SOURCE_DENSITY,int_vals);
409 }
410
411 mpStrategies[model_i]->Solve();
412 SetVariable1ToVarible2(mpVMModelParts[model_i],HELMHOLTZ_VAR_DENSITY,FD);
413 }

416 void ComputePhyiscalDensity(){
417
418 for(int model_i=0;model_i<mpVMModelParts.size();model_i++){
419 //now do the projection and then set the PD
420 for(auto& node_i : mpVMModelParts[model_i]->Nodes()){
421 const auto& filtered_density = node_i.FastGetSolutionStepValue(FD);
422 auto& physical_density = node_i.FastGetSolutionStepValue(PD);
423 auto& physical_density_der = node_i.FastGetSolutionStepValue(D_PD_D_FD);
424 physical_density = ProjectForward(filtered_density,beta);
425 physical_density_der = FirstFilterDerivative(filtered_density,beta);
426 }
427 }
428
429 // update elements' density
430 for(auto& control_obj : mControlSettings["controlling_objects"]){
431 ModelPart& r_controlling_object = mrModel.GetModelPart(control_obj.GetString());
432 for (int i = 0; i < (int)r_controlling_object.Elements().size(); i++) {
433 ModelPart::ElementsContainerType::iterator it = r_controlling_object.ElementsBegin() + i;
434 double elem_i_density = 0.0;
435 for(unsigned int node_element = 0; node_element<it->GetGeometry().size(); node_element++)
436 elem_i_density += it->GetGeometry()[node_element].FastGetSolutionStepValue(PD);
437 elem_i_density /= it->GetGeometry().size();
438 it->GetProperties().SetValue(DENSITY,elem_i_density);
439 }
440 }

443 double ProjectForward(double x,double beta){
444
445 double x_c = 1.0;
446 double x_f = 0.0;
447
448 double pow_val = -2.0*beta*(x-(x_c+x_f)/2);
449 double value = (x_c-x_f)/(1+std::exp(pow_val)) + x_f;
450 if(value>1.0)
451 value = 1.0;
452 if(value<0.0)
453 value = 0.0;
454 return value;
455 }
456
457 double ProjectBackward(double y,double beta){
458
459 double y_c = 1.0;
460 double y_f = 0.0;
461
462 if((y>y_f) && (y<y_c))
463 return ((y_f+y_c)/2.0) + (1.0/(-2.0*beta)) * std::log(((y_c-y_f)/(y-y_f))-1);
464 else if(y==y_c)
465 return y_c;
466 else if(y==y_f)
467 return y_f;
468 else if(y>y_c)
469 return y_c;
470 else if(y<y_f)
471 return y_f;
472 }

474 double FirstFilterDerivative(double x, double beta){
475
476 double x_c = 1.0;
477 double x_f = 0.0;
478
479 double pow_val = -2.0*beta*(x-(x_c+x_f)/2);
480 return (1.0/(1+std::exp(pow_val))) * (1.0/(1+std::exp(pow_val))) * 2.0 * beta * std::exp(pow_val);
481 }
