71 void DamageExtrapolationAndSmoothing(NodeStresses *pNodeStressesVector)
72 {
73 double damage;
74 // Loop over elements to extrapolate the stress to the nodes
75 for (ElementsArrayType::ptr_iterator it = mr_model_part.Elements().ptr_begin(); it != mr_model_part.Elements().ptr_end(); ++it)
76 {
77 bool condition_is_active = true;
78 if ((*it)->IsDefined(ACTIVE))
79 {
80 condition_is_active = (*it)->Is(ACTIVE);
81 }
82
83 if (condition_is_active)
84 {
85 if ((*it)->GetGeometry().PointsNumber() == 3)
86 {
87 damage = (*it)->GetValue(DAMAGE_ELEMENT);
88
89 for (int i = 0; i < 3; i++)
90 {
91 pNodeStressesVector[(*it)->GetGeometry().GetPoint(i).Id() - 1].NElems += 1;
92 pNodeStressesVector[(*it)->GetGeometry().GetPoint(i).Id() - 1].Damage += damage;
93 }
94 }
95 else
96 {
97 damage = (*it)->GetValue(DAMAGE_ELEMENT);
98
99 for (int i = 0; i < 4; i++)
100 {
101 pNodeStressesVector[(*it)->GetGeometry().GetPoint(i).Id() - 1].NElems += 1;
102 pNodeStressesVector[(*it)->GetGeometry().GetPoint(i).Id() - 1].Damage += damage;
103 }
104 }
105 }
106 }
107
108 // Ponderate over the elements coincident on that node
109 for (unsigned int i = 0; i < mNNodes; i++)
110 {
111 pNodeStressesVector[i].Damage = pNodeStressesVector[i].Damage / pNodeStressesVector[i].NElems;
112 }
113
114 // Loop to compute the max eq. stress in order to normalize
115 for (ModelPart::NodeIterator it = mr_model_part.NodesBegin(); it != mr_model_part.NodesEnd(); ++it)
116 {
117 const int Id = (*it).Id();
118 double &damage = it->GetSolutionStepValue(NODAL_DAMAGE);
119 damage = pNodeStressesVector[Id - 1].Damage;
120 }
121 }
