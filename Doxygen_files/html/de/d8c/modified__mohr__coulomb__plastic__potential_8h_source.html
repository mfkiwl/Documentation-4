116 static void CalculatePlasticPotentialDerivative(
117 const array_1d<double, VoigtSize>& rPredictiveStressVector,
118 const array_1d<double, VoigtSize>& rDeviator,
119 const double J2,
120 array_1d<double, VoigtSize>& rGFlux,
121 ConstitutiveLaw::Parameters& rValues
122 )
123 {
124 const Properties& r_material_properties = rValues.GetMaterialProperties();
125
126 array_1d<double, VoigtSize> first_vector, second_vector, third_vector;
127
128 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateFirstVector(first_vector);
129 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateSecondVector(rDeviator, J2, second_vector);
130 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateThirdVector(rDeviator, J2, third_vector);
131
132 double J3, lode_angle;
133 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(rDeviator, J3);
134 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
135
136 const double checker = std::abs(lode_angle * 180.0 / Globals::Pi);
137
138 const double dilatancy = r_material_properties[DILATANCY_ANGLE] * Globals::Pi / 180.0;
139 const double sin_dil = std::sin(dilatancy);
140 const double cos_dil = std::cos(dilatancy);
141 const double sin_theta = std::sin(lode_angle);
142 const double cos_theta = std::cos(lode_angle);
143 const double cos_3theta = std::cos(3.0 * lode_angle);
144 const double tan_theta = std::tan(lode_angle);
145 const double tan_3theta = std::tan(3.0 * lode_angle);
146 const double Root3 = std::sqrt(3.0);
147
148 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
149 const double compr_yield = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
150 const double tensi_yield = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
151 const double n = compr_yield / tensi_yield;
152
153 const double angle_phi = (Globals::Pi * 0.25) + dilatancy * 0.5;
154 const double alpha = n / (std::tan(angle_phi) * std::tan(angle_phi));
155
156 const double CFL = 2.0 * std::tan(angle_phi) / cos_dil;
157
158 const double K1 = 0.5 * (1 + alpha) - 0.5 * (1 - alpha) * sin_dil;
159 const double K2 = 0.5 * (1 + alpha) - 0.5 * (1 - alpha) / sin_dil;
160 const double K3 = 0.5 * (1 + alpha) * sin_dil - 0.5 * (1 - alpha);
161
162 double c1, c2, c3;
163 if (std::abs(sin_dil) > tolerance)
164 c1 = CFL * K3 / 3.0;
165 else
166 c1 = 0.0; // check
167
168 if (checker < 29.0) {
169 c2 = cos_theta * CFL * (K1 * (1 + tan_theta * tan_3theta) + K2 * sin_dil * (tan_3theta - tan_theta) / Root3);
170 c3 = CFL * (K1 * Root3 * sin_theta + K2 * sin_dil * cos_theta) / (2.0 * J2 * cos_3theta);
171 } else {
172 c3 = 0.0;
173 double Aux = 1.0;
174 if (std::abs(lode_angle) > tolerance)
175 Aux = -1.0;
176 c2 = 0.5 * CFL * (K1 * Root3 + Aux * K2 * sin_dil / Root3);
177 }
178
179 noalias(rGFlux) = c1 * first_vector + c2 * second_vector + c3 * third_vector;
180 }

186 static int Check(const Properties& rMaterialProperties)
187 {
188 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(DILATANCY_ANGLE)) << "DILATANCY_ANGLE is not a defined value" << std::endl;
189 if (!rMaterialProperties.Has(YIELD_STRESS)) {
190 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
191 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
192
193 const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
194 const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
195
196 KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
197 KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
198 } else {
199 const double yield_stress = rMaterialProperties[YIELD_STRESS];
200
201 KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
202 }
203
204 return 0;
205 }
