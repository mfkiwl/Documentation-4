389  bool SolveSolutionStep() override
390  {
391  KRATOS_TRY;
392 
393  bool is_converged = false;
394 
395  // Getting model part
396  ModelPart& r_model_part = StrategyBaseType::GetModelPart();
397 
398  // We get the process info
399  ProcessInfo& r_process_info = r_model_part.GetProcessInfo();
400 
401  if (r_process_info.Is(INTERACTION)) {
402  // We get the system
403  TSystemMatrixType& rA = *BaseType::mpA;
404  TSystemVectorType& rDx = *BaseType::mpDx;
405  TSystemVectorType& rb = *BaseType::mpb;
406 
407  int inner_iteration = 0;
408  const SizeType echo_level_convergence_criteria = BaseType::mpConvergenceCriteria->GetEchoLevel();
409  while (!is_converged && inner_iteration < mThisParameters["inner_loop_iterations"].GetInt()) {
410  ++inner_iteration;
411 
412  if (echo_level_convergence_criteria > 0 && r_model_part.GetCommunicator().MyPID() == 0 ) {
413  KRATOS_INFO("Simplified semi-smooth strategy") << BOLDFONT("INNER ITERATION: ") << inner_iteration << std::endl;
414  }
415 
416  // We solve one loop
417  r_process_info[NL_ITERATION_NUMBER] = 1;
418  is_converged = AuxiliarySolveSolutionStep();
419 
420  // We check the convergence
421  if (r_process_info[NL_ITERATION_NUMBER] == 1) r_process_info[NL_ITERATION_NUMBER] = 2; // Trigger check
422  is_converged = mpMPCContactCriteria->PostCriteria(r_model_part, BaseType::GetBuilderAndSolver()->GetDofSet(), rA, rDx, rb);
423 
424  if (echo_level_convergence_criteria > 0 && r_model_part.GetCommunicator().MyPID() == 0 ) {
425  if (is_converged) KRATOS_INFO("Simplified semi-smooth strategy") << BOLDFONT("Simplified semi-smooth strategy. INNER ITERATION: ") << BOLDFONT(FGRN("CONVERGED")) << std::endl;
426  else KRATOS_INFO("Simplified semi-smooth strategy") << BOLDFONT("INNER ITERATION: ") << BOLDFONT(FRED("NOT CONVERGED")) << std::endl;
427  }
428  }
429  } else {
430  is_converged = AuxiliarySolveSolutionStep();
431  }
432 
433  return is_converged;
434 
435  KRATOS_CATCH("");
436  }

442  bool AuxiliarySolveSolutionStep()
443  {
444  // Getting flag INTERACTION
445  ModelPart& r_model_part = StrategyBaseType::GetModelPart();
446  const bool update_each_nl_iteration = mThisParameters["update_each_nl_iteration"].GetBool();
447  VariableUtils().SetFlag(INTERACTION, update_each_nl_iteration, r_model_part.GetSubModelPart("ComputingContact").Conditions());
448 
449  // Pointers needed in the solution
450  typename TSchemeType::Pointer p_scheme = this->GetScheme();
451  typename TBuilderAndSolverType::Pointer p_builder_and_solver = this->GetBuilderAndSolver();
452  auto& r_dof_set = p_builder_and_solver->GetDofSet();
453 
454  TSystemMatrixType& rA = *BaseType::mpA;
455  TSystemVectorType& rDx = *BaseType::mpDx;
456  TSystemVectorType& rb = *BaseType::mpb;
457 
458  // Initializing the parameters of the Newton-Raphson cycle
459  unsigned int iteration_number = 1;
460  r_model_part.GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
461  bool is_converged = false;
462  bool residual_is_updated = false;
463 
464  // Computing nodal weights
465  ComputeNodalWeights();
466 
467  p_scheme->InitializeNonLinIteration(r_model_part, rA, rDx, rb);
468  BaseType::mpConvergenceCriteria->InitializeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
469  is_converged = BaseType::mpConvergenceCriteria->PreCriteria(r_model_part, p_builder_and_solver->GetDofSet(), rA, rDx, rb);
470 
471 // // If enforcing NTN
472 // const bool enforce_ntn = mThisParameters["enforce_ntn"].GetBool();
473 // if (enforce_ntn) {
474 // EnforcingNTN();
475 // }
476 
477  // Function to perform the building and the solving phase.
478  if (StrategyBaseType::mRebuildLevel > 0 || StrategyBaseType::mStiffnessMatrixIsBuilt == false) {
479  TSparseSpace::SetToZero(rA);
480  TSparseSpace::SetToZero(rDx);
481  TSparseSpace::SetToZero(rb);
482 
483  p_builder_and_solver->BuildAndSolve(p_scheme, r_model_part, rA, rDx, rb);
484  } else {
485  TSparseSpace::SetToZero(rDx); //Dx=0.00;
486  TSparseSpace::SetToZero(rb);
487 
488  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
489  }
490 
491  // Debugging info
492  BaseType::EchoInfo(iteration_number);
493 
494  // Updating the results stored in the database
495  BaseType::UpdateDatabase(rA, rDx, rb, StrategyBaseType::MoveMeshFlag());
496 
497  p_scheme->FinalizeNonLinIteration(r_model_part, rA, rDx, rb);
498  BaseType::mpConvergenceCriteria->FinalizeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
499 
500  // Calculate reactions if required
501  if (BaseType::mCalculateReactionsFlag)
502  p_builder_and_solver->CalculateReactions(p_scheme, r_model_part, rA, rDx, rb);
503 
504  if (is_converged) {
505  if (BaseType::mpConvergenceCriteria->GetActualizeRHSflag()) {
506  TSparseSpace::SetToZero(rb);
507 
508  p_builder_and_solver->BuildRHS(p_scheme, r_model_part, rb);
509  }
510 
511  is_converged = BaseType::mpConvergenceCriteria->PostCriteria(r_model_part, p_builder_and_solver->GetDofSet(), rA, rDx, rb);
512  }
513 
514  // Iteration Cycle... performed only for NonLinearProblems
515  while (!is_converged && iteration_number++ < BaseType::mMaxIterationNumber) {
516  // Setting the number of iteration
517  r_model_part.GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
518 
519  // Computing nodal weights
520  ComputeNodalWeights();
521 
522  // Calling InitializeNonLinIteration
523  p_scheme->InitializeNonLinIteration(r_model_part, rA, rDx, rb);
524  BaseType::mpConvergenceCriteria->InitializeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
525 
526  // Shaping correctly the system
527  if (update_each_nl_iteration) {
528  p_builder_and_solver->SetUpDofSet(p_scheme, r_model_part);
529  p_builder_and_solver->SetUpSystem(r_model_part);
530  p_builder_and_solver->ResizeAndInitializeVectors(p_scheme, BaseType::mpA, BaseType::mpDx, BaseType::mpb, r_model_part);
531  }
532 
533  is_converged = BaseType::mpConvergenceCriteria->PreCriteria(r_model_part, p_builder_and_solver->GetDofSet(), rA, rDx, rb);
534 
535  // Call the linear system solver to find the correction mDx for the it is not called if there is no system to solve
536  if (SparseSpaceType::Size(rDx) != 0) {
537  if (StrategyBaseType::mRebuildLevel > 1 || !StrategyBaseType::mStiffnessMatrixIsBuilt) {
538  if (!BaseType::GetKeepSystemConstantDuringIterations()) {
539  //A = 0.00;
540  TSparseSpace::SetToZero(rA);
541  TSparseSpace::SetToZero(rDx);
542  TSparseSpace::SetToZero(rb);
543 
544  p_builder_and_solver->BuildAndSolve(p_scheme, r_model_part, rA, rDx, rb);
545  } else {
546  TSparseSpace::SetToZero(rDx);
547  TSparseSpace::SetToZero(rb);
548 
549  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
550  }
551  } else {
552  TSparseSpace::SetToZero(rDx);
553  TSparseSpace::SetToZero(rb);
554 
555  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
556  }
557  } else {
558  KRATOS_WARNING("NO DOFS") << "ATTENTION: no free DOFs!! " << std::endl;
559  }
560 
561  // Debugging info
562  BaseType::EchoInfo(iteration_number);
563 
564  // Updating the results stored in the database
565  BaseType::UpdateDatabase(rA, rDx, rb, StrategyBaseType::MoveMeshFlag());
566 
567  p_scheme->FinalizeNonLinIteration(r_model_part, rA, rDx, rb);
568  BaseType::mpConvergenceCriteria->FinalizeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
569 
570  residual_is_updated = false;
571 
572  // Calculate reactions if required
573  if (BaseType::mCalculateReactionsFlag)
574  p_builder_and_solver->CalculateReactions(p_scheme, r_model_part, rA, rDx, rb);
575 
576  if (is_converged) {
577  if (BaseType::mpConvergenceCriteria->GetActualizeRHSflag()) {
578  TSparseSpace::SetToZero(rb);
579 
580  p_builder_and_solver->BuildRHS(p_scheme, r_model_part, rb);
581  residual_is_updated = true;
582  }
583 
584  is_converged = BaseType::mpConvergenceCriteria->PostCriteria(r_model_part, p_builder_and_solver->GetDofSet(), rA, rDx, rb);
585  }
586  }
587 
588  // Plots a warning if the maximum number of iterations is exceeded
589  if (iteration_number >= BaseType::mMaxIterationNumber) {
590  BaseType::MaxIterationsExceeded();
591  } else {
592  KRATOS_INFO_IF("NR-Strategy", this->GetEchoLevel() > 0) << "Convergence achieved after " << iteration_number << " / " << BaseType::mMaxIterationNumber << " iterations" << std::endl;
593  }
594 
595  // Recalculate residual if needed (note that some convergence criteria need it to be recalculated)
596  if (!residual_is_updated) {
597  // NOTE:
598  // The following part will be commented because it is time consuming
599  // and there is no obvious reason to be here. If someone need this
600  // part please notify the community via mailing list before uncommenting it.
601  // Pooyan.
602 
603 // TSparseSpace::SetToZero(mb);
604 // p_builder_and_solver->BuildRHS(p_scheme, r_model_part, mb);
605  }
606 
607  // Calculate reactions if required
608  if (BaseType::mCalculateReactionsFlag)
609  p_builder_and_solver->CalculateReactions(p_scheme, r_model_part, rA, rDx, rb);
610 
611  return is_converged;
612  }
