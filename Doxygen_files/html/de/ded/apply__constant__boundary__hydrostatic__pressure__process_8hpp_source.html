88 void ExecuteInitialize() override
89 {
90 KRATOS_TRY
91
92 if (mrModelPart.NumberOfNodes() > 0) {
93 const Variable<double> &var = KratosComponents< Variable<double> >::Get(mVariableName);
94
95 block_for_each(mrModelPart.Nodes(), [&var, this](Node<3>& rNode) {
96 if (mIsFixed) rNode.Fix(var);
97 else if (mIsFixedProvided) rNode.Free(var);
98
99 const double pressure = mSpecificWeight * (mReferenceCoordinate - rNode.Coordinates()[mGravityDirection]);
100
101 if (pressure > 0.0) {
102 rNode.FastGetSolutionStepValue(var) = pressure;
103 } else {
104 rNode.FastGetSolutionStepValue(var) = 0.0;
105 }
106 });
107 }
108
109 KRATOS_CATCH("")
110 }
