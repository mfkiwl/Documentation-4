91 void CalculateResidualNorm(
92 ModelPart& rModelPart,
93 TDataType& rResidualSolutionNorm,
94 typename BaseType::SizeType& rDofNum,
95 typename BaseType::DofsArrayType& rDofSet,
96 const typename BaseType::TSystemVectorType& rB) override
97 {
98 // Initialize
99 TDataType residual_solution_norm = TDataType();
100 long int local_dof_num = 0;
101
102 const int rank = rB.Comm().MyPID();
103
104 // Loop over Dofs
105 #pragma omp parallel for reduction(+:residual_solution_norm,local_dof_num)
106 for (int i = 0; i < static_cast<int>(rDofSet.size()); i++) {
107 auto it_dof = rDofSet.begin() + i;
108
109 typename BaseType::IndexType dof_id;
110 TDataType residual_dof_value;
111
112 if (it_dof->IsFree() && (it_dof->GetSolutionStepValue(PARTITION_INDEX) == rank)) {
113 dof_id = it_dof->EquationId();
114 residual_dof_value = TSparseSpace::GetValue(rB,dof_id);
115 residual_solution_norm += residual_dof_value * residual_dof_value;
116 local_dof_num++;
117 }
118 }
119
120 // Combine local contributions
121 // Note that I'm not merging the two calls because one adds doubles and the other ints (JC)
122 rB.Comm().SumAll(&residual_solution_norm,&rResidualSolutionNorm,1);
123 // SizeType is long unsigned int in linux, but EpetraComm does not support unsigned types
124 long int global_dof_num = 0;
125 rB.Comm().SumAll(&local_dof_num,&global_dof_num,1);
126 rDofNum = static_cast<typename BaseType::SizeType>(global_dof_num);
127
128 rResidualSolutionNorm = std::sqrt(rResidualSolutionNorm);
129 }
