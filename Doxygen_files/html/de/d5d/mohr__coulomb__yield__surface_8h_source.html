117 static void CalculateEquivalentStress(
118 const array_1d<double, VoigtSize>& rPredictiveStressVector,
119 const Vector& rStrainVector,
120 double& rEquivalentStress,
121 ConstitutiveLaw::Parameters& rValues
122 )
123 {
124 double I1, J2, J3, lode_angle;
125 array_1d<double, VoigtSize> deviator = ZeroVector(VoigtSize);
126
127 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateI1Invariant(rPredictiveStressVector, I1);
128 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ2Invariant(rPredictiveStressVector, I1, deviator, J2);
129 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(deviator, J3);
130 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
131
132 const Properties& r_material_properties = rValues.GetMaterialProperties();
133 const double friction_angle = r_material_properties[FRICTION_ANGLE] * Globals::Pi / 180.0;
134
135 rEquivalentStress = (std::cos(lode_angle) - std::sin(lode_angle) * std::sin(friction_angle) / std::sqrt(3.0)) * std::sqrt(J2) +
136 I1 * std::sin(friction_angle) / 3.0;
137 }

162 static void CalculateDamageParameter(
163 ConstitutiveLaw::Parameters& rValues,
164 double& rAParameter,
165 const double CharacteristicLength
166 )
167 {
168 const Properties& r_material_properties = rValues.GetMaterialProperties();
169 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
170 const double young_modulus = r_material_properties[YOUNG_MODULUS];
171 double equivalent_yield;
172 GetInitialUniaxialThreshold(rValues, equivalent_yield);
173 if (r_material_properties[SOFTENING_TYPE] == static_cast<int>(SofteningType::Exponential)) {
174 rAParameter = 1.00 / (fracture_energy * young_modulus / (CharacteristicLength * std::pow(equivalent_yield, 2)) - 0.5);
175 KRATOS_ERROR_IF(rAParameter < 0.0) << "Fracture Energy is too low, increase FRACTURE_ENERGY..." << std::endl;
176 } else { // linear
177 rAParameter = -std::pow(equivalent_yield, 2) / (2.0 * young_modulus * fracture_energy / CharacteristicLength);
178 }
179 }

211 static void CalculateYieldSurfaceDerivative(
212 const array_1d<double, VoigtSize>& rPredictiveStressVector,
213 const array_1d<double, VoigtSize>& rDeviator,
214 const double J2,
215 array_1d<double, VoigtSize>& rFFlux,
216 ConstitutiveLaw::Parameters& rValues
217 )
218 {
219 array_1d<double, VoigtSize> first_vector, second_vector, third_vector;
220 const Properties& r_material_properties = rValues.GetMaterialProperties();
221 const double friction_angle = r_material_properties[FRICTION_ANGLE] * Globals::Pi / 180.0;
222
223 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateFirstVector(first_vector);
224 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateSecondVector(rDeviator, J2, second_vector);
225 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateThirdVector(rDeviator, J2, third_vector);
226
227 double J3, lode_angle;
228 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateJ3Invariant(rDeviator, J3);
229 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateLodeAngle(J2, J3, lode_angle);
230
231 double c1, c3, c2;
232 double checker = std::abs(lode_angle * 180.0 / Globals::Pi);
233
234 if (std::abs(checker) < 29.0) { // If it is not the edge
235 c1 = std::sin(friction_angle) / 3.0;
236 c3 = (std::sqrt(3.0) * std::sin(lode_angle) + std::sin(friction_angle) * std::cos(lode_angle)) /
237 (2.0 * J2 * std::cos(3.0 * lode_angle));
238 c2 = 0.5 * std::cos(lode_angle)*(1.0 + std::tan(lode_angle) * std::sin(3.0 * lode_angle) +
239 std::sin(friction_angle) * (std::tan(3.0 * lode_angle) - std::tan(lode_angle)) / std::sqrt(3.0));
240 } else { // smoothing with drucker-praguer
241 c1 = 3.0 * (2.0 * std::sin(friction_angle) / (std::sqrt(3.0) * (3.0 - std::sin(friction_angle))));
242 c2 = 1.0;
243 c3 = 0.0;
244 }
245
246 noalias(rFFlux) = c1 * first_vector + c2 * second_vector + c3 * third_vector;
247 }
