165 void Update(
166 ModelPart& rModelPart,
167 DofsArrayType& rDofSet,
168 TSystemMatrixType& rA,
169 TSystemVectorType& rDx,
170 TSystemVectorType& rb
171 ) override
172 {
173 KRATOS_TRY;
174
175 DerivedBaseType::mpDofUpdater->UpdateDofs(rDofSet, rDx);
176
177 // Updating time derivatives (nodally for efficiency)
178 array_1d<double, 3 > delta_displacement;
179 block_for_each(rModelPart.Nodes(), delta_displacement, [&](Node<3>& rNode, array_1d<double,3>& rDeltaDisplacementTLS){
180
181 noalias(rDeltaDisplacementTLS) = rNode.FastGetSolutionStepValue(DISPLACEMENT) - rNode.FastGetSolutionStepValue(DISPLACEMENT, 1);
182
183 array_1d<double, 3>& r_current_velocity = rNode.FastGetSolutionStepValue(VELOCITY);
184 const array_1d<double, 3>& r_previous_velocity = rNode.FastGetSolutionStepValue(VELOCITY, 1);
185
186 noalias(r_current_velocity) = (this->mBossak.c1 * rDeltaDisplacementTLS - this->mBossak.c4 * r_previous_velocity);
187 });
188
189 KRATOS_CATCH( "" );
190 }

201 void Predict(
202 ModelPart& rModelPart,
203 DofsArrayType& rDofSet,
204 TSystemMatrixType& rA,
205 TSystemVectorType& rDx,
206 TSystemVectorType& rb
207 ) override
208 {
209 KRATOS_TRY;
210
211 // Process info
212 const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
213
214 // Delta time
215 const double delta_time = r_current_process_info[DELTA_TIME];
216
217 // Updating time derivatives (nodally for efficiency)
218 const auto it_node_begin = rModelPart.Nodes().begin();
219
220 // Auxiliar variables
221 const array_1d<double, 3> zero_array = ZeroVector(3);
222 array_1d<double, 3 > delta_displacement = zero_array;
223
224 // Getting position
225 const int disppos_x = it_node_begin->HasDofFor(DISPLACEMENT_X) ? static_cast<int>(it_node_begin->GetDofPosition(DISPLACEMENT_X)) : -1;
226 const int velpos_x = it_node_begin->HasDofFor(VELOCITY_X) ? static_cast<int>(it_node_begin->GetDofPosition(VELOCITY_X)) : -1;
227 const int disppos_y = it_node_begin->HasDofFor(DISPLACEMENT_Y) ? static_cast<int>(it_node_begin->GetDofPosition(DISPLACEMENT_Y)) : -1;
228 const int velpos_y = it_node_begin->HasDofFor(VELOCITY_Y) ? static_cast<int>(it_node_begin->GetDofPosition(VELOCITY_Y)) : -1;
229 const int disppos_z = it_node_begin->HasDofFor(DISPLACEMENT_Z) ? static_cast<int>(it_node_begin->GetDofPosition(DISPLACEMENT_Z)) : -1;
230 const int velpos_z = it_node_begin->HasDofFor(VELOCITY_Z) ? static_cast<int>(it_node_begin->GetDofPosition(VELOCITY_Z)) : -1;
231
232 block_for_each(rModelPart.Nodes(), delta_displacement, [&](Node<3>& rNode, array_1d<double,3>& rDeltaDisplacementTLS){
233
234 bool predicted_x = false;
235 bool predicted_y = false;
236 bool predicted_z = false;
237
238 //Predicting: r_current_displacement = r_previous_displacement + r_previous_velocity * delta_time;
239 //ATTENTION::: the prediction is performed only on free nodes
240
241 const array_1d<double, 3>& r_previous_velocity = rNode.FastGetSolutionStepValue(VELOCITY, 1);
242 const array_1d<double, 3>& r_previous_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT, 1);
243 array_1d<double, 3>& r_current_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION);
244 array_1d<double, 3>& r_current_velocity = rNode.FastGetSolutionStepValue(VELOCITY);
245 array_1d<double, 3>& r_current_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT);
246
247 if (velpos_x > -1) {
248 if (rNode.GetDof(VELOCITY_X, velpos_x).IsFixed()) {
249 rDeltaDisplacementTLS[0] = (r_current_velocity[0] + this->mBossak.c4 * r_previous_velocity[0])/this->mBossak.c1;
250 r_current_displacement[0] = r_previous_displacement[0] + rDeltaDisplacementTLS[0];
251 predicted_x = true;
252 }
253 }
254 if (disppos_x > -1 && !predicted_x) {
255 if (!rNode.GetDof(DISPLACEMENT_X, disppos_x).IsFixed() && !predicted_x) {
256 r_current_displacement[0] = r_previous_displacement[0] + delta_time * r_previous_velocity[0];
257 }
258 }
259
260 if (velpos_y > -1) {
261 if (rNode.GetDof(VELOCITY_Y, velpos_y).IsFixed()) {
262 rDeltaDisplacementTLS[1] = (r_current_velocity[1] + this->mBossak.c4 * r_previous_velocity[1])/this->mBossak.c1;
263 r_current_displacement[1] = r_previous_displacement[1] + rDeltaDisplacementTLS[1];
264 predicted_y = true;
265 }
266 }
267 if (disppos_y > -1 && !predicted_y) {
268 if (!rNode.GetDof(DISPLACEMENT_Y, disppos_y).IsFixed() && !predicted_y) {
269 r_current_displacement[1] = r_previous_displacement[1] + delta_time * r_previous_velocity[1];
270 }
271 }
272
273 if (velpos_z > -1) {
274 if (rNode.GetDof(VELOCITY_Z, velpos_z).IsFixed()) {
275 rDeltaDisplacementTLS[2] = (r_current_velocity[2] + this->mBossak.c4 * r_previous_velocity[2])/this->mBossak.c1;
276 r_current_displacement[2] = r_previous_displacement[2] + rDeltaDisplacementTLS[2];
277 predicted_z = true;
278 }
279 }
280 if (disppos_z > -1 && !predicted_z) {
281 if (!rNode.GetDof(DISPLACEMENT_Z, disppos_z).IsFixed() && !predicted_z) {
282 r_current_displacement[2] = r_previous_displacement[2] + delta_time * r_previous_velocity[2];
283 }
284 }
285
286 // Updating time derivatives
287 noalias(r_current_acceleration) = zero_array;
288 noalias(r_current_velocity) = r_previous_velocity;
289 });
290
291 KRATOS_CATCH( "" );
292 }

380 LocalSystemMatrixType& rLHSContribution,
381 LocalSystemMatrixType& rD,
382 LocalSystemMatrixType& rM,
383 const ProcessInfo& rCurrentProcessInfo
384 ) override
385 {
386 // Adding damping contribution

388 noalias(rLHSContribution) += rD * this->mBossak.c1;
389 else if (rM.size1() != 0) {

391 noalias(rLHSContribution) += rM * beta * this->mBossak.c1;
392 }
