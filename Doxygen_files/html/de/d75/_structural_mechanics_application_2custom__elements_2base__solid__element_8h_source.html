1023 template<class TType>
1024 void CalculateOnConstitutiveLaw(
1025 const Variable<TType>& rVariable,
1026 std::vector<TType>& rOutput,
1027 const ProcessInfo& rCurrentProcessInfo
1028 )
1029 {
1030 const bool is_rotated = IsElementRotated();
1031 const GeometryType::IntegrationPointsArrayType& integration_points = GetGeometry().IntegrationPoints( this->GetIntegrationMethod() );
1032
1033 const SizeType number_of_nodes = GetGeometry().size();
1034 const SizeType dimension = GetGeometry().WorkingSpaceDimension();
1035 const SizeType strain_size = mConstitutiveLawVector[0]->GetStrainSize();
1036
1037 KinematicVariables this_kinematic_variables(strain_size, dimension, number_of_nodes);
1038 ConstitutiveVariables this_constitutive_variables(strain_size);
1039
1040 // Create constitutive law parameters:
1041 ConstitutiveLaw::Parameters Values(GetGeometry(),GetProperties(),rCurrentProcessInfo);
1042
1043 // Set constitutive law flags:
1044 Flags& ConstitutiveLawOptions=Values.GetOptions();
1045 ConstitutiveLawOptions.Set(ConstitutiveLaw::USE_ELEMENT_PROVIDED_STRAIN, UseElementProvidedStrain());
1046 ConstitutiveLawOptions.Set(ConstitutiveLaw::COMPUTE_STRESS, true);
1047 ConstitutiveLawOptions.Set(ConstitutiveLaw::COMPUTE_CONSTITUTIVE_TENSOR, false);
1048
1049 Values.SetStrainVector(this_constitutive_variables.StrainVector);
1050
1051 for (IndexType point_number = 0; point_number < integration_points.size(); ++point_number) {
1052 // Compute element kinematics B, F, DN_DX ...
1053 this->CalculateKinematicVariables(this_kinematic_variables, point_number, this->GetIntegrationMethod());
1054
1055 // Compute material reponse
1056 this->SetConstitutiveVariables(this_kinematic_variables, this_constitutive_variables, Values, point_number, integration_points);
1057
1058 // rotate to local axes strain/F
1059 if (is_rotated)
1060 RotateToLocalAxes(Values, this_kinematic_variables);
1061
1062 rOutput[point_number] = mConstitutiveLawVector[point_number]->CalculateValue( Values, rVariable, rOutput[point_number] );
1063 }
1064 }
