121 const Properties& r_material_properties = rValues.GetMaterialProperties();

123 array_1d<double, VoigtSize> first_vector, second_vector, third_vector;
124
125 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateFirstVector(first_vector);
126 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculateSecondVector(rDeviator, J2, second_vector);
127
128 const double dilatancy = r_material_properties[DILATANCY_ANGLE] * Globals::Pi / 180.0;
129 const double sin_dil = std::sin(dilatancy);
130 const double Root3 = std::sqrt(3.0);
131
132 const double CFL = -Root3 * (3.0 - sin_dil) / (3.0 * sin_dil - 3.0);
133 const double c1 = CFL * 2.0 * sin_dil / (Root3 * (3.0 - sin_dil));
134 const double c2 = CFL;
135
136 noalias(rGFlux) = c1 * first_vector + c2 * second_vector;
137 }
