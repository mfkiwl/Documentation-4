185 void CalculateConstitutiveMatrix(Matrix& rConstitutiveMatrix,
186 const Properties& rProperties) override
187 {
188 KRATOS_TRY
189
190 // Lame constants
191 const double& rYoungModulus = rProperties[YOUNG_MODULUS];
192 const double& rPoissonCoefficient = rProperties[POISSON_RATIO];
193
194 // Plane strain constitutive matrix
195 rConstitutiveMatrix ( 0 , 0 ) = (rYoungModulus*(1.0-rPoissonCoefficient)/((1.0+rPoissonCoefficient)*(1.0-2.0*rPoissonCoefficient)));
196 rConstitutiveMatrix ( 1 , 1 ) = rConstitutiveMatrix ( 0 , 0 );
197
198 rConstitutiveMatrix ( 2 , 2 ) = rConstitutiveMatrix ( 0 , 0 )*(1.0-2.0*rPoissonCoefficient)/(2.0*(1.0-rPoissonCoefficient));
199
200 rConstitutiveMatrix ( 0 , 1 ) = rConstitutiveMatrix ( 0 , 0 )*rPoissonCoefficient/(1.0-rPoissonCoefficient);
201 rConstitutiveMatrix ( 1 , 0 ) = rConstitutiveMatrix ( 0 , 1 );
202
203 rConstitutiveMatrix ( 0 , 2 ) = 0.0;
204 rConstitutiveMatrix ( 2 , 0 ) = 0.0;
205 rConstitutiveMatrix ( 1 , 2 ) = 0.0;
206 rConstitutiveMatrix ( 2 , 1 ) = 0.0;
