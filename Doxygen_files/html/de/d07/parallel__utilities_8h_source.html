140template<
141 class TContainerType,
142 class TIteratorType=decltype(std::declval<TContainerType>().begin()),
143 int TMaxThreads=Globals::MaxAllowedThreads
144 >
145class BlockPartition
146{
147public:
152 BlockPartition(TIteratorType it_begin,
153 TIteratorType it_end,
154 int Nchunks = ParallelUtilities::GetNumThreads())
155 {
156 KRATOS_ERROR_IF(Nchunks < 1) << "Number of chunks must be > 0 (and not " << Nchunks << ")" << std::endl;
157
158 const std::ptrdiff_t size_container = it_end-it_begin;
159
160 if (size_container == 0) {
161 mNchunks = Nchunks;
162 } else {
163 // in case the container is smaller than the number of chunks
164 mNchunks = std::min(static_cast<int>(size_container), Nchunks);
165 }
166 const std::ptrdiff_t block_partition_size = size_container / mNchunks;
167 mBlockPartition[0] = it_begin;
168 mBlockPartition[mNchunks] = it_end;
169 for (int i=1; i<mNchunks; i++) {
170 mBlockPartition[i] = mBlockPartition[i-1] + block_partition_size;
171 }
172 }
173
177 template <class TData>
178 BlockPartition(TData &&rData, int Nchunks = ParallelUtilities::GetNumThreads())
179 : BlockPartition(rData.begin(), rData.end(), Nchunks)
180 {}

182 virtual ~BlockPartition() = default;

187 template <class TUnaryFunction>
188 inline void for_each(TUnaryFunction&& f)
189 {
190 KRATOS_PREPARE_CATCH_THREAD_EXCEPTION
191
192 #pragma omp parallel for
193 for (int i=0; i<mNchunks; ++i) {
194 KRATOS_TRY
195 for (auto it = mBlockPartition[i]; it != mBlockPartition[i+1]; ++it) {
196 f(*it); //note that we pass the value to the function, not the iterator
197 }
198 KRATOS_CATCH_THREAD_EXCEPTION
199 }
200
201 KRATOS_CHECK_AND_THROW_THREAD_EXCEPTION
202 }

382 //pure c++11 version (can handle exceptions)
383 template <class TUnaryFunction>
384 inline void for_pure_c11(TUnaryFunction &&f)
385 {
386 std::vector< std::future<void> > runners(mNchunks);
387 const auto& partition = mBlockPartition;
388 for (int i=0; i<mNchunks; ++i) {
389 runners[i] = std::async(std::launch::async, [&partition, i, &f]()
390 {
391 for (auto k = partition[i]; k < partition[i+1]; ++k) {
392 f(k);
393 }
394 });
395 }
396
397 //here we impose a syncronization and we check the exceptions
398 for(int i=0; i<mNchunks; ++i) {
399 try {
400 runners[i].get();
401 }
402 catch(Exception& e) {
403 KRATOS_ERROR << std::endl << "THREAD number: " << i << " caught exception " << e.what() << std::endl;
404 } catch(std::exception& e) {
405 KRATOS_ERROR << std::endl << "THREAD number: " << i << " caught exception " << e.what() << std::endl;
406 } catch(...) {
407 KRATOS_ERROR << std::endl << "unknown error" << std::endl;
408 }
409 }
