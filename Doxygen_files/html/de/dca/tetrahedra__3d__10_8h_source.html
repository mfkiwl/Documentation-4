486 ) const override

492 if ( (rResult[1] >= (0.0 - Tolerance)) && (rResult[1] <= (1.0 + Tolerance)) )

698 double ShapeFunctionValue( IndexType ShapeFunctionIndex,
699 const CoordinatesArrayType& rPoint ) const override
700 {
701 double fourthCoord = 1.0 - ( rPoint[0] + rPoint[1] + rPoint[2] );
702
703 switch ( ShapeFunctionIndex )
704 {
705 case 0:
706 return( fourthCoord*( 2*fourthCoord - 1 ) );
707 case 1:
708 return( rPoint[0]*( 2*rPoint[0] - 1 ) );
709 case 2:
710 return( rPoint[1]*( 2*rPoint[1] - 1 ) );
711 case 3:
712 return( rPoint[2]*( 2*rPoint[2] - 1 ) );
713 case 4:
714 return( 4*fourthCoord*rPoint[0] );
715 case 5:
716 return( 4*rPoint[0]*rPoint[1] );
717 case 6:
718 return( 4*fourthCoord*rPoint[1] );
719 case 7:
720 return( 4*fourthCoord*rPoint[2] );
721 case 8:
722 return( 4*rPoint[0]*rPoint[2] );
723 case 9:
724 return( 4*rPoint[1]*rPoint[2] );
725 default:
726 KRATOS_ERROR << "Wrong index of shape function!" << *this << std::endl;
727 }
728
729 return 0;
730 }

732 Matrix& ShapeFunctionsLocalGradients(Matrix& rResult,
733 const CoordinatesArrayType& rPoint) const override
734 {
735 const double fourthCoord = 1.0 - (rPoint[0] + rPoint[1] + rPoint[2]);
736 if (rResult.size1() != this->size() || rResult.size2() != this->LocalSpaceDimension())
737 rResult.resize(this->size(), this->LocalSpaceDimension(), false);
738
739 rResult(0, 0) = -(4.0 * fourthCoord - 1.0);
740 rResult(0, 1) = -(4.0 * fourthCoord - 1.0);
741 rResult(0, 2) = -(4.0 * fourthCoord - 1.0);
742 rResult(1, 0) = 4.0 * rPoint[0] - 1.0;
743 rResult(1, 1) = 0.0;
744 rResult(1, 2) = 0.0;
745 rResult(2, 0) = 0.0;
746 rResult(2, 1) = 4.0 * rPoint[1] - 1.0;
747 rResult(2, 2) = 0.0;
748 rResult(3, 0) = 0.0;
749 rResult(3, 1) = 0.0;
750 rResult(3, 2) = 4.0 * rPoint[2] - 1.0;
751 rResult(4, 0) = -4.0 * rPoint[0] + 4.0 * fourthCoord;
752 rResult(4, 1) = -4.0 * rPoint[0];
753 rResult(4, 2) = -4.0 * rPoint[0];
754 rResult(5, 0) = 4.0 * rPoint[1];
755 rResult(5, 1) = 4.0 * rPoint[0];
756 rResult(5, 2) = 0.0;
757 rResult(6, 0) = -4.0 * rPoint[1];
758 rResult(6, 1) = -4.0 * rPoint[1] + 4.0 * fourthCoord;
759 rResult(6, 2) = -4.0 * rPoint[1];
760 rResult(7, 0) = -4.0 * rPoint[2];
761 rResult(7, 1) = -4.0 * rPoint[2];
762 rResult(7, 2) = -4.0 * rPoint[2] + 4.0 * fourthCoord;
763 rResult(8, 0) = 4.0 * rPoint[2];
764 rResult(8, 1) = 0.0;
765 rResult(8, 2) = 4.0 * rPoint[0];
766 rResult(9, 0) = 0.0;
767 rResult(9, 1) = 4.0 * rPoint[2];
768 rResult(9, 2) = 4.0 * rPoint[1];

888 static void ShapeFunctionsValuesImpl(Vector &rResult, const CoordinatesArrayType& rCoordinates)
889 {
890 if (rResult.size() != 10)
891 rResult.resize(10, false);
892 const double fourthCoord = 1.0 - rCoordinates[0] - rCoordinates[1] - rCoordinates[2];
893 rResult[0] = fourthCoord * (2.0 * fourthCoord - 1.0);
894 rResult[1] = rCoordinates[0] * (2.0 * rCoordinates[0] - 1.0);
895 rResult[2] = rCoordinates[1] * (2.0 * rCoordinates[1] - 1.0);
896 rResult[3] = rCoordinates[2] * (2.0 * rCoordinates[2] - 1.0);
897 rResult[4] = 4.0 * fourthCoord * rCoordinates[0];
898 rResult[5] = 4.0 * rCoordinates[0] * rCoordinates[1];
899 rResult[6] = 4.0 * rCoordinates[1] * fourthCoord;
900 rResult[7] = 4.0 * rCoordinates[2] * fourthCoord;
901 rResult[8] = 4.0 * rCoordinates[0] * rCoordinates[2];
902 rResult[9] = 4.0 * rCoordinates[1] * rCoordinates[2];

914 {
915 const std::size_t points_number = 10;

917 IntegrationPointsArrayType integration_points = all_integration_points[static_cast<int>(ThisMethod)];
918 Matrix shape_function_values(integration_points.size(), points_number);

939 static ShapeFunctionsGradientsType
940 CalculateShapeFunctionsIntegrationPointsLocalGradients(
941 typename BaseType::IntegrationMethod ThisMethod )
942 {
943 IntegrationPointsContainerType all_integration_points =
944 AllIntegrationPoints();
945 IntegrationPointsArrayType integration_points = all_integration_points[static_cast<int>(ThisMethod)];
946 //number of integration points
947 const int integration_points_number = integration_points.size();
948 ShapeFunctionsGradientsType d_shape_f_values( integration_points_number );
949 //initialising container
950 //loop over all integration points
951
952 for ( int pnt = 0; pnt < integration_points_number; pnt++ )
953 {
954 double fourthCoord = 1.0 - ( integration_points[pnt].X() + integration_points[pnt].Y() + integration_points[pnt].Z() );
955 double fourthCoord_DX = -1.0;
956 double fourthCoord_DY = -1.0;
957 double fourthCoord_DZ = -1.0;
958
959 Matrix result = ZeroMatrix( 10, 3 );
960 result( 0, 0 ) = ( 4 * fourthCoord - 1.0 ) * fourthCoord_DX;
961 result( 0, 1 ) = ( 4 * fourthCoord - 1.0 ) * fourthCoord_DY;
962 result( 0, 2 ) = ( 4 * fourthCoord - 1.0 ) * fourthCoord_DZ;
963 result( 1, 0 ) = 4 * integration_points[pnt].X() - 1.0;
964 result( 1, 1 ) = 0.0;
965 result( 1, 2 ) = 0.0;
966 result( 2, 0 ) = 0.0;
967 result( 2, 1 ) = 4 * integration_points[pnt].Y() - 1.0;
968 result( 2, 2 ) = 0.0;
969 result( 3, 0 ) = 0.0;
970 result( 3, 1 ) = 0.0;
971 result( 3, 2 ) = 4 * integration_points[pnt].Z() - 1.0 ;
972 result( 4, 0 ) = 4 * fourthCoord_DX * integration_points[pnt].X() + 4 * fourthCoord;
973 result( 4, 1 ) = 4 * fourthCoord_DY * integration_points[pnt].X();
974 result( 4, 2 ) = 4 * fourthCoord_DZ * integration_points[pnt].X();
975 result( 5, 0 ) = 4 * integration_points[pnt].Y();
976 result( 5, 1 ) = 4 * integration_points[pnt].X();
977 result( 5, 2 ) = 0.0;
978 result( 6, 0 ) = 4 * fourthCoord_DX * integration_points[pnt].Y();
979 result( 6, 1 ) = 4 * fourthCoord_DY * integration_points[pnt].Y() + 4 * fourthCoord;
980 result( 6, 2 ) = 4 * fourthCoord_DZ * integration_points[pnt].Y();
981 result( 7, 0 ) = 4 * fourthCoord_DX * integration_points[pnt].Z();
982 result( 7, 1 ) = 4 * fourthCoord_DY * integration_points[pnt].Z();
983 result( 7, 2 ) = 4 * fourthCoord_DZ * integration_points[pnt].Z() + 4 * fourthCoord;
984 result( 8, 0 ) = 4 * integration_points[pnt].Z();
985 result( 8, 1 ) = 0.0;
986 result( 8, 2 ) = 4 * integration_points[pnt].X();
987 result( 9, 0 ) = 0.0;
988 result( 9, 1 ) = 4 * integration_points[pnt].Z();
989 result( 9, 2 ) = 4 * integration_points[pnt].Y();
990
991 d_shape_f_values[pnt] = result;
992 }
993
994 return d_shape_f_values;
