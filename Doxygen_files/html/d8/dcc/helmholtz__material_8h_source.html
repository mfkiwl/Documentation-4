325 void CalculateNodeNeighbourCount()
326 {
327 for(int model_i =0;model_i<mpVMModelParts.size();model_i++)
328 {
329 ModelPart* mpVMModePart = mpVMModelParts[model_i];
330 auto& r_nodes = mpVMModePart->Nodes();
331 int mNumNodes = r_nodes.size();
332
333 VariableUtils variable_utils;
334 variable_utils.SetFlag(STRUCTURE,true,r_nodes);
335
336 // Note: this should not be parallel, the operation is not threadsafe if the variable is uninitialized
337 for (auto& r_node : r_nodes)
338 {
339 r_node.SetValue(NUMBER_OF_NEIGHBOUR_ELEMENTS,0);
340 }
341
342 mNumNodes = mpVMModePart->GetCommunicator().GetDataCommunicator().SumAll(mNumNodes);
343
344 auto& r_elements = mpVMModePart->Elements();
345 const int num_elements = r_elements.size();
346
347 #pragma omp parallel for
348 for (int i = 0; i < num_elements; i++)
349 {
350 auto i_elem = r_elements.begin() + i;
351 auto& r_geom = i_elem->GetGeometry();
352 for (unsigned int i = 0; i < r_geom.PointsNumber(); i++)
353 {
354 auto& r_node = r_geom[i];
355 if (r_node.Is(STRUCTURE))
356 {
357 r_node.SetLock();
358 r_node.GetValue(NUMBER_OF_NEIGHBOUR_ELEMENTS) += 1;
359 r_node.UnSetLock();
360 }
361 }
362 }
363
364 mpVMModePart->GetCommunicator().AssembleNonHistoricalData(NUMBER_OF_NEIGHBOUR_ELEMENTS);
365
366 }

372 for(auto& control_obj : mControlSettings["controlling_objects"]){
373 ModelPart& r_controlling_object = mrModel.GetModelPart(control_obj.GetString());

379 if (root_model_part.HasSubModelPart(vm_model_part_name)){

381 for(int i =0; i<mpVMModelParts.size(); i++)
382 if(mpVMModelParts[i]->Name()==p_vm_model_part->Name())

387 p_vm_model_part_property = p_vm_model_part->CreateNewProperties(root_model_part.NumberOfProperties()+1);

435 void ComputeFilteredDensity(){
436
437 for(int model_i=0;model_i<mpVMModelParts.size();model_i++){
438
439 //first update control density
440 AddVariable1ToVarible2(mpVMModelParts[model_i],D_CD,CD);
441
442 //now filter nodal control desnity
443 //first we need to multiply with mass matrix
444 SetVariable(mpVMModelParts[model_i],HELMHOLTZ_VAR_DENSITY,0.0);
445 SetVariable(mpVMModelParts[model_i],HELMHOLTZ_SOURCE_DENSITY,0.0);
446 //now we need to multiply with the mass matrix
447 #pragma omp parallel for
448 for(auto& elem_i : mpVMModelParts[model_i]->Elements())
449 {
450 VectorType origin_values;
451 GetElementVariableValuesVector(elem_i,CD,origin_values);
452 MatrixType mass_matrix;
453 elem_i.Calculate(HELMHOLTZ_MASS_MATRIX,mass_matrix,mpVMModelParts[model_i]->GetProcessInfo());
454 VectorType int_vals = prod(mass_matrix,origin_values);
455 AddElementVariableValuesVector(elem_i,HELMHOLTZ_SOURCE_DENSITY,int_vals);
456 }
457
458 //now we need to apply BCs
459 #pragma omp parallel for
460 for(auto& node_i : mpVMModelParts[model_i]->Nodes())
461 {
462 auto& r_nodal_variable_hl_var = node_i.FastGetSolutionStepValue(HELMHOLTZ_VAR_DENSITY);
463 if(node_i.IsFixed(HELMHOLTZ_VAR_DENSITY))
464 r_nodal_variable_hl_var = node_i.FastGetSolutionStepValue(FD);
465 }
466 mpStrategies[model_i]->Solve();
467 SetVariable1ToVarible2(mpVMModelParts[model_i],HELMHOLTZ_VAR_DENSITY,FD);
468 }

471 void ComputePhyiscalDensity(){
472
473 for(int model_i=0;model_i<mpVMModelParts.size();model_i++){
474 //now do the projection and then set the PD
475 #pragma omp parallel for
476 for(auto& node_i : mpVMModelParts[model_i]->Nodes()){
477 const auto& filtered_density = node_i.FastGetSolutionStepValue(FD);
478 auto& physical_density = node_i.FastGetSolutionStepValue(PD);
479 auto& physical_density_der = node_i.FastGetSolutionStepValue(D_PD_D_FD);
480 physical_density = ProjectForward(filtered_density,filtered_densities,physical_densities,beta);
481 physical_density_der = ProjectionDerivative(filtered_density,filtered_densities,physical_densities,beta);
482 }
483 }
484
485 // update elements' density
486 for(auto& control_obj : mControlSettings["controlling_objects"]){
487 ModelPart& r_controlling_object = mrModel.GetModelPart(control_obj.GetString());
488 #pragma omp parallel for
489 for (int i = 0; i < (int)r_controlling_object.Elements().size(); i++) {
490 ModelPart::ElementsContainerType::iterator it = r_controlling_object.ElementsBegin() + i;
491 double elem_i_density = 0.0;
492 for(unsigned int node_element = 0; node_element<it->GetGeometry().size(); node_element++)
493 elem_i_density += it->GetGeometry()[node_element].FastGetSolutionStepValue(PD);
494 elem_i_density /= it->GetGeometry().size();
495 it->GetProperties().SetValue(DENSITY,elem_i_density);
496 }
497 }

529 double ProjectForward(double x, Vector x_limits, Vector y_limits, double beta, int penal_fac = 1){
530
531 double x1,x2,y1,y2;
532 int index_x1 = 0;
533 if(x>=x_limits[x_limits.size()-1]){
534 x1=x_limits[x_limits.size()-2];
535 index_x1 = x_limits.size()-2;
536 x2=x_limits[x_limits.size()-1];
537 y1=y_limits[y_limits.size()-2];
538 y2=y_limits[y_limits.size()-1];
539 }
540 else if(x<=x_limits[0]){
541 x1=x_limits[0];
542 index_x1 = 0;
543 x2=x_limits[1];
544 y1=y_limits[0];
545 y2=y_limits[1];
546 }
547 else{
548 for(int i=0;i<x_limits.size()-1;i++)
549 if((x>=x_limits[i]) && (x<=x_limits[i+1]))
550 {
551 y1 = y_limits[i];
552 y2 = y_limits[i+1];
553 x1 = x_limits[i];
554 index_x1 = i;
555 x2 = x_limits[i+1];
556 break;
557 }
558 }
559
560 double pow_val = -2.0*beta*(x-(x1+x2)/2);
561
562 return (y2-y1)/(std::pow(1+std::exp(pow_val),penal_fac)) + y1;
563 }

566 double ProjectBackward(double y,Vector x_limits,Vector y_limits,double beta){
567
568 double x = 0;
569 if(y>=y_limits[y_limits.size()-1])
570 x = x_limits[y_limits.size()-1];
571 else if(y<=y_limits[0])
572 x = x_limits[0];
573 else{
574 for(int i=0;i<y_limits.size()-1;i++)
575 if((y>y_limits[i]) && (y<y_limits[i+1]))
576 {
577 double y1 = y_limits[i];
578 double y2 = y_limits[i+1];
579 double x1 = x_limits[i];
580 double x2 = x_limits[i+1];
581 x = ((x2+x1)/2.0) + (1.0/(-2.0*beta)) * std::log(((y2-y1)/(y-y1))-1);
582 break;
583 }
584 else if(y==y_limits[i]){
585 x = x_limits[i];
586 break;
587 }
588 else if(y==y_limits[i+1]){
589 x = x_limits[i+1];
590 break;
591 }
592 }
593 return x;
594 }

596 double ProjectionDerivative(double x,Vector x_limits,Vector y_limits,double beta,int penal_fac = 1){
597
598 double dfdx = 0;
599 double x1,x2,y1,y2;
600 if(x>=x_limits[x_limits.size()-1]){
601 x1=x_limits[x_limits.size()-2];
602 x2=x_limits[x_limits.size()-1];
603 y1=y_limits[y_limits.size()-2];
604 y2=y_limits[y_limits.size()-1];
605 }
606 else if(x<=x_limits[0]){
607 x1=x_limits[0];
608 x2=x_limits[1];
609 y1=y_limits[0];
610 y2=y_limits[1];
611 }
612 else{
613 for(int i=0;i<x_limits.size()-1;i++)
614 if((x>=x_limits[i]) && (x<=x_limits[i+1]))
615 {
616 y1 = y_limits[i];
617 y2 = y_limits[i+1];
618 x1 = x_limits[i];
619 x2 = x_limits[i+1];
620 break;
621 }
622 }
623
624 double pow_val = -2.0*beta*(x-(x1+x2)/2);
625 double dydx = (y2-y1) * (1.0/std::pow(1+std::exp(pow_val),penal_fac+1)) * penal_fac * 2.0 * beta * std::exp(pow_val);
626
627 // if (y2<y1)
628 // dydx *=-1;
629
630 return dydx;
