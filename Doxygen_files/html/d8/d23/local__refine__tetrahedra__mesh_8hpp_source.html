82 unsigned int CreateCenterNode(Geometry<Node < 3 > >& geom, ModelPart& model_part)
83 {
84 // Determine a new unique id
85 unsigned int new_id = (model_part.NodesEnd() - 1)->Id() + 1;
86
87 if( model_part.Nodes().find(new_id) != model_part.NodesEnd() )
88 {
89 KRATOS_THROW_ERROR(std::logic_error, "adding a center node with an already existing id","");
90 }
91
92 // Determine the coordinates of the new node
93 double X = (geom[0].X() + geom[1].X() + geom[2].X() + geom[3].X()) / 4.0;
94 double Y = (geom[0].Y() + geom[1].Y() + geom[2].Y() + geom[3].Y()) / 4.0;
95 double Z = (geom[0].Z() + geom[1].Z() + geom[2].Z() + geom[3].Z()) / 4.0;
96
97 double X0 = (geom[0].X0() + geom[1].X0() + geom[2].X0() + geom[3].X0()) / 4.0;
98 double Y0 = (geom[0].Y0() + geom[1].Y0() + geom[2].Y0() + geom[3].Y0()) / 4.0;
99 double Z0 = (geom[0].Z0() + geom[1].Z0() + geom[2].Z0() + geom[3].Z0()) / 4.0;
100
101 // Generate the new node
102 Node < 3 > ::Pointer pnode = model_part.CreateNewNode(new_id, X, Y, Z);
103
104 unsigned int buffer_size = model_part.NodesBegin()->GetBufferSize();
105 pnode->SetBufferSize(buffer_size);
106
107 pnode->X0() = X0;
108 pnode->Y0() = Y0;
109 pnode->Z0() = Z0;
110
111 // Add the dofs
112 Node < 3 > ::DofsContainerType& reference_dofs = (model_part.NodesBegin())->GetDofs();
113 unsigned int step_data_size = model_part.GetNodalSolutionStepDataSize();
114
115 for (Node < 3 > ::DofsContainerType::iterator iii = reference_dofs.begin(); iii != reference_dofs.end(); iii++)
116 {
117 Node < 3 > ::DofType& rDof = **iii;
118 Node < 3 > ::DofType::Pointer p_new_dof = pnode->pAddDof(rDof);
119
120 // The variables are left as free for the internal node
121 (p_new_dof)->FreeDof();
122 }
123
124 // Intepolating the data
125 for (unsigned int step = 0; step < buffer_size; step++)
126 {
127 double* new_step_data = pnode->SolutionStepData().Data(step);
128 double* step_data1 = geom[0].SolutionStepData().Data(step);
129 double* step_data2 = geom[1].SolutionStepData().Data(step);
130 double* step_data3 = geom[2].SolutionStepData().Data(step);
131 double* step_data4 = geom[3].SolutionStepData().Data(step);
132
133 // Copying this data in the position of the vector we are interested in
134 for (unsigned int j = 0; j < step_data_size; j++)
135 {
136 new_step_data[j] = 0.25 * (step_data1[j] + step_data2[j] + step_data3[j] + step_data4[j]);
137 }
138 }
139
140 pnode->GetValue(FATHER_NODES).resize(0);
141 pnode->GetValue(FATHER_NODES).push_back( Node< 3 >::WeakPointer( geom(0) ) );
142 pnode->GetValue(FATHER_NODES).push_back( Node< 3 >::WeakPointer( geom(1) ) );
143 pnode->GetValue(FATHER_NODES).push_back( Node< 3 >::WeakPointer( geom(2) ) );
144 pnode->GetValue(FATHER_NODES).push_back( Node< 3 >::WeakPointer( geom(3) ) );
145
146 return new_id;
147 }
