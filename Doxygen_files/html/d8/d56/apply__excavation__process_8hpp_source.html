76 if (mrModelPart.NumberOfElements() > 0) {

94 }

96 }

100 if (mDeactivateSoilPart) {
101 block_for_each(mrModelPart.Conditions(), [&](Condition& rCondition) {
102 rCondition.Set(ACTIVE, false);
103 });
104 } else {
105 block_for_each(mrModelPart.Conditions(), [&](Condition& rCondition) {
106 rCondition.Set(ACTIVE, true);
107 });
108 }
