176 void Execute() override
177 {
178
179 KRATOS_TRY
180
181 ProcessInfo& rCurrentProcessInfo = this->mrModelPart.GetProcessInfo();
182
183 const double& rCurrentTime = rCurrentProcessInfo[TIME];
184
185 if( this->mEntity == EntityType::NODES || this->mEntity == EntityType::CONDITIONS ){
186
187 if( KratosComponents< Variable<double> >::Has( this->mvariable_name ) ) //case of double variable
188 {
189 AssignValueToNodes<>(KratosComponents< Variable<double> >::Get(this->mvariable_name), rCurrentTime);
190 }
191 else if( KratosComponents< Variable<Vector> >::Has( this->mvariable_name ) ) //case of vector variable
192 {
193 AssignValueToConditions<>(KratosComponents< Variable<Vector> >::Get(this->mvariable_name), rCurrentTime);
194 }

197 KRATOS_ERROR << "Not able to set the variable. Attempting to set variable:" << this->mvariable_name << std::endl;
198 }
199

202
203 if( KratosComponents< Variable<Vector> >::Has( this->mvariable_name ) ) //case of vector variable

205 AssignValueToElements<>(KratosComponents< Variable<Vector> >::Get(this->mvariable_name), rCurrentTime);
206 }
207 else
208 {
209 KRATOS_ERROR << "Not able to set the variable. Attempting to set variable:" << this->mvariable_name << std::endl;

348 void LocalAxesTransform(const double& rX_global, const double& rY_global, const double& rZ_global,
349 double& rx_local, double& ry_local, double& rz_local)
350 {
351 rx_local = rX_global;
352 ry_local = rY_global;
353 rz_local = rZ_global;
354
355 if( mHasLocalOrigin || mHasLocalAxes ){
356
357 //implement global to local axes transformation
358 Vector GlobalPosition(3);
359 GlobalPosition[0] = rX_global;
360 GlobalPosition[1] = rY_global;
361 GlobalPosition[2] = rZ_global;
362
363 if( mHasLocalOrigin )
364 GlobalPosition -= mLocalOrigin;
365
366 Vector LocalPosition(3);
367 noalias(LocalPosition) = ZeroVector(3);
368
369 if( mHasLocalAxes )
370 noalias(LocalPosition) = prod(mTransformationMatrix,GlobalPosition);
371
372 rx_local = LocalPosition[0];
373 ry_local = LocalPosition[1];
374 rz_local = LocalPosition[2];
375
376 }

380 {
381
