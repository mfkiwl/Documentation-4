209 }

211 }
212
213 double meanFreeSurfaceElementSize=freeSurfaceLength/numberOfFreeSurfaceElements;
214 double offset = 0;
215
216 if (freeSurfaceLength > 0)
217 {
218 offset = volumeLoss / freeSurfaceLength;
219
220 double ratioOffeset = fabs(offset) / freeSurfaceLength;
221 ratioOffeset = fabs(offset) / meanFreeSurfaceElementSize;
222 if(ratioOffeset>0.10){
223 // std::cout << "OFFSET is higher than 10.0% of free - surface element length " << offset << std::endl;

228
229 // if( mEchoLevel > 0 )

232 for (ModelPart::NodesContainerType::iterator i_node = mrModelPart.NodesBegin(); i_node != mrModelPart.NodesEnd(); i_node++)

234 if (i_node->Is(BOUNDARY) && i_node->IsNot(RIGID))

240 if (dimension == 3)

242 i_node->Z() += Normal[2] * offset;

244 }
245 }
246 }

248 if (mEchoLevel > 0)
249 std::cout << " RECOVER VOLUME LOSSES ]; " << std::endl;

312 double Compute3dArea(array_1d<double, 3> PointA, array_1d<double, 3> PointB, array_1d<double, 3> PointC)
313 {
314 double a = MathUtils<double>::Norm3(PointA - PointB);
315 double b = MathUtils<double>::Norm3(PointB - PointC);
316 double c = MathUtils<double>::Norm3(PointC - PointA);
317 double s = (a + b + c) / 2.0;
318 double Area = std::sqrt(s * (s - a) * (s - b) * (s - c));
319 return Area;
320 }
