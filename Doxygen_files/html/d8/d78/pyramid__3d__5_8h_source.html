456 ) const override
457 {
458 if ( std::abs( rPointLocalCoordinates[0] ) <= (1.0 + Tolerance) ) {

462 std::abs(rPointLocalCoordinates[1]) +

556 static Matrix CalculateShapeFunctionsIntegrationPointsValues(typename BaseType::IntegrationMethod ThisMethod)
557 {
558 IntegrationPointsContainerType all_integration_points = AllIntegrationPoints();
559 IntegrationPointsArrayType integration_points = all_integration_points[static_cast<int>(ThisMethod)];
560 //number of integration points
561 const std::size_t integration_points_number = integration_points.size();
562 //number of nodes in current geometry
563 const std::size_t points_number = 5;
564 //setting up return matrix
565 Matrix shape_function_values( integration_points_number, points_number );
566

569 for (std::size_t i=0; i<points_number; ++i) {
570 shape_function_values( pnt, i ) = ShapeFunctionValueImpl(i, integration_points[pnt]);

605 static Matrix& CalculateShapeFunctionsLocalGradients(
606 Matrix& rResult,
607 const CoordinatesArrayType& rPoint
608 )
609 {
610 rResult.resize( 5, 3, false );
611 noalias( rResult ) = ZeroMatrix( 5, 3 );
612
613 rResult( 0, 0 ) = (-0.125) * ( 1 - rPoint[1] ) * ( 1 - rPoint[2] ) ;
614 rResult( 0, 1 ) = (-0.125) * ( 1 - rPoint[0] ) * ( 1 - rPoint[2] ) ;
615 rResult( 0, 2 ) = (-0.125) * ( 1 - rPoint[0] ) * ( 1 - rPoint[1] ) ;
616
617 rResult( 1, 0 ) = (+0.125) * ( 1 - rPoint[1] ) * ( 1 - rPoint[2] ) ;
618 rResult( 1, 1 ) = (-0.125) * ( 1 + rPoint[0] ) * ( 1 - rPoint[2] ) ;
619 rResult( 1, 2 ) = (-0.125) * ( 1 + rPoint[0] ) * ( 1 - rPoint[1] ) ;
620
621 rResult( 2, 0 ) = (+0.125) * ( 1 + rPoint[1] ) * ( 1 - rPoint[2] ) ;
622 rResult( 2, 1 ) = (+0.125) * ( 1 + rPoint[0] ) * ( 1 - rPoint[2] ) ;
623 rResult( 2, 2 ) = (-0.125) * ( 1 + rPoint[0] ) * ( 1 + rPoint[1] ) ;
624
625 rResult( 3, 0 ) = (-0.125) * ( 1 + rPoint[1] ) * ( 1 - rPoint[2] ) ;
626 rResult( 3, 1 ) = (+0.125) * ( 1 - rPoint[0] ) * ( 1 - rPoint[2] ) ;
627 rResult( 3, 2 ) = (-0.125) * ( 1 - rPoint[0] ) * ( 1 + rPoint[1] ) ;
628
629 rResult( 4, 0 ) = 0.00 ;
630 rResult( 4, 1 ) = 0.00 ;
631 rResult( 4, 2 ) = +0.50 ;
632
633 return rResult;
634 }

734 // see ShapeFunctionValue
735 // special function such that it can also be used in the static methods
736 static double ShapeFunctionValueImpl(
737 IndexType ShapeFunctionIndex,
738 const CoordinatesArrayType& rPoint)
739 {
740 switch ( ShapeFunctionIndex )
741 {
742 case 0:
743 return( (0.125) * (1 - rPoint[0]) * (1 - rPoint[1]) * (1 - rPoint[2]) );
744 case 1:
745 return( (0.125) * (1 + rPoint[0]) * (1 - rPoint[1]) * (1 - rPoint[2]) );
746 case 2:
747 return( (0.125) * (1 + rPoint[0]) * (1 + rPoint[1]) * (1 - rPoint[2]) );
748 case 3:
749 return( (0.125) * (1 - rPoint[0]) * (1 + rPoint[1]) * (1 - rPoint[2]) );
750 case 4:
751 return( (0.5) * (1 + rPoint[2]) );
752 default:
753 KRATOS_ERROR << "Wrong index of shape function:" << ShapeFunctionIndex << std::endl;
754 }
755
756 return 0;
757 }
