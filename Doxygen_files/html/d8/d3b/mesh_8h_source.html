273 SizeType WorkingSpaceDimension() const
274 {
275 SizeType dimension = 3;
276
277 // NOTE: possible segmentacion fault if a Element or Condition
278 // is created using the base class of geometry, then the mpGeometryData
279 // of the geometry is a null pointer and has not any mWorkingSpaceDimension
280 if(NumberOfElements()!=0)
281 dimension = (mpElements->begin())->GetGeometry().WorkingSpaceDimension();
282 else if(NumberOfConditions()!=0)
283 dimension = (mpConditions->begin())->GetGeometry().WorkingSpaceDimension();
284 else if(NumberOfNodes()!=0)
285 dimension = (mpNodes->begin())->Dimension();
286
287 return dimension;
288 }
