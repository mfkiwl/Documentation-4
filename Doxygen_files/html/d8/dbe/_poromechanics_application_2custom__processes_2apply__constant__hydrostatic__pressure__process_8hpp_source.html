80 void ExecuteInitialize() override
81 {
82 KRATOS_TRY;
83
84 const Variable<double>& var = KratosComponents< Variable<double> >::Get(mvariable_name);
85
86 const int nnodes = static_cast<int>(mr_model_part.Nodes().size());
87
88 if(nnodes != 0)
89 {
90 ModelPart::NodesContainerType::iterator it_begin = mr_model_part.NodesBegin();
91
92 array_1d<double,3> Coordinates;
93
94 #pragma omp parallel for private(Coordinates)
95 for(int i = 0; i<nnodes; i++)
96 {
97 ModelPart::NodesContainerType::iterator it = it_begin + i;
98
99 if(mis_fixed)
100 {
101 it->Fix(var);
102 }
103
104 noalias(Coordinates) = it->Coordinates();
105
106 const double pressure = mspecific_weight*( mreference_coordinate - Coordinates[mgravity_direction] );
107
108 if(pressure > 0.0)
109 {
110 it->FastGetSolutionStepValue(var) = pressure;
111 }
112 else
113 {
114 it->FastGetSolutionStepValue(var) = 0.0;
115 }
116 }
117 }
118
119 KRATOS_CATCH("");
