344 void BuildROM(
345 typename TSchemeType::Pointer pScheme,
346 ModelPart &rModelPart,
347 LSPGSystemMatrixType &rA,
348 LSPGSystemVectorType &rb) override
349 {
350 KRATOS_TRY
351 // Define a dense matrix to hold the reduced problem
352 rA = ZeroMatrix(BaseType::GetEquationSystemSize(), this->GetNumberOfROMModes());
353 rb = ZeroVector(BaseType::GetEquationSystemSize());
354
355 // Build the system matrix by looping over elements and conditions and assembling to A
356 KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
357
358 // Get ProcessInfo from main model part
359 const auto& r_current_process_info = rModelPart.GetProcessInfo();
360
361
362 // Assemble all entities
363 const auto assembling_timer = BuiltinTimer();
364
365 AssemblyTLS assembly_tls_container;
366
367 const auto& r_elements = rModelPart.Elements();
368
369 if(!r_elements.empty())
370 {
371 block_for_each(r_elements, assembly_tls_container,
372 [&](Element& r_element, AssemblyTLS& r_thread_prealloc)
373 {
374 CalculateLocalContributionLSPG(r_element, rA, rb, r_thread_prealloc, *pScheme, r_current_process_info);
375 });
376 }
377
378
379 const auto& r_conditions = rModelPart.Conditions();
380
381 if(!r_conditions.empty())
382 {
383 block_for_each(r_conditions, assembly_tls_container,
384 [&](Condition& r_condition, AssemblyTLS& r_thread_prealloc)
385 {
386 CalculateLocalContributionLSPG(r_condition, rA, rb, r_thread_prealloc, *pScheme, r_current_process_info);
387 });
388 }
389
390 KRATOS_INFO_IF("LeastSquaresPetrovGalerkinROMBuilderAndSolver", (this->GetEchoLevel() > 0)) << "Build time: " << assembling_timer.ElapsedSeconds() << std::endl;
391 KRATOS_INFO_IF("LeastSquaresPetrovGalerkinROMBuilderAndSolver", (this->GetEchoLevel() > 2)) << "Finished parallel building" << std::endl;
392
393 KRATOS_CATCH("")

399 void SolveROM(
400 ModelPart &rModelPart,
401 LSPGSystemMatrixType &rA,
402 LSPGSystemVectorType &rb,
403 TSystemVectorType &rDx) override
404 {
405 KRATOS_TRY
406
407 LSPGSystemVectorType dxrom(this->GetNumberOfROMModes());
408
409 const auto solving_timer = BuiltinTimer();
410 // Calculate the QR decomposition
411 DenseHouseholderQRDecomposition<TDenseSpace> qr_decomposition;
412 // ^Correct after properly defining LSPGSystemMatrixType
413 qr_decomposition.Compute(rA);
414 qr_decomposition.Solve(rb, dxrom);
415 KRATOS_INFO_IF("LeastSquaresPetrovGalerkinROMBuilderAndSolver", (this->GetEchoLevel() > 0)) << "Solve reduced system time: " << solving_timer.ElapsedSeconds() << std::endl;
416
417 // Save the ROM solution increment in the root modelpart database
418 auto& r_root_mp = rModelPart.GetRootModelPart();
419 noalias(r_root_mp.GetValue(ROM_SOLUTION_INCREMENT)) += dxrom;
420
421 // project reduced solution back to full order model
422 const auto backward_projection_timer = BuiltinTimer();
423 this->ProjectToFineBasis(dxrom, rModelPart, rDx);
424 KRATOS_INFO_IF("LeastSquaresPetrovGalerkinROMBuilderAndSolver", (this->GetEchoLevel() > 0)) << "Project to fine basis time: " << backward_projection_timer.ElapsedSeconds() << std::endl;
425
426 KRATOS_CATCH("")

506 template<typename TEntity>
507 void CalculateLocalContributionLSPG(
508 TEntity& rEntity,
509 LSPGSystemMatrixType& rAglobal,
510 LSPGSystemVectorType& rBglobal,
511 AssemblyTLS& rPreAlloc,
512 TSchemeType& rScheme,
513 const ProcessInfo& rCurrentProcessInfo)
514 {
515 if (rEntity.IsDefined(ACTIVE) && rEntity.IsNot(ACTIVE)) return;
516
517 // Calculate elemental contribution
518 rScheme.CalculateSystemContributions(rEntity, rPreAlloc.lhs, rPreAlloc.romB, rPreAlloc.eq_id, rCurrentProcessInfo);
519 rEntity.GetDofList(rPreAlloc.dofs, rCurrentProcessInfo);
520
521 const SizeType ndofs = rPreAlloc.dofs.size();
522 ResizeIfNeeded(rPreAlloc.phiE, ndofs, this->GetNumberOfROMModes());
523 ResizeIfNeeded(rPreAlloc.romA, ndofs, this->GetNumberOfROMModes());
524
525 const auto &r_geom = rEntity.GetGeometry();
526 RomAuxiliaryUtilities::GetPhiElemental(rPreAlloc.phiE, rPreAlloc.dofs, r_geom, this->mMapPhi);
527
528 noalias(rPreAlloc.romA) = prod(rPreAlloc.lhs, rPreAlloc.phiE);
529
530
531 // Assembly
532 for(SizeType row=0; row < ndofs; ++row)
533 {
534 const SizeType global_row = rPreAlloc.eq_id[row];
535
536 double& r_bi = rBglobal(global_row);
537 AtomicAdd(r_bi, rPreAlloc.romB[row]);
538
539 if(rPreAlloc.dofs[row]->IsFree())
540 {
541 for(SizeType col=0; col < this->GetNumberOfROMModes(); ++col)
542 {
543 // const SizeType global_col = rPreAlloc.eq_id[col];
544 const SizeType global_col = col;
545 double& r_aij = rAglobal(global_row, global_col);
546 AtomicAdd(r_aij, rPreAlloc.romA(row, col));
547 }
548 }
549 }
550 }

555 template<typename TEntity>
556 void CalculateAssembledResiduals(
557 TEntity& rEntity,
558 LSPGSystemVectorType& rBglobal,
559 AssemblyTLS& rPreAlloc,
560 TSchemeType& rScheme,
561 const ProcessInfo& rCurrentProcessInfo)
562 {
563 if (rEntity.IsDefined(ACTIVE) && rEntity.IsNot(ACTIVE)) return;
564
565 // Calculate elemental contribution
566 rScheme.CalculateRHSContribution(rEntity, rPreAlloc.romB, rPreAlloc.eq_id, rCurrentProcessInfo);
567 rEntity.GetDofList(rPreAlloc.dofs, rCurrentProcessInfo);
568
569 const SizeType ndofs = rPreAlloc.dofs.size();
570
571 // Assembly
572 for(SizeType row=0; row < ndofs; ++row)
573 {
574 const SizeType global_row = rPreAlloc.eq_id[row];
575
576 if(rPreAlloc.dofs[row]->IsFree())
577 {
578 double& r_bi = rBglobal(global_row);
579 AtomicAdd(r_bi, rPreAlloc.romB[row]);
580 }
581 }
582 }
