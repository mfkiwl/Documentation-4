239 Vector CalculateAveragePlasticStrain()
240 {
241 Vector average_plastic_strain(VoigtSize);
242 noalias(average_plastic_strain) = ZeroVector(VoigtSize);
243 for (IndexType i = 0; i < NumberOfEdges; ++i)
244 average_plastic_strain += mPlasticStrains[i];
245 return average_plastic_strain / NumberOfEdges;
246 }
247
248 double CalculateAverageAcumulatedPlasticStrain()
249 {
250 double acumulated_strain = 0.0;
251 for (IndexType i = 0; i < NumberOfEdges; ++i)
252 acumulated_strain += mAcumulatedPlasticStrains[i];
253 return acumulated_strain / NumberOfEdges;
254 }
