178 template<class TMatrixType>
179 static TDataType Cofactor(const TMatrixType& rMat, IndexType i, IndexType j)
180 {
181 static_assert(std::is_same<typename TMatrixType::value_type, TDataType>::value, "Bad value type.");
182
183 KRATOS_ERROR_IF(rMat.size1() != rMat.size2() || rMat.size1() == 0) << "Bad matrix dimensions." << std::endl;
184
185 if (rMat.size1() == 1)
186 return 1;
187
188 IndirectArrayType ia1(rMat.size1() - 1), ia2(rMat.size2() - 1);
189
190 // Construct the submatrix structure for the first minor.
191 unsigned i_sub = 0;
192 for (unsigned k = 0; k < rMat.size1(); ++k)
193 if (k != i)
194 ia1(i_sub++) = k;
195
196 unsigned j_sub = 0;
197 for (unsigned k = 0; k < rMat.size2(); ++k)
198 if (k != j)
199 ia2(j_sub++) = k;
200
201#ifdef KRATOS_USE_AMATRIX // This macro definition is for the migration period and to be removed afterward please do not use it
202 PermutationMatrix<const TMatrixType, IndirectArrayType> sub_mat(rMat, ia1, ia2);
203#else
204 boost::numeric::ublas::matrix_indirect<const TMatrixType, IndirectArrayType> sub_mat(rMat, ia1, ia2);
205#endif // KRATOS_USE_AMATRIX
206 const TDataType first_minor = Det(sub_mat);
207 return ((i + j) % 2) ? -first_minor : first_minor;
208 }

239 static inline BoundedMatrix<TDataType, TDim, TDim> InvertMatrix(
240 const BoundedMatrix<TDataType, TDim, TDim>& rInputMatrix,
241 TDataType& rInputMatrixDet,
242 const TDataType Tolerance = ZeroTolerance
243 )
244 {
245 BoundedMatrix<TDataType, TDim, TDim> inverted_matrix;
246
247 /* Compute Determinant of the matrix */
248 rInputMatrixDet = Det(rInputMatrix);
249
250 if constexpr (TDim == 1) {
251 inverted_matrix(0,0) = 1.0/rInputMatrix(0,0);
252 rInputMatrixDet = rInputMatrix(0,0);
253 } else if constexpr (TDim == 2) {
254 InvertMatrix2(rInputMatrix, inverted_matrix, rInputMatrixDet);
255 } else if constexpr (TDim == 3) {
256 InvertMatrix3(rInputMatrix, inverted_matrix, rInputMatrixDet);
257 } else if constexpr (TDim == 4) {
258 InvertMatrix4(rInputMatrix, inverted_matrix, rInputMatrixDet);
259 } else {
260 KRATOS_ERROR << "Size not implemented. Size: " << TDim << std::endl;
261 }
262
263 // Checking condition number
264 if (Tolerance > 0.0) { // Check is skipped for negative tolerances
265 CheckConditionNumber(rInputMatrix, inverted_matrix, Tolerance);
266 }
267
268 return inverted_matrix;

277 template<class TMatrix1, class TMatrix2>
278 static inline bool CheckConditionNumber(
279 const TMatrix1& rInputMatrix,
280 TMatrix2& rInvertedMatrix,
281 const TDataType Tolerance = std::numeric_limits<double>::epsilon(),
282 const bool ThrowError = true
283 )
284 {
285 // We want at least 4 significant digits
286 const TDataType max_condition_number = (1.0/Tolerance) * 1.0e-4;
287
288 // Find the condition number to define is inverse is OK
289 const double input_matrix_norm = norm_frobenius(rInputMatrix);
290 const double inverted_matrix_norm = norm_frobenius(rInvertedMatrix);
291
292 // Now the condition number is the product of both norms
293 const double cond_number = input_matrix_norm * inverted_matrix_norm ;
294 // Finally check if the condition number is low enough
295 if (cond_number > max_condition_number) {
296 if (ThrowError) {
297 KRATOS_WATCH(rInputMatrix);
298 KRATOS_ERROR << " Condition number of the matrix is too high!, cond_number = " << cond_number << std::endl;
299 }
300 return false;
301 }
302
303 return true;
304 }

315 template<class TMatrix1, class TMatrix2>
316 static void GeneralizedInvertMatrix(
317 const TMatrix1& rInputMatrix,
318 TMatrix2& rInvertedMatrix,
319 TDataType& rInputMatrixDet,
320 const TDataType Tolerance = ZeroTolerance
321 )
322 {
323 const SizeType size_1 = rInputMatrix.size1();
324 const SizeType size_2 = rInputMatrix.size2();
325
326 if (size_1 == size_2) {
327 InvertMatrix(rInputMatrix, rInvertedMatrix, rInputMatrixDet, Tolerance);
328 } else if (size_1 < size_2) { // Right inverse
329 if (rInvertedMatrix.size1() != size_2 || rInvertedMatrix.size2() != size_1) {
330 rInvertedMatrix.resize(size_2, size_1, false);
331 }
332 const Matrix aux = prod(rInputMatrix, trans(rInputMatrix));
333 Matrix auxInv;
334 InvertMatrix(aux, auxInv, rInputMatrixDet, Tolerance);
335 rInputMatrixDet = std::sqrt(rInputMatrixDet);
336 noalias(rInvertedMatrix) = prod(trans(rInputMatrix), auxInv);
337 } else { // Left inverse
338 if (rInvertedMatrix.size1() != size_2 || rInvertedMatrix.size2() != size_1) {
339 rInvertedMatrix.resize(size_2, size_1, false);
340 }
341 const Matrix aux = prod(trans(rInputMatrix), rInputMatrix);
342 Matrix auxInv;
343 InvertMatrix(aux, auxInv, rInputMatrixDet, Tolerance);
344 rInputMatrixDet = std::sqrt(rInputMatrixDet);
345 noalias(rInvertedMatrix) = prod(auxInv, trans(rInputMatrix));
346 }

385 template<class TMatrix1, class TMatrix2>
386 static void InvertMatrix(
387 const TMatrix1& rInputMatrix,
388 TMatrix2& rInvertedMatrix,
389 TDataType& rInputMatrixDet,
390 const TDataType Tolerance = ZeroTolerance
391 )
392 {
393 KRATOS_DEBUG_ERROR_IF_NOT(rInputMatrix.size1() == rInputMatrix.size2()) << "Matrix provided is non-square" << std::endl;
394
395 const SizeType size = rInputMatrix.size2();
396
397 if(size == 1) {
398 if(rInvertedMatrix.size1() != 1 || rInvertedMatrix.size2() != 1) {
399 rInvertedMatrix.resize(1,1,false);
400 }
401 rInvertedMatrix(0,0) = 1.0/rInputMatrix(0,0);
402 rInputMatrixDet = rInputMatrix(0,0);
403 } else if (size == 2) {
404 InvertMatrix2(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
405 } else if (size == 3) {
406 InvertMatrix3(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
407 } else if (size == 4) {
408 InvertMatrix4(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
409 } else if (std::is_same<TMatrix1, Matrix>::value) {
410
411 const SizeType size1 = rInputMatrix.size1();
412 const SizeType size2 = rInputMatrix.size2();
413 if(rInvertedMatrix.size1() != size1 || rInvertedMatrix.size2() != size2) {
414 rInvertedMatrix.resize(size1, size2,false);
415 }
416
417 Matrix A(rInputMatrix);
418#ifdef KRATOS_USE_AMATRIX // This macro definition is for the migration period and to be removed afterward please do not use it
419 AMatrix::LUFactorization<MatrixType, DenseVector<std::size_t> > lu_factorization(A);
420 rInputMatrixDet = lu_factorization.determinant();
421 KRATOS_ERROR_IF(std::abs(rInputMatrixDet) <= ZeroTolerance) << "Matrix is singular: " << rInputMatrix << std::endl;
422 rInvertedMatrix = lu_factorization.inverse();
423#else
424 typedef permutation_matrix<SizeType> pmatrix;
425 pmatrix pm(A.size1());
426 const int singular = lu_factorize(A,pm);
427 rInvertedMatrix.assign( IdentityMatrix(A.size1()));
428 KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
429 lu_substitute(A, pm, rInvertedMatrix);
430
431 // Calculating determinant
432 rInputMatrixDet = 1.0;
433
434 for (IndexType i = 0; i < size1;++i) {
435 IndexType ki = pm[i] == i ? 0 : 1;
436 rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
437 }
438 #endif // ifdef KRATOS_USE_AMATRIX
439 } else { // Bounded-matrix case
440 const SizeType size1 = rInputMatrix.size1();
441 const SizeType size2 = rInputMatrix.size2();
442
443 Matrix A(rInputMatrix);
444 Matrix invA(rInvertedMatrix);
445
446 #ifdef KRATOS_USE_AMATRIX // This macro definition is for the migration period and to be removed afterward please do not use it
447 AMatrix::LUFactorization<MatrixType, DenseVector<std::size_t> > lu_factorization(A);
448 rInputMatrixDet = lu_factorization.determinant();
449 KRATOS_ERROR_IF(std::abs(rInputMatrixDet) <= ZeroTolerance) << "Matrix is singular: " << rInputMatrix << std::endl;
450 invA = lu_factorization.inverse();
451 #else
452 typedef permutation_matrix<SizeType> pmatrix;
453 pmatrix pm(size1);
454 const int singular = lu_factorize(A,pm);
455 invA.assign( IdentityMatrix(size1));
456 KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
457 lu_substitute(A, pm, invA);
458
459 // Calculating determinant
460 rInputMatrixDet = 1.0;
461
462 for (IndexType i = 0; i < size1;++i) {
463 IndexType ki = pm[i] == i ? 0 : 1;
464 rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
465 }
466 #endif // ifdef KRATOS_USE_AMATRIX
467
468 for (IndexType i = 0; i < size1;++i) {
469 for (IndexType j = 0; j < size2;++j) {
470 rInvertedMatrix(i,j) = invA(i,j);
471 }
472 }
473 }
474
475 // Checking condition number
476 if (Tolerance > 0.0) { // Check is skipped for negative tolerances
477 CheckConditionNumber(rInputMatrix, rInvertedMatrix, Tolerance);
478 }
479 }

487 template<class TMatrix1, class TMatrix2>
488 static void InvertMatrix2(
489 const TMatrix1& rInputMatrix,
490 TMatrix2& rInvertedMatrix,
491 TDataType& rInputMatrixDet
492 )
493 {
494 KRATOS_TRY;
495
496 KRATOS_DEBUG_ERROR_IF_NOT(rInputMatrix.size1() == rInputMatrix.size2()) << "Matrix provided is non-square" << std::endl;
497
498 if(rInvertedMatrix.size1() != 2 || rInvertedMatrix.size2() != 2) {
499 rInvertedMatrix.resize(2,2,false);
500 }
501
502 rInputMatrixDet = rInputMatrix(0,0)*rInputMatrix(1,1)-rInputMatrix(0,1)*rInputMatrix(1,0);
503
504 rInvertedMatrix(0,0) = rInputMatrix(1,1);
505 rInvertedMatrix(0,1) = -rInputMatrix(0,1);
506 rInvertedMatrix(1,0) = -rInputMatrix(1,0);
507 rInvertedMatrix(1,1) = rInputMatrix(0,0);
508
509 rInvertedMatrix/=rInputMatrixDet;
510
511 KRATOS_CATCH("");
512 }

520 template<class TMatrix1, class TMatrix2>
521 static void InvertMatrix3(
522 const TMatrix1& rInputMatrix,
523 TMatrix2& rInvertedMatrix,
524 TDataType& rInputMatrixDet
525 )
526 {
527 KRATOS_TRY;
528
529 KRATOS_DEBUG_ERROR_IF_NOT(rInputMatrix.size1() == rInputMatrix.size2()) << "Matrix provided is non-square" << std::endl;
530
531 if(rInvertedMatrix.size1() != 3 || rInvertedMatrix.size2() != 3) {
532 rInvertedMatrix.resize(3,3,false);
533 }
534
535 // Filling the inverted matrix with the algebraic complements
536 // First column
537 rInvertedMatrix(0,0) = rInputMatrix(1,1)*rInputMatrix(2,2) - rInputMatrix(1,2)*rInputMatrix(2,1);
538 rInvertedMatrix(1,0) = -rInputMatrix(1,0)*rInputMatrix(2,2) + rInputMatrix(1,2)*rInputMatrix(2,0);
539 rInvertedMatrix(2,0) = rInputMatrix(1,0)*rInputMatrix(2,1) - rInputMatrix(1,1)*rInputMatrix(2,0);
540
541 // Second column
542 rInvertedMatrix(0,1) = -rInputMatrix(0,1)*rInputMatrix(2,2) + rInputMatrix(0,2)*rInputMatrix(2,1);
543 rInvertedMatrix(1,1) = rInputMatrix(0,0)*rInputMatrix(2,2) - rInputMatrix(0,2)*rInputMatrix(2,0);
544 rInvertedMatrix(2,1) = -rInputMatrix(0,0)*rInputMatrix(2,1) + rInputMatrix(0,1)*rInputMatrix(2,0);
545
546 // Third column
547 rInvertedMatrix(0,2) = rInputMatrix(0,1)*rInputMatrix(1,2) - rInputMatrix(0,2)*rInputMatrix(1,1);
548 rInvertedMatrix(1,2) = -rInputMatrix(0,0)*rInputMatrix(1,2) + rInputMatrix(0,2)*rInputMatrix(1,0);
549 rInvertedMatrix(2,2) = rInputMatrix(0,0)*rInputMatrix(1,1) - rInputMatrix(0,1)*rInputMatrix(1,0);
550
551 // Calculation of determinant (of the input matrix)
552 rInputMatrixDet = rInputMatrix(0,0)*rInvertedMatrix(0,0) + rInputMatrix(0,1)*rInvertedMatrix(1,0) + rInputMatrix(0,2)*rInvertedMatrix(2,0);
553
554 // Finalizing the calculation of the inverted matrix
555 rInvertedMatrix /= rInputMatrixDet;
556
557 KRATOS_CATCH("")
558 }

816 const TDataType abs_x = std::abs(x);
817

826 }
827 }

1673 static inline bool GaussSeidelEigenSystem(
1674 const TMatrixType1& rA,
1675 TMatrixType2& rEigenVectorsMatrix,
1676 TMatrixType2& rEigenValuesMatrix,
1677 const TDataType Tolerance = 1.0e-18,
1678 const SizeType MaxIterations = 20
1679 )
1680 {
1681 bool is_converged = false;
1682
1683 const SizeType size = rA.size1();
1684
1685#ifdef KRATOS_USE_AMATRIX // This macro definition is for the migration period and to be removed afterward please do not use it
1686 KRATOS_WARNING_IF("EigenSystem", rEigenVectorsMatrix.size1() != size || rEigenVectorsMatrix.size2() != size) << "EigenSystem has detected an incorrect size of your EigenVectorsMatrix matrix. Please resize before compute" << std::endl;

1688#else

1690 rEigenVectorsMatrix.resize(size, size, false);
1691 if (rEigenValuesMatrix.size1() != size || rEigenValuesMatrix.size2() != size)
1692 rEigenValuesMatrix.resize(size, size, false);
1693#endif // KRATOS_USE_AMATRIX
1694
1695 const TMatrixType2 identity_matrix = IdentityMatrix(size);
1696 noalias(rEigenVectorsMatrix) = identity_matrix;
1697 noalias(rEigenValuesMatrix) = rA;
1698
1699 // Auxiliar values
1700 TMatrixType2 aux_A, aux_V_matrix, rotation_matrix;
1701 TDataType a, u, c, s, gamma, teta;
1702 IndexType index1, index2;
1703
1704 aux_A.resize(size,size,false);
1705 aux_V_matrix.resize(size,size,false);
1706 rotation_matrix.resize(size,size,false);
1707
1708 for(IndexType iterations = 0; iterations < MaxIterations; ++iterations) {
1709 is_converged = true;
1710
1711 a = 0.0;
1712 index1 = 0;
1713 index2 = 1;
1714
1715 for(IndexType i = 0; i < size; ++i) {
1716 for(IndexType j = (i + 1); j < size; ++j) {
1717 if((std::abs(rEigenValuesMatrix(i, j)) > a ) && (std::abs(rEigenValuesMatrix(i, j)) > Tolerance)) {
1718 a = std::abs(rEigenValuesMatrix(i,j));
1719 index1 = i;
1720 index2 = j;
1721 is_converged = false;
1722 }
1723 }
1724 }
1725
1726 if(is_converged) {
1727 break;
1728 }
1729
1730 // Calculation of Rotation angle
1731 gamma = (rEigenValuesMatrix(index2, index2)-rEigenValuesMatrix(index1, index1)) / (2 * rEigenValuesMatrix(index1, index2));
1732 u = 1.0;
1733
1734 if(std::abs(gamma) > Tolerance && std::abs(gamma)< (1.0/Tolerance)) {
1735 u = gamma / std::abs(gamma) * 1.0 / (std::abs(gamma) + std::sqrt(1.0 + gamma * gamma));
1736 } else {
1737 if (std::abs(gamma) >= (1.0/Tolerance)) {
1738 u = 0.5 / gamma;
1739 }
1740 }
1741
1742 c = 1.0 / (std::sqrt(1.0 + u * u));
1743 s = c * u;
1744 teta = s / (1.0 + c);
1745
1746 // Rotation of the Matrix
1747 noalias(aux_A) = rEigenValuesMatrix;
1748 aux_A(index2, index2) = rEigenValuesMatrix(index2,index2) + u * rEigenValuesMatrix(index1, index2);
1749 aux_A(index1, index1) = rEigenValuesMatrix(index1,index1) - u * rEigenValuesMatrix(index1, index2);
1750 aux_A(index1, index2) = 0.0;
1751 aux_A(index2, index1) = 0.0;
1752
1753 for(IndexType i = 0; i < size; ++i) {
1754 if((i!= index1) && (i!= index2)) {
1755 aux_A(index2, i) = rEigenValuesMatrix(index2, i) + s * (rEigenValuesMatrix(index1, i)- teta * rEigenValuesMatrix(index2, i));
1756 aux_A(i, index2) = rEigenValuesMatrix(index2, i) + s * (rEigenValuesMatrix(index1, i)- teta * rEigenValuesMatrix(index2, i));
1757 aux_A(index1, i) = rEigenValuesMatrix(index1, i) - s * (rEigenValuesMatrix(index2, i) + teta * rEigenValuesMatrix(index1, i));
1758 aux_A(i, index1) = rEigenValuesMatrix(index1, i) - s * (rEigenValuesMatrix(index2, i) + teta * rEigenValuesMatrix(index1, i));
1759 }
1760 }
1761
1762 noalias(rEigenValuesMatrix) = aux_A;
1763
1764 // Calculation of the eigeneigen vector matrix V
1765 noalias(rotation_matrix) = identity_matrix;
1766 rotation_matrix(index2, index1) = -s;
1767 rotation_matrix(index1, index2) = s;
1768 rotation_matrix(index1, index1) = c;
1769 rotation_matrix(index2, index2) = c;
1770
1771 noalias(aux_V_matrix) = ZeroMatrix(size, size);
1772
1773 for(IndexType i = 0; i < size; ++i) {
1774 for(IndexType j = 0; j < size; ++j) {
1775 for(IndexType k = 0; k < size; ++k) {
1776 aux_V_matrix(i, j) += rEigenVectorsMatrix(i, k) * rotation_matrix(k, j);
1777 }
1778 }
1779 }
1780 noalias(rEigenVectorsMatrix) = aux_V_matrix;
1781 }
1782
1783 KRATOS_WARNING_IF("MathUtils::EigenSystem", !is_converged) << "Spectral decomposition not converged " << std::endl;
1784
1785 return is_converged;
1786 }
