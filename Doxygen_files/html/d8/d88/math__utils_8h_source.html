375  static void InvertMatrix(
376  const TMatrix1& rInputMatrix,
377  TMatrix2& rInvertedMatrix,
378  TDataType& rInputMatrixDet,
379  const TDataType Tolerance = ZeroTolerance
380  )
381  {
382  KRATOS_DEBUG_ERROR_IF_NOT(rInputMatrix.size1() == rInputMatrix.size2()) << "Matrix provided is non-square" << std::endl;
383 
384  const SizeType size = rInputMatrix.size2();
385 
386  if(size == 1) {
387  if(rInvertedMatrix.size1() != 1 || rInvertedMatrix.size2() != 1) {
388  rInvertedMatrix.resize(1,1,false);
389  }
390  rInvertedMatrix(0,0) = 1.0/rInputMatrix(0,0);
391  rInputMatrixDet = rInputMatrix(0,0);
392  } else if (size == 2) {
393  InvertMatrix2(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
394  } else if (size == 3) {
395  InvertMatrix3(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
396  } else if (size == 4) {
397  InvertMatrix4(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
398  } else if (std::is_same<TMatrix1, Matrix>::value) {
399 
400  const SizeType size1 = rInputMatrix.size1();
401  const SizeType size2 = rInputMatrix.size2();
402  if(rInvertedMatrix.size1() != size1 || rInvertedMatrix.size2() != size2) {
403  rInvertedMatrix.resize(size1, size2,false);
404  }
405 
406  Matrix A(rInputMatrix);
407  typedef permutation_matrix<SizeType> pmatrix;
408  pmatrix pm(A.size1());
409  const int singular = lu_factorize(A,pm);
410  rInvertedMatrix.assign( IdentityMatrix(A.size1()));
411  KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
412  lu_substitute(A, pm, rInvertedMatrix);
413 
414  // Calculating determinant
415  rInputMatrixDet = 1.0;
416 
417  for (IndexType i = 0; i < size1;++i) {
418  IndexType ki = pm[i] == i ? 0 : 1;
419  rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
420  }
421  } else { // Bounded-matrix case
422  const SizeType size1 = rInputMatrix.size1();
423  const SizeType size2 = rInputMatrix.size2();
424 
425  Matrix A(rInputMatrix);
426  Matrix invA(rInvertedMatrix);
427 
428  typedef permutation_matrix<SizeType> pmatrix;
429  pmatrix pm(size1);
430  const int singular = lu_factorize(A,pm);
431  invA.assign( IdentityMatrix(size1));
432  KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
433  lu_substitute(A, pm, invA);
434 
435  // Calculating determinant
436  rInputMatrixDet = 1.0;
437 
438  for (IndexType i = 0; i < size1;++i) {
439  IndexType ki = pm[i] == i ? 0 : 1;
440  rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
441  }
442 
443  for (IndexType i = 0; i < size1;++i) {
444  for (IndexType j = 0; j < size2;++j) {
445  rInvertedMatrix(i,j) = invA(i,j);
446  }
447  }
448  }
449 
450  // Checking condition number
451  if (Tolerance > 0.0) { // Check is skipped for negative tolerances
452  CheckConditionNumber(rInputMatrix, rInvertedMatrix, Tolerance);
453  }
454  }

495  template<class TMatrix1, class TMatrix2>
496  static void InvertMatrix3(
497  const TMatrix1& rInputMatrix,
498  TMatrix2& rInvertedMatrix,
499  TDataType& rInputMatrixDet

505

518  rInvertedMatrix(1,1) = rInputMatrix(0,0)*rInputMatrix(2,2) - rInputMatrix(0,2)*rInputMatrix(2,0);
519  rInvertedMatrix(2,1) = -rInputMatrix(0,0)*rInputMatrix(2,1) + rInputMatrix(0,1)*rInputMatrix(2,0);
520 
521  // Third column
522  rInvertedMatrix(0,2) = rInputMatrix(0,1)*rInputMatrix(1,2) - rInputMatrix(0,2)*rInputMatrix(1,1);
523  rInvertedMatrix(1,2) = -rInputMatrix(0,0)*rInputMatrix(1,2) + rInputMatrix(0,2)*rInputMatrix(1,0);
524  rInvertedMatrix(2,2) = rInputMatrix(0,0)*rInputMatrix(1,1) - rInputMatrix(0,1)*rInputMatrix(1,0);
525 
526  // Calculation of determinant (of the input matrix)
527  rInputMatrixDet = rInputMatrix(0,0)*rInvertedMatrix(0,0) + rInputMatrix(0,1)*rInvertedMatrix(1,0) + rInputMatrix(0,2)*rInvertedMatrix(2,0);
528 
529  // Finalizing the calculation of the inverted matrix
530  rInvertedMatrix /= rInputMatrixDet;
531 
532  KRATOS_CATCH("")
533  }
