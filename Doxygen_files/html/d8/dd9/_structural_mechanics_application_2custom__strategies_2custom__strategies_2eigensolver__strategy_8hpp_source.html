301 void InitializeSolutionStep() override
302 {
303 KRATOS_TRY
304
305 ModelPart& rModelPart = BaseType::GetModelPart();
306 const int rank = rModelPart.GetCommunicator().MyPID();
307
308 KRATOS_INFO_IF("EigensolverStrategy", BaseType::GetEchoLevel() > 2 && rank == 0)
309 << "Entering InitializeSolutionStep" << std::endl;
310
311 BuilderAndSolverPointerType& pBuilderAndSolver = this->pGetBuilderAndSolver();
312 SchemePointerType& pScheme = this->pGetScheme();
313 SparseMatrixPointerType& pStiffnessMatrix = this->pGetStiffnessMatrix();
314 SparseMatrixType& rStiffnessMatrix = this->GetStiffnessMatrix();
315
316 // Initialize dummy vectors
317 SparseVectorPointerType pDx = SparseSpaceType::CreateEmptyVectorPointer();
318 SparseVectorPointerType pb = SparseSpaceType::CreateEmptyVectorPointer();
319 auto& rDx = *pDx;
320 auto& rb = *pb;
321
322 // Reset solution dofs
323 BuiltinTimer system_construction_time;
324 if (pBuilderAndSolver->GetDofSetIsInitializedFlag() == false ||
325 pBuilderAndSolver->GetReshapeMatrixFlag() == true)
326 {
327 // Set up list of dofs
328 BuiltinTimer setup_dofs_time;
329 pBuilderAndSolver->SetUpDofSet(pScheme, rModelPart);
330
331 KRATOS_INFO_IF("Setup Dofs Time", BaseType::GetEchoLevel() > 0 && rank == 0)
332 << setup_dofs_time.ElapsedSeconds() << std::endl;
333
334 // Set global equation ids
335 BuiltinTimer setup_system_time;
336 pBuilderAndSolver->SetUpSystem(rModelPart);
337
338 KRATOS_INFO_IF("Setup System Time", BaseType::GetEchoLevel() > 0 && rank == 0)
339 << setup_system_time.ElapsedSeconds() << std::endl;
340
341 // Resize and initialize system matrices
342 BuiltinTimer system_matrix_resize_time;
343 SparseMatrixPointerType& pMassMatrix = this->pGetMassMatrix();
344
345 // Mass matrix
346 pBuilderAndSolver->ResizeAndInitializeVectors(
347 pScheme, pMassMatrix, pDx, pb, rModelPart);
348
349 // Stiffness matrix
350 pBuilderAndSolver->ResizeAndInitializeVectors(
351 pScheme, pStiffnessMatrix, pDx, pb, rModelPart);
352
353 KRATOS_INFO_IF("System Matrix Resize Time", BaseType::GetEchoLevel() > 0 && rank == 0)
354 << system_matrix_resize_time.ElapsedSeconds() << std::endl;
355 }
356 else
357 {
358 SparseSpaceType::Resize(rb, SparseSpaceType::Size1(rStiffnessMatrix));
359 SparseSpaceType::Set(rb, 0.0);
360 SparseSpaceType::Resize(rDx, SparseSpaceType::Size1(rStiffnessMatrix));
361 SparseSpaceType::Set(rDx, 0.0);
362 }
363
364 KRATOS_INFO_IF("System Construction Time", BaseType::GetEchoLevel() > 0 && rank == 0)
365 << system_construction_time.ElapsedSeconds() << std::endl;
366
367 // Initial operations ... things that are constant over the solution
368 // step
369 pBuilderAndSolver->InitializeSolutionStep(BaseType::GetModelPart(),
370 rStiffnessMatrix, rDx, rb);
371
372 // Initial operations ... things that are constant over the solution
373 // step
374 pScheme->InitializeSolutionStep(BaseType::GetModelPart(), rStiffnessMatrix, rDx, rb);
375
376 KRATOS_INFO_IF("EigensolverStrategy", BaseType::GetEchoLevel() > 2 && rank == 0)
377 << "Exiting InitializeSolutionStep" << std::endl;
378
379 KRATOS_CATCH("")
380 }

382 bool SolveSolutionStep() override
383 {
384 KRATOS_TRY;
385
386 ModelPart& rModelPart = BaseType::GetModelPart();
387
388 SchemePointerType& pScheme = this->pGetScheme();
389 SparseMatrixType& rMassMatrix = this->GetMassMatrix();
390 SparseMatrixType& rStiffnessMatrix = this->GetStiffnessMatrix();
391
392 // Initialize dummy rhs vector
393 SparseVectorType b;
394 SparseSpaceType::Resize(b,SparseSpaceType::Size1(rMassMatrix));
395 SparseSpaceType::Set(b,0.0);
396
397 // if the size of the Matrix is the same as the size of the dofset, then also the dirichlet-dofs are in the matrix
398 // i.e. a BlockBuilder is used
399 const bool matrix_contains_dirichlet_dofs = SparseSpaceType::Size1(rMassMatrix) == this->pGetBuilderAndSolver()->GetDofSet().size();
400
401 const bool master_slave_constraints_defined = rModelPart.NumberOfMasterSlaveConstraints() != 0;
402
403 rModelPart.GetProcessInfo()[BUILD_LEVEL] = 1;
404 TSparseSpace::SetToZero(rMassMatrix);
405
406 EntitiesUtilities::InitializeNonLinearIterationAllEntities(rModelPart);
407
408 this->pGetBuilderAndSolver()->Build(pScheme,rModelPart,rMassMatrix,b);
409 if (master_slave_constraints_defined) {
410 this->pGetBuilderAndSolver()->ApplyConstraints(pScheme, rModelPart, rMassMatrix, b);
411 }
412 if (matrix_contains_dirichlet_dofs) {
413 this->ApplyDirichletConditions(rMassMatrix, mMassMatrixDiagonalValue);
414 }
415
416 if (BaseType::GetEchoLevel() == 4) {
417 TSparseSpace::WriteMatrixMarketMatrix("MassMatrix.mm", rMassMatrix, false);
418 }
419
420 rModelPart.GetProcessInfo()[BUILD_LEVEL] = 2;
421 TSparseSpace::SetToZero(rStiffnessMatrix);
422 this->pGetBuilderAndSolver()->Build(pScheme,rModelPart,rStiffnessMatrix,b);
423 if (master_slave_constraints_defined) {
424 this->pGetBuilderAndSolver()->ApplyConstraints(pScheme, rModelPart, rStiffnessMatrix, b);
425 }
426
427 if (matrix_contains_dirichlet_dofs) {
428 this->ApplyDirichletConditions(rStiffnessMatrix, mStiffnessMatrixDiagonalValue);
429 }
430
431 EntitiesUtilities::FinalizeNonLinearIterationAllEntities(rModelPart);
432
433 if (BaseType::GetEchoLevel() == 4) {
434 TSparseSpace::WriteMatrixMarketMatrix("StiffnessMatrix.mm", rStiffnessMatrix, false);
435 }
436
437 // Eigenvector matrix and eigenvalue vector are initialized by the solver
438 DenseVectorType Eigenvalues;
439 DenseMatrixType Eigenvectors;
440
441 // Solve for eigenvalues and eigenvectors
442 BuiltinTimer system_solve_time;
443 this->pGetBuilderAndSolver()->GetLinearSystemSolver()->Solve(
444 rStiffnessMatrix,
445 rMassMatrix,
446 Eigenvalues,
447 Eigenvectors);
448
449 KRATOS_INFO_IF("System Solve Time", BaseType::GetEchoLevel() > 0)
450 << system_solve_time.ElapsedSeconds() << std::endl;
451
452 if (master_slave_constraints_defined){
453 this->ReconstructSlaveSolution(Eigenvectors);
454 }
455
456 this->AssignVariables(Eigenvalues,Eigenvectors);
457
458
459 if (mComputeModalDecompostion) {
460 ComputeModalDecomposition(Eigenvectors);
461 }
462
463 return true;
464 KRATOS_CATCH("")
465 }

601 void ReconstructSlaveSolution(
602 DenseMatrixType& rEigenvectors
603 )
604 {
605 KRATOS_TRY
606
607 auto& r_model_part = BaseType::GetModelPart();
608 const std::size_t number_of_eigenvalues = rEigenvectors.size1();
609
610 struct TLS{
611 Matrix relation_matrix;
612 Vector constant_vector;
613 Vector master_dofs_values;
614 };
615
616 for (std::size_t i_eigenvalue = 0; i_eigenvalue < number_of_eigenvalues; ++i_eigenvalue){
617
618 // Reset slave dofs
619 block_for_each(r_model_part.MasterSlaveConstraints(), [&i_eigenvalue, &rEigenvectors](const MasterSlaveConstraint& r_master_slave_constraint){
620 const auto& r_slave_dofs_vector = r_master_slave_constraint.GetSlaveDofsVector();
621 for (const auto& r_slave_dof: r_slave_dofs_vector){
622 AtomicMult(rEigenvectors(i_eigenvalue, r_slave_dof->EquationId()), 0.0);
623 }
624 });
625
626 // Apply constraints
627 block_for_each(r_model_part.MasterSlaveConstraints(), TLS(), [&i_eigenvalue, &rEigenvectors, &r_model_part](const MasterSlaveConstraint& r_master_slave_constraint, TLS& rTLS){
628 // Detect if the constraint is active or not. If the user did not make any choice the constraint
629 // It is active by default
630 bool constraint_is_active = true;
631 if (r_master_slave_constraint.IsDefined(ACTIVE))
632 constraint_is_active = r_master_slave_constraint.Is(ACTIVE);
633 if (constraint_is_active) {
634 // Saving the master dofs values
635 const auto& r_master_dofs_vector = r_master_slave_constraint.GetMasterDofsVector();
636 const auto& r_slave_dofs_vector = r_master_slave_constraint.GetSlaveDofsVector();
637 rTLS.master_dofs_values.resize(r_master_dofs_vector.size());
638 for (IndexType i = 0; i < r_master_dofs_vector.size(); ++i) {
639 rTLS.master_dofs_values[i] = rEigenvectors(i_eigenvalue, r_master_dofs_vector[i]->EquationId());
640 }
641 // Apply the constraint to the slave dofs
642 r_master_slave_constraint.GetLocalSystem(rTLS.relation_matrix, rTLS.constant_vector, r_model_part.GetProcessInfo());
643 double aux;
644 for (IndexType i = 0; i < rTLS.relation_matrix.size1(); ++i) {
645 aux = rTLS.constant_vector[i];
646 for(IndexType j = 0; j < rTLS.relation_matrix.size2(); ++j) {
647 aux += rTLS.relation_matrix(i,j) * rTLS.master_dofs_values[j];
648 }
649 AtomicAdd(rEigenvectors(i_eigenvalue, r_slave_dofs_vector[i]->EquationId()), aux);
650 }
651 }
652 });
653 }
654
655 KRATOS_CATCH("")
656 }

664 void ApplyDirichletConditions(
665 SparseMatrixType& rA,
666 double Factor)
667 {
668 KRATOS_TRY
669
670 const int rank = BaseType::GetModelPart().GetCommunicator().MyPID();
671
672 KRATOS_INFO_IF("EigensolverStrategy", BaseType::GetEchoLevel() > 2 && rank == 0)
673 << "Entering ApplyDirichletConditions" << std::endl;
674
675 const std::size_t SystemSize = rA.size1();
676 std::vector<double> ScalingFactors(SystemSize);
677 auto& rDofSet = this->pGetBuilderAndSolver()->GetDofSet();
678 const int NumDofs = static_cast<int>(rDofSet.size());
679
680 // NOTE: dofs are assumed to be numbered consecutively
681 IndexPartition(NumDofs).for_each([&rDofSet, &ScalingFactors](std::size_t k){
682 auto dof_iterator = std::begin(rDofSet) + k;
683 ScalingFactors[k] = (dof_iterator->IsFixed()) ? 0.0 : 1.0;
684 });
685
686 double* AValues = std::begin(rA.value_data());
687 std::size_t* ARowIndices = std::begin(rA.index1_data());
688 std::size_t* AColIndices = std::begin(rA.index2_data());
689
690 // if there is a line of all zeros, put one on the diagonal
691 // #pragma omp parallel for firstprivate(SystemSize)
692 // for(int k = 0; k < static_cast<int>(SystemSize); ++k)
693 // {
694 // std::size_t ColBegin = ARowIndices[k];
695 // std::size_t ColEnd = ARowIndices[k+1];
696 // bool empty = true;
697 // for (auto j = ColBegin; j < ColEnd; ++j)
698 // if(AValues[j] != 0.0)
699 // {
700 // empty = false;
701 // break;
702 // }
703 // if(empty == true)
704 // rA(k,k) = 1.0;
705 // }
706
707 IndexPartition(SystemSize).for_each([&](std::size_t k){
708 std::size_t ColBegin = ARowIndices[k];
709 std::size_t ColEnd = ARowIndices[k+1];
710 if (ScalingFactors[k] == 0.0)
711 {
712 // row dof is fixed. zero off-diagonal columns and factor diagonal
713 for (std::size_t j = ColBegin; j < ColEnd; ++j)
714 {
715 if (AColIndices[j] != k)
716 {
717 AValues[j] = 0.0;
718 }
719 else
720 {
721 AValues[j] *= Factor;
722 }
723 }
724 }
725 else
726 {
727 // row dof is not fixed. zero columns associated with fixed dofs
728 for (std::size_t j = ColBegin; j < ColEnd; ++j)
729 {
730 AValues[j] *= ScalingFactors[AColIndices[j]];
731 }
732 }
733 });
734
735 KRATOS_INFO_IF("EigensolverStrategy", BaseType::GetEchoLevel() > 2 && rank == 0)
736 << "Exiting ApplyDirichletConditions" << std::endl;
737
738 KRATOS_CATCH("")
739 }

742 void AssignVariables(DenseVectorType& rEigenvalues, DenseMatrixType& rEigenvectors)
743 {
744 ModelPart& rModelPart = BaseType::GetModelPart();
745 const std::size_t NumEigenvalues = rEigenvalues.size();
746
747 // store eigenvalues in process info
748 rModelPart.GetProcessInfo()[EIGENVALUE_VECTOR] = rEigenvalues;
749
750 const auto& r_dof_set = this->pGetBuilderAndSolver()->GetDofSet();
751
752 for (ModelPart::NodeIterator itNode = rModelPart.NodesBegin(); itNode!= rModelPart.NodesEnd(); itNode++) {
753 ModelPart::NodeType::DofsContainerType& NodeDofs = itNode->GetDofs();
754 const std::size_t NumNodeDofs = NodeDofs.size();
755 Matrix& rNodeEigenvectors = itNode->GetValue(EIGENVECTOR_MATRIX);
756 if (rNodeEigenvectors.size1() != NumEigenvalues || rNodeEigenvectors.size2() != NumNodeDofs) {
757 rNodeEigenvectors.resize(NumEigenvalues,NumNodeDofs,false);
758 }
759
760 // fill the EIGENVECTOR_MATRIX
761 for (std::size_t i = 0; i < NumEigenvalues; i++) {
762 for (std::size_t j = 0; j < NumNodeDofs; j++)
763 {
764 const auto itDof = std::begin(NodeDofs) + j;
765 bool is_active = !(r_dof_set.find(**itDof) == r_dof_set.end());
766 if ((*itDof)->IsFree() && is_active) {
767 rNodeEigenvectors(i,j) = rEigenvectors(i,(*itDof)->EquationId());
768 }
769 else {
770 rNodeEigenvectors(i,j) = 0.0;
771 }
772 }
773 }
774 }

783 void ComputeModalDecomposition(const DenseMatrixType& rEigenvectors)
784 {
785 const SparseMatrixType& rMassMatrix = this->GetMassMatrix();
786 SparseMatrixType m_temp = ZeroMatrix(rEigenvectors.size1(),rEigenvectors.size2());
787 boost::numeric::ublas::axpy_prod(rEigenvectors,rMassMatrix,m_temp,true);
788 Matrix modal_mass_matrix = ZeroMatrix(m_temp.size1(),m_temp.size1());
789 boost::numeric::ublas::axpy_prod(m_temp,trans(rEigenvectors),modal_mass_matrix);
790
791 const SparseMatrixType& rStiffnessMatrix = this->GetStiffnessMatrix();
792 SparseMatrixType k_temp = ZeroMatrix(rEigenvectors.size1(),rEigenvectors.size2());
793 boost::numeric::ublas::axpy_prod(rEigenvectors,rStiffnessMatrix,k_temp,true);
794 Matrix modal_stiffness_matrix = ZeroMatrix(k_temp.size1(),k_temp.size1());
795 boost::numeric::ublas::axpy_prod(k_temp,trans(rEigenvectors),modal_stiffness_matrix);
796
