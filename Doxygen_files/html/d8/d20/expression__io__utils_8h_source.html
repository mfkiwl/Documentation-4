72  template <class TContainerType, class TContainerDataIO, class TVariableType>
73  static void WriteFromExpression(
74  TContainerType& rContainer,
75  Communicator& rCommunicator,
76  const Expression& rExpression,
77  TVariableType pVariable)
78  {
79  KRATOS_TRY
80 
81  const IndexType number_of_entities = rContainer.size();
82 
83  if (number_of_entities > 0) {
84  KRATOS_TRY
85  std::visit([&, number_of_entities](auto pVariable) {
86  using data_type = typename std::remove_const_t<std::remove_pointer_t<decltype(pVariable)>>::Type;
87 
88  VariableExpressionDataIO<data_type> variable_flatten_data_io(rExpression.GetItemShape());
89 
90  // initialize the container variables first
91  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
92  // initializes ghost nodes as for the later synchronization
93  // only, the nodal non historical values needs to be set unless
94  // they are properly initialized. Otherwise, in synchronization, the variables will
95  // not be there in the ghost nodes hence seg faults.
96 
97  // the vectors and matrices needs to be initialized in historical and non-historical
98  // data containers because they need to be initialized with the correct size for synchronization
99  if constexpr(std::is_same_v<data_type, Vector> || std::is_same_v<data_type, Matrix>) {
100  data_type dummy_value{};
101  variable_flatten_data_io.Assign(dummy_value, rExpression, 0);
102  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
103  VariableUtils().SetVariable(*pVariable, dummy_value, rCommunicator.GhostMesh().Nodes());
104  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
105  VariableUtils().SetNonHistoricalVariable(*pVariable, dummy_value, rCommunicator.GhostMesh().Nodes());
106  }
107  } else {
108  // if it is a static type, then it only needs to be initialized in the non-historical container with zeros.
109  // historical container should be initialized with the default values when the container is created.
110  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
111  VariableUtils().SetNonHistoricalVariableToZero(*pVariable, rCommunicator.GhostMesh().Nodes());
112  }
113  }
114  }
115 
116  IndexPartition<IndexType>(number_of_entities).for_each(data_type{}, [&rContainer, &pVariable, &rExpression, &variable_flatten_data_io](const IndexType Index, data_type& rValue){
117  variable_flatten_data_io.Assign(rValue, rExpression, Index);
118  TContainerDataIO::SetValue(*(rContainer.begin() + Index), *pVariable, rValue);
119  });
120 
121  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
122  // synchronize nodal values
123  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
124  rCommunicator.SynchronizeVariable(*pVariable);
125  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
126  rCommunicator.SynchronizeNonHistoricalVariable(*pVariable);
127  }
128  }
129  }, pVariable);
130  KRATOS_CATCH(" Variable: " + std::visit([](const auto& rpVariable) -> std::string {return rpVariable->Name();}, pVariable))
131  }

133  KRATOS_CATCH("");
134  }
