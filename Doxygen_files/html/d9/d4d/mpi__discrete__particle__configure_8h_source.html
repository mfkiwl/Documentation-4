126 static inline bool Intersection(const PointerType& rObj_1, const PointerType& rObj_2){
127 array_1d<double, 3> rObj_2_to_rObj_1 = rObj_1->GetGeometry().GetPoint(0) - rObj_2->GetGeometry().GetPoint(0);
128 double distance_2 = inner_prod(rObj_2_to_rObj_1, rObj_2_to_rObj_1);
129 //distance_2 is the inter-center distance squared (from the definition of distance in search-structure.h, with operator (,))
130 const double& radius_1 = rObj_1->GetGeometry()[0].GetSolutionStepValue(RADIUS);
131 const double& radius_2 = rObj_2->GetGeometry()[0].GetSolutionStepValue(RADIUS);
132 double radius_sum = radius_1 + radius_2;
133 bool intersect = (distance_2 - radius_sum * radius_sum) <= 0;
134 return intersect;
135 }

155 static inline bool IntersectionBox(const PointerType& rObject, const PointType& rLowPoint, const PointType& rHighPoint)
156 {
157
158// double separation_from_particle_radius_ratio = 0.1;
159
160 array_1d<double, 3> center_of_particle = rObject->GetGeometry().GetPoint(0);
161
162 const double& radius = rObject->GetGeometry()[0].GetSolutionStepValue(RADIUS);
163
164 bool intersect = (rLowPoint[0] - radius <= center_of_particle[0] && rLowPoint[1] - radius <= center_of_particle[1] && rLowPoint[2] - radius <= center_of_particle[2] &&
165 rHighPoint[0] + radius >= center_of_particle[0] && rHighPoint[1] + radius >= center_of_particle[1] && rHighPoint[2] + radius >= center_of_particle[2]);
166
167 return intersect;

183 static inline void Distance(const PointerType& rObj_1, const PointerType& rObj_2, double& distance)
184 {
185 array_1d<double, 3> center_of_particle1 = rObj_1->GetGeometry().GetPoint(0);
186 array_1d<double, 3> center_of_particle2 = rObj_2->GetGeometry().GetPoint(0);
187
188 distance = sqrt((center_of_particle1[0] - center_of_particle2[0]) * (center_of_particle1[0] - center_of_particle2[0]) +
189 (center_of_particle1[1] - center_of_particle2[1]) * (center_of_particle1[1] - center_of_particle2[1]) +
190 (center_of_particle1[2] - center_of_particle2[2]) * (center_of_particle1[2] - center_of_particle2[2]) );
191 }

194 static inline void ReduceIds(int& total_elements, int& first_element)
195 {
196 int mpi_rank;
197 int mpi_size;
198
199 MPI_Comm_rank(MPI_COMM_WORLD, &mpi_rank);
200 MPI_Comm_size(MPI_COMM_WORLD, &mpi_size);
201
202 std::vector<int> reduceArray(mpi_size+1);
203
204 MPI_Allgather(&total_elements,1,MPI_INT,&reduceArray[1],1,MPI_INT,MPI_COMM_WORLD);
205
206 reduceArray[0] = 1;
207 for(int i = 1; i <= mpi_size; i++)
208 reduceArray[i] += reduceArray[i-1];
209

225 template<class TObjectType>
226 static inline void AsyncSendAndReceive(std::vector<TObjectType>& SendObjects,
227 std::vector<TObjectType>& RecvObjects,
228 int * msgSendSize,
229 int * msgRecvSize)
230 {
231 Timer::Start("ASYNC1");
232
233 int mpi_rank;
234 int mpi_size;
235
236 MPI_Comm_rank(MPI_COMM_WORLD, &mpi_rank);
237 MPI_Comm_size(MPI_COMM_WORLD, &mpi_size);
238
239 std::stringstream * serializer_buffer;
240 std::vector<std::string> buffer(mpi_size);
241// std::string buffer[mpi_size];
242
243 for(int i = 0; i < mpi_size; i++)
244 {
245 if(mpi_rank != i)
246 {
247 Kratos::Serializer particleSerializer;
248 particleSerializer.save("nodes",SendObjects[i]);
249
250 serializer_buffer = (std::stringstream *)particleSerializer.pGetBuffer();
251 buffer[i] = std::string(serializer_buffer->str());
252 msgSendSize[i] = buffer[i].size();
253 }
254 }
255
256 MPI_Alltoall(msgSendSize,1,MPI_INT,msgRecvSize,1,MPI_INT,MPI_COMM_WORLD);
257
258 int NumberOfCommunicationEvents = 0;
259 int NumberOfCommunicationEventsIndex = 0;
260
261 char * message[mpi_size];
262 char * mpi_send_buffer[mpi_size];
263
264 for(int j = 0; j < mpi_size; j++)
265 {
266 if(j != mpi_rank && msgRecvSize[j]) NumberOfCommunicationEvents++;
267 if(j != mpi_rank && msgSendSize[j]) NumberOfCommunicationEvents++;
268 }
269
270 MPI_Request reqs[NumberOfCommunicationEvents];
271 MPI_Status stats[NumberOfCommunicationEvents];
272
273 //Set up all receive and send events
274 for(int i = 0; i < mpi_size; i++)
275 {
276 if(i != mpi_rank && msgRecvSize[i])
277 {
278 message[i] = (char *)malloc(sizeof(char) * msgRecvSize[i]);
279
280 MPI_Irecv(message[i],msgRecvSize[i],MPI_CHAR,i,0,MPI_COMM_WORLD,&reqs[NumberOfCommunicationEventsIndex++]);
281 }
282
283 if(i != mpi_rank && msgSendSize[i])
284 {
285 mpi_send_buffer[i] = (char *)malloc(sizeof(char) * msgSendSize[i]);
286 memcpy(mpi_send_buffer[i],buffer[i].c_str(),msgSendSize[i]);
287 MPI_Isend(mpi_send_buffer[i],msgSendSize[i],MPI_CHAR,i,0,MPI_COMM_WORLD,&reqs[NumberOfCommunicationEventsIndex++]);
288 }
289 }
290 //wait untill all communications finish
291 MPI_Waitall(NumberOfCommunicationEvents, reqs, stats);
292
293 MPI_Barrier(MPI_COMM_WORLD);
294
295 for(int i = 0; i < mpi_size; i++)
296 {
297 if (i != mpi_rank && msgRecvSize[i])
298 {
299 Kratos::Serializer particleSerializer;
300 std::stringstream * serializer_buffer;
301 serializer_buffer = (std::stringstream *)particleSerializer.pGetBuffer();
302 serializer_buffer->write(message[i], msgRecvSize[i]);
303 particleSerializer.load("nodes",RecvObjects[i]);
304 }
305
306 MPI_Barrier(MPI_COMM_WORLD);
307 }
308
309 // Free buffers
310 for(int i = 0; i < mpi_size; i++)
311 {
312 if(mpi_rank != i && msgRecvSize[i])
313 free(message[i]);
314
315 if(i != mpi_rank && msgSendSize[i])
316 free(mpi_send_buffer[i]);
317 }
318 Timer::Stop("ASYNC1");
319 }
