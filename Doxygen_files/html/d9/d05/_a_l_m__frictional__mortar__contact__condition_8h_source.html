413 IndexType GetActiveInactiveValue(const GeometryType& CurrentGeometry) const override
414 {
415 IndexType value = 0;
416 for (IndexType i_node = 0; i_node < TNumNodes; ++i_node) {
417 if (CurrentGeometry[i_node].Is(ACTIVE) == true) {
418 if (CurrentGeometry[i_node].Is(SLIP) == true)
419 value += std::pow(3, i_node);
420 else
421 value += 2 * std::pow(3, i_node);
422 }
423 }
424
425 return value;
