56 inline void ConstructNeighbourRelations(GeometryType& rGeom, const ModelPart& rBackgroundGridModelPart)
57 {
58 std::vector<typename Geometry<Node<3>>::Pointer> geometry_neighbours;
59 for (IndexType j = 0; j < rBackgroundGridModelPart.NumberOfElements(); j++)
60 {
61 auto p_geometry_neighbour = (rBackgroundGridModelPart.ElementsBegin() + j)->pGetGeometry();
62 if (p_geometry_neighbour->Id() != rGeom.Id()) // dont add the parent as its own neighbour
63 {
64 for (IndexType n = 0; n < p_geometry_neighbour->size(); n++)
65 {
66 for (IndexType k = 0; k < rGeom.size(); k++)
67 {
68 if (rGeom[k].Id() == (*p_geometry_neighbour)[n].Id()) {
69 // Prevent duplicate additions
70 bool add_entry = true;
71 for (size_t i = 0; i < geometry_neighbours.size(); i++)
72 {
73 if (geometry_neighbours[i]->Id() == p_geometry_neighbour->Id())
74 {
75 add_entry = false;
76 break;
77 }
78 }
79 if (add_entry)
80 {
81 geometry_neighbours.push_back(p_geometry_neighbour);
82 }
83 break;
84 }
85 }
86 }
87 }
88
89 }
90 #pragma omp critical
91 rGeom.SetValue(GEOMETRY_NEIGHBOURS, geometry_neighbours);
92 }

96 {

112 inline GeometryType& FindGridGeom(GeometryType& rParentGeom,
113 const ModelPart& rBackgroundGridModelPart,
114 const double Tolerance,
115 const array_1d<double, 3>& xg,
116 array_1d<double, 3>& rLocalCoords,
117 const ProcessInfo& rProcessInfo,
118 bool& IsFound)
119 {
120 IsFound = false;
121
122 if (CheckIsInside(rParentGeom, rLocalCoords, xg, Tolerance)) {
123 IsFound = true;
124 return rParentGeom;
125 }
126 else
127 {
128 if (!rParentGeom.Has(GEOMETRY_NEIGHBOURS))
129 ConstructNeighbourRelations(rParentGeom, rBackgroundGridModelPart);
130
131 auto& geometry_neighbours = rParentGeom.GetValue(GEOMETRY_NEIGHBOURS);
132 for (IndexType k = 0; k < geometry_neighbours.size(); ++k) {
133 if (CheckIsInside(*geometry_neighbours[k], rLocalCoords, xg, Tolerance)) {
134 IsFound = true;
135 return *(geometry_neighbours[k].get());
136 }
137 }
138 }
139
140 return rParentGeom;
141 }

161 inline void NeighbourSearchElements(const ModelPart& rMPMModelPart,
162 const ModelPart& rBackgroundGridModelPart,
163 std::vector<typename Element::Pointer>& rMissingElements,
164 const double Tolerance)
165 {
166 #pragma omp parallel for
167 for (int i = 0; i < static_cast<int>(rMPMModelPart.Elements().size()); ++i) {
168 auto element_itr = (rMPMModelPart.ElementsBegin() + i);
169 array_1d<double, 3> local_coordinates;
170 bool is_found = false;
171 std::vector<array_1d<double, 3>> xg;
172 element_itr->CalculateOnIntegrationPoints(MP_COORD, xg, rBackgroundGridModelPart.GetProcessInfo());
173
174 GeometryType& r_found_geom = FindGridGeom(element_itr->GetGeometry().GetGeometryParent(0),
175 rBackgroundGridModelPart, Tolerance, xg[0], local_coordinates,
176 rMPMModelPart.GetProcessInfo(), is_found);
177
178 if (is_found)
179 {
180 const bool is_pqmpm = (rBackgroundGridModelPart.GetProcessInfo().Has(IS_PQMPM))
181 ? rBackgroundGridModelPart.GetProcessInfo().GetValue(IS_PQMPM) : false;
182 if (is_pqmpm)
183 {
184 // Updates the quadrature point geometry.
185 (*element_itr).GetGeometry().SetGeometryParent(&r_found_geom);
186 PQMPMPartitionUtilities::PartitionMasterMaterialPointsIntoSubPoints(rBackgroundGridModelPart, xg[0],
187 local_coordinates, *element_itr, element_itr->pGetGeometry(), Tolerance);
188 }
189 else
190 {
191 CreateQuadraturePointsUtility<Node<3>>::UpdateFromLocalCoordinates(
192 element_itr->pGetGeometry(), local_coordinates,
193 element_itr->GetGeometry().IntegrationPoints()[0].Weight(), r_found_geom);
194 }
195
196 if (IsExplicitAndNeedsCorrection(element_itr->pGetGeometry(), rBackgroundGridModelPart.GetProcessInfo()))
197 is_found = false;
198 else {
199 for (IndexType j = 0; j < r_found_geom.PointsNumber(); ++j)
200 r_found_geom.Points()[j].Set(ACTIVE);
201 }
202 }
203 if(!is_found)
204 {
205 #pragma omp critical
206 rMissingElements.push_back(&*element_itr);
207 }
208 }
209 }

215 inline void NeighbourSearchConditions(const ModelPart& rMPMModelPart,
216 const ModelPart& rBackgroundGridModelPart,
217 std::vector<typename Condition::Pointer>& rMissingConditions,
218 const double Tolerance)
219 {
220 #pragma omp parallel for
221 for (int i = 0; i < static_cast<int>(rMPMModelPart.Conditions().size()); ++i) {
222 auto condition_itr = rMPMModelPart.Conditions().begin() + i;
223
224 std::vector<array_1d<double, 3>> xg;
225 condition_itr->CalculateOnIntegrationPoints(MPC_COORD, xg, rMPMModelPart.GetProcessInfo());
226
227 if (xg.size() > 0 && condition_itr->Is(BOUNDARY))
228 {
229 array_1d<double, 3> local_coordinates;
230 bool is_found = false;
231
232 GeometryType& r_found_geom = FindGridGeom(condition_itr->GetGeometry().GetGeometryParent(0),
233 rBackgroundGridModelPart, Tolerance, xg[0], local_coordinates,
234 rMPMModelPart.GetProcessInfo(), is_found);
235
236 if (is_found)
237 {
238 CreateQuadraturePointsUtility<Node<3>>::UpdateFromLocalCoordinates(
239 condition_itr->pGetGeometry(), local_coordinates,
240 condition_itr->GetGeometry().IntegrationPoints()[0].Weight(), r_found_geom);
241
242 for (IndexType j = 0; j < r_found_geom.PointsNumber(); ++j)
243 r_found_geom[j].Set(ACTIVE);
244 }
245 else
246 {
247 #pragma omp critical
248 rMissingConditions.push_back(&*condition_itr);
249 }
250 }
251 }

255 inline bool IsFixExplicitAndOnElementEdge(const Vector& N, const ProcessInfo& rProcessInfo)
256 {
257 if (rProcessInfo.Has(IS_FIX_EXPLICIT_MP_ON_GRID_EDGE)) {
258 if (rProcessInfo.GetValue(IS_FIX_EXPLICIT_MP_ON_GRID_EDGE)) {
259 // check if MP is exactly on the edge of the element, this gives spurious strains in explicit
260 for (SizeType i = 0; i < N.size(); ++i) {
261 if (std::abs(N[i]) < std::numeric_limits<double>::epsilon()) {
262 return true;
263 }
264 }
265 }
266 }
267
268 return false;

272 template <std::size_t TDimension>
273 void BinBasedSearchElementsAndConditions(ModelPart& rMPMModelPart,
274 ModelPart& rBackgroundGridModelPart,
275 std::vector<typename Element::Pointer>& rMissingElements,
276 std::vector<typename Condition::Pointer>& rMissingConditions,
277 const std::size_t MaxNumberOfResults, const double Tolerance)
278 {
279 const ProcessInfo& r_process_info = rBackgroundGridModelPart.GetProcessInfo();
280 bool is_pqmpm = (r_process_info.Has(IS_PQMPM))
281 ? r_process_info.GetValue(IS_PQMPM) : false;
282
283 // Search background grid and make element active
284 Vector N;
285 const int max_result = 1000;
286
287 #pragma omp parallel
288 {
289 BinBasedFastPointLocator<TDimension> SearchStructure(rBackgroundGridModelPart);
290 SearchStructure.UpdateSearchDatabase();
291 typename BinBasedFastPointLocator<TDimension>::ResultContainerType results(max_result);
292
293 // Element search and assign background grid
294 #pragma omp for
295 for (int i = 0; i < static_cast<int>(rMissingElements.size()); ++i) {
296 auto element_itr = *(rMissingElements.begin() + i);
297 std::vector<array_1d<double, 3>> xg;
298 element_itr->CalculateOnIntegrationPoints(MP_COORD, xg, rMPMModelPart.GetProcessInfo());
299 typename BinBasedFastPointLocator<TDimension>::ResultIteratorType result_begin = results.begin();
300 Element::Pointer pelem;
301
302 // FindPointOnMesh find the background element in which a given point falls and the relative shape functions
303 bool is_found = SearchStructure.FindPointOnMesh(xg[0], N, pelem, result_begin, MaxNumberOfResults, Tolerance);
304
305 if (is_found == true) {
306 if (IsFixExplicitAndOnElementEdge(N, r_process_info) && !is_pqmpm) {
307 // MP is exactly on the edge. Now we give it a little 'nudge'
308 array_1d<double, 3> xg_nudged = array_1d<double, 3>(xg[0]);
309 std::vector<array_1d<double, 3>> mp_vel;
310 element_itr->CalculateOnIntegrationPoints(MP_VELOCITY, mp_vel, rMPMModelPart.GetProcessInfo());
311 xg_nudged += r_process_info[DELTA_TIME] / 1000.0 * mp_vel[0];
312 if (SearchStructure.FindPointOnMesh(xg_nudged, N, pelem, result_begin, MaxNumberOfResults, Tolerance)) {
313 element_itr->SetValuesOnIntegrationPoints(MP_COORD, { xg_nudged }, rMPMModelPart.GetProcessInfo());
314 KRATOS_INFO("MPMSearchElementUtility") << "WARNING: To prevent spurious explicit stresses, Material Point "
315 << element_itr->Id() << " was nudged." << std::endl;
316 } else {
317 is_found = SearchStructure.FindPointOnMesh(xg[0], N, pelem, result_begin, MaxNumberOfResults, Tolerance);
318 KRATOS_INFO("MPMSearchElementUtility") << "WARNING: Material Point " << element_itr->Id()
319 << " lies exactly on an element edge and may give spurious results." << std::endl;
320 }
321 }
322 pelem->Set(ACTIVE);
323
324 const bool is_pqmpm = (rBackgroundGridModelPart.GetProcessInfo().Has(IS_PQMPM))
325 ? rBackgroundGridModelPart.GetProcessInfo().GetValue(IS_PQMPM) : false;
326 if (is_pqmpm)
327 {
328 // Updates the quadrature point geometry.
329 (*element_itr).GetGeometry().SetGeometryParent((pelem->pGetGeometry().get()));
330 UpdatePartitionedQuadraturePoint(rBackgroundGridModelPart, xg[0],
331 *element_itr, pelem->pGetGeometry(), Tolerance);
332 }
333 else
334 {
335 auto p_quadrature_point_geometry = element_itr->pGetGeometry();
336 array_1d<double, 3> local_coordinates;
337 p_quadrature_point_geometry->PointLocalCoordinates(local_coordinates, xg[0]);
338 CreateQuadraturePointsUtility<Node<3>>::UpdateFromLocalCoordinates(
339 p_quadrature_point_geometry, local_coordinates,
340 p_quadrature_point_geometry->IntegrationPoints()[0].Weight(), pelem->GetGeometry());
341 }
342
343 auto& r_geometry = element_itr->GetGeometry();
344 for (IndexType j = 0; j < r_geometry.PointsNumber(); ++j)
345 r_geometry[j].Set(ACTIVE);
346 }
347 else {
348 KRATOS_INFO("MPMSearchElementUtility") << "WARNING: Search Element for Material Point: " << element_itr->Id()
349 << " is failed. Geometry is cleared." << std::endl;
350
351 element_itr->GetGeometry().clear();
352 element_itr->Reset(ACTIVE);
353 element_itr->Set(TO_ERASE);
354 }
355 }
356
357 // Condition search and assign background grid
358 #pragma omp for
359 for (int i = 0; i < static_cast<int>(rMissingConditions.size()); ++i) {
360 auto condition_itr = *(rMissingConditions.begin() + i);
361 std::vector<array_1d<double, 3>> xg;
362 condition_itr->CalculateOnIntegrationPoints(MPC_COORD, xg, rMPMModelPart.GetProcessInfo());
363
364 if (xg.size() > 0) {
365 // Only search for particle based BCs!
366 // Grid BCs are still applied on MP_model_part but we don't want to search for them.
367 typename BinBasedFastPointLocator<TDimension>::ResultIteratorType result_begin = results.begin();
368 Element::Pointer pelem;
369
370 // FindPointOnMesh find the background element in which a given point falls and the relative shape functions
371 bool is_found = SearchStructure.FindPointOnMesh(xg[0], N, pelem, result_begin, MaxNumberOfResults, Tolerance);
372
373 if (is_found == true) {
374 auto p_quadrature_point_geometry = condition_itr->pGetGeometry();
375 array_1d<double, 3> local_coordinates;
376 p_quadrature_point_geometry->PointLocalCoordinates(local_coordinates, xg[0]);
377 CreateQuadraturePointsUtility<Node<3>>::UpdateFromLocalCoordinates(
378 p_quadrature_point_geometry, local_coordinates,
379 p_quadrature_point_geometry->IntegrationPoints()[0].Weight(), pelem->GetGeometry());
380
381 auto& r_geometry = condition_itr->GetGeometry();
382
383 for (IndexType j = 0; j < r_geometry.PointsNumber(); ++j)
384 r_geometry[j].Set(ACTIVE);
385 } else {
386 KRATOS_INFO("MPMSearchElementUtility") << "WARNING: Search Element for Material Point Condition: " << condition_itr->Id()
387 << " is failed. Geometry is cleared." << std::endl;
388
389 condition_itr->GetGeometry().clear();
390 condition_itr->Reset(ACTIVE);
391 condition_itr->Set(TO_ERASE);
392 }
393 }
394 }
395 }

423 template<std::size_t TDimension>
424 void SearchElement(ModelPart& rBackgroundGridModelPart, ModelPart& rMPMModelPart, const std::size_t MaxNumberOfResults,
425 const double Tolerance)
426 {
427 ResetElementsAndNodes(rBackgroundGridModelPart);
428
429 std::vector<typename Element::Pointer> missing_elements;
430 std::vector<typename Condition::Pointer> missing_conditions;
431
432 NeighbourSearchElements(rMPMModelPart, rBackgroundGridModelPart, missing_elements, Tolerance);
433 NeighbourSearchConditions(rMPMModelPart, rBackgroundGridModelPart, missing_conditions, Tolerance);
434
435 if (missing_conditions.size() > 0 || missing_elements.size() > 0)
436 BinBasedSearchElementsAndConditions<TDimension>(rMPMModelPart,
437 rBackgroundGridModelPart, missing_elements, missing_conditions,
438 MaxNumberOfResults, Tolerance);
439 }
