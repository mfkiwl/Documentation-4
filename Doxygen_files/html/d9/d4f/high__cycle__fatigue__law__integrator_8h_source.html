126  static double CalculateTensionCompressionFactor(const Vector& rStressVector)
127  {
128  array_1d<double,3> principal_stresses;
129  AdvancedConstitutiveLawUtilities<6>::CalculatePrincipalStresses(principal_stresses, rStressVector);
130 
131 
132  double abs_component = 0.0, average_component = 0.0, sum_abs = 0.0, sum_average = 0.0;
133  for (unsigned int i = 0; i < principal_stresses.size(); ++i) {
134  abs_component = std::abs(principal_stresses[i]);
135  average_component = 0.5 * (principal_stresses[i] + abs_component);
136  sum_average += average_component;
137  sum_abs += abs_component;
138  }
139  const double pre_indicator = sum_average / sum_abs;

166  static void CalculateFatigueParameters(const double MaxStress,
167  double ReversionFactor,
168  const Properties& rMaterialParameters,
169  double& rB0,
170  double& rSth,
171  double& rAlphat,
172  double& rN_f)
173  {
174  const Vector& r_fatigue_coefficients = rMaterialParameters[HIGH_CYCLE_FATIGUE_COEFFICIENTS];
175  double ultimate_stress = rMaterialParameters.Has(YIELD_STRESS) ? rMaterialParameters[YIELD_STRESS] : rMaterialParameters[YIELD_STRESS_TENSION];
176  const double yield_stress = ultimate_stress;
177 
178  // The calculation is prepared to update the rN_f value when using a softening curve which initiates with hardening.
179  // The jump in the advance in time process is done in these cases to the Syield rather to Sult.
180  const int softening_type = rMaterialParameters[SOFTENING_TYPE];
181  const int curve_by_points = static_cast<int>(SofteningType::CurveFittingDamage);
182  if (softening_type == curve_by_points) {
183  const Vector& stress_damage_curve = rMaterialParameters[STRESS_DAMAGE_CURVE]; //Integrated_stress points of the fitting curve
184  const SizeType curve_points = stress_damage_curve.size() - 1;
185 
186  ultimate_stress = 0.0;
187  for (IndexType i = 1; i <= curve_points; ++i) {
188  ultimate_stress = std::max(ultimate_stress, stress_damage_curve[i-1]);
189  }
190  }
191 
192  //These variables have been defined following the model described by S. Oller et al. in A continuum mechanics model for mechanical fatigue analysis (2005), equation 13 on page 184.
193  const double Se = r_fatigue_coefficients[0] * ultimate_stress;
194  const double STHR1 = r_fatigue_coefficients[1];
195  const double STHR2 = r_fatigue_coefficients[2];
196  const double ALFAF = r_fatigue_coefficients[3];
197  const double BETAF = r_fatigue_coefficients[4];
198  const double AUXR1 = r_fatigue_coefficients[5];
199  const double AUXR2 = r_fatigue_coefficients[6];
200 
201  if (std::abs(ReversionFactor) < 1.0) {
202  rSth = Se + (ultimate_stress - Se) * std::pow((0.5 + 0.5 * ReversionFactor), STHR1);
203  rAlphat = ALFAF + (0.5 + 0.5 * ReversionFactor) * AUXR1;
204  } else {
205  rSth = Se + (ultimate_stress - Se) * std::pow((0.5 + 0.5 / ReversionFactor), STHR2);
206  rAlphat = ALFAF - (0.5 + 0.5 / ReversionFactor) * AUXR2;
207  }
208 
209  const double square_betaf = std::pow(BETAF, 2.0);
210  if (MaxStress > rSth && MaxStress <= ultimate_stress) {
211  rN_f = std::pow(10.0,std::pow(-std::log((MaxStress - rSth) / (ultimate_stress - rSth))/rAlphat,(1.0/BETAF)));
212  rB0 = -(std::log(MaxStress / ultimate_stress) / std::pow((std::log10(rN_f)), square_betaf));
213 
214  if (softening_type == curve_by_points) {
215  rN_f = std::pow(rN_f, std::pow(std::log(MaxStress / yield_stress) / std::log(MaxStress / ultimate_stress), 1.0 / square_betaf));
216  }
217  }
218  }
