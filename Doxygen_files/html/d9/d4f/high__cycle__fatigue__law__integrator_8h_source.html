106 const Vector& PreviousStresses,
107 bool& rMaxIndicator,
108 bool& rMinIndicator)
109 {
110 const double stress_1 = PreviousStresses[1];

112 const double stress_increment_1 = stress_1 - stress_2;
113 const double stress_increment_2 = CurrentStress - stress_1;

127 static double CalculateTensionCompressionFactor(const Vector& rStressVector)
128 {
129 array_1d<double,3> principal_stresses;
130 AdvancedConstitutiveLawUtilities<6>::CalculatePrincipalStresses(principal_stresses, rStressVector);
131
132
133 double abs_component = 0.0, average_component = 0.0, sum_abs = 0.0, sum_average = 0.0;
134 for (unsigned int i = 0; i < principal_stresses.size(); ++i) {
135 abs_component = std::abs(principal_stresses[i]);
136 average_component = 0.5 * (principal_stresses[i] + abs_component);
137 sum_average += average_component;
138 sum_abs += abs_component;
139 }
140 const double pre_indicator = sum_average / sum_abs;
141 if (pre_indicator < 0.5) {
142 return -1.0;
143 } else {
144 return 1.0;
145 }
146 }

167 static void CalculateFatigueParameters(const double MaxStress,
168 double ReversionFactor,
169 const Properties& rMaterialParameters,
170 double& rB0,
171 double& rSth,
172 double& rAlphat,
173 double& rN_f)
174 {
175 const Vector& r_fatigue_coefficients = rMaterialParameters[HIGH_CYCLE_FATIGUE_COEFFICIENTS];
176 double ultimate_stress = rMaterialParameters.Has(YIELD_STRESS) ? rMaterialParameters[YIELD_STRESS] : rMaterialParameters[YIELD_STRESS_TENSION];
177 const double yield_stress = ultimate_stress;
178
179 // The calculation is prepared to update the rN_f value when using a softening curve which initiates with hardening.
180 // The jump in the advance in time process is done in these cases to the Syield rather to Sult.
181 const int softening_type = rMaterialParameters[SOFTENING_TYPE];
182 const int curve_by_points = static_cast<int>(SofteningType::CurveFittingDamage);
183 if (softening_type == curve_by_points) {
184 const Vector& stress_damage_curve = rMaterialParameters[STRESS_DAMAGE_CURVE]; //Integrated_stress points of the fitting curve
185 const SizeType curve_points = stress_damage_curve.size() - 1;
186
187 ultimate_stress = 0.0;
188 for (IndexType i = 1; i <= curve_points; ++i) {
189 ultimate_stress = std::max(ultimate_stress, stress_damage_curve[i-1]);
190 }
191 }
192
193 //These variables have been defined following the model described by S. Oller et al. in A continuum mechanics model for mechanical fatigue analysis (2005), equation 13 on page 184.
194 const double Se = r_fatigue_coefficients[0] * ultimate_stress;
195 const double STHR1 = r_fatigue_coefficients[1];
196 const double STHR2 = r_fatigue_coefficients[2];
197 const double ALFAF = r_fatigue_coefficients[3];
198 const double BETAF = r_fatigue_coefficients[4];
199 const double AUXR1 = r_fatigue_coefficients[5];
200 const double AUXR2 = r_fatigue_coefficients[6];
201
202 if (std::abs(ReversionFactor) < 1.0) {
203 rSth = Se + (ultimate_stress - Se) * std::pow((0.5 + 0.5 * ReversionFactor), STHR1);
204 rAlphat = ALFAF + (0.5 + 0.5 * ReversionFactor) * AUXR1;
205 } else {
206 rSth = Se + (ultimate_stress - Se) * std::pow((0.5 + 0.5 / ReversionFactor), STHR2);
207 rAlphat = ALFAF - (0.5 + 0.5 / ReversionFactor) * AUXR2;
208 }
209
210 const double square_betaf = std::pow(BETAF, 2.0);
211 if (MaxStress > rSth && MaxStress <= ultimate_stress) {
212 rN_f = std::pow(10.0,std::pow(-std::log((MaxStress - rSth) / (ultimate_stress - rSth))/rAlphat,(1.0/BETAF)));
213 rB0 = -(std::log(MaxStress / ultimate_stress) / std::pow((std::log10(rN_f)), square_betaf));
214
215 if (softening_type == curve_by_points) {
216 rN_f = std::pow(rN_f, std::pow(std::log(MaxStress / yield_stress) / std::log(MaxStress / ultimate_stress), 1.0 / square_betaf));
217 }
218 }
219 }

233 static void CalculateFatigueReductionFactorAndWohlerStress(const Properties& rMaterialParameters,
234 const double MaxStress,
235 unsigned int LocalNumberOfCycles,
236 unsigned int GlobalNumberOfCycles,
237 const double B0,
238 const double Sth,
239 const double Alphat,
240 double& rFatigueReductionFactor,
241 double& rWohlerStress)
242 {
243 const double BETAF = rMaterialParameters[HIGH_CYCLE_FATIGUE_COEFFICIENTS][4];
244 if (GlobalNumberOfCycles > 2){
245 double ultimate_stress = rMaterialParameters.Has(YIELD_STRESS) ? rMaterialParameters[YIELD_STRESS] : rMaterialParameters[YIELD_STRESS_TENSION];
246
247 // The calculation is prepared to update the rN_f value when using a softening curve which initiates with hardening.
248 // The jump in the advance in time process is done in these cases to the Syield rather to Sult.
249 const int softening_type = rMaterialParameters[SOFTENING_TYPE];
250 const int curve_by_points = static_cast<int>(SofteningType::CurveFittingDamage);
251 if (softening_type == curve_by_points) {
252 const Vector& stress_damage_curve = rMaterialParameters[STRESS_DAMAGE_CURVE]; //Integrated_stress points of the fitting curve
253 const SizeType curve_points = stress_damage_curve.size() - 1;
254
255 ultimate_stress = 0.0;
256 for (IndexType i = 1; i <= curve_points; ++i) {
257 ultimate_stress = std::max(ultimate_stress, stress_damage_curve[i-1]);
258 }
259 }
260 rWohlerStress = (Sth + (ultimate_stress - Sth) * std::exp(-Alphat * (std::pow(std::log10(static_cast<double>(LocalNumberOfCycles)), BETAF)))) / ultimate_stress;
261 }
262 if (MaxStress > Sth) {
263 rFatigueReductionFactor = std::exp(-B0 * std::pow(std::log10(static_cast<double>(LocalNumberOfCycles)), (BETAF * BETAF)));
264 rFatigueReductionFactor = (rFatigueReductionFactor < 0.01) ? 0.01 : rFatigueReductionFactor;
265 }
