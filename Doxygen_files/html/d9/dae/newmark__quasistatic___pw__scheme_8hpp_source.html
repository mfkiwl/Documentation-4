63 int Check(const ModelPart& rModelPart) const override
64 {
65 KRATOS_TRY
66
67 BaseType::Check(rModelPart);
68
69 //check that variables are correctly allocated
70 for (const auto& rNode : rModelPart.Nodes())
71 {
72 if (rNode.SolutionStepsDataHas(WATER_PRESSURE) == false)
73 KRATOS_ERROR << "WATER_PRESSURE variable is not allocated for node "
74 << rNode.Id()
75 << std::endl;
76
77 if (rNode.SolutionStepsDataHas(DT_WATER_PRESSURE) == false)
78 KRATOS_ERROR << "DT_WATER_PRESSURE variable is not allocated for node "
79 << rNode.Id()
80 << std::endl;
81
82 if (rNode.HasDofFor(WATER_PRESSURE) == false)
83 KRATOS_ERROR << "missing WATER_PRESSURE dof on node "
84 << rNode.Id()
85 << std::endl;
86 }
87
88 //check for minimum value of the buffer index.
89 if (rModelPart.GetBufferSize() < 2)
90 KRATOS_ERROR << "insufficient buffer size. Buffer size should be greater than 2. Current size is "
91 << rModelPart.GetBufferSize()
92 << std::endl;
93
94 // Check beta, gamma and theta
95 if (mBeta <= 0.0 || mGamma<= 0.0 || mTheta <= 0.0)
96 KRATOS_ERROR << "Some of the scheme variables: beta, gamma or theta has an invalid value "
97 << std::endl;
98
99 return 0;
100
101 KRATOS_CATCH( "" )
102 }

127 inline void UpdateVariablesDerivatives(ModelPart& rModelPart) override
128 {
129 KRATOS_TRY
130
131 //Update DtPressure
132 block_for_each(rModelPart.Nodes(), [&](Node<3>& rNode){
133 const double DeltaPressure = rNode.FastGetSolutionStepValue(WATER_PRESSURE)
134 - rNode.FastGetSolutionStepValue(WATER_PRESSURE, 1);
135 const auto &PreviousDtPressure = rNode.FastGetSolutionStepValue(DT_WATER_PRESSURE, 1);
136
137 rNode.FastGetSolutionStepValue(DT_WATER_PRESSURE) = 1.0/(mTheta*mDeltaTime)*(DeltaPressure - (1.0-mTheta)*mDeltaTime*PreviousDtPressure);
138 });
139
140 KRATOS_CATCH( "" )
141 }
