209 template< std::size_t TDim, std::size_t TNumNodes>
210 static BoundedMatrix<double, TNumNodes, TDim> ComputeTangentMatrixSlip(
211 const GeometryType& rGeometry,
212 const std::size_t StepSlip = 1
213 )
214 {
215 /* DEFINITIONS */
216 // Zero tolerance
217 const double zero_tolerance = std::numeric_limits<double>::epsilon();
218 // Tangent matrix
219 BoundedMatrix<double, TNumNodes, TDim> tangent_matrix;
220
221 for (IndexType i_node = 0; i_node < TNumNodes; ++i_node) {
222 const array_1d<double, 3>& r_gt = rGeometry[i_node].FastGetSolutionStepValue(WEIGHTED_SLIP, StepSlip);
223 const double norm_slip = norm_2(r_gt);
224 if (norm_slip > zero_tolerance) { // Non zero r_gt
225 const array_1d<double, 3> tangent_slip = r_gt/norm_slip;
226 for (std::size_t i_dof = 0; i_dof < TDim; ++i_dof)
227 tangent_matrix(i_node, i_dof) = tangent_slip[i_dof];
228 } else { // We consider the tangent direction as auxiliary
229 const array_1d<double, 3>& r_normal = rGeometry[i_node].FastGetSolutionStepValue(NORMAL);
230 array_1d<double, 3> tangent_xi, tangent_eta;
231 MathUtils<double>::OrthonormalBasis(r_normal, tangent_xi, tangent_eta);
232 if constexpr (TDim == 3) {
233 for (std::size_t i_dof = 0; i_dof < 3; ++i_dof)
234 tangent_matrix(i_node, i_dof) = tangent_xi[i_dof];
235 } else {
236 if (std::abs(tangent_xi[2]) > std::numeric_limits<double>::epsilon()) {
237 for (std::size_t i_dof = 0; i_dof < 2; ++i_dof)
238 tangent_matrix(i_node, i_dof) = tangent_eta[i_dof];
239 } else {
240 for (std::size_t i_dof = 0; i_dof < 2; ++i_dof)
241 tangent_matrix(i_node, i_dof) = tangent_xi[i_dof];
242 }
243 }
244 }
245 }
246
247 return tangent_matrix;
248 }
