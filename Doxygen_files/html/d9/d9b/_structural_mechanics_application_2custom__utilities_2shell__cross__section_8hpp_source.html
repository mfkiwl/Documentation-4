532
533 void InitializeIntegrationPoints(const Properties& rProps, const int NumIntegrationPoints)
534 {
535 KRATOS_TRY
536
537 const ConstitutiveLaw::Pointer& pMaterial = rProps[CONSTITUTIVE_LAW];
538 KRATOS_ERROR_IF(pMaterial == nullptr) << "A Ply needs a constitutive law to be set. "
539 << "Missing constitutive law in property: " << rProps.Id() << std::endl;;
540
541 // generate the integration points
542 mIntegrationPoints.clear();
543 mIntegrationPoints.resize(NumIntegrationPoints);
544 for (int i=0; i<NumIntegrationPoints; ++i) {
545 mIntegrationPoints[i].SetConstitutiveLaw(pMaterial->Clone());
546 }
547
548 KRATOS_CATCH("")
549 }
550 void UpdateIntegrationPoints(const Properties& rProps)
551 {
552 KRATOS_TRY
553
554 const SizeType num_int_points = mIntegrationPoints.size();
555
556 // generate the weights (composite simpson rule)
557 Vector ip_w(num_int_points, 1.0);
558 if (num_int_points >= 3) {
559 for (IndexType i=1; i<num_int_points-1; ++i) {
560 double iw = (i % 2 == 0) ? 2.0 : 4.0;
561 ip_w(i) = iw;
562 }
563 ip_w /= sum(ip_w);
564 }
565
566 // generate locations (direction: top(+thickness/2) to bottom(-thickness/2)
567 const double location = GetLocation(rProps);
568 const double thickness = GetThickness(rProps);
569
570 Vector ip_loc(num_int_points, 0.0);
571 if (num_int_points >= 3) {
572 double loc_start = location + 0.5 * thickness;
573 double loc_incr = thickness / double(num_int_points-1);
574 for (IndexType i=0; i<num_int_points; ++i) {
575 ip_loc(i) = loc_start;
576 loc_start -= loc_incr;
577 }
578 }
579
580 for (IndexType i=0; i<num_int_points; ++i) {
581 IntegrationPoint& r_int_point = mIntegrationPoints[i];
582 r_int_point.SetWeight(ip_w(i) * thickness);
583 r_int_point.SetLocation(ip_loc(i));
584 }
585
586 KRATOS_CATCH("")
587 }

957 inline void GetRotationMatrixForGeneralizedStrains(double radians, Matrix& T)
958 {
959 double c = std::cos(radians);
960 double s = std::sin(radians);
961
962 SizeType strain_size = GetStrainSize();
963
964 if (T.size1() != strain_size || T.size2() != strain_size) {
965 T.resize(strain_size, strain_size, false);
966 }
967 noalias(T) = ZeroMatrix(strain_size, strain_size);
968
969 T(0, 0) = c * c;
970 T(0, 1) = s * s;
971 T(0, 2) = - s * c;
972 T(1, 0) = s * s;
973 T(1, 1) = c * c;
974 T(1, 2) = s * c;
975 T(2, 0) = 2.0 * s * c;
976 T(2, 1) = - 2.0 * s * c;
977 T(2, 2) = c * c - s * s;
978
979 project(T, range(3, 6), range(3, 6)) = project(T, range(0, 3), range(0, 3));
980
981 if (strain_size == 8) {
982 T(6, 6) = c;
983 T(6, 7) = s;
984 T(7, 6) = - s;
985 T(7, 7) = c;
986 }
987 }
