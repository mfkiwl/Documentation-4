123 static void IntegrateStressVector(
124 array_1d<double, VoigtSize>& rPredictiveStressVector,
125 const double UniaxialStress,
126 double& rDamage,
127 double& rThreshold,
128 ConstitutiveLaw::Parameters& rValues,
129 const double CharacteristicLength
130 )
131 {
132 const Properties& r_material_properties = rValues.GetMaterialProperties();
133
134 const int softening_type = r_material_properties[SOFTENING_TYPE];
135 double damage_parameter;
136 TYieldSurfaceType::CalculateDamageParameter(rValues, damage_parameter, CharacteristicLength);
137
138 switch (softening_type)
139 {
140 case static_cast<int>(SofteningType::Linear):
141 CalculateLinearDamage(UniaxialStress, rThreshold, damage_parameter, CharacteristicLength, rValues, rDamage);
142 break;
143 case static_cast<int>(SofteningType::Exponential):
144 CalculateExponentialDamage(UniaxialStress, rThreshold, damage_parameter, CharacteristicLength, rValues, rDamage);
145 break;
146 default:
147 KRATOS_ERROR << "SOFTENING_TYPE not defined or wrong..." << softening_type << std::endl;
148 break;
149 }
150 rPredictiveStressVector *= (1.0 - rDamage);
151 }

161 static void CalculateExponentialDamage(
162 const double UniaxialStress,
163 const double Threshold,
164 const double DamageParameter,
165 const double CharacteristicLength,
166 ConstitutiveLaw::Parameters& rValues,
167 double& rDamage
168 )
169 {
170 double initial_threshold;
171 TYieldSurfaceType::GetInitialUniaxialThreshold(rValues, initial_threshold);
172 rDamage = 1.0 - (initial_threshold / UniaxialStress) * std::exp(DamageParameter *
173 (1.0 - UniaxialStress / initial_threshold));
174 }

184 static void CalculateLinearDamage(
185 const double UniaxialStress,
186 const double Threshold,
187 const double DamageParameter,
188 const double CharacteristicLength,

190 double& rDamage

195 rDamage = (1.0 - initial_threshold / UniaxialStress) / (1.0 + DamageParameter);
