104 void ExecuteInitialize() override
105 {
106 KRATOS_TRY
107
108 if (mrModelPart.NumberOfNodes() > 0) {
109 const Variable<double> &var = KratosComponents< Variable<double> >::Get(mVariableName);
110
111 if (mIsSeepage) {
112 block_for_each(mrModelPart.Nodes(), [&var, this](Node<3>& rNode) {
113 const double pressure = CalculatePressure(rNode);
114
115 if ((PORE_PRESSURE_SIGN_FACTOR * pressure) < 0.0) {
116 rNode.FastGetSolutionStepValue(var) = pressure;
117 if (mIsFixed) rNode.Fix(var);
118 } else {
119 rNode.Free(var);
120 }
121 });
122 } else {
123 block_for_each(mrModelPart.Nodes(), [&var, this](Node<3>& rNode) {
124 if (mIsFixed) rNode.Fix(var);
125 else if (mIsFixedProvided) rNode.Free(var);
126
127 const double pressure = CalculatePressure(rNode);
128
129 if ((PORE_PRESSURE_SIGN_FACTOR * pressure) < mPressureTensionCutOff) {
130 rNode.FastGetSolutionStepValue(var) = pressure;
131 } else {
132 rNode.FastGetSolutionStepValue(var) = mPressureTensionCutOff;
133 }
134 });
135 }
136 }

175 double CalculatePressure(const Node<3> &rNode)
176 {
177 // find top boundary
178 std::vector< Node<3>* > TopBoundaryNodes;
179 FindTopBoundaryNodes(rNode, TopBoundaryNodes);
180 double PressureTop;
181 double CoordinateTop;
182 CalculateBoundaryPressure(rNode, TopBoundaryNodes, PressureTop, CoordinateTop);
183
184 // find bottom boundary
185 std::vector< Node<3>* > BottomBoundaryNodes;
186 FindBottomBoundaryNodes(rNode, BottomBoundaryNodes);
187 double PressureBottom;
188 double CoordinateBottom;
189 CalculateBoundaryPressure(rNode, BottomBoundaryNodes, PressureBottom, CoordinateBottom, true);
190
191 // calculate pressure
192 if (std::abs(CoordinateTop - CoordinateBottom) > TINY) {
193 const double slopeP = (PressureTop - PressureBottom) / (CoordinateTop - CoordinateBottom);
194 const double pressure = slopeP * (rNode.Coordinates()[mGravityDirection] - CoordinateBottom ) + PressureBottom;
195 return pressure;
196 } else {
197 return PressureBottom;
198 }
199 }

211 void CalculateBoundaryPressure( const Node<3> &rNode,
212 const std::vector< Node<3>*> &BoundaryNodes,
213 double &pressure,
214 double &coordinate,
215 bool isBottom=false )
216 {
217 // find top boundary
218 std::vector< Node<3>*> LeftBoundaryNodes;
219 FindLeftBoundaryNodes(rNode, BoundaryNodes, LeftBoundaryNodes);
220
221 std::vector< Node<3>*> RightBoundaryNodes;
222 FindRightBoundaryNodes(rNode, BoundaryNodes, RightBoundaryNodes);
223
224 if (LeftBoundaryNodes.size() > 0 && RightBoundaryNodes.size() > 0) {
225 Node<3> *LeftNode;
226 LeftNode = FindClosestNodeOnBoundaryNodes(rNode, LeftBoundaryNodes, isBottom);
227 Node<3> *RightNode;
228 RightNode = FindClosestNodeOnBoundaryNodes(rNode, RightBoundaryNodes, isBottom);
229
230 InterpolateBoundaryPressure(rNode, LeftNode, RightNode, pressure, coordinate);
231 return;
232
233 } else if (LeftBoundaryNodes.size() > 0) {
234 InterpolateBoundaryPressureWithOneContainer(rNode, LeftBoundaryNodes, pressure, coordinate);
235 return;
236
237 } else if (RightBoundaryNodes.size() > 0) {
238 InterpolateBoundaryPressureWithOneContainer(rNode, RightBoundaryNodes, pressure, coordinate);
239 return;
240
241 } else {
242 KRATOS_ERROR << "There is not enough points around interpolation, node Id" << rNode.Id() << std::endl;
243 }

247 void InterpolateBoundaryPressureWithOneContainer(const Node<3> &rNode,
248 const std::vector< Node<3>*> &BoundaryNodes,
249 double &pressure,
250 double &coordinate )
251 {
252 std::vector< Node<3>*> FoundNodes;
253 FindTwoClosestNodeOnBoundaryNodes(rNode, BoundaryNodes, FoundNodes);
254
255 const Variable<double> &var = KratosComponents< Variable<double> >::Get(mVariableName);
256
257 const double &pressureLeft = FoundNodes[0]->FastGetSolutionStepValue(var);
258 Vector3 CoordinatesLeft;
259 noalias(CoordinatesLeft) = FoundNodes[0]->Coordinates();
260
261 const double &pressureRight = FoundNodes[1]->FastGetSolutionStepValue(var);
262 Vector3 CoordinatesRight;
263 noalias(CoordinatesRight) = FoundNodes[1]->Coordinates();
264
265 // calculate pressure
266 if (std::abs(CoordinatesRight[mHorizontalDirection] - CoordinatesLeft[mHorizontalDirection]) > TINY) {
267 const double slopeP = (pressureRight - pressureLeft) / (CoordinatesRight[mHorizontalDirection] - CoordinatesLeft[mHorizontalDirection]);
268 pressure = slopeP * (rNode.Coordinates()[mHorizontalDirection] - CoordinatesLeft[mHorizontalDirection]) + pressureLeft;
269
270 const double slopeY = (CoordinatesRight[mGravityDirection] - CoordinatesLeft[mGravityDirection]) / (CoordinatesRight[mHorizontalDirection] - CoordinatesLeft[mHorizontalDirection]);
271 coordinate = slopeY * (rNode.Coordinates()[mHorizontalDirection] - CoordinatesLeft[mHorizontalDirection]) + CoordinatesLeft[mGravityDirection];
272 } else {
273 pressure = pressureLeft;
274 coordinate = CoordinatesLeft[mGravityDirection];
275 }

307 void FindTwoClosestNodeOnBoundaryNodes(const Node<3> &rNode,
308 const std::vector< Node<3>*> &BoundaryNodes,
309 std::vector< Node<3>*> &FoundNodes)
310 {
311 const double HorizontalCoordiante = rNode.Coordinates()[mHorizontalDirection];
312 FoundNodes.resize(2);
313
314 unsigned int nFound = 0;
315 double horizontalDistanceClosest_1 = LARGE;
316 for (unsigned int i = 0; i < BoundaryNodes.size(); ++i) {
317 Vector3 CoordinatesBoundary;
318 noalias(CoordinatesBoundary) = BoundaryNodes[i]->Coordinates();
319
320 if (std::abs(CoordinatesBoundary[mHorizontalDirection] - HorizontalCoordiante) <= horizontalDistanceClosest_1) {
321 horizontalDistanceClosest_1 = std::abs(CoordinatesBoundary[mHorizontalDirection] - HorizontalCoordiante);
322 FoundNodes[0] = BoundaryNodes[i];
323 nFound++;
324 }
325 }
326
327 double horizontalDistanceClosest_2 = LARGE;
328 for (unsigned int i = 0; i < BoundaryNodes.size(); ++i) {
329 Vector3 CoordinatesBoundary;
330 noalias(CoordinatesBoundary) = BoundaryNodes[i]->Coordinates();
331
332 if (std::abs(CoordinatesBoundary[mHorizontalDirection] - HorizontalCoordiante) <= horizontalDistanceClosest_2 &&
333 std::abs(CoordinatesBoundary[mHorizontalDirection] - HorizontalCoordiante) > horizontalDistanceClosest_1) {
334 horizontalDistanceClosest_2 = std::abs(CoordinatesBoundary[mHorizontalDirection] - HorizontalCoordiante);
335 FoundNodes[1] = BoundaryNodes[i];
336 nFound++;
337 }
338 }
339
340 KRATOS_ERROR_IF(nFound < 2) << "Not enough points for interpolation: Coordinates"<< rNode.Coordinates() << std::endl;
341 }

344 Node<3>*
345 FindClosestNodeOnBoundaryNodes(const Node<3> &rNode,
346 const std::vector< Node<3>* > &BoundaryNodes,
347 const bool isBottom)
348 {
349 const double HorizontalCoordiante = rNode.Coordinates()[mHorizontalDirection];
350 Node<3> *pNode;
351 std::vector< Node<3>*> FoundNodes;
352
353 double horizontalDistance = LARGE;
354 for (unsigned int i = 0; i < BoundaryNodes.size(); ++i) {
355 if (std::abs(BoundaryNodes[i]->Coordinates()[mHorizontalDirection] - HorizontalCoordiante) <= horizontalDistance) {
356 horizontalDistance = std::abs(BoundaryNodes[i]->Coordinates()[mHorizontalDirection] - HorizontalCoordiante);
357 FoundNodes.push_back(BoundaryNodes[i]);
358 }
359 }
360
361 if (isBottom) {
362 double height = LARGE;
363 for (unsigned int i = 0; i < FoundNodes.size(); ++i) {
364 if (FoundNodes[i]->Coordinates()[mGravityDirection] < height) {
365 pNode = FoundNodes[i];
366 height = FoundNodes[i]->Coordinates()[mGravityDirection];
367 }
368 }
369 } else {
370 double height = -LARGE;
371 for (unsigned int i = 0; i < FoundNodes.size(); ++i) {
372 if (FoundNodes[i]->Coordinates()[mGravityDirection] > height) {
373 pNode = FoundNodes[i];
374 height = FoundNodes[i]->Coordinates()[mGravityDirection];
375 }
376 }
377 }
378
379 return pNode;
380 }

465 void FillListOfBoundaryNodesFast(std::vector<int> &BoundaryNodes)
466 {
467 const int ID_UNDEFINED = -1;
468 const int N_ELEMENT = 10;
469
470 std::vector<std::vector<int>> ELementsOfNodes;
471 std::vector<int> ELementsOfNodesSize;
472
473 int MaxNodeID = GetMaxNodeID();
474
475 ELementsOfNodes.resize(MaxNodeID);
476 ELementsOfNodesSize.resize(MaxNodeID);
477
478 for (unsigned int i=0; i < ELementsOfNodes.size(); ++i) {
479 ELementsOfNodes[i].resize(N_ELEMENT);
480 ELementsOfNodesSize[i] = 0;
481 std::fill(ELementsOfNodes[i].begin(), ELementsOfNodes[i].end(), ID_UNDEFINED);
482 }
483
484 const unsigned int nElements = mrModelPart.NumberOfElements();
485 ModelPart::ElementsContainerType::iterator it_begin_elements = mrModelPart.ElementsBegin();
486
487 for (unsigned int i=0; i < nElements; ++i) {
488 ModelPart::ElementsContainerType::iterator pElemIt = it_begin_elements + i;
489 for (unsigned int iPoint=0; iPoint < pElemIt->GetGeometry().PointsNumber(); ++iPoint) {
490 int NodeID = pElemIt->GetGeometry()[iPoint].Id();
491 int ElementId = pElemIt->Id();
492
493 int index = NodeID-1;
494 ELementsOfNodesSize[index]++;
495 if (ELementsOfNodesSize[index] > N_ELEMENT-1) {
496 ELementsOfNodes[index].push_back(ElementId);
497 } else {
498 ELementsOfNodes[index][ELementsOfNodesSize[index]-1] = ElementId;
499 }
500 }
501 }
502
503 for (unsigned int i=0; i < nElements; ++i) {
504 ModelPart::ElementsContainerType::iterator pElemIt = it_begin_elements + i;
505
506 int nEdges = pElemIt->GetGeometry().EdgesNumber();
507 for (int iEdge = 0; iEdge < nEdges; ++iEdge) {
508 const unsigned int nPoints = pElemIt->GetGeometry().GenerateEdges()[iEdge].PointsNumber();
509 std::vector<int> FaceID(nPoints);
510 for (unsigned int iPoint = 0; iPoint < nPoints; ++iPoint) {
511 FaceID[iPoint] = pElemIt->GetGeometry().GenerateEdges()[iEdge].GetPoint(iPoint).Id();
512 }
513
514 if (!IsMoreThanOneElementWithThisEdgeFast(FaceID, ELementsOfNodes, ELementsOfNodesSize)) {
515 // boundary nodes:
516 for (unsigned int iPoint = 0; iPoint < nPoints; ++iPoint) {
517 std::vector<int>::iterator it = std::find(BoundaryNodes.begin(), BoundaryNodes.end(), FaceID[iPoint]);
518 if (it == BoundaryNodes.end()) {
519 BoundaryNodes.push_back(FaceID[iPoint]);
520 }
521 }
522 }
523 }
524 }
525 // KRATOS_INFO("FillListOfBoundaryNodesFast:BoundaryNodes") << BoundaryNodes << std::endl;
526
527 KRATOS_ERROR_IF(BoundaryNodes.size()==0)
528 << "No boundary node is found for interpolate line pressure process" << std::endl;
529
530 }

532 bool IsMoreThanOneElementWithThisEdgeFast(const std::vector<int> &FaceID,
533 const std::vector<std::vector<int>> &ELementsOfNodes,
534 const std::vector<int> &ELementsOfNodesSize)
535
536 {
537 const int ID_UNDEFINED = -1;
538 int nMaxElements = 0;
539 for (unsigned int iPoint = 0; iPoint < FaceID.size(); ++iPoint) {
540 int NodeID = FaceID[iPoint];
541 int index = NodeID-1;
542 nMaxElements += ELementsOfNodesSize[index];
543 }
544
545 if (nMaxElements > 0) {
546 std::vector<vector<int>> ElementIDs;
547 ElementIDs.resize(FaceID.size());
548 for (unsigned int i=0; i<ElementIDs.size(); ++i) {
549 ElementIDs[i].resize(nMaxElements);
550 std::fill(ElementIDs[i].begin(), ElementIDs[i].end(), ID_UNDEFINED);
551 }
552
553
554 for (unsigned int iPoint = 0; iPoint < FaceID.size(); ++iPoint) {
555 int NodeID = FaceID[iPoint];
556 int index = NodeID-1;
557 for (int i=0; i < ELementsOfNodesSize[index]; ++i) {
558 int iElementID = ELementsOfNodes[index][i];
559 ElementIDs[iPoint][i] = iElementID;
560 }
561 }
562
563
564 std::vector<int> SharedElementIDs;
565 for (unsigned int iPoint = 0; iPoint < FaceID.size(); ++iPoint) {
566 for (unsigned int i=0; i < ElementIDs[iPoint].size(); ++i) {
567 int iElementID = ElementIDs[iPoint][i];
568 bool found = false;
569 if (iElementID !=ID_UNDEFINED) {
570 for (unsigned int iPointInner = 0; iPointInner < FaceID.size(); ++iPointInner) {
571 if (iPointInner != iPoint) {
572 for (unsigned int j = 0; j < ElementIDs[iPointInner].size(); ++j) {
573 if (ElementIDs[iPointInner][j]==iElementID) found = true;
574 }
575 }
576 }
577 }
578
579 if (found) {
580 std::vector<int>::iterator it = std::find(SharedElementIDs.begin(), SharedElementIDs.end(), iElementID);
581 if (it == SharedElementIDs.end()) {
582 SharedElementIDs.push_back(iElementID);
583 }
584 }
585 }
586 }
587
588 if (SharedElementIDs.size() > 1)
589 return true;
590 else
591 return false;
592 }
593
594 return false;
595 }
