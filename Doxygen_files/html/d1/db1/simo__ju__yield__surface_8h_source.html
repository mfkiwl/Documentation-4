114 static void CalculateEquivalentStress(
115 const array_1d<double, VoigtSize>& rPredictiveStressVector,
116 const Vector& rStrainVector,
117 double& rEquivalentStress,
118 ConstitutiveLaw::Parameters& rValues
119 )
120 {
121 const Properties& r_material_properties = rValues.GetMaterialProperties();
122
123 // It compares with fc / sqrt(E)
124 array_1d<double, Dimension> principal_stress_vector;
125 AdvancedConstitutiveLawUtilities<VoigtSize>::CalculatePrincipalStresses(principal_stress_vector, rPredictiveStressVector);
126
127 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
128 const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
129 const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
130 const double n = std::abs(yield_compression / yield_tension);
131
132 double SumA = 0.0, SumB = 0.0, SumC = 0.0, ere0, ere1;
133 for (std::size_t cont = 0; cont < 2; ++cont) {
134 SumA += std::abs(principal_stress_vector[cont]);
135 SumB += 0.5 * (principal_stress_vector[cont] + std::abs(principal_stress_vector[cont]));
136 SumC += 0.5 * (-principal_stress_vector[cont] + std::abs(principal_stress_vector[cont]));
137 }
138 ere0 = SumB / SumA;
139 ere1 = SumC / SumA;
140
141 double auxf = 0.0;
142 for (std::size_t cont = 0; cont < VoigtSize; ++cont) {
143 auxf += rStrainVector[cont] * rPredictiveStressVector[cont]; // E:S
144 }
145 rEquivalentStress = std::sqrt(auxf);
146 rEquivalentStress *= (ere0 * n + ere1);
147 }

171 static void CalculateDamageParameter(
172 ConstitutiveLaw::Parameters& rValues,
173 double& rAParameter,
174 const double CharacteristicLength
175 )
176 {
177 const Properties& r_material_properties = rValues.GetMaterialProperties();
178
179 const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
180 const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
181 const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
182 const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
183 const double n = yield_compression / yield_tension;
184
185 if (r_material_properties[SOFTENING_TYPE] == static_cast<int>(SofteningType::Exponential)) {
186 rAParameter = 1.0 / (fracture_energy * n * n / (CharacteristicLength * std::pow(yield_compression, 2)) - 0.5);
187 KRATOS_ERROR_IF(rAParameter < 0.0) << "Fracture energy is too low, increase FRACTURE_ENERGY..." << std::endl;
188 } else { // linear
189 rAParameter = -std::pow(yield_compression, 2) / (2.0 * fracture_energy * n * n / CharacteristicLength);
190 }
191 }

238 static int Check(const Properties& rMaterialProperties)
239 {
240 if (!rMaterialProperties.Has(YIELD_STRESS)) {
241 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
242 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
243
244 const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
245 const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
246
247 KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
248 KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
249 } else {
250 const double yield_stress = rMaterialProperties[YIELD_STRESS];
251
252 KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
253 }
254 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRACTURE_ENERGY)) << "FRACTURE_ENERGY is not a defined value" << std::endl;
255 KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "YOUNG_MODULUS is not a defined value" << std::endl;
256
257 return TPlasticPotentialType::Check(rMaterialProperties);
258 }

271 static double GetScaleFactorTension(const Properties& rMaterialProperties)
272 {
273 const double yield_compression = rMaterialProperties.Has(YIELD_STRESS) ? rMaterialProperties[YIELD_STRESS] : rMaterialProperties[YIELD_STRESS_COMPRESSION];
274 const double yield_tension = rMaterialProperties.Has(YIELD_STRESS) ? rMaterialProperties[YIELD_STRESS] : rMaterialProperties[YIELD_STRESS_TENSION];
