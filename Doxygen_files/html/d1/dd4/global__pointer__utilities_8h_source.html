263  template< class TContainerType >
264  static GlobalPointersVector< typename TContainerType::value_type > RetrieveGlobalIndexedPointers(
265  const TContainerType& container,
266  const std::vector<int>& id_list,
267  const DataCommunicator& rDataCommunicator
268  )
269  {
270  auto global_pointers_list = RetrieveGlobalIndexedPointersMap(container, id_list, rDataCommunicator);
271 
272  int current_rank = rDataCommunicator.Rank();
273 
274  //compute final array
275  GlobalPointersVector< typename TContainerType::value_type > result;
276  result.reserve(id_list.size());
277  for(unsigned int i=0; i<id_list.size(); ++i)
278  {
279  auto it = global_pointers_list.find(id_list[i]);
280  if(it != global_pointers_list.end())
281  result.push_back( it->second );
282  else
283  KRATOS_ERROR << "id " << id_list[i] << " not found for processor " << current_rank << std::endl;
284  }
285 
286  return result;
287
