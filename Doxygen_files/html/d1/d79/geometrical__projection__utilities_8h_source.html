90 template<class TGeometryType>
91 static inline double FastProjectDirection(
92 const TGeometryType& rGeom,
93 const PointType& rPointToProject,
94 PointType& rPointProjected,
95 const array_1d<double,3>& rNormal,
96 const array_1d<double,3>& rVector,
97 const SizeType EchoLevel = 0
98 )
99 {
100 // Zero tolerance
101 const double zero_tolerance = std::numeric_limits<double>::epsilon();
102
103 // We define the distance
104 double distance = 0.0;
105
106 const array_1d<double,3> vector_points = rGeom[0].Coordinates() - rPointToProject.Coordinates();
107
108 if( norm_2( rVector ) < zero_tolerance && norm_2( rNormal ) > zero_tolerance ) {
109 distance = inner_prod(vector_points, rNormal)/norm_2(rNormal);
110 noalias(rPointProjected.Coordinates()) = rPointToProject.Coordinates() + rVector * distance;
111 KRATOS_WARNING_IF("GeometricalProjectionUtilities", EchoLevel > 0) << "WARNING:: Zero projection vector. Projection using the condition vector instead." << std::endl;
112 } else if (std::abs(inner_prod(rVector, rNormal) ) > zero_tolerance) {
113 distance = inner_prod(vector_points, rNormal)/inner_prod(rVector, rNormal);
114 noalias(rPointProjected.Coordinates()) = rPointToProject.Coordinates() + rVector * distance;
115 } else {
116 noalias(rPointProjected.Coordinates()) = rPointToProject.Coordinates();
117 KRATOS_WARNING_IF("GeometricalProjectionUtilities", EchoLevel > 0) << "WARNING:: The line and the plane are coplanar. Something wrong happened " << std::endl;
118 }
119
120 return distance;
121 }

194 template<class TGeometryType>
195 static inline double FastProjectOnLine(const TGeometryType& rGeometry,
196 const PointType& rPointToProject,
197 PointType& rPointProjected)
198 {
199 const array_1d<double, 3>& r_p_a = rGeometry[0].Coordinates();
200 const array_1d<double, 3>& r_p_b = rGeometry[1].Coordinates();
201 const array_1d<double, 3> ab = r_p_b - r_p_a;
202
203 const array_1d<double, 3>& p_c = rPointToProject.Coordinates();
204
205 const double factor = (inner_prod(r_p_b, p_c) - inner_prod(r_p_a, p_c) - inner_prod(r_p_b, r_p_a) + inner_prod(r_p_a, r_p_a)) / inner_prod(ab, ab);
206
207 rPointProjected.Coordinates() = r_p_a + factor * ab;
208
209 return norm_2(rPointProjected.Coordinates()-p_c);
210 }

222 template<class TGeometryType>
223 static inline double FastMinimalDistanceOnLine(
224 const TGeometryType& rGeometry,
225 const PointType& rPoint,
226 const double Tolerance = 1.0e-9
227 )
228 {
229 PointType projected_point;
230 const double projected_distance = FastProjectOnLine(rGeometry, rPoint, projected_point);
231 typename TGeometryType::CoordinatesArrayType projected_local;
232 if (rGeometry.IsInside(projected_point.Coordinates(), projected_local, Tolerance)) {
233 return projected_distance;
234 } else {
235 const double distance_a = rPoint.Distance(rGeometry[0]);
236 const double distance_b = rPoint.Distance(rGeometry[1]);
237 return std::min(distance_a, distance_b);
238 }

269 template<class TGeometryType, class TPointClass1, class TPointClass2 = TPointClass1>
270 static inline double FastProjectOnLine2D(
271 const TGeometryType& rGeometry,
272 const TPointClass1& rPointToProject,
273 TPointClass2& rPointProjected
274 )
275 {
276 // Node coordinates
277 const array_1d<double, 3>& r_p_a = rGeometry[0];
278 const array_1d<double, 3>& r_p_b = rGeometry[1];
279 const array_1d<double, 3>& r_p_c = rPointToProject;
280
281 // We compute the normal
282 array_1d<double,3> normal;
283 normal[0] = r_p_b[1] - r_p_a[1];
284 normal[1] = r_p_a[0] - r_p_b[0];
285 normal[2] = 0.0;
286
287 const double norm_normal = norm_2(normal);
288 KRATOS_ERROR_IF(norm_normal <= std::numeric_limits<double>::epsilon()) << "Zero norm normal: X: " << normal[0] << "\tY: " << normal[1] << std::endl;
289 normal /= norm_normal;
290
291 const array_1d<double,3> vector_points = r_p_a - r_p_c;
292 const double distance = inner_prod(vector_points, normal);
293 noalias(rPointProjected) = r_p_c + normal * distance;
294
295 return distance;
296 }

306 template<class TGeometryType>
307 static inline bool ProjectIterativeLine2D(
308 TGeometryType& rGeomOrigin,
309 const array_1d<double,3>& rPointDestiny,
310 array_1d<double,3>& rResultingPoint,
311 const array_1d<double, 3>& rNormal,
312 const double Tolerance = 1.0e-8,
313 double DeltaXi = 0.5
314 )
315 {
316// rResultingPoint.clear();
317
318 double old_delta_xi = 0.0;
319
320 array_1d<double, 3> current_global_coords;
321
322 KRATOS_DEBUG_CHECK_VARIABLE_IN_NODAL_DATA(NORMAL, rGeomOrigin[0])
323 KRATOS_DEBUG_CHECK_VARIABLE_IN_NODAL_DATA(NORMAL, rGeomOrigin[1])
324
325 array_1d<array_1d<double, 3>, 2> normals;
326 normals[0] = rGeomOrigin[0].FastGetSolutionStepValue(NORMAL);
327 normals[1] = rGeomOrigin[1].FastGetSolutionStepValue(NORMAL);
328
329 BoundedMatrix<double,2,2> X;
330 BoundedMatrix<double,2,1> DN;
331 for(unsigned int i=0; i<2;++i) {
332 X(0,i) = rGeomOrigin[i].X();
333 X(1,i) = rGeomOrigin[i].Y();
334 }
335
336 Matrix J = ZeroMatrix( 1, 1 );
337
338 //Newton iteration:
339
340 const unsigned int max_iter = 20;
341
342 for ( unsigned int k = 0; k < max_iter; ++k ) {
343 array_1d<double, 2> N_origin;
344 N_origin[0] = 0.5 * ( 1.0 - rResultingPoint[0]);
345 N_origin[1] = 0.5 * ( 1.0 + rResultingPoint[0]);
346
347 array_1d<double,3> normal_xi = N_origin[0] * normals[0] + N_origin[1] * normals[1];
348 normal_xi = normal_xi/norm_2(normal_xi);
349
350 current_global_coords = ZeroVector(3);
351 for( IndexType i_node = 0; i_node < 2; ++i_node )
352 current_global_coords += N_origin[i_node] * rGeomOrigin[i_node].Coordinates();
353
354 const array_1d<double,3> VectorPoints = rGeomOrigin.Center() - rPointDestiny;
355 const double distance = inner_prod(VectorPoints, rNormal)/inner_prod(-normal_xi, rNormal);
356 const array_1d<double, 3> current_destiny_global_coords = rPointDestiny - normal_xi * distance;
357
358 // Derivatives of shape functions
359 Matrix ShapeFunctionsGradients;
360 ShapeFunctionsGradients = rGeomOrigin.ShapeFunctionsLocalGradients(ShapeFunctionsGradients, rResultingPoint );
361
362 #ifdef KRATOS_USE_AMATRIX // This macro definition is for the migration period and to be removed afterward please do not use it
363 DN = prod(X,ShapeFunctionsGradients);
364
365 J = prod(trans(DN),DN); // TODO: Add the non linearity concerning the normal
366 #else
367 noalias(DN) = prod(X,ShapeFunctionsGradients);
368
369 noalias(J) = prod(trans(DN),DN); // TODO: Add the non linearity concerning the normal
370 #endif // ifdef KRATOS_USE_AMATRIX
371
372 const Vector RHS = prod(trans(DN),subrange(current_destiny_global_coords - current_global_coords,0,2));
373
374 old_delta_xi = DeltaXi;
375 DeltaXi = RHS[0]/J(0, 0);
376
377 rResultingPoint[0] += DeltaXi;
378
379 if (rResultingPoint[0] <= -1.0)
380 rResultingPoint[0] = -1.0;
381 else if (rResultingPoint[0] >= 1.0)
382 rResultingPoint[0] = 1.0;
383
384 if ( std::abs(DeltaXi - old_delta_xi) < Tolerance )
385 return true;
386 }
387
388 return false;
389 }
