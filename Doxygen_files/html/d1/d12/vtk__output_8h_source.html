413 template <typename TData>
414 void WriteScalarDataToFile(const TData& rData, std::ofstream& rFileStream) const
415 {
416 if (mFileFormat == VtkOutput::FileFormat::VTK_ASCII) {
417 rFileStream << rData;
418 } else if (mFileFormat == VtkOutput::FileFormat::VTK_BINARY) {
419 TData data = rData;
420 ForceBigEndian(reinterpret_cast<unsigned char *>(&data));
421 rFileStream.write(reinterpret_cast<char *>(&data), sizeof(TData));
422 }
423 }
424
431 template <typename TData>
432 void WriteVectorDataToFile(const TData& rData, std::ofstream& rFileStream) const
433 {
434 if (mFileFormat == VtkOutput::FileFormat::VTK_ASCII) {
435 for (const auto& r_data_comp : rData) {
436 rFileStream << float(r_data_comp) << " ";
437 }
438 } else if (mFileFormat == VtkOutput::FileFormat::VTK_BINARY) {
439 for (const auto& r_data_comp : rData ) {
440 float data_comp_local = (float)r_data_comp; // should not be const or a reference for enforcing big endian
441 ForceBigEndian(reinterpret_cast<unsigned char *>(&data_comp_local));
442 rFileStream.write(reinterpret_cast<char *>(&data_comp_local), sizeof(float));
443 }
444 }
445 }
