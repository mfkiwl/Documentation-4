578  void ComputeSurfaceNormals(){
579 
580  for(auto& control_obj : mControlSettings["controlling_objects"]){
581  ModelPart& r_controlling_object = mrModel.GetModelPart(control_obj.GetString());
582  VariableUtils().SetHistoricalVariableToZero(NORMAL, r_controlling_object.Nodes());
583  const array_1d<double,3> local_coords = ZeroVector(3);
584  for(auto& cond_i : r_controlling_object.Conditions()){
585  const auto& r_geom = cond_i.GetGeometry();
586 
587  const array_1d<double,3> normal = r_geom.Normal(local_coords);
588  const double coeff = 1.0/r_geom.size();
589 
590  for(auto& node_i : r_geom)
591  {
592  node_i.SetLock();
593  noalias(node_i.FastGetSolutionStepValue(NORMAL)) += coeff * normal;
594  node_i.UnSetLock();
595  }
596  }
597  for (auto& node_i : r_controlling_object.Nodes())
598  {
599  array_1d<double,3>& area_normal = node_i.FastGetSolutionStepValue(NORMAL);
600  double& nodal_area = node_i.FastGetSolutionStepValue(NODAL_AREA);
601 
602  const double norm2 = norm_2(area_normal);
603  KRATOS_ERROR_IF(norm2<1e-10) << "CalculateUnitNormals: Norm2 of normal for node "
604  << node_i.Id() << " is < 1e-10!" << std::endl;
605 
606  noalias(area_normal) = area_normal/norm2;
607  nodal_area = norm2;
608  }
609  }

612  void AdjustFilterSizes()
613  {
614  for(int obj_i=0;obj_i<mpVMModelPartsTypes.size();obj_i++)
615  if(mpVMModelPartsTypes[obj_i]=="solid"){
616  ModelPart* mpVMModePart = mpVMModelParts[obj_i];
617  Properties::Pointer p_vm_model_part_property = mpVMModelPartsProperties[obj_i];
618 
619  p_vm_model_part_property->SetValue(HELMHOLTZ_BULK_RADIUS_SHAPE,1.0);
620 
621  double total_volume_strain_energy = 0.0;
622  #pragma omp parallel for reduction(+:total_volume_strain_energy)
623  for(auto& elem_i : mpVMModePart->Elements()){
624  double elem_strain_energy;
625  elem_i.Calculate(ELEMENT_STRAIN_ENERGY,elem_strain_energy,mpVMModePart->GetProcessInfo());
626  total_volume_strain_energy += elem_strain_energy;
627  }
628 
629  double total_surface_strain_energy = 0.0;
630  #pragma omp parallel for reduction(+:total_surface_strain_energy)
631  for(auto& cond_i : mpVMModePart->Conditions()){
632  double cond_strain_energy;
633  cond_i.Calculate(ELEMENT_STRAIN_ENERGY,cond_strain_energy,mpVMModePart->GetProcessInfo());
634  total_surface_strain_energy += cond_strain_energy;
635  }
636  double bulk_surface_ratio = mTechniqueSettings["surface_bulk_ratio"].GetDouble();
637  double bulk_filter_size = (total_surface_strain_energy)/(bulk_surface_ratio * total_volume_strain_energy);
638  bulk_filter_size = std::pow(bulk_filter_size,0.1/0.1);
639  p_vm_model_part_property->SetValue(HELMHOLTZ_BULK_RADIUS_SHAPE,bulk_filter_size);
640 
641  KRATOS_INFO("ImplicitVertexMorphing:AdjustFilterSizes") << " ++++ total_volume_strain_energy: "<<total_volume_strain_energy<<" ++++ "<<std::endl;
642  KRATOS_INFO("ImplicitVertexMorphing:AdjustFilterSizes") << " ++++ total_surface_strain_energy: "<<total_surface_strain_energy<<" ++++ "<<std::endl;
643  KRATOS_INFO("ImplicitVertexMorphing:AdjustFilterSizes") << " ++++ bulk_filter_size: "<<bulk_filter_size<<" ++++ "<<std::endl;
644  }

658  // // calculate (K+M) * X element wise, here we treat CX as an auxilary
659  rCurrentProcessInfo[COMPUTE_CONTROL_POINTS_SHAPE] = false;
660  #pragma omp parallel for
661  for(auto& node_i : mpVMModePart->Nodes())
662  {
663  array_3d& r_nodal_variable_hl_vars = node_i.FastGetSolutionStepValue(HELMHOLTZ_VARS_SHAPE);
664  r_nodal_variable_hl_vars(0) = node_i.X0();
665  r_nodal_variable_hl_vars(1) = node_i.Y0();
666  r_nodal_variable_hl_vars(2) = node_i.Z0();
667  }
668 
669  #pragma omp parallel for
670  for(auto& elem_i : mpVMModePart->Elements())
671  {
672  VectorType rhs;
673  MatrixType lhs;
674  elem_i.Initialize(mpVMModePart->GetProcessInfo());
675  elem_i.CalculateLocalSystem(lhs,rhs,mpVMModePart->GetProcessInfo());
676  AddElementVariableValuesVector(elem_i,CX,rhs,-1.0);
677  }
678 
679  #pragma omp parallel for
680  for(auto& cond_i : mpVMModePart->Conditions())
681  {
682  VectorType rhs;
683  MatrixType lhs;
684  cond_i.Initialize(mpVMModePart->GetProcessInfo());
685  cond_i.CalculateLocalSystem(lhs,rhs,mpVMModePart->GetProcessInfo());
686  AddConditionVariableValuesVector(cond_i,CX,rhs,-1.0);
687  }
688 
689 
690  // here we fill the RHS of M * S = (K+M) * X
691  SetVariable1ToVarible2(mpVMModePart,CX,HELMHOLTZ_SOURCE_SHAPE);
692  SetVariableZero(mpVMModePart,HELMHOLTZ_VARS_SHAPE);
693 
694  // apply BC on the RHS and unassign BC
695  #pragma omp parallel for
696  for(auto& node_i : mpVMModePart->Nodes())
697  {
698  array_3d& r_nodal_variable_hl_vars = node_i.FastGetSolutionStepValue(HELMHOLTZ_VARS_SHAPE);
699 
700  if(node_i.IsFixed(HELMHOLTZ_VARS_SHAPE_X))
701  r_nodal_variable_hl_vars(0) = node_i.X0();
702 
703  if(node_i.IsFixed(HELMHOLTZ_VARS_SHAPE_Y))
704  r_nodal_variable_hl_vars(1) = node_i.Y0();
705 
706  if(node_i.IsFixed(HELMHOLTZ_VARS_SHAPE_Z))
707  r_nodal_variable_hl_vars(2) = node_i.Z0();
708  }
709 
710  // here we compute S = M-1 * (K+M) * X
711  rCurrentProcessInfo[COMPUTE_CONTROL_POINTS_SHAPE] = true;
712  mpStrategies[model_i]->Solve();
713  // here we clear/reset the problem for the clearaty's sake
714  mpStrategies[model_i]->GetStrategy()->Clear();
715  SetVariable1ToVarible2(mpVMModePart,HELMHOLTZ_VARS_SHAPE,CX);
716  rCurrentProcessInfo[COMPUTE_CONTROL_POINTS_SHAPE] = false;
717  }
