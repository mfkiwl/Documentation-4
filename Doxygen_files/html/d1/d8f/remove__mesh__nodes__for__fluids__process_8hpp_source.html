280 bool RemoveNodesOnDistance(int &inside_nodes_removed, int &boundary_nodes_removed)
281 {
282 KRATOS_TRY
283
284 const unsigned int dimension = mrModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
285
286 //***SIZES :::: parameters do define the tolerance in mesh size:
287 bool some_node_is_removed = false;
288
289 // bucket size definition:
290 unsigned int bucket_size = 20;
291
292 // create the list of the nodes to be check during the search
293 std::vector<Node<3>::Pointer> list_of_nodes;
294 list_of_nodes.reserve(mrModelPart.NumberOfNodes());
295 for (ModelPart::NodesContainerType::iterator i_node = mrModelPart.NodesBegin(); i_node != mrModelPart.NodesEnd(); i_node++)
296 {
297 (list_of_nodes).push_back(*(i_node.base()));
298 }
299
300 KdtreeType nodes_tree(list_of_nodes.begin(), list_of_nodes.end(), bucket_size);
302 // all of the nodes in this list will be preserved
303 unsigned int num_neighbours = 100;
304 std::vector<Node<3>::Pointer> neighbours(num_neighbours);
305 std::vector<double> neighbour_distances(num_neighbours);
306
307 // radius means the distance, if the distance between two nodes is closer to radius -> mark for removing
308 double radius = 0;
309 Node<3> work_point(0, 0.0, 0.0, 0.0);
310 unsigned int n_points_in_radius;
311
312 const ProcessInfo &rCurrentProcessInfo = mrModelPart.GetProcessInfo();
313 const double currentTime = rCurrentProcessInfo[TIME];
314 double meshSize = mrRemesh.Refine->CriticalRadius;
315
316 bool refiningBox = false;
317 for (unsigned int index = 0; index < mrRemesh.UseRefiningBox.size(); index++)
318 {
319 if (mrRemesh.UseRefiningBox[index] == true && currentTime > mrRemesh.RefiningBoxInitialTime[index] && currentTime < mrRemesh.RefiningBoxFinalTime[index])
320 {
321 refiningBox = true;
322 }
323 }
324
325 const unsigned int principalModelPartId = rCurrentProcessInfo[MAIN_MATERIAL_PROPERTY];
326 unsigned int erased_nodes = 0;
327 for (ModelPart::ElementsContainerType::const_iterator ie = mrModelPart.ElementsBegin();
328 ie != mrModelPart.ElementsEnd(); ie++)
329 {
330 unsigned int rigidNodes = 0;
331 // coordinates
332 for (unsigned int i = 0; i < ie->GetGeometry().size(); i++)
333 {
334 if ((ie->GetGeometry()[i].Is(RIGID) && ie->GetGeometry()[i].IsNot(INLET)) || ie->GetGeometry()[i].Is(SOLID))
335 {
336 rigidNodes++;
337 }
338 }
339 if (dimension == 2)
340 {
341 if (rigidNodes > 0)
342 EraseCriticalNodes2D(ie->GetGeometry(), erased_nodes, inside_nodes_removed);
343 }
344 else if (dimension == 3)
345 {
346 if (rigidNodes > 1)
347 EraseCriticalNodes3D(ie->GetGeometry(), erased_nodes, inside_nodes_removed, rigidNodes);
348 }
349 }
350
351 for (ModelPart::NodesContainerType::const_iterator in = mrModelPart.NodesBegin(); in != mrModelPart.NodesEnd(); in++)
352 {
353
354 const unsigned int propertyIdNode = in->FastGetSolutionStepValue(PROPERTY_ID);
355 meshSize = mrRemesh.Refine->CriticalRadius;
356 double rigidNodeLocalMeshSize = 0;
357 double rigidNodeMeshCounter = 0;
358 NodeWeakPtrVectorType &rN = in->GetValue(NEIGHBOUR_NODES);
359
360 for (unsigned int i = 0; i < rN.size(); i++)
361 {
362 if (rN[i].Is(RIGID))
363 {
364 rigidNodeLocalMeshSize += rN[i].FastGetSolutionStepValue(NODAL_H_WALL);
365 rigidNodeMeshCounter += 1.0;
366 }
367 }
368
369 if (refiningBox == true)
370 {
371 array_1d<double, 3> NodeCoordinates = in->Coordinates();
372 if (dimension == 2)
373 {
374 bool insideTransitionZone = false;
375 mMesherUtilities.DefineMeshSizeInTransitionZones2D(mrRemesh, currentTime, NodeCoordinates, meshSize, insideTransitionZone);
376 }
377 else if (dimension == 3)
378 {
379 bool insideTransitionZone = false;
380 mMesherUtilities.DefineMeshSizeInTransitionZones3D(mrRemesh, currentTime, NodeCoordinates, meshSize, insideTransitionZone);
381 }
382 }
383
384 if (rigidNodeMeshCounter > 0)
385 {
386 const double rigidWallMeshSize = rigidNodeLocalMeshSize / rigidNodeMeshCounter;
387 const double ratio = rigidWallMeshSize / meshSize;
388 const double tolerance = 2.0;
389 if (ratio > tolerance)
390 {
391 meshSize *= 0.5;
392 meshSize += 0.5 * rigidWallMeshSize;
393 }
394 }
395
396 const double size_for_distance_boundary = 0.6 * meshSize;
397 const double size_for_wall_tip_contact_side = 0.15 * mrRemesh.Refine->CriticalSide;
398
399 if (in->Is(TO_ERASE))
400 {
401 some_node_is_removed = true;
402 }
403 bool on_contact_tip = false;
404
405 if (in->Is(TO_SPLIT) || in->Is(CONTACT))
406 on_contact_tip = true;
407
408 if (in->IsNot(NEW_ENTITY) && in->IsNot(INLET) && in->IsNot(ISOLATED) && in->IsNot(RIGID) && in->IsNot(SOLID))
409 {
410 unsigned int neighErasedNodes = 0;
411 radius = 0.6 * meshSize;
412
413 work_point[0] = in->X();
414 work_point[1] = in->Y();
415 work_point[2] = in->Z();
416 unsigned int freeSurfaceNeighNodes = 0;
417 // unsigned int rigidNeighNodes=0;
418 bool interfaceElement = false;
419
420 if (in->Is(FREE_SURFACE))
421 {
422 // it must be more difficult to erase a free_surface node, otherwise, lot of volume is lost
423 // this value has a strong effect on volume variation due to remeshing
424 radius = 0.475 * meshSize; // compared with element radius
425 NodeWeakPtrVectorType &neighb_nodes = in->GetValue(NEIGHBOUR_NODES);
426 unsigned int countRigid = 0;
427
428 for (NodeWeakPtrVectorType::iterator nn = neighb_nodes.begin(); nn != neighb_nodes.end(); nn++)
429 {
430 if ((nn)->Is(RIGID) || (nn)->Is(SOLID))
431 {
432 countRigid++;
433 }
434 if ((nn)->Is(TO_ERASE))
435 {
436 neighErasedNodes++;
437 }
438 }
439 if (countRigid == neighb_nodes.size())
440 {
441 radius = 0.15 * meshSize;
442 }
443 }
444 else
445 {
446 NodeWeakPtrVectorType &neighb_nodes = in->GetValue(NEIGHBOUR_NODES);
447 for (NodeWeakPtrVectorType::iterator nn = neighb_nodes.begin(); nn != neighb_nodes.end(); nn++)
448 {
449 const unsigned int propertyIdSecondNode = (nn)->FastGetSolutionStepValue(PROPERTY_ID);
450 if ((nn)->Is(FREE_SURFACE))
451 {
452 freeSurfaceNeighNodes++;
453 }
454 if ((nn)->Is(TO_ERASE))
455 {
456 neighErasedNodes++;
457 }
458 if (propertyIdNode != propertyIdSecondNode && (nn)->IsNot(RIGID))
459 {
460 interfaceElement = true;
461 }
462 }
463 }
464
465 if (freeSurfaceNeighNodes > 1)
466 {
467 radius = 0.5 * meshSize;
468 }
469 else if (interfaceElement == true)
470 {
471 if (dimension == 2)
472 radius = 0.54 * meshSize; // 10% less than normal nodes
473 if (dimension == 3)
474 radius = 0.48 * meshSize; // 20% less than normal nodes
475 }
476
477 if (in->Is(INLET))
478 {
479 radius = 0.3 * meshSize; // compared with element radius
480 }
481 n_points_in_radius = nodes_tree.SearchInRadius(work_point, radius, neighbours.begin(), neighbour_distances.begin(), num_neighbours);
482
483 if (n_points_in_radius > 1 && neighErasedNodes == 0)
484 {
485
486 if (in->IsNot(INLET) && in->IsNot(RIGID) && in->IsNot(SOLID) && in->IsNot(ISOLATED))
487 {
488 if (mrRemesh.Refine->RemovingOptions.Is(MesherUtilities::REMOVE_NODES_ON_DISTANCE))
489 {
490
491 if (in->IsNot(FREE_SURFACE) && in->IsNot(RIGID) && freeSurfaceNeighNodes == dimension)
492 {
493 NodeWeakPtrVectorType &neighb_nodes = in->GetValue(NEIGHBOUR_NODES);
494 array_1d<double, 3> sumOfCoordinates = in->Coordinates();
495 array_1d<double, 3> sumOfCurrentVelocities = in->FastGetSolutionStepValue(VELOCITY, 0);
496 array_1d<double, 3> sumOfPreviousVelocities = in->FastGetSolutionStepValue(VELOCITY, 1);
497 double sumOfPressures = in->FastGetSolutionStepValue(PRESSURE, 0);
498 double counter = 1.0;
499
500 for (NodeWeakPtrVectorType::iterator nn = neighb_nodes.begin(); nn != neighb_nodes.end(); nn++)
501 {
502 counter += 1.0;
503 noalias(sumOfCoordinates) += (nn)->Coordinates();
504 noalias(sumOfCurrentVelocities) += (nn)->FastGetSolutionStepValue(VELOCITY, 0);
505 noalias(sumOfPreviousVelocities) += (nn)->FastGetSolutionStepValue(VELOCITY, 1);
506 sumOfPressures += (nn)->FastGetSolutionStepValue(PRESSURE, 0);
507 }
508 in->X() = sumOfCoordinates[0] / counter;
509 in->Y() = sumOfCoordinates[1] / counter;
510 in->X0() = sumOfCoordinates[0] / counter;
511 in->Y0() = sumOfCoordinates[1] / counter;
512 in->FastGetSolutionStepValue(DISPLACEMENT_X, 0) = 0;
513 in->FastGetSolutionStepValue(DISPLACEMENT_Y, 0) = 0;
514 in->FastGetSolutionStepValue(DISPLACEMENT_X, 1) = 0;
515 in->FastGetSolutionStepValue(DISPLACEMENT_Y, 1) = 0;
516 in->FastGetSolutionStepValue(VELOCITY_X, 0) = sumOfCurrentVelocities[0] / counter;
517 in->FastGetSolutionStepValue(VELOCITY_Y, 0) = sumOfCurrentVelocities[1] / counter;
518 in->FastGetSolutionStepValue(VELOCITY_X, 1) = sumOfPreviousVelocities[0] / counter;
519 in->FastGetSolutionStepValue(VELOCITY_Y, 1) = sumOfPreviousVelocities[1] / counter;
520 in->FastGetSolutionStepValue(PRESSURE, 0) = sumOfPressures / counter;
521 if (dimension == 3)
522 {
523 in->Z() = sumOfCoordinates[2] / counter;
524 in->Z0() = sumOfCoordinates[2] / counter;
525 in->FastGetSolutionStepValue(DISPLACEMENT_Z, 0) = 0;
526 in->FastGetSolutionStepValue(DISPLACEMENT_Z, 1) = 0;
527 in->FastGetSolutionStepValue(VELOCITY_Z, 0) = sumOfCurrentVelocities[2] / counter;
528 in->FastGetSolutionStepValue(VELOCITY_Z, 1) = sumOfPreviousVelocities[2] / counter;
529 }
530 }
531 else
532 {
533 in->Set(TO_ERASE);
534 some_node_is_removed = true;
535 inside_nodes_removed++;
536 if (propertyIdNode != principalModelPartId) // this is to conserve the number of nodes of the smaller domain in case of a two-fluid analysis
537 {
538 mrRemesh.Info->BalancePrincipalSecondaryPartsNodes += -1;
539 }
540 }
541 }
542 }
543 else if (in->IsNot(INLET))
544 {
545 unsigned int k = 0;
546 unsigned int counter = 0;
547 for (std::vector<Node<3>::Pointer>::iterator nn = neighbours.begin(); nn != neighbours.begin() + n_points_in_radius; nn++)
548 {
549 bool nn_on_contact_tip = false;
550
551 if ((*nn)->Is(TO_SPLIT) || (*nn)->Is(CONTACT))
552 nn_on_contact_tip = true;
553
554 if ((*nn)->Is(BOUNDARY) && !nn_on_contact_tip && neighbour_distances[k] < size_for_distance_boundary && neighbour_distances[k] > 0.0)
555 {
556 if ((*nn)->IsNot(TO_ERASE))
557 {
558 counter += 1;
559 }
560 }
561
562 if ((*nn)->Is(BOUNDARY) && nn_on_contact_tip && neighbour_distances[k] < size_for_wall_tip_contact_side)
563 {
564 if ((*nn)->IsNot(TO_ERASE))
565 {
566 counter += 1;
567 }
568 }
569 k++;
570 }
571
572 if (counter > 1 && in->IsNot(RIGID) && in->IsNot(SOLID) && in->IsNot(NEW_ENTITY) && !on_contact_tip)
573 { // Can be inserted in the boundary refine
574 in->Set(TO_ERASE);
575 if (mEchoLevel > 1)
576 std::cout << " Removed Boundary Node [" << in->Id() << "] on Distance " << std::endl;
577 some_node_is_removed = true;
578 boundary_nodes_removed++;
579
580 if (propertyIdNode != principalModelPartId) // this is to conserve the number of nodes of the smaller domain in case of a two-fluid analysis
581 {
582 mrRemesh.Info->BalancePrincipalSecondaryPartsNodes += -1;
583 }
584 }
585 }
586 }
587 }
588 }
589
590 if (erased_nodes > 0)
591 {
592 if (mEchoLevel > 1)
593 std::cout << "layer_nodes_removed " << erased_nodes << std::endl;
594 some_node_is_removed = true;
595 }
596 // Build boundary after removing boundary nodes due distance criterion
597 if (mEchoLevel > 1)
598 {
599 std::cout << "boundary_nodes_removed " << boundary_nodes_removed << std::endl;
600 std::cout << "inside_nodes_removed " << inside_nodes_removed << std::endl;
601 }
602 return some_node_is_removed;
603
604 KRATOS_CATCH(" ")
605 }

607 void EraseCriticalNodes2D(Element::GeometryType &eElement, unsigned int &erased_nodes, int &inside_nodes_removed)
608 {
609
610 KRATOS_TRY
611
612 double safetyCoefficient2D = 0.5;
613 double elementVolume = eElement.Area();
614 unsigned int numNodes = eElement.size();
615
616 const ProcessInfo &rCurrentProcessInfo = mrModelPart.GetProcessInfo();
617 const unsigned int principalModelPartId = rCurrentProcessInfo[MAIN_MATERIAL_PROPERTY];
618
619 array_1d<double, 3> Edges(3, 0.0);
620 array_1d<unsigned int, 3> FirstEdgeNode(3, 0);
621 array_1d<unsigned int, 3> SecondEdgeNode(3, 0);
622 double wallLength = 0;
623 array_1d<double, 3> CoorDifference = eElement[1].Coordinates() - eElement[0].Coordinates();
624 double SquaredLength = CoorDifference[0] * CoorDifference[0] + CoorDifference[1] * CoorDifference[1];
625 Edges[0] = std::sqrt(SquaredLength);
626 FirstEdgeNode[0] = 0;
627 SecondEdgeNode[0] = 1;
628 if (eElement[0].Is(RIGID) && eElement[1].Is(RIGID))
629 {
630 wallLength = Edges[0];
631 }
632 unsigned int counter = 0;
633 for (unsigned int i = 2; i < eElement.size(); i++)
634 {
635 for (unsigned int j = 0; j < i; j++)
636 {
637 noalias(CoorDifference) = eElement[i].Coordinates() - eElement[j].Coordinates();
638 SquaredLength = CoorDifference[0] * CoorDifference[0] + CoorDifference[1] * CoorDifference[1];
639 counter += 1;
640 Edges[counter] = std::sqrt(SquaredLength);
641 FirstEdgeNode[counter] = j;
642 SecondEdgeNode[counter] = i;
643 if (eElement[i].Is(RIGID) && eElement[j].Is(RIGID) && Edges[counter] > wallLength)
644 {
645 wallLength = Edges[counter];
646 }
647 }
648 }
649
651 for (unsigned int i = 0; i < eElement.size(); i++)
652 {
653 if (eElement[i].IsNot(RIGID) && eElement[i].IsNot(TO_ERASE) && eElement[i].IsNot(SOLID) && eElement[i].IsNot(ISOLATED))
654 {
655 double height = elementVolume * 2.0 / wallLength;
656
658 if (eElement[i].Is(FREE_SURFACE))
659 {
660 NodeWeakPtrVectorType &neighb_nodes = eElement[i].GetValue(NEIGHBOUR_NODES);
661 unsigned int countRigid = 0;
662 unsigned int countFreeSurface = 0;
663 for (NodeWeakPtrVectorType::iterator nn = neighb_nodes.begin(); nn != neighb_nodes.end(); nn++)
664 {
665 if ((nn)->Is(RIGID) || (nn)->Is(SOLID))
666 {
667 countRigid++;
668 }
669 if ((nn)->Is(FREE_SURFACE) && (nn)->IsNot(RIGID) && (nn)->IsNot(SOLID))
670 {
671 countFreeSurface++;
672 }
673 }
674 if ((countRigid + countFreeSurface) == neighb_nodes.size() && countRigid > 0)
675 {
676 safetyCoefficient2D = 0.25;
677 }
678 }
679
681 if (height < (0.5 * safetyCoefficient2D * wallLength))
682 {
683 eElement[i].Set(TO_ERASE);
684 erased_nodes += 1;
685 inside_nodes_removed++;
686
687 const unsigned int propertyIdNode = eElement[i].FastGetSolutionStepValue(PROPERTY_ID);
688 if (propertyIdNode != principalModelPartId) // this is to conserve the number of nodes of the smaller domain in case of a two-fluid analysis
689 {
690 mrRemesh.Info->BalancePrincipalSecondaryPartsNodes += -1;
691 }
692 }
693 }
694 }
695
696 bool longDamBreak = false; // to attivate in case of long dam breaks to avoid separeted elements in the water front
697 if (longDamBreak == true)
698 {
699 for (unsigned int i = 0; i < numNodes; i++)
700 {
701 if (eElement[i].Is(FREE_SURFACE) && eElement[i].IsNot(RIGID))
702 {
703
704 GlobalPointersVector<Element> &neighb_elems = eElement[i].GetValue(NEIGHBOUR_ELEMENTS);
705 GlobalPointersVector<Node<3>> &neighb_nodes = eElement[i].GetValue(NEIGHBOUR_NODES);
