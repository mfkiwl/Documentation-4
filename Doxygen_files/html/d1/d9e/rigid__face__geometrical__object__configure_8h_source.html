173 static inline bool IntersectionBox(const PointerType& rObject, const PointType& rLowPoint, const PointType& rHighPoint) {
174
175 double xyz_min[3] = { 1e20, 1e20, 1e20};
176 double xyz_max[3] = {-1e20, -1e20, -1e20};
177
178 for (std::size_t inode = 0; inode < rObject->GetGeometry().size(); inode++) {
179 const array_1d<double, 3>& Coord = rObject->GetGeometry()[inode].Coordinates();
180 xyz_min[0] = (xyz_min[0] > Coord[0]) ? Coord[0] : xyz_min[0];
181 xyz_min[1] = (xyz_min[1] > Coord[1]) ? Coord[1] : xyz_min[1];
182 xyz_min[2] = (xyz_min[2] > Coord[2]) ? Coord[2] : xyz_min[2];
183
184 xyz_max[0] = (xyz_max[0] < Coord[0]) ? Coord[0] : xyz_max[0];
185 xyz_max[1] = (xyz_max[1] < Coord[1]) ? Coord[1] : xyz_max[1];
186 xyz_max[2] = (xyz_max[2] < Coord[2]) ? Coord[2] : xyz_max[2];
187 }
188
189 bool intersect = (rLowPoint [0] <= xyz_max[0] && rLowPoint [1] <= xyz_max[1] && rLowPoint [2] <= xyz_max[2] &&
190 rHighPoint[0] >= xyz_min[0] && rHighPoint[1] >= xyz_min[1] && rHighPoint[2] >= xyz_min[2]);
191
192 return intersect;

202 floatle(rLowPoint[1] - radius,center_of_particle[1]) &&
203 floatle(rLowPoint[2] - radius,center_of_particle[2]) &&

205 floatge(rHighPoint[1] + radius,center_of_particle[1]) &&
206 floatge(rHighPoint[2] + radius,center_of_particle[2]));

226 static inline bool FastIntersection3D(const GeometryType& DE_Geom,const GeometryType& FE_Geom, const double& Radius)
227 {
228 //rObj_1 is particle, and rObj_2 is condition
229 bool ContactExists = false;
230
231 unsigned int FE_size = FE_Geom.size();
232 std::vector< array_1d<double,3> > Coord(FE_size);
233 //Coord.resize(FE_size); //, array_1d<double,3>(3,0.0) );
234
235 for (unsigned int i = 0; i<FE_size; i++) {
236 for (unsigned int j = 0; j<3; j++) {
237 Coord[i][j] = FE_Geom[i].Coordinates()[j];
238 }
239 }
240
241 double distance_point_to_plane = 0.0;
242 unsigned int current_edge_index = 0;
243
244 ContactExists = GeometryFunctions::FastFacetCheck( Coord, DE_Geom[0].Coordinates(), Radius, distance_point_to_plane, current_edge_index);
245
246 if (ContactExists){return true;}
247
248 //The key here is to see that we only need to check for further contact if, when not having contact with plane, the distance_point_to_plane is lower than the radius.
249 //In this case it might have contact with edges or vertices, otherwise no contact is possible.
250
251 else if (distance_point_to_plane < Radius) {
252 bool local_contact_exists = false;
253 for (unsigned int e = current_edge_index; e < FE_size; e++ ) {
254 local_contact_exists = GeometryFunctions::FastEdgeVertexCheck( Coord[e], Coord[(e+1)%FE_size], DE_Geom[0].Coordinates(), Radius );
255 if (local_contact_exists) {return true;}
256 }//for every edge
257 }//no plane contact found
258
259 return false;
260 }//FastIntersection3D

313 static inline void Distance(const PointerType& rObj_1, const PointerType& rObj_2, double& distance) {
314 const array_1d<double, 3>& center_of_particle1 = rObj_1->GetGeometry()[0];
315 const array_1d<double, 3>& center_of_particle2 = rObj_2->GetGeometry()[0];
316

319 (center_of_particle1[2] - center_of_particle2[2]) * (center_of_particle1[2] - center_of_particle2[2]) );

323 static inline bool DistanceHierarchy(SphericParticle* rObj_1,
324 DEMWall* rObj_2,
325 const double LocalCoordSystem[3][3],
326 const double DistPToB,
327 std::vector<double> Weight,
328 int ContactType,
329 std::vector< double > & Distance_Array,
330 std::vector< array_1d<double,3> >& Normal_Array,
331 std::vector< array_1d<double,4> >& Weight_Array,
332 std::vector<int> & Id_Array,
333 std::vector<int> & ContactTypes
334 )
335 {
336
337 int new_ID = rObj_2->Id();
338 std::size_t i_current = Normal_Array.size();
339 std::size_t neighbor_size = Normal_Array.size();
340
341 bool substitute = false;
342 int position = i_current;
343
344 for (std::size_t i_old_neigh = 0; i_old_neigh < neighbor_size; i_old_neigh++)
345 {
346 //int i_Old_RF_position = i_old_neigh * 16;
347
348 double Old_Normal_Vector[3] = {0.0};
349 Old_Normal_Vector[0] = Normal_Array[i_old_neigh][0];
350 Old_Normal_Vector[1] = Normal_Array[i_old_neigh][1];
351 Old_Normal_Vector[2] = Normal_Array[i_old_neigh][2];
352
353 double New_Dist = DistPToB;
354 double Old_dist = Distance_Array[i_old_neigh];
355
356 double New_projected_on_old = DEM_INNER_PRODUCT_3(LocalCoordSystem[2], Old_Normal_Vector);
357 double New_projected_distance = New_projected_on_old * New_Dist;
358 double Old_projected_distance = New_projected_on_old * Old_dist;
359
360 if (New_projected_distance - Old_dist > -1.0e-6 * std::abs(Old_dist)) {//old has hierarchy over new //DO NOT SAVE NEW NEIGH
361 return false;
362 }
363
364 if (Old_projected_distance - New_Dist > -1.0e-6 * std::abs(New_Dist)) { //new has hierarchy over old
365
366 int old_ID = Id_Array[i_old_neigh];
367 if (new_ID == old_ID) {//SUBSTITUTE
368 position = i_old_neigh;
369 substitute = true;
370 }
371
372 else { //DISABLE THE OLD ONE
373 ContactTypes[i_old_neigh] = -1;
374 }
375
376 } //new has hierarchy over
377
378 }//Loop over Old Neigh
379
380 std::vector<DEMWall*>& neighbour_rigid_faces = rObj_1->mNeighbourRigidFaces;
381
382 if(!substitute) { //if substitute we wont resize or pushback
383 Distance_Array.resize(neighbor_size+1);
384 Weight_Array.resize(neighbor_size+1);
385 Normal_Array.resize(neighbor_size+1);
386 Id_Array.resize(neighbor_size+1);
387 ContactTypes.resize(neighbor_size+1);
388
389 neighbour_rigid_faces.push_back(rObj_2);
390 }
391
392 Normal_Array[position][0] = LocalCoordSystem[2][0];
393 Normal_Array[position][1] = LocalCoordSystem[2][1];
394 Normal_Array[position][2] = LocalCoordSystem[2][2];
395 Weight_Array[position][0] = Weight[0];
396 Weight_Array[position][1] = Weight[1];
397 Weight_Array[position][2] = Weight[2];
398 Weight_Array[position][3] = Weight[3];
399 Distance_Array[position] = DistPToB;
400
401 Id_Array[position] = new_ID;
402 ContactTypes[position] = ContactType;
403
404
405 return true;
406 }//DistanceHierarchy

425 DoubleHierarchyMethod2D(rObj_1,rObj_2, Distance_Array, Normal_Array,Weight_Array, Id_Array,ContactType_Array);

433 static inline void DoubleHierarchyMethod3D(SphericParticle* rObj_1,
434 DEMWall* rObj_2,
435 std::vector< double > & Distance_Array,
436 std::vector< array_1d<double,3> >& Normal_Array,
437 std::vector< array_1d<double,4> >& Weight_Array,
438 std::vector< int >& Id_Array,
439 std::vector< int >& ContactType_Array
440 )
441
442 {
443 //rObj_1 is particle, and rObj_2 is condition
444 //TYPE HIERARCHY
445 int ContactType = -1;
446 //-1: No contact;
447 // 1: Plane;
448 // 2: Edge;
449 // 3: Point.
450 // 4: Edge or Point
451
452 bool ContactExists = false;
453 const GeometryType& DE_Geom = rObj_1->GetGeometry();
454
455 double Radius = rObj_1->GetInteractionRadius();
456
457 const GeometryType& FE_Geom = rObj_2->GetGeometry();
458 unsigned int FE_size = FE_Geom.size();
459
460 double local_coord_system[3][3] = { {0.0},{0.0},{0.0} };
461 std::vector<double> Weight(4,0.0);
462
463 double distance_point_to_plane = 0.0;
464 unsigned int current_edge_index = 0;
465 bool inside = false;
466 ContactExists = GeometryFunctions::FacetCheck(FE_Geom, DE_Geom[0].Coordinates(), Radius, local_coord_system,
467 distance_point_to_plane, Weight, current_edge_index, inside);
468 if (ContactExists == true) {
469 ContactType = 1;
470 ContactExists = DistanceHierarchy(rObj_1,rObj_2, local_coord_system, distance_point_to_plane, Weight, ContactType, Distance_Array, Normal_Array,Weight_Array, Id_Array, ContactType_Array);
471 return;
472 }
473 //The key here is to see that we only need to check for further contact if, when not having contact with plane, the distance_point_to_plane is lower than the radius.
474 //In this case it might have contact with edges or vertices, otherwise no contact is possible.
475
476 //The check should avoid the edges which yielded a OUTSIDE value in the inside-outside check. i.e. we will check from the current edge to the last.
477
478 if (!ContactExists) {
479
481 if (distance_point_to_plane < Radius) {
482 bool local_contact_exists = false;
483 for (unsigned int e = current_edge_index; e < FE_size; e++) {
484 double eta = 0.5; // dummy initialize
485 double distance_point_to_edge = 2.0 * Radius; //dummy big initialization
486
487 local_contact_exists = GeometryFunctions::EdgeCheck(FE_Geom[e], FE_Geom[(e + 1) % FE_size], DE_Geom[0].Coordinates(), Radius, local_coord_system,
488 distance_point_to_edge, eta);
489 if (local_contact_exists) {
490 //save data
491 ContactType = 2;
492 Weight[e] = 1 - eta;
493 Weight[(e + 1) % FE_size] = eta; //the rest remain 0 (valid for triangles and quadrilateral)
494 if (FE_size > 4) { KRATOS_WATCH("WEIGHTS ALONG EDGE CANT BE CALCULATED WITH SUKUMAR FORMULAE") }
495 ContactExists = DistanceHierarchy(rObj_1, rObj_2, local_coord_system, distance_point_to_edge, Weight, ContactType, Distance_Array, Normal_Array, Weight_Array, Id_Array, ContactType_Array);
496 continue; //skip vertex check
497 }
498 if ((local_contact_exists == false) && (distance_point_to_edge < Radius)) {
499 unsigned int vertex_to_check = -1;
500 if (eta < 0.0) { vertex_to_check = e; }
501 else if (eta > 1.0) { vertex_to_check = (e + 1) % FE_size; }
502 else { continue; }
503 double distance_point_to_vertex = 0.0;
504 local_contact_exists = GeometryFunctions::VertexCheck(FE_Geom[vertex_to_check], DE_Geom[0].Coordinates(), Radius, local_coord_system, distance_point_to_vertex);
505
506 if (local_contact_exists) {
507 ContactType = 3;
508 Weight[vertex_to_check] = 1.0; //the rest weights stay 0.0;
509 ContactExists = DistanceHierarchy(rObj_1, rObj_2, local_coord_system, distance_point_to_vertex, Weight, ContactType, Distance_Array, Normal_Array, Weight_Array, Id_Array, ContactType_Array);
510 }
511 } // (ContactExists == false) && (distance_point_to_edge < Radius )
512 }//for every edge
513 }
514
515 // Add to non contact list the faces intersected by the normal direction between the particle and the FE
516 else {
517 if (inside)
518 rObj_1->mNeighbourNonContactRigidFaces.push_back(rObj_2);
519 }
520 }
521
522 return;
523 } //DoubleHierarchyMethod3D

526 static inline void DoubleHierarchyMethod2D(SphericParticle* rObj_1,
527 DEMWall* rObj_2,
528 std::vector< double > & Distance_Array,
529 std::vector< array_1d<double,3> >& Normal_Array,
530 std::vector< array_1d<double,4> >& Weight_Array,
531 std::vector< int > & Id_Array,
532 std::vector< int > & ContactType_Array
533 )
534 {
535 //rObj_1 is particle, and rObj_2 is condition
536 int ContactType = -1;
537 //-1: No contact;
538 // 2: Edge;
539 // 3: Vertex.
540
541 bool ContactExists = false;
542
543 const GeometryType& DE_Geom = rObj_1->GetGeometry();
544
545 double Radius = rObj_1->GetInteractionRadius();
546
547 const GeometryType& FE_Geom = rObj_2->GetGeometry();
548
549 double local_coord_system[3][3] = { {0.0},{0.0},{0.0} };
550 std::vector<double> Weight(4,0.0);
551 std::vector< array_1d<double,3> > Coord(2);
552
553 for (unsigned int i = 0; i<2; i++) {
554 for (unsigned int j = 0; j<3; j++) {
555 Coord[i][j] = FE_Geom[i].Coordinates()[j];
556 }
557 }
558
559 double eta = 0.5; // dummy initialize
560 double distance_point_to_edge = 2*Radius; //dummy big initialization
561
562 ContactExists = GeometryFunctions::EdgeCheck( Coord[0], Coord[1], DE_Geom[0].Coordinates(), Radius, local_coord_system,
563 distance_point_to_edge, eta);
564 if (ContactExists) { //save data
565 ContactType = 2;
566 Weight[0] = 1-eta;
567 Weight[1] = eta; //the rest remain 0 (valid for triangles and quadrilateral)
568
569 ContactExists = DistanceHierarchy(rObj_1,rObj_2, local_coord_system, distance_point_to_edge, Weight, ContactType, Distance_Array, Normal_Array, Weight_Array, Id_Array, ContactType_Array);
570 return;
571 }
572
573 if (ContactExists == false) {
574 if (distance_point_to_edge < Radius) {
575 unsigned int vertex_to_check = -1;
576 if (eta < 0.0) { vertex_to_check = 0; }
577 else if (eta > 1.0) { vertex_to_check = 1; }
578 double distance_point_to_vertex = 0.0;
579 ContactExists = GeometryFunctions::VertexCheck(Coord[vertex_to_check], DE_Geom[0].Coordinates(), Radius, local_coord_system, distance_point_to_vertex);
580
581 if (ContactExists) {
582 ContactType = 3;
583 Weight[vertex_to_check] = 1.0; //the rest weights stay 0.0;
584 ContactExists = DistanceHierarchy(rObj_1, rObj_2, local_coord_system, distance_point_to_vertex, Weight, ContactType, Distance_Array, Normal_Array, Weight_Array, Id_Array, ContactType_Array);
585 return;
586 }
587 }
588 else { // noncontact rigid face (distance_point_to_edge >= Radius)
589 if ((eta >= 0.0) && (eta <= 1.0))
590 rObj_1->mNeighbourNonContactRigidFaces.push_back(rObj_2);
591 }
592 }
593
594 return;
595 }//DoubleHierarchyMethod2D
