74 template <class TContainerType, class TContainerDataIO>
75 template<class TDataType>
76 void SpecializedContainerExpression<TContainerType, TContainerDataIO>::Evaluate(const Variable<TDataType>& rVariable)
77 {
78  KRATOS_TRY
79 
80  auto& r_container = this->GetContainer();
81  const IndexType number_of_entities = r_container.size();
82 
83  if (number_of_entities > 0) {
84  const auto& r_expression = this->GetExpression();
85 
86  VariableExpressionDataIO<TDataType> variable_flatten_data_io(r_expression.GetShape());
87 
88  // initialize the container variables first
89  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
90  // initializes ghost nodes as for the later synchronization
91  // only, the nodal non historical values needs to be set unless
92  // they are properly initialized. Otherwise, in synchronization, the variables will
93  // not be there in the ghost nodes hence seg faults.
94 
95  // the vectors and matrices needs to be initialized in historical and non-historical
96  // data containers because they need to be initialized with the correct size for synchronization
97  if constexpr(std::is_same_v<TDataType, Vector> || std::is_same_v<TDataType, Matrix>) {
98  TDataType dummy_value{};
99  variable_flatten_data_io.Assign(dummy_value, r_expression, 0);
100  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
101  VariableUtils().SetVariable(rVariable, dummy_value, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
102  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
103  VariableUtils().SetNonHistoricalVariable(rVariable, dummy_value, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
104  }
105  } else {
106  // if it is a static type, then it only needs to be initialized in the non-historical container with zeros.
107  // historical container should be initialized with the default values when the container is created.
108  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
109  VariableUtils().SetNonHistoricalVariableToZero(rVariable, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
110  }
111  }
112  }
113 
114  IndexPartition<IndexType>(number_of_entities).for_each(TDataType{}, [&r_container, &rVariable, &r_expression, &variable_flatten_data_io](const IndexType Index, TDataType& rValue){
115  variable_flatten_data_io.Assign(rValue, r_expression, Index);
116  TContainerDataIO::SetValue(*(r_container.begin() + Index), rVariable, rValue);
117  });
118 
119  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
120  // synchronize nodal values
121  auto& r_communicator = this->GetModelPart().GetCommunicator();
122 
123  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
124  r_communicator.SynchronizeVariable(rVariable);
125  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
126  r_communicator.SynchronizeNonHistoricalVariable(rVariable);
127  }
128  }
129  }
130 
131  KRATOS_CATCH("");
132 }
