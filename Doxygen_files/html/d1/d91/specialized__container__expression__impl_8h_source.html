54 {
55  KRATOS_TRY
56 
57  const auto& r_container = this->GetContainer();
58  const IndexType number_of_entities = r_container.size();
59 
60  using raw_data_type = std::conditional_t<
61  std::is_same_v<TDataType, char>, char,
62  std::conditional_t<
63  std::is_same_v<TDataType, int>, int,
64  double
65  >
66  >;
67 
68  if (number_of_entities != 0) {
69  // initialize the shape with the first entity value
70  VariableExpressionDataIO<TDataType> variable_flatten_data_io(TContainerDataIO::GetValue(*r_container.begin(), rVariable));
71 
72  auto p_expression = LiteralFlatExpression<raw_data_type>::Create(number_of_entities, variable_flatten_data_io.GetItemShape());
73  auto& r_expression = *p_expression;
74  this->mpExpression = p_expression;
75 
76  IndexPartition<IndexType>(number_of_entities).for_each([&r_container, &rVariable, &variable_flatten_data_io, &r_expression](const IndexType Index){
77  const auto& values = TContainerDataIO::GetValue(*(r_container.begin() + Index), rVariable);
78  variable_flatten_data_io.Read(r_expression, Index, values);
79  });
80  }
81 
82  KRATOS_CATCH("")
83 }

85 template <class TContainerType, class TContainerDataIO, class TMeshType>
86 template<class TDataType>
87 void SpecializedContainerExpression<TContainerType, TContainerDataIO, TMeshType>::Evaluate(const Variable<TDataType>& rVariable)
88 {
89  KRATOS_TRY
90 
91  auto& r_container = this->GetContainer();
92  const IndexType number_of_entities = r_container.size();
93 
94  if (number_of_entities > 0) {
95  const auto& r_expression = this->GetExpression();
96 
97  VariableExpressionDataIO<TDataType> variable_flatten_data_io(r_expression.GetItemShape());
98 
99  // initialize the container variables first
100  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
101  // initializes ghost nodes as for the later synchronization
102  // only, the nodal non historical values needs to be set unless
103  // they are properly initialized. Otherwise, in synchronization, the variables will
104  // not be there in the ghost nodes hence seg faults.
105 
106  // the vectors and matrices needs to be initialized in historical and non-historical
107  // data containers because they need to be initialized with the correct size for synchronization
108  if constexpr(std::is_same_v<TDataType, Vector> || std::is_same_v<TDataType, Matrix>) {
109  TDataType dummy_value{};
110  variable_flatten_data_io.Assign(dummy_value, r_expression, 0);
111  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
112  VariableUtils().SetVariable(rVariable, dummy_value, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
113  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
114  VariableUtils().SetNonHistoricalVariable(rVariable, dummy_value, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
115  }
116  } else {
117  // if it is a static type, then it only needs to be initialized in the non-historical container with zeros.
118  // historical container should be initialized with the default values when the container is created.
119  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
120  VariableUtils().SetNonHistoricalVariableToZero(rVariable, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
121  }
122  }
123  }
124 
125  IndexPartition<IndexType>(number_of_entities).for_each(TDataType{}, [&r_container, &rVariable, &r_expression, &variable_flatten_data_io](const IndexType Index, TDataType& rValue){
126  variable_flatten_data_io.Assign(rValue, r_expression, Index);
127  TContainerDataIO::SetValue(*(r_container.begin() + Index), rVariable, rValue);
128  });
129 
130  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
131  // synchronize nodal values
132  auto& r_communicator = this->GetModelPart().GetCommunicator();
133 
134  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
135  r_communicator.SynchronizeVariable(rVariable);
136  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
137  r_communicator.SynchronizeNonHistoricalVariable(rVariable);
138  }
139  }
140  }
141 
142  KRATOS_CATCH("");
143 }
