67 if (mrModelPart.NumberOfNodes() > 0) {
68 const Variable<double> &var = KratosComponents< Variable<double> >::Get(mVariableName);
69 const double Time = mrModelPart.GetProcessInfo()[TIME]/mTimeUnitConverter;
70 const double deltaH = mpTable->GetValue(Time);
71
72 Vector3 direction=ZeroVector(3);
73 direction[mGravityDirection] = 1.0;
74
75 block_for_each(mrModelPart.Nodes(), [&deltaH, &var, &direction, this](Node<3>& rNode){
76 double distance = 0.0;
77 double d = 0.0;
78 for (unsigned int j=0; j < rNode.Coordinates().size(); ++j) {
79 distance += mNormalVector[j]*rNode.Coordinates()[j];
80 d += mNormalVector[j]*direction[j];
81 }
82 distance = -(distance - mEqRHS) / d;
83 distance += deltaH;
84
85 const double pressure = mSpecificWeight * distance;
86
87 if (pressure > 0.0) {
88 rNode.FastGetSolutionStepValue(var) = pressure;
89 } else {
90 rNode.FastGetSolutionStepValue(var) = 0.0;
91 }
92 });
